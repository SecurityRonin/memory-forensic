1//! Windows Credential Manager credential extraction from LSASS memory.
2//!
3//! The Windows Credential Manager stores encrypted credential blobs in the
4//! user profile (`%APPDATA%\Microsoft\Credentials\`). When a user is logged
5//! in, the Credential Manager SSP in LSASS decrypts and caches these entries
6//! in a `_KIWI_CREDMAN_LIST_ENTRY` linked list in LSASS process memory.
7//!
8//! This walker reads those in-memory (already-decrypted) entries directly from
9//! the LSASS dump, recovering the plaintext credentials that were loaded for
10//! the active logon session.
11//!
12//! # Attribution
13//!
14//! Structure layout reverse-engineered and documented by:
15//!   Benjamin Delpy & Vincent Le Toux, Mimikatz `sekurlsa::credman` (MIT License)
16//!   <https://github.com/gentilkiwi/mimikatz>
17//!   Offsets independently re-implemented in Rust for read-only forensic analysis.
18//!
19//! # Forensic guarantee
20//!
21//! Read-only — no live process access, no Win32 API calls, no state modification.
22

28/// Maximum list entries to visit before aborting (cycle protection).
29const MAX_ITERATIONS: usize = 10_000;
30
31/// Walk the Credential Manager `_KIWI_CREDMAN_LIST_ENTRY` doubly-linked list
32/// in LSASS memory.
33///
34/// # Arguments
35/// * `reader` — object reader backed by the full physical dump
36/// * `lsass_cr3` — CR3 value of the LSASS process (page table root)
37/// * `credman_list_head_vaddr` — virtual address of the list head in LSASS VA space
38pub fn walk_credman<P: PhysicalMemoryProvider + Clone>(
39    reader: &ObjectReader<P>,
40    lsass_cr3: u64,
41    credman_list_head_vaddr: u64,
42) -> Result<Vec<CredManEntry>> {
43    let proc_reader = reader.with_cr3(lsass_cr3);
44    let vas = proc_reader.vas();
45
46    let mut ptr_buf = [0u8; 8];
47    vas.read_virt(credman_list_head_vaddr, &mut ptr_buf)
48        .map_err(crate::Error::Core)?;
49    let mut current = u64::from_le_bytes(ptr_buf);
50
51    let mut results = Vec::new();
52    let mut iterations = 0usize;
53
54    while current != credman_list_head_vaddr {
55        if iterations >= MAX_ITERATIONS {
56            break;
57        }
58        iterations += 1;
59
60        // Username at current+0x60
61        let username_bytes = read_unicode_string_raw(vas, current + 0x60).unwrap_or_default();
62        // Password at current+0x80
63        let password_bytes = read_unicode_string_raw(vas, current + 0x80).unwrap_or_default();
64        // Target/Server at current+0xA0
65        let target_bytes = read_unicode_string_raw(vas, current + 0xA0).unwrap_or_default();
66
67        if !username_bytes.is_empty() {
68            if let Some(username) = decode_utf16le_or_none(&username_bytes) {
69                if !username.is_empty() {
70                    let target = decode_utf16le_or_none(&target_bytes).unwrap_or_default();
71
72                    let (password, credential_blob) = if password_bytes.is_empty() {
73                        (None, None)
74                    } else if let Some(plain) = decode_utf16le_or_none(&password_bytes) {
75                        (Some(plain), None)
76                    } else {
77                        (None, Some(password_bytes))
78                    };
79
80                    results.push(CredManEntry {
81                        target,
82                        username,
83                        password,
84                        credential_blob,
85                    });
86                }
87            }
88        }
89
90        // Advance: read Flink at current+0x00
91        vas.read_virt(current, &mut ptr_buf)
92            .map_err(crate::Error::Core)?;
93        current = u64::from_le_bytes(ptr_buf);
94    }
95
96    Ok(results)
97}

101/// Read a `_UNICODE_STRING` at `us_vaddr` and return the raw byte buffer.
102///
103/// Layout (x64):
104///   +0x00  Length        u16   byte count of string data
105///   +0x02  MaximumLength u16
106///   +0x04  _padding      u32
107///   +0x08  Buffer        u64   pointer to UTF-16LE data
108fn read_unicode_string_raw<P: PhysicalMemoryProvider>(
109    vas: &memf_core::vas::VirtualAddressSpace<P>,
110    us_vaddr: u64,
111) -> Result<Vec<u8>> {
112    let mut len_buf = [0u8; 2];
113    vas.read_virt(us_vaddr, &mut len_buf)
114        .map_err(crate::Error::Core)?;
115    let len = u16::from_le_bytes(len_buf) as usize;
116
117    if len == 0 {
118        return Ok(Vec::new());
119    }
120
121    let mut ptr_buf = [0u8; 8];
122    vas.read_virt(us_vaddr + 8, &mut ptr_buf)
123        .map_err(crate::Error::Core)?;
124    let buf_ptr = u64::from_le_bytes(ptr_buf);
125
126    if buf_ptr == 0 {
127        return Ok(Vec::new());
128    }
129
130    let mut data = vec![0u8; len];
131    vas.read_virt(buf_ptr, &mut data)
132        .map_err(crate::Error::Core)?;
133    Ok(data)
134}

136/// Decode raw bytes as UTF-16LE if every code unit is a valid Unicode scalar.
137/// Returns `Some(String)` for plaintext, `None` for encrypted/binary data.
138fn decode_utf16le_or_none(bytes: &[u8]) -> Option<String> {
139    if bytes.len() < 2 || bytes.len() % 2 != 0 {
140        return None;
141    }
142    let units: Vec<u16> = bytes
143        .chunks_exact(2)
144        .map(|c| u16::from_le_bytes([c[0], c[1]]))
145        .collect();
146    if units.iter().any(|&u| (0xD800..=0xDFFF).contains(&u)) {
147        return None;
148    }
149    String::from_utf16(&units).ok()
150}

160    fn utf16le(s: &str) -> Vec<u8> {
161        s.encode_utf16().flat_map(u16::to_le_bytes).collect()
162    }
163
164    fn make_unicode_string(str_addr: u64, s: &str) -> (Vec<u8>, Vec<u8>) {
165        let str_bytes = utf16le(s);
166        let len = str_bytes.len() as u16;
167        let mut us = vec![0u8; 16];
168        us[0..2].copy_from_slice(&len.to_le_bytes());
169        us[2..4].copy_from_slice(&len.to_le_bytes());
170        us[8..16].copy_from_slice(&str_addr.to_le_bytes());
171        (us, str_bytes)
172    }

174    #[test]
175    fn walk_credman_empty_list_returns_nothing() {
176        let list_head: u64 = 0x1000;
177
178        let mut head_page = vec![0u8; 0x1000];
179        // Self-referencing list head (Flink = Blink = self)
180        head_page[0..8].copy_from_slice(&list_head.to_le_bytes());
181        head_page[8..16].copy_from_slice(&list_head.to_le_bytes());
182
183        let head_paddr: u64 = 0x0010_0000;
184
185        let (cr3, mem) = PageTableBuilder::new()
186            .map_4k(list_head, head_paddr, flags::WRITABLE)
187            .write_phys(head_paddr, &head_page)
188            .build();
189
190        let isf = IsfBuilder::new().build_json();
191        let resolver = IsfResolver::from_value(&isf).expect("valid ISF");
192        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
193        let reader = ObjectReader::new(vas, Box::new(resolver));
194
195        let result = walk_credman(&reader, cr3, list_head).unwrap();
196        assert!(result.is_empty());
197    }

199    #[test]
200    fn walk_credman_one_entry() {
201        let list_head: u64 = 0x1000;
202        let entry_addr: u64 = 0x2000;
203        let username_str: u64 = 0x3000;
204        let password_str: u64 = 0x3100;
205        let target_str: u64 = 0x3200;
206
207        let (us_username, username_bytes) = make_unicode_string(username_str, "jdoe");
208        let (us_password, password_bytes) = make_unicode_string(password_str, "Vault!Pass1");
209        let (us_target, target_bytes) = make_unicode_string(target_str, "https://intranet.corp");
210
211        // Page 0x1000: list head — Flink → entry
212        let mut head_page = vec![0u8; 0x1000];
213        head_page[0..8].copy_from_slice(&entry_addr.to_le_bytes());
214        head_page[8..16].copy_from_slice(&entry_addr.to_le_bytes());
215
216        // Page 0x2000: the _KIWI_CREDMAN_LIST_ENTRY
217        let mut entry_page = vec![0u8; 0x1000];
218        // Flink at +0x00 → back to list_head (terminates walk)
219        entry_page[0x00..0x08].copy_from_slice(&list_head.to_le_bytes());
220        entry_page[0x08..0x10].copy_from_slice(&list_head.to_le_bytes());
221        // Username UNICODE_STRING at +0x60
222        entry_page[0x60..0x70].copy_from_slice(&us_username);
223        // Password UNICODE_STRING at +0x80
224        entry_page[0x80..0x90].copy_from_slice(&us_password);
225        // Target UNICODE_STRING at +0xA0
226        entry_page[0xA0..0xB0].copy_from_slice(&us_target);
227
228        // Page 0x3000: string data
229        let mut str_page = vec![0u8; 0x1000];
230        str_page[0x000..username_bytes.len()].copy_from_slice(&username_bytes);
231        str_page[0x100..0x100 + password_bytes.len()].copy_from_slice(&password_bytes);
232        str_page[0x200..0x200 + target_bytes.len()].copy_from_slice(&target_bytes);
233
234        let head_paddr: u64 = 0x0010_0000;
235        let entry_paddr: u64 = 0x0020_0000;
236        let str_paddr: u64 = 0x0030_0000;
237
238        let (cr3, mem) = PageTableBuilder::new()
239            .map_4k(0x1000, head_paddr, flags::WRITABLE)
240            .write_phys(head_paddr, &head_page)
241            .map_4k(0x2000, entry_paddr, flags::WRITABLE)
242            .write_phys(entry_paddr, &entry_page)
243            .map_4k(0x3000, str_paddr, flags::WRITABLE)
244            .write_phys(str_paddr, &str_page)
245            .build();
246
247        let isf = IsfBuilder::new().build_json();
248        let resolver = IsfResolver::from_value(&isf).expect("valid ISF");
249        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
250        let reader = ObjectReader::new(vas, Box::new(resolver));
251
252        let result = walk_credman(&reader, cr3, list_head).unwrap();
253        assert_eq!(result.len(), 1);
254        assert_eq!(result[0].username, "jdoe");
255        assert_eq!(result[0].password.as_deref(), Some("Vault!Pass1"));
256        assert_eq!(result[0].target, "https://intranet.corp");
257        assert!(result[0].credential_blob.is_none());
258    }
