1//! Browser credential forensics walker.
2//!
3//! Scans Chromium-based browser root process heap regions for plaintext
4//! credential records that the browser's password manager writes to committed,
5//! writeable heap pages.
6//!
7//! Supported browsers (all Chromium-based, share the same credential layout):
8//! `msedge.exe`, `chrome.exe`, `brave.exe`, `opera.exe`, `vivaldi.exe`,
9//! `chromium.exe`.
10//!
11//! # Memory layout
12//!
13//! The browser's credential manager transiently materialises saved passwords in
14//! the form:
15//!
16//! ```text
17//! <alpha-prefix>https?<SP><username><SP><password><SP><NUL>
18//! ```
19//!
20//! Each credential record is typically preceded (in the same region) by a URL
21//! record of the form:
22//!
23//! ```text
24//! <NUL><NUL><NUL><url-chars>https?<SP><username><SP><password>
25//! ```
26//!
27//! # Attribution
28//!
29//! Technique originally documented and demonstrated (C#) by:
30//!   L1v1ng0ffTh3L4N, "EdgeSavedPasswordsDumper" (2024)
31//!   <https://github.com/L1v1ng0ffTh3L4N/EdgeSavedPasswordsDumper>
32//!   (MIT License; regex patterns independently re-implemented in Rust for
33//!   read-only forensic analysis of memory dumps)
34//!
35//! # Forensic guarantee
36//!
37//! This walker only reads bytes from a memory dump — it opens no live process,
38//! calls no Win32 API, and modifies no state.
39

48static CRED_RE: OnceLock<regex::Regex> = OnceLock::new();
49
50fn cred_re() -> &'static regex::Regex {
51    CRED_RE.get_or_init(|| {
52        // Single compile-time-constant pattern in a OnceLock initializer; the
53        // expect enforces a programmer contract (the literal is known-valid),
54        // not trust in untrusted input — hence the targeted allow.
55        #[allow(clippy::expect_used)]
56        regex::Regex::new(
57            r"(?-u)[a-zA-Z]https?[ ]([a-zA-Z0-9\-_\.@?]{3,20})[ ]([a-zA-Z0-9!#$%^&*()\-+=\[\]{};:<>?/~\t ]{6,40})[ ]\x00",
58        )
59        .expect("cred_re is a valid compile-time pattern")
60    })
61}

63/// Chromium-based browser process names whose credential layout is supported.
64///
65/// All entries share the same Chromium password manager codebase and therefore
66/// the same in-memory credential record format. Root-process detection is
67/// performed per entry so child processes of one browser are not confused with
68/// those of another.
69pub const CHROMIUM_BROWSERS: &[&str] = &[
70    "msedge.exe",
71    "chrome.exe",
72    "brave.exe",
73    "opera.exe",
74    "vivaldi.exe",
75    "chromium.exe",
76];
77
78/// Walk committed, writeable heap regions of the root Chromium-based browser
79/// process(es) in the dump and extract any plaintext credential records.
80///
81/// # Root process selection
82///
83/// Chromium spawns many child processes under the same executable name.
84/// Saved passwords reside only in the **root** (browser) process — the one
85/// whose parent PID does not belong to another process of the same image name.
86/// Root detection is performed independently for each browser in
87/// [`CHROMIUM_BROWSERS`], so multiple browsers can coexist in the same dump.
88///
89/// # Arguments
90///
91/// * `reader` — kernel-space `ObjectReader` (uses kernel CR3 / symbol table).
92/// * `ps_head_vaddr` — virtual address of `PsActiveProcessHead`.
93pub fn walk_browser_credentials<P: PhysicalMemoryProvider + Clone>(
94    reader: &ObjectReader<P>,
95    ps_head_vaddr: u64,
96) -> Result<Vec<BrowserCredentialInfo>> {
97    // Build root PID set: for each browser, root = process whose ppid is not in
98    // the same browser's PID set.
99    let procs = crate::process::walk_processes(reader, ps_head_vaddr)?;
100    let mut root_pids = HashSet::new();
101    for &browser in CHROMIUM_BROWSERS {
102        let browser_pids: HashSet<u64> = procs
103            .iter()
104            .filter(|p| p.image_name.eq_ignore_ascii_case(browser))
105            .map(|p| p.pid)
106            .collect();
107        for p in procs.iter().filter(|p| {
108            p.image_name.eq_ignore_ascii_case(browser) && !browser_pids.contains(&p.ppid)
109        }) {
110            root_pids.insert(p.pid);
111        }
112    }
113
114    // for_each_heap_region internally calls walk_processes again (acceptable double-call
115    // for read-only forensic analysis).
116    let wr = crate::heap_walker::for_each_heap_region(
117        reader,
118        ps_head_vaddr,
119        |proc| root_pids.contains(&proc.pid),
120        |bytes, proc| {
121            scan_region(bytes)
122                .into_iter()
123                .map(|(url, username, password)| BrowserCredentialInfo {
124                    pid: proc.pid,
125                    image_name: proc.image_name.clone(),
126                    url,
127                    username,
128                    password,
129                })
130                .collect()
131        },
132        |info: &BrowserCredentialInfo| (info.pid, info.username.clone(), info.password.clone()),
133    )?;
134    Ok(wr.items)
135}

137/// Scan raw bytes from a memory region for Edge credential records.
138///
139/// Returns `(url, username, password)` tuples. `url` is empty when the URL
140/// pattern did not match in the same buffer.
141///
142/// # Pattern source
143///
144/// Adapted from the regex in:
145///   L1v1ng0ffTh3L4N, "EdgeSavedPasswordsDumper"
146///   <https://github.com/L1v1ng0ffTh3L4N/EdgeSavedPasswordsDumper>
147pub(crate) fn scan_region(data: &[u8]) -> Vec<(String, String, String)> {
148    // Convert to UTF-8 with lossy replacement; NUL bytes are preserved as-is.
149    let text = String::from_utf8_lossy(data);
150
151    // Credential record pattern:
152    //   <alpha>https?<SP><username>{3,20}<SP><password>{6,40}<SP><NUL>
153    let cred_re = cred_re();
154
155    let mut out = Vec::new();
156
157    for caps in cred_re.captures_iter(&text) {
158        let username = caps[1].to_string();
159        let password = caps[2].trim_end().to_string();
160
161        // URL record pattern:
162        //   <NUL><NUL><NUL><url_chars>https?<SP><username><SP><password>
163        let url_pat = format!(
164            r"(?-u)\x00\x00\x00([A-Za-z0-9\-._~:/?#\[\]@!$&'()*+,;=%]+)https?[ ]{}[ ]{}",
165            regex::escape(&username),
166            regex::escape(&password),
167        );
168
169        let url = regex::Regex::new(&url_pat)
170            .ok()
171            .and_then(|re| re.captures(&text))
172            .map(|m| m[1].to_string())
173            .unwrap_or_default();
174
175        out.push((url, username, password));
176    }
177
178    out
179}

181#[cfg(test)]
182mod tests {
183    use super::*;
184
185    #[test]
186    fn chromium_browsers_includes_all_supported() {
187        assert!(CHROMIUM_BROWSERS.contains(&"msedge.exe"));
188        assert!(CHROMIUM_BROWSERS.contains(&"chrome.exe"));
189        assert!(CHROMIUM_BROWSERS.contains(&"brave.exe"));
190        assert!(CHROMIUM_BROWSERS.contains(&"opera.exe"));
191        assert!(CHROMIUM_BROWSERS.contains(&"vivaldi.exe"));
192        assert!(CHROMIUM_BROWSERS.contains(&"chromium.exe"));
193    }
194
195    #[test]
196    fn scan_region_empty_returns_nothing() {
197        assert!(scan_region(b"").is_empty());
198    }
199
200    #[test]
201    fn scan_region_no_pattern_returns_nothing() {
202        assert!(scan_region(b"hello world this is random text with no credentials").is_empty());
203    }
204
205    #[test]
206    fn scan_region_extracts_username_and_password() {
207        // Minimal credential record: B + https + SP + username + SP + password + SP + NUL
208        let mut buf = Vec::new();
209        buf.extend_from_slice(b"Bhttps admin Secret123! \x00");
210        let results = scan_region(&buf);
211        assert_eq!(results.len(), 1, "expected 1 credential match");
212        let (url, username, password) = &results[0];
213        assert_eq!(username, "admin");
214        assert_eq!(password, "Secret123!");
215        assert_eq!(url, "", "no URL pattern in buffer");
216    }
217
218    #[test]
219    fn scan_region_extracts_url_when_present() {
220        // URL record immediately before credential record:
221        //   NUL NUL NUL <url_chars> https SP <username> SP <password>
222        //   <alpha> https SP <username> SP <password> SP NUL
223        let mut buf = Vec::new();
224        buf.extend_from_slice(b"\x00\x00\x00example.com/https admin MyPass1! ");
225        buf.extend_from_slice(b"Bhttps admin MyPass1! \x00");
226        let results = scan_region(&buf);
227        assert_eq!(results.len(), 1);
228        let (url, username, password) = &results[0];
229        assert_eq!(username, "admin");
230        assert_eq!(password, "MyPass1!");
231        assert_eq!(url, "example.com/");
232    }
233
234    #[test]
235    fn scan_region_multiple_credentials() {
236        let mut buf = Vec::new();
237        buf.extend_from_slice(b"Bhttps alice Passw0rd! \x00");
238        buf.extend_from_slice(b"Bhttps bob Hunter2!! \x00");
239        let results = scan_region(&buf);
240        assert_eq!(results.len(), 2);
241        let usernames: Vec<_> = results.iter().map(|(_, u, _)| u.as_str()).collect();
242        assert!(usernames.contains(&"alice"));
243        assert!(usernames.contains(&"bob"));
244    }
245
246    #[test]
247    fn scan_region_is_deterministic() {
248        // Regression guard: sequential calls must return identical results.
249        let buf = b"Bhttps admin Secret123! \x00";
250        let r1 = scan_region(buf);
251        let r2 = scan_region(buf);
252        assert_eq!(
253            r1.len(),
254            r2.len(),
255            "second call must return same number of results"
256        );
257        if !r1.is_empty() {
258            assert_eq!(r1[0].1, r2[0].1, "username must be identical across calls");
259            assert_eq!(r1[0].2, r2[0].2, "password must be identical across calls");
260        }
261    }
