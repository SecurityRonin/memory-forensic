1//! TLS (Thread Local Storage) callback validation walker.
2//!
3//! Walks the TLS directory of each loaded PE image to enumerate TLS
4//! callback function pointers and detect those that resolve outside the
5//! module's mapped range — a sign of callback hijacking.
6//! MITRE ATT&CK T1055 / T1106.
7
8use memf_core::object_reader::ObjectReader;
9use memf_format::PhysicalMemoryProvider;
10
11use crate::{types::TlsCallbackInfo, Result};
12
13/// Classify whether a TLS callback virtual address is unbacked — i.e. it does
14/// not fall within any known loaded module's address range.
15///
16/// Returns `true` if `callback_va` is not covered by any `(start, end)` range
17/// in `module_ranges`, indicating the callback has been redirected to
18/// injected or shellcode memory.
19pub fn is_unbacked_tls_callback(callback_va: u64, module_ranges: &[(u64, u64)]) -> bool {
20    !module_ranges
21        .iter()
22        .any(|&(start, end)| callback_va >= start && callback_va < end)
23}
24
25/// Walk TLS directories of all loaded modules in all processes and validate
26/// callback addresses.
27///
28/// For each `_IMAGE_TLS_DIRECTORY` found, extracts the callback array and
29/// checks that each entry falls within `[module_base, module_base + size)`.
30/// Callbacks that resolve outside this range are flagged as
31/// `is_outside_module = true`.
32///
33/// # MITRE ATT&CK
34/// T1055 — Process Injection (TLS callback abuse)
35pub fn scan_tls_callbacks<P: PhysicalMemoryProvider>(
36    reader: &ObjectReader<P>,
37) -> Result<Vec<TlsCallbackInfo>> {
38    let _ = reader;
39    Ok(vec![])
40}

50    fn make_minimal_reader(
51    ) -> ObjectReader<memf_core::test_builders::SyntheticPhysMem> {
52        let isf = IsfBuilder::new().build_json();
53        let resolver = IsfResolver::from_value(&isf).unwrap();
54        let (cr3, mem) = PageTableBuilder::new().build();
55        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
56        ObjectReader::new(vas, Box::new(resolver))
57    }
58
59    #[test]
60    fn empty_memory_returns_ok_empty() {
61        let reader = make_minimal_reader();
62        let result = scan_tls_callbacks(&reader);
63        assert!(result.is_ok(), "should succeed with minimal reader");
64        assert!(
65            result.unwrap().is_empty(),
66            "empty memory → no TLS callback hits"
67        );
68    }
69
70    #[test]
71    fn result_is_vec_of_tls_callback_info() {
72        let reader = make_minimal_reader();
73        let result: Result<Vec<TlsCallbackInfo>> = scan_tls_callbacks(&reader);
74        assert!(result.is_ok());
75    }
