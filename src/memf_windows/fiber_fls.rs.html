1//! Fiber and Fiber Local Storage (FLS) abuse detection — MITRE ATT&CK T1055.
2//!
3//! Detects threads that have been converted to fibers and FLS callback
4//! pointers that resolve outside loaded module ranges.
5
6use memf_core::object_reader::ObjectReader;
7use memf_format::PhysicalMemoryProvider;
8
9use crate::{types::FiberInfo, Result};

11/// Classify whether a fiber's saved instruction pointer (`RIP`) points outside
12/// all known loaded module address ranges.
13///
14/// Returns `true` if `rip` does not fall within any `(start, end)` range in
15/// `module_ranges`, indicating the fiber is executing injected or shellcode
16/// that is not backed by a mapped module.
17pub fn is_suspicious_fiber_rip(rip: u64, module_ranges: &[(u64, u64)]) -> bool {
18    !module_ranges
19        .iter()
20        .any(|&(start, end)| rip >= start && rip < end)
21}
22
23/// Scan all threads for fiber conversion and suspicious FLS callbacks.
24///
25/// Walks `_ETHREAD` structures looking for threads whose `Fiber` flag is
26/// set (indicating a call to `ConvertThreadToFiber`). Extracts the saved
27/// fiber context `RIP` and checks each FLS callback slot against the set
28/// of loaded module address ranges, flagging any that fall outside.
29///
30/// # MITRE ATT&CK
31/// T1055 — Process Injection (Fiber / FLS sub-technique)
32pub fn scan_fiber_fls<P: PhysicalMemoryProvider>(
33    reader: &ObjectReader<P>,
34) -> Result<Vec<FiberInfo>> {
35    let _ = reader;
36    Ok(vec![])
37}

47    fn make_minimal_reader(
48    ) -> ObjectReader<memf_core::test_builders::SyntheticPhysMem> {
49        let isf = IsfBuilder::new().build_json();
50        let resolver = IsfResolver::from_value(&isf).unwrap();
51        let (cr3, mem) = PageTableBuilder::new().build();
52        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
53        ObjectReader::new(vas, Box::new(resolver))
54    }

56    #[test]
57    fn empty_memory_returns_ok_empty() {
58        let reader = make_minimal_reader();
59        let result = scan_fiber_fls(&reader);
60        assert!(result.is_ok(), "should succeed with minimal reader");
61        assert!(result.unwrap().is_empty(), "empty memory → no fiber/FLS hits");
62    }
