1//! Windows UNICODE_STRING reader.
2
3use memf_core::object_reader::ObjectReader;
4use memf_format::PhysicalMemoryProvider;
5
6/// Read a Windows `_UNICODE_STRING` at the given virtual address.
7///
8/// Reads `Length` (u16 at offset 0) and `Buffer` (pointer at offset 8),
9/// then reads `Length` bytes of UTF-16LE from `Buffer` and converts to
10/// a Rust `String`.
11pub fn read_unicode_string<P: PhysicalMemoryProvider>(
12    reader: &ObjectReader<P>,
13    ustr_vaddr: u64,
14) -> crate::Result<String> {
15    // Read Length (u16 at offset 0 of _UNICODE_STRING)
16    let length: u16 = reader.read_field(ustr_vaddr, "_UNICODE_STRING", "Length")?;
17
18    // Read Buffer pointer (u64 at offset 8 of _UNICODE_STRING)
19    let buffer: u64 = reader.read_field(ustr_vaddr, "_UNICODE_STRING", "Buffer")?;
20
21    // Empty string: Length == 0 or null Buffer pointer
22    if length == 0 || buffer == 0 {
23        return Ok(String::new());
24    }
25
26    // Read `length` bytes of UTF-16LE data from the Buffer virtual address
27    let raw = reader.read_bytes(buffer, length as usize)?;
28
29    // Convert pairs of bytes to u16 (little-endian)
30    let u16_vec: Vec<u16> = raw
31        .chunks_exact(2)
32        .map(|pair| u16::from_le_bytes([pair[0], pair[1]]))
33        .collect();
34
35    // Convert to Rust String, lossy for any invalid surrogates
36    let s = String::from_utf16_lossy(&u16_vec);
37
38    // Trim trailing null characters
39    Ok(s.trim_end_matches('\0').to_string())
40}

50    /// Encode a Rust &str as UTF-16LE bytes.
51    fn utf16le_bytes(s: &str) -> Vec<u8> {
52        s.encode_utf16().flat_map(u16::to_le_bytes).collect()
53    }
54
55    /// Build a _UNICODE_STRING struct in memory (16 bytes):
56    /// [0..2]: Length (u16 LE) — number of bytes, not chars
57    /// [2..4]: MaximumLength (u16 LE)
58    /// [8..16]: Buffer (u64 LE pointer to UTF-16LE data)
59    fn build_unicode_string_bytes(length: u16, max_length: u16, buffer_ptr: u64) -> Vec<u8> {
60        let mut data = vec![0u8; 16];
61        data[0..2].copy_from_slice(&length.to_le_bytes());
62        data[2..4].copy_from_slice(&max_length.to_le_bytes());
63        data[8..16].copy_from_slice(&buffer_ptr.to_le_bytes());
64        data
65    }
66
67    fn make_unicode_reader(data: &[u8], vaddr: u64, paddr: u64) -> ObjectReader<SyntheticPhysMem> {
68        let isf = IsfBuilder::windows_kernel_preset().build_json();
69        let resolver = IsfResolver::from_value(&isf).unwrap();
70        let (cr3, mem) = PageTableBuilder::new()
71            .map_4k(vaddr, paddr, flags::WRITABLE)
72            .write_phys(paddr, data)
73            .build();
74        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
75        ObjectReader::new(vas, Box::new(resolver))
76    }

78    #[test]
79    fn read_simple_unicode_string() {
80        // "ntoskrnl.exe" — 12 chars, 24 bytes as UTF-16LE
81        let text = "ntoskrnl.exe";
82        let utf16 = utf16le_bytes(text);
83        let length = utf16.len() as u16; // 24
84
85        // Layout in one 4KB page at paddr 0x10_0000, vaddr 0xFFFF_8000_0000_0000:
86        // offset 0..16: _UNICODE_STRING struct
87        // offset 256..256+24: UTF-16LE string data
88        let vaddr_base = 0xFFFF_8000_0000_0000u64;
89        let paddr_base = 0x10_0000u64;
90        let string_data_offset = 256u64;
91        let buffer_vaddr = vaddr_base + string_data_offset;
92
93        let ustr_bytes = build_unicode_string_bytes(length, length, buffer_vaddr);
94
95        // Build the page data: unicode_string struct at start, UTF-16 data at offset 256
96        let mut page_data = vec![0u8; 4096];
97        page_data[0..16].copy_from_slice(&ustr_bytes);
98        page_data[string_data_offset as usize..string_data_offset as usize + utf16.len()]
99            .copy_from_slice(&utf16);
100
101        let reader = make_unicode_reader(&page_data, vaddr_base, paddr_base);
102        let result = read_unicode_string(&reader, vaddr_base).unwrap();
103        assert_eq!(result, "ntoskrnl.exe");
104    }
