1//! Windows process/module memory extraction.
2//!
3//! Provides raw-byte extraction of PE images from live or snapshot memory,
4//! reconstruction of on-disk PE layout from in-memory layout, and enumeration
5//! of memory-mapped (non-private) VAD regions.
6
7use crate::types::{WinDllInfo, WinDriverInfo, WinProcessInfo, WinVadInfo};
8use memf_core::object_reader::ObjectReader;
9use memf_format::PhysicalMemoryProvider;
10
11/// A dumped module or process image extracted from memory.
12#[derive(Debug, Clone)]
13pub struct ModuleDump {
14    /// Base name of the module (e.g. `"ntdll.dll"`).
15    pub name: String,
16    /// Virtual base address where the image is loaded.
17    pub base_addr: u64,
18    /// Raw bytes read directly from virtual memory (in-memory layout).
19    pub raw_bytes: Vec<u8>,
20    /// Reconstructed on-disk PE layout, if reconstruction succeeded.
21    pub reconstructed: Option<Vec<u8>>,
22}
23
24/// A memory-mapped file region extracted from VAD information.
25#[derive(Debug, Clone)]
26pub struct MappedFileRegion {
27    /// Start virtual address of the mapped region.
28    pub start_vaddr: u64,
29    /// End virtual address of the mapped region (inclusive last byte).
30    pub end_vaddr: u64,
31    /// File path backing this mapping (empty if unavailable).
32    pub file_path: String,
33    /// Page protection value.
34    pub protection: u32,
35}
36
37/// Abstraction over a memory source for testability.
38pub(crate) trait MemReader {
39    /// Read `len` bytes starting at virtual address `vaddr`.
40    fn read_region(&self, vaddr: u64, len: usize) -> crate::Result<Vec<u8>>;
41}
42
43impl<P: PhysicalMemoryProvider> MemReader for ObjectReader<P> {
44    fn read_region(&self, vaddr: u64, len: usize) -> crate::Result<Vec<u8>> {
45        Ok(self.read_bytes(vaddr, len)?)
46    }
47}

49/// Read `size` bytes from virtual address `vaddr` using the given reader.
50pub fn dump_memory_region<P: PhysicalMemoryProvider>(
51    reader: &ObjectReader<P>,
52    vaddr: u64,
53    size: usize,
54) -> crate::Result<Vec<u8>> {
55    dump_region_inner(reader, vaddr, size)
56}
57
58/// Inner implementation of region dumping, testable via `MemReader`.
59pub(crate) fn dump_region_inner(
60    reader: &impl MemReader,
61    vaddr: u64,
62    size: usize,
63) -> crate::Result<Vec<u8>> {
64    reader.read_region(vaddr, size)
65}

67/// Dump a loaded DLL from process virtual memory.
68pub fn moddump<P: PhysicalMemoryProvider>(
69    reader: &ObjectReader<P>,
70    dll: &WinDllInfo,
71) -> crate::Result<ModuleDump> {
72    moddump_inner(reader, &dll.name, dll.base_addr, dll.size as usize)
73}

75/// Dump a loaded kernel driver from kernel virtual memory.
76pub fn moddump_driver<P: PhysicalMemoryProvider>(
77    reader: &ObjectReader<P>,
78    drv: &WinDriverInfo,
79) -> crate::Result<ModuleDump> {
80    moddump_inner(reader, &drv.name, drv.base_addr, drv.size as usize)

83/// Inner implementation of module dumping.
84pub(crate) fn moddump_inner(
85    reader: &impl MemReader,
86    name: &str,
87    base_addr: u64,
88    size: usize,
89) -> crate::Result<ModuleDump> {
90    let raw_bytes = reader.read_region(base_addr, size)?;
91    let reconstructed = reconstruct_pe(&raw_bytes).ok();
92    Ok(ModuleDump {
93        name: name.to_string(),
94        base_addr,
95        raw_bytes,
96        reconstructed,
97    })
98}

100/// Dump all memory regions belonging to a process (minidump-style).
101pub fn procdump<P: PhysicalMemoryProvider>(
102    reader: &ObjectReader<P>,
103    process: &WinProcessInfo,
104    dlls: &[WinDllInfo],
105) -> crate::Result<ModuleDump> {
106    procdump_inner(reader, process, dlls)
107}
108
109/// Inner implementation of process dumping.
110pub(crate) fn procdump_inner(
111    reader: &impl MemReader,
112    process: &WinProcessInfo,
113    dlls: &[WinDllInfo],
114) -> crate::Result<ModuleDump> {
115    let exe = dlls
116        .iter()
117        .min_by_key(|d| d.load_order)
118        .ok_or_else(|| crate::Error::WalkFailed {
119            walker: "procdump",
120            reason: format!("no DLLs found for PID {}", process.pid),
121        })?;
122    moddump_inner(reader, &exe.name, exe.base_addr, exe.size as usize)
123}

125/// Enumerate all memory-mapped (non-private) VAD regions.
126///
127/// Private VADs (anonymous allocations) are excluded; only file-backed
128/// mappings are returned. The `file_path` field is left empty because
129/// path resolution requires additional kernel symbol walking.
130pub fn list_mapped_files(vads: &[WinVadInfo]) -> Vec<MappedFileRegion> {
131    vads.iter()
132        .filter(|v| !v.is_private)
133        .map(|v| MappedFileRegion {
134            start_vaddr: v.start_vaddr,
135            end_vaddr: v.end_vaddr,
136            file_path: String::new(),
137            protection: v.protection,
138        })
139        .collect()
140}

142/// Reconstruct an on-disk PE layout from an in-memory PE image.
143///
144/// In-memory PEs have sections mapped to their `VirtualAddress` offsets;
145/// on-disk PEs place sections at their `PointerToRawData` offsets.
146/// This function copies headers verbatim and remaps each section.
147///
148/// # Errors
149///
150/// Returns `WalkFailed` if the input is not a valid PE image.
151pub fn reconstruct_pe(in_memory: &[u8]) -> crate::Result<Vec<u8>> {
152    if in_memory.len() < 0x40 || &in_memory[0..2] != b"MZ" {
153        return Err(crate::Error::WalkFailed {
154            walker: "reconstruct_pe",
155            reason: "not a PE image (no MZ header)".into(),
156        });
157    }
158    let e_lfanew = u32::from_le_bytes([
159        in_memory[0x3C],
160        in_memory[0x3D],
161        in_memory[0x3E],
162        in_memory[0x3F],
163    ]) as usize;
164    if e_lfanew + 24 > in_memory.len() || &in_memory[e_lfanew..e_lfanew + 4] != b"PE\0\0" {
165        return Err(crate::Error::WalkFailed {
166            walker: "reconstruct_pe",
167            reason: "not a valid PE (missing PE signature)".into(),
168        });
169    }
170
171    let num_sections =
172        u16::from_le_bytes([in_memory[e_lfanew + 6], in_memory[e_lfanew + 7]]) as usize;
173    let opt_size =
174        u16::from_le_bytes([in_memory[e_lfanew + 20], in_memory[e_lfanew + 21]]) as usize;
175    let sh_offset = e_lfanew + 24 + opt_size;
176
177    if sh_offset + num_sections * 40 > in_memory.len() {
178        return Err(crate::Error::WalkFailed {
179            walker: "reconstruct_pe",
180            reason: "section headers out of bounds".into(),
181        });
182    }
183
184    // Determine output size from section file extents
185    let mut out_size = sh_offset + num_sections * 40;
186    for i in 0..num_sections {
187        let s = sh_offset + i * 40;
188        let ptr_raw = u32::from_le_bytes([
189            in_memory[s + 20],
190            in_memory[s + 21],
191            in_memory[s + 22],
192            in_memory[s + 23],
193        ]) as usize;
194        let raw_sz = u32::from_le_bytes([
195            in_memory[s + 16],
196            in_memory[s + 17],
197            in_memory[s + 18],
198            in_memory[s + 19],
199        ]) as usize;
200        out_size = out_size.max(ptr_raw.saturating_add(raw_sz));
201    }
202
203    // Copy headers verbatim
204    let header_sz = (sh_offset + num_sections * 40).min(in_memory.len());
205    let mut out = vec![0u8; out_size];
206    out[..header_sz].copy_from_slice(&in_memory[..header_sz]);
207
208    // Copy each section from VirtualAddress in source to PointerToRawData in output
209    for i in 0..num_sections {
210        let s = sh_offset + i * 40;
211        let virt_addr = u32::from_le_bytes([
212            in_memory[s + 12],
213            in_memory[s + 13],
214            in_memory[s + 14],
215            in_memory[s + 15],
216        ]) as usize;
217        let virt_size = u32::from_le_bytes([
218            in_memory[s + 8],
219            in_memory[s + 9],
220            in_memory[s + 10],
221            in_memory[s + 11],
222        ]) as usize;
223        let ptr_raw = u32::from_le_bytes([
224            in_memory[s + 20],
225            in_memory[s + 21],
226            in_memory[s + 22],
227            in_memory[s + 23],
228        ]) as usize;
229        let raw_sz = u32::from_le_bytes([
230            in_memory[s + 16],
231            in_memory[s + 17],
232            in_memory[s + 18],
233            in_memory[s + 19],
234        ]) as usize;
235
236        if ptr_raw == 0 || raw_sz == 0 {
237            continue;
238        }
239        let copy_len = virt_size.min(raw_sz);
240        let src_end = virt_addr.saturating_add(copy_len).min(in_memory.len());
241        let actual = src_end.saturating_sub(virt_addr);
242        if actual > 0 && ptr_raw + actual <= out.len() {
243            out[ptr_raw..ptr_raw + actual]
244                .copy_from_slice(&in_memory[virt_addr..virt_addr + actual]);
245        }
246    }
247
248    Ok(out)
249}

265            Self { base, data }
266        }

269    impl MemReader for FakeReader {
270        fn read_region(&self, vaddr: u64, len: usize) -> crate::Result<Vec<u8>> {
271            if vaddr < self.base {
272                return Ok(vec![0u8; len]);
273            }
274            let off = (vaddr - self.base) as usize;
275            if off >= self.data.len() {
276                return Ok(vec![0u8; len]);
277            }
278            let n = len.min(self.data.len() - off);
279            let mut out = vec![0u8; len];
280            out[..n].copy_from_slice(&self.data[off..off + n]);
281            Ok(out)
282        }
283    }

287    /// Build a valid AMD64 PE in memory layout (sections at VirtualAddress offsets).
288    fn build_memory_pe() -> Vec<u8> {
289        let mut buf = vec![0u8; 0x3000];
290
291        // DOS header
292        buf[0] = b'M';
293        buf[1] = b'Z';
294        // e_lfanew at 0x3C
295        let e_lfanew: u32 = 0x80;
296        buf[0x3C..0x40].copy_from_slice(&e_lfanew.to_le_bytes());
297
298        // PE signature at 0x80
299        let pe_sig_offset = e_lfanew as usize;
300        buf[pe_sig_offset..pe_sig_offset + 4].copy_from_slice(b"PE\0\0");
301
302        // COFF header at pe_sig_offset + 4
303        let coff = pe_sig_offset + 4;
304        // Machine: AMD64 = 0x8664
305        buf[coff..coff + 2].copy_from_slice(&0x8664u16.to_le_bytes());
306        // NumberOfSections = 2
307        buf[coff + 2..coff + 4].copy_from_slice(&2u16.to_le_bytes());
308        // SizeOfOptionalHeader = 240 (0xF0)
309        buf[coff + 16..coff + 18].copy_from_slice(&240u16.to_le_bytes());
310        // Characteristics
311        buf[coff + 18..coff + 20].copy_from_slice(&0x0022u16.to_le_bytes());
312
313        // Optional header at coff + 20
314        let opt = coff + 20;
315        // Magic: PE32+ = 0x20B
316        buf[opt..opt + 2].copy_from_slice(&0x020Bu16.to_le_bytes());
317        // SectionAlignment at opt+56
318        buf[opt + 56..opt + 60].copy_from_slice(&0x1000u32.to_le_bytes());
319        // FileAlignment at opt+60
320        buf[opt + 60..opt + 64].copy_from_slice(&0x0200u32.to_le_bytes());
321        // SizeOfImage at opt+56+0x38 = opt+112 ... actually layout:
322        // opt+56: SectionAlignment, opt+60: FileAlignment
323        // SizeOfImage at opt+56 (PE32+ optional header layout):
324        //   +0:  Magic(2), MajorLinkerVersion(1), MinorLinkerVersion(1)
325        //   +4:  SizeOfCode(4)
326        //   +8:  SizeOfInitializedData(4)
327        //   +12: SizeOfUninitializedData(4)
328        //   +16: AddressOfEntryPoint(4)
329        //   +20: BaseOfCode(4)
330        //   +24: ImageBase(8)
331        //   +32: SectionAlignment(4)
332        //   +36: FileAlignment(4)
333        //   ...
334        //   +56: SizeOfImage(4)
335        //   +60: SizeOfHeaders(4)
336        //   +64: CheckSum(4)
337        //   ...
338        //   +92: NumberOfRvaAndSizes(4)
339        //   +96: DataDirectory[16]
340
341        // Fix: reset and use correct PE32+ optional header field offsets
342        // Magic at opt+0
343        buf[opt..opt + 2].copy_from_slice(&0x020Bu16.to_le_bytes());
344        // ImageBase at opt+24
345        buf[opt + 24..opt + 32].copy_from_slice(&0x0000000140000000u64.to_le_bytes());
346        // SectionAlignment at opt+32
347        buf[opt + 32..opt + 36].copy_from_slice(&0x1000u32.to_le_bytes());
348        // FileAlignment at opt+36
349        buf[opt + 36..opt + 40].copy_from_slice(&0x0200u32.to_le_bytes());
350        // SizeOfImage at opt+56
351        buf[opt + 56..opt + 60].copy_from_slice(&0x4000u32.to_le_bytes());
352        // SizeOfHeaders at opt+60
353        buf[opt + 60..opt + 64].copy_from_slice(&0x0400u32.to_le_bytes());
354        // NumberOfRvaAndSizes at opt+92
355        buf[opt + 92..opt + 96].copy_from_slice(&16u32.to_le_bytes());
356
357        // Section headers start at opt + 240 = pe_sig_offset+4+20+240
358        let sect = opt + 240; // = 0x80 + 4 + 20 + 240 = 0x80 + 264 = 0x188
359
360        // Section 0: .text
361        // Name (8 bytes)
362        buf[sect..sect + 8].copy_from_slice(b".text\0\0\0");
363        // VirtualSize at sect+8
364        buf[sect + 8..sect + 12].copy_from_slice(&0x0200u32.to_le_bytes());
365        // VirtualAddress at sect+12
366        buf[sect + 12..sect + 16].copy_from_slice(&0x1000u32.to_le_bytes());
367        // SizeOfRawData at sect+16
368        buf[sect + 16..sect + 20].copy_from_slice(&0x0200u32.to_le_bytes());
369        // PointerToRawData at sect+20
370        buf[sect + 20..sect + 24].copy_from_slice(&0x0400u32.to_le_bytes());
371
372        // Section 1: .data (40 bytes per section header)
373        let sect1 = sect + 40;
374        buf[sect1..sect1 + 8].copy_from_slice(b".data\0\0\0");
375        // VirtualSize
376        buf[sect1 + 8..sect1 + 12].copy_from_slice(&0x0100u32.to_le_bytes());
377        // VirtualAddress
378        buf[sect1 + 12..sect1 + 16].copy_from_slice(&0x2000u32.to_le_bytes());
379        // SizeOfRawData
380        buf[sect1 + 16..sect1 + 20].copy_from_slice(&0x0200u32.to_le_bytes());
381        // PointerToRawData
382        buf[sect1 + 20..sect1 + 24].copy_from_slice(&0x0600u32.to_le_bytes());
383
384        // Section data in memory layout
385        // .text at VirtualAddress 0x1000
386        for b in &mut buf[0x1000..0x1200] {
387            *b = 0xCC;
388        }
389        // .data at VirtualAddress 0x2000
390        for b in &mut buf[0x2000..0x2100] {
391            *b = 0xDD;
392        }
393
394        buf
395    }
