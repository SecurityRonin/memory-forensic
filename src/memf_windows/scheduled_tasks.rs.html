1//! Windows scheduled task persistence detection.
2//!
3//! Scheduled tasks are a common persistence mechanism — malware creates tasks
4//! that survive reboot via `schtasks.exe` or COM-based APIs. The Task Scheduler
5//! service stores task definitions in `_TASK_ENTRY` structures accessible
6//! through the `TaskSchedulerService` or `ubpm` kernel objects.
7//!
8//! Enumerating scheduled tasks from memory captures tasks that may have been
9//! deleted from disk but remain in the service's in-memory cache.
10
11use memf_core::object_reader::ObjectReader;
12use memf_format::PhysicalMemoryProvider;
13
14use crate::unicode::read_unicode_string;
15
16/// Maximum number of scheduled task entries to walk (safety limit).
17const MAX_TASKS: usize = 4096;
18
19/// Information about a scheduled task recovered from kernel memory.
20#[derive(Debug, Clone, serde::Serialize)]
21pub struct ScheduledTaskInfo {
22    /// Task name (e.g., "\Microsoft\Windows\UpdateOrchestrator\Schedule Scan").
23    pub name: String,
24    /// Task path in the Task Scheduler namespace.
25    pub path: String,
26    /// Action command line (what the task executes).
27    pub action: String,
28    /// Task author.
29    pub author: String,
30    /// Whether the task is currently enabled.
31    pub enabled: bool,
32    /// Last run time (FILETIME).
33    pub last_run_time: u64,
34    /// Next run time (FILETIME).
35    pub next_run_time: u64,
36    /// Whether this task looks suspicious.
37    pub is_suspicious: bool,
38}
39
40/// Classify a scheduled task as suspicious.
41///
42/// Returns `true` for tasks that match patterns commonly used for persistence:
43/// - Action contains encoded commands (powershell -enc, cmd /c)
44/// - Task in unusual paths (not under \Microsoft\Windows)
45/// - Action references temp directories or user-writable locations
46pub fn classify_scheduled_task(name: &str, action: &str) -> bool {
47    if name.is_empty() || action.is_empty() {
48        return false;
49    }
50
51    let action_lower = action.to_ascii_lowercase();
52
53    // Encoded/obfuscated commands
54    if action_lower.contains("-enc") || action_lower.contains("-encodedcommand") {
55        return true;
56    }
57
58    // Commands from suspicious locations
59    if action_lower.contains("\\temp\\")
60        || action_lower.contains("\\tmp\\")
61        || action_lower.contains("\\appdata\\")
62        || action_lower.contains("\\downloads\\")
63        || action_lower.contains("\\public\\")
64    {
65        return true;
66    }
67
68    // Known suspicious patterns
69    if action_lower.contains("mshta") || action_lower.contains("regsvr32 /s /n /u /i:") {
70        return true;
71    }
72
73    // Task in non-standard path with script execution
74    let name_lower = name.to_ascii_lowercase();
75    if !name_lower.starts_with("\\microsoft\\") && !name_lower.starts_with("microsoft\\") {
76        if action_lower.contains("powershell")
77            || action_lower.contains("wscript")
78            || action_lower.contains("cscript")
79        {
80            return true;
81        }
82    }
83
84    false
85}

87/// Enumerate scheduled tasks from the Task Scheduler service memory.
88///
89/// Looks up `UbpmTaskEnumerator` or `TaskSchedulerService` to find
90/// in-memory task definitions. Returns an empty `Vec` if the required
91/// symbols are not present.
92pub fn walk_scheduled_tasks<P: PhysicalMemoryProvider>(
93    reader: &ObjectReader<P>,
94) -> crate::Result<Vec<ScheduledTaskInfo>> {
95    // Try UbpmTaskEnumerator first, then TaskSchedulerTaskList.
96    let list_head = match reader.symbols().symbol_address("UbpmTaskEnumerator") {
97        Some(addr) => addr,
98        None => match reader.symbols().symbol_address("TaskSchedulerTaskList") {
99            Some(addr) => addr,
100            None => return Ok(Vec::new()),
101        },
102    };
103
104    // Resolve _TASK_ENTRY field offsets.
105    let task_list_entry_off = reader
106        .symbols()
107        .field_offset("_TASK_ENTRY", "TaskListEntry")
108        .unwrap_or(0x00);
109
110    let name_off = reader
111        .symbols()
112        .field_offset("_TASK_ENTRY", "Name")
113        .unwrap_or(0x10);
114
115    let path_off = reader
116        .symbols()
117        .field_offset("_TASK_ENTRY", "Path")
118        .unwrap_or(0x20);
119
120    let action_off = reader
121        .symbols()
122        .field_offset("_TASK_ENTRY", "Action")
123        .unwrap_or(0x30);
124
125    let author_off = reader
126        .symbols()
127        .field_offset("_TASK_ENTRY", "Author")
128        .unwrap_or(0x40);
129
130    let enabled_off = reader
131        .symbols()
132        .field_offset("_TASK_ENTRY", "Enabled")
133        .unwrap_or(0x50);
134
135    let last_run_off = reader
136        .symbols()
137        .field_offset("_TASK_ENTRY", "LastRunTime")
138        .unwrap_or(0x58);
139
140    let next_run_off = reader
141        .symbols()
142        .field_offset("_TASK_ENTRY", "NextRunTime")
143        .unwrap_or(0x60);
144
145    // Read head Flink.
146    let first = match reader.read_bytes(list_head, 8) {
147        Ok(bytes) if bytes.len() == 8 => u64::from_le_bytes(bytes[..8].try_into().unwrap()),
148        _ => return Ok(Vec::new()),
149    };
150
151    if first == 0 || first == list_head {
152        return Ok(Vec::new());
153    }
154
155    let mut tasks = Vec::new();
156    let mut current = first;
157    let mut seen = std::collections::HashSet::new();
158
159    while current != list_head && current != 0 && tasks.len() < MAX_TASKS {
160        if !seen.insert(current) {
161            break; // Cycle detection.
162        }
163
164        // current points to TaskListEntry within _TASK_ENTRY.
165        let task_addr = current.wrapping_sub(task_list_entry_off);
166
167        // Read UNICODE_STRING fields.
168        let name = read_unicode_string(reader, task_addr + name_off).unwrap_or_default();
169        let path = read_unicode_string(reader, task_addr + path_off).unwrap_or_default();
170        let action = read_unicode_string(reader, task_addr + action_off).unwrap_or_default();
171        let author = read_unicode_string(reader, task_addr + author_off).unwrap_or_default();
172
173        // Read enabled flag (u32, nonzero = enabled).
174        let enabled = match reader.read_bytes(task_addr + enabled_off, 4) {
175            Ok(bytes) if bytes.len() == 4 => {
176                u32::from_le_bytes(bytes[..4].try_into().unwrap()) != 0
177            }
178            _ => false,
179        };
180
181        // Read timestamps (FILETIME u64).
182        let last_run_time = match reader.read_bytes(task_addr + last_run_off, 8) {
183            Ok(bytes) if bytes.len() == 8 => u64::from_le_bytes(bytes[..8].try_into().unwrap()),
184            _ => 0,
185        };
186
187        let next_run_time = match reader.read_bytes(task_addr + next_run_off, 8) {
188            Ok(bytes) if bytes.len() == 8 => u64::from_le_bytes(bytes[..8].try_into().unwrap()),
189            _ => 0,
190        };
191
192        let is_suspicious = classify_scheduled_task(&name, &action);
193
194        tasks.push(ScheduledTaskInfo {
195            name,
196            path,
197            action,
198            author,
199            enabled,
200            last_run_time,
201            next_run_time,
202            is_suspicious,
203        });
204
205        // Follow Flink to next entry.
206        current = match reader.read_bytes(current, 8) {
207            Ok(bytes) if bytes.len() == 8 => u64::from_le_bytes(bytes[..8].try_into().unwrap()),
208            _ => break,
209        };
210    }
211
212    Ok(tasks)
213}

488    /// Walk body: list_head → entry → list_head (one entry, then returns to head).
489    ///
490    /// Uses default field offsets (no _TASK_ENTRY fields in ISF), so all
491    /// strings are empty and the task is pushed with empty fields.
492    #[test]
493    fn walk_scheduled_tasks_one_entry_in_loop() {
494        let list_vaddr: u64 = 0xFFFF_8000_0080_0000;
495        let list_paddr: u64 = 0x0080_0000;
496        let entry_vaddr: u64 = 0xFFFF_8000_0081_0000;
497        let entry_paddr: u64 = 0x0081_0000;
498
499        // List head page: [Flink=entry_vaddr, ...]
500        let mut head_page = vec![0u8; 0x1000];
501        head_page[0..8].copy_from_slice(&entry_vaddr.to_le_bytes());
502
503        // Entry page: [Flink=list_vaddr (loop terminator), ...task fields]
504        let mut entry_page = vec![0u8; 0x1000];
505        // Flink at offset 0 (task_list_entry_off = 0)
506        entry_page[0..8].copy_from_slice(&list_vaddr.to_le_bytes());
507        // LastRunTime at +0x58 = 0x1234
508        entry_page[0x58..0x60].copy_from_slice(&0x1234u64.to_le_bytes());
509        // NextRunTime at +0x60 = 0x5678
510        entry_page[0x60..0x68].copy_from_slice(&0x5678u64.to_le_bytes());
511
512        let isf = IsfBuilder::new()
513            .add_struct("_TASK_ENTRY", 0x80)
514            .add_symbol("UbpmTaskEnumerator", list_vaddr)
515            .build_json();
516        let resolver = IsfResolver::from_value(&isf).unwrap();
517        let (cr3, mem) = PageTableBuilder::new()
518            .map_4k(list_vaddr, list_paddr, flags::WRITABLE)
519            .write_phys(list_paddr, &head_page)
520            .map_4k(entry_vaddr, entry_paddr, flags::WRITABLE)
521            .write_phys(entry_paddr, &entry_page)
522            .build();
523        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
524        let reader = ObjectReader::new(vas, Box::new(resolver));
525
526        let result = walk_scheduled_tasks(&reader).unwrap();
527        assert_eq!(result.len(), 1, "exactly one task entry should be produced");
528        let task = &result[0];
529        assert_eq!(task.last_run_time, 0x1234);
530        assert_eq!(task.next_run_time, 0x5678);
531        assert!(!task.enabled);
532        assert!(!task.is_suspicious);
533    }

535    /// Walk body: two entries, second entry has Flink = list_head → loop stops.
536    /// Tests the Flink-following code path with two iterations.
537    #[test]
538    fn walk_scheduled_tasks_two_entries_in_loop() {
539        let list_vaddr: u64 = 0xFFFF_8000_0082_0000;
540        let entry1_vaddr: u64 = 0xFFFF_8000_0083_0000;
541        let entry2_vaddr: u64 = 0xFFFF_8000_0084_0000;
542
543        let list_paddr: u64 = 0x0082_0000;
544        let entry1_paddr: u64 = 0x0083_0000;
545        let entry2_paddr: u64 = 0x0084_0000;
546
547        let mut head_page = vec![0u8; 0x1000];
548        head_page[0..8].copy_from_slice(&entry1_vaddr.to_le_bytes());
549
550        let mut e1_page = vec![0u8; 0x1000];
551        e1_page[0..8].copy_from_slice(&entry2_vaddr.to_le_bytes());
552        // Enabled at +0x50 = 1
553        e1_page[0x50..0x54].copy_from_slice(&1u32.to_le_bytes());
554
555        let mut e2_page = vec![0u8; 0x1000];
556        e2_page[0..8].copy_from_slice(&list_vaddr.to_le_bytes()); // back to head
557
558        let isf = IsfBuilder::new()
559            .add_struct("_TASK_ENTRY", 0x80)
560            .add_symbol("UbpmTaskEnumerator", list_vaddr)
561            .build_json();
562        let resolver = IsfResolver::from_value(&isf).unwrap();
563        let (cr3, mem) = PageTableBuilder::new()
564            .map_4k(list_vaddr, list_paddr, flags::WRITABLE)
565            .write_phys(list_paddr, &head_page)
566            .map_4k(entry1_vaddr, entry1_paddr, flags::WRITABLE)
567            .write_phys(entry1_paddr, &e1_page)
568            .map_4k(entry2_vaddr, entry2_paddr, flags::WRITABLE)
569            .write_phys(entry2_paddr, &e2_page)
570            .build();
571        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
572        let reader = ObjectReader::new(vas, Box::new(resolver));
573
574        let result = walk_scheduled_tasks(&reader).unwrap();
575        assert_eq!(result.len(), 2, "two entries should be produced");
576        assert!(result[0].enabled, "first entry should be enabled");
577        assert!(!result[1].enabled, "second entry should not be enabled");
578    }
