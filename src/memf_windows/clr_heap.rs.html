1//! Reflective .NET (CLR) assembly detection walker — MITRE ATT&CK T1620.
2//!
3//! Scans CLR heap domains for assemblies that were loaded reflectively
4//! (no backing file on disk) and optionally retain an `MZ`/`PE` header
5//! in memory.
6
7use memf_core::object_reader::ObjectReader;
8use memf_format::PhysicalMemoryProvider;
9
10use crate::{types::ClrAssemblyInfo, Result};
11
12/// Classify whether a CLR assembly was loaded dynamically (reflectively) with
13/// no backing file on disk.
14///
15/// Returns `true` if the assembly is flagged as in-memory (`is_in_memory`) and
16/// has no associated PE path on disk (`!has_pe_path`). An assembly that is
17/// in memory but has a known disk path is a normal compiled assembly; one
18/// that is in memory without a path is a dynamically emitted / reflectively
19/// loaded assembly.
20pub fn is_dynamic_assembly(is_in_memory: bool, has_pe_path: bool) -> bool {
21    is_in_memory && !has_pe_path
22}
23
24/// Scan CLR `AppDomain` heaps for dynamically loaded (fileless) assemblies.
25///
26/// Locates the CLR data structure roots via the `mscorwks.dll` or
27/// `clr.dll` export table, then walks `AppDomain->AssemblyList` and each
28/// `Assembly->Module` to extract metadata. Reports any assembly whose
29/// `ModuleFile.Path` is empty (dynamic) or whose mapped memory contains an
30/// `MZ` signature indicating an in-memory PE.
31///
32/// # MITRE ATT&CK
33/// T1620 — Reflective Code Loading
34pub fn scan_clr_heap<P: PhysicalMemoryProvider>(
35    reader: &ObjectReader<P>,
36) -> Result<Vec<ClrAssemblyInfo>> {
37    let _ = reader;
38    Ok(vec![])
39}

49    fn make_minimal_reader(
50    ) -> ObjectReader<memf_core::test_builders::SyntheticPhysMem> {
51        let isf = IsfBuilder::new().build_json();
52        let resolver = IsfResolver::from_value(&isf).unwrap();
53        let (cr3, mem) = PageTableBuilder::new().build();
54        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
55        ObjectReader::new(vas, Box::new(resolver))
56    }
57
58    #[test]
59    fn empty_memory_returns_ok_empty() {
60        let reader = make_minimal_reader();
61        let result = scan_clr_heap(&reader);
62        assert!(result.is_ok(), "should succeed with minimal reader");
63        assert!(
64            result.unwrap().is_empty(),
65            "empty memory → no CLR assembly hits"
66        );
67    }
68
69    #[test]
70    fn result_is_vec_of_clr_assembly_info() {
71        let reader = make_minimal_reader();
72        let result: Result<Vec<ClrAssemblyInfo>> = scan_clr_heap(&reader);
73        assert!(result.is_ok());
74    }
