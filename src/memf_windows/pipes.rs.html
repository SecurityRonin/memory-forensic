1//! Windows kernel named pipe enumeration for C2/lateral-movement detection.
2//!
3//! Walks the kernel Object Manager namespace tree starting from
4//! `ObpRootDirectoryObject`, navigates to `\Device\NamedPipe`, and
5//! enumerates all pipe objects within that directory.  Each pipe name
6//! is checked against known-suspicious patterns (Cobalt Strike beacon
7//! pipes, PsExec service pipes, Meterpreter post-exploitation pipes,
8//! GUID-like random pipe names, etc.).
9
10use memf_core::object_reader::ObjectReader;
11use memf_format::PhysicalMemoryProvider;
12
13use crate::object_directory::walk_directory;
14
15/// Maximum recursion depth when walking nested object directories to
16/// reach `\Device\NamedPipe`.
17const MAX_DIR_DEPTH: usize = 8;
18
19/// Information about a single named pipe found in kernel memory.
20#[derive(Debug, Clone, serde::Serialize)]
21pub struct NamedPipeInfo {
22    /// The name of the pipe.
23    pub name: String,
24    /// Whether this pipe name matches a known-suspicious pattern.
25    pub is_suspicious: bool,
26    /// Human-readable reason for flagging, if suspicious.
27    pub suspicion_reason: Option<String>,
28}
29
30/// Enumerate named pipes from the object directory.
31///
32/// Resolves `ObpRootDirectoryObject`, walks through `\Device\NamedPipe`,
33/// and returns information about each pipe found.  Returns an empty `Vec`
34/// if the root directory symbol is missing or the path cannot be resolved.
35pub fn walk_named_pipes<P: PhysicalMemoryProvider>(
36    reader: &ObjectReader<P>,
37) -> crate::Result<Vec<NamedPipeInfo>> {
38    let root_dir_sym_addr = match reader.symbols().symbol_address("ObpRootDirectoryObject") {
39        Some(v) => v,
40        None => return Ok(Vec::new()),
41    };
42
43    // Read the pointer stored at the symbol address
44    let root_dir_addr: u64 = match reader.read_bytes(root_dir_sym_addr, 8) {
45        Ok(bytes) => u64::from_le_bytes(bytes[..8].try_into().expect("8 bytes")),
46        Err(_) => return Ok(Vec::new()),
47    };
48
49    if root_dir_addr == 0 {
50        return Ok(Vec::new());
51    }
52
53    let named_pipe_dir = match find_subdir_by_path(reader, root_dir_addr, &["Device", "NamedPipe"])
54    {
55        Some(addr) => addr,
56        None => return Ok(Vec::new()),
57    };
58
59    let entries = walk_directory(reader, named_pipe_dir).unwrap_or_default();
60
61    let mut results = Vec::new();
62    for (name, _body_addr) in entries {
63        let (is_suspicious, suspicion_reason) = match classify_pipe(&name) {
64            Some(reason) => (true, Some(reason)),
65            None => (false, None),
66        };
67        results.push(NamedPipeInfo {
68            name,
69            is_suspicious,
70            suspicion_reason,
71        });
72    }
73
74    Ok(results)
75}

77/// Walk a path of subdirectory names from a starting directory address.
78///
79/// Returns the object body address of the final directory in the path,
80/// or `None` if any segment is not found.
81fn find_subdir_by_path<P: PhysicalMemoryProvider>(
82    reader: &ObjectReader<P>,
83    mut dir_addr: u64,
84    segments: &[&str],
85) -> Option<u64> {
86    for (_depth, segment) in segments.iter().enumerate().take(MAX_DIR_DEPTH) {
87        let entries = walk_directory(reader, dir_addr).unwrap_or_default();
88        let found = entries
89            .into_iter()
90            .find(|(name, _)| name.eq_ignore_ascii_case(segment));
91        match found {
92            Some((_name, body_addr)) => dir_addr = body_addr,
93            None => return None,
94        }
95    }
96    Some(dir_addr)
97}

99/// Check if a pipe name matches known C2/lateral-movement patterns.
100///
101/// Returns `Some(reason)` if the name is suspicious, `None` otherwise.
102/// Patterns are checked in order of specificity to avoid false positives.
103pub fn classify_pipe(name: &str) -> Option<String> {
104    let lower = name.to_lowercase();
105
106    // Cobalt Strike: MSSE-*-server
107    if lower.starts_with("msse-") && lower.ends_with("-server") {
108        return Some("Cobalt Strike beacon pipe (MSSE-*-server)".to_string());
109    }
110    // msagent_* — Cobalt Strike post-ex
111    if lower.starts_with("msagent_") {
112        return Some("Cobalt Strike msagent post-ex pipe".to_string());
113    }
114    // postex_ssh_ takes priority over postex_
115    if lower.starts_with("postex_ssh_") {
116        return Some("Cobalt Strike postex_ssh pipe".to_string());
117    }
118    // postex_ — Cobalt Strike post-ex
119    if lower.starts_with("postex_") {
120        return Some("Cobalt Strike postex pipe".to_string());
121    }
122    // PsExec pipes
123    if lower.starts_with("psexesvc") || lower.starts_with("psexec") {
124        return Some("PsExec service pipe".to_string());
125    }
126    // RemCom
127    if lower.starts_with("remcom_") {
128        return Some("RemCom lateral movement pipe".to_string());
129    }
130    // CSExec
131    if lower.starts_with("csexec") {
132        return Some("CSExec lateral movement pipe".to_string());
133    }
134    // GUID-like pipe names (random C2 named pipes)
135    if is_guid_like(&lower) {
136        return Some("GUID-like pipe name (likely random C2 pipe)".to_string());
137    }
138    None
139}

141/// Check whether a string matches the GUID format: `xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx`
142/// where each `x` is a hex digit.
143fn is_guid_like(s: &str) -> bool {
144    // GUID: 8-4-4-4-12 hex digits with hyphens = 36 chars total
145    if s.len() != 36 {
146        return false;
147    }
148    let bytes = s.as_bytes();
149    // Hyphen positions: 8, 13, 18, 23
150    if bytes[8] != b'-' || bytes[13] != b'-' || bytes[18] != b'-' || bytes[23] != b'-' {
151        return false;
152    }
153    // All other positions must be hex digits
154    for (i, &b) in bytes.iter().enumerate() {
155        if i == 8 || i == 13 || i == 18 || i == 23 {
156            continue;
157        }
158        if !b.is_ascii_hexdigit() {
159            return false;
160        }
161    }
162    true
163}

370    /// Helper: encode a Rust &str as UTF-16LE bytes.
371    fn utf16le(s: &str) -> Vec<u8> {
372        s.encode_utf16().flat_map(|c| c.to_le_bytes()).collect()
373    }

375    /// walk_named_pipes: full two-level directory traversal.
376    ///
377    /// Layout uses page-aligned body addresses: each object's body sits at offset 0
378    /// of its own 4K-aligned page, so it can be used directly as a directory page.
379    /// Object header (name_info + _OBJECT_HEADER) lives in a separate page before it.
380    ///
381    ///   HDR page (vaddr H):  name_info(+0x00) + _OBJECT_HEADER(+0x20)
382    ///                        string inline at +0x100
383    ///                        body_vaddr = H + 0x50 → must be page-aligned
384    ///
385    /// We achieve this by placing object headers at page_base + 0xFB0 so that
386    /// body = page_base + 0xFB0 + 0x50 = page_base + 0x1000 = next_page_base.
387    /// That means the body IS the start of the next 4K-aligned page.
388    ///
389    ///   Layout per object (two consecutive 4K pages A, B):
390    ///     A + 0xFB0: _OBJECT_HEADER_NAME_INFO (0x20 bytes) + _OBJECT_HEADER (0x30 bytes)
391    ///                → body = A + 0xFB0 + 0x50 = A + 0x1000 = B
392    ///     A + 0xE00: inline string for the name
393    ///     B + 0x000...: used as directory (37 bucket pointers)
394    #[test]
395    fn walk_named_pipes_full_path_finds_suspicious_pipe() {
396        // Each object takes two consecutive 4K pages (header page + body/dir page).
397        // Naming: DEV_H = Device header page, DEV_B = Device body/dir page, etc.
398
399        const SYM_VADDR: u64 = OBP_ROOT_DIR_OBJ_VADDR;
400        const SYM_PADDR: u64 = 0x00A2_0000;
401        // Root directory page
402        const ROOT_VADDR: u64 = 0xFFFF_8000_0040_0000;
403        const ROOT_PADDR: u64 = 0x0040_0000;
404        // Root entry (points to Device object body)
405        const ROOT_E_VADDR: u64 = 0xFFFF_8000_0041_0000;
406        const ROOT_E_PADDR: u64 = 0x0041_0000;
407        // Device object: header at DEV_H, body at DEV_B (= DEV_H + 0x1000)
408        // Header page base addresses. Body = hdr_vaddr + 0x1000 (next 4K page).
409        // HDR_OFF=0xFB0, body = hdr_vaddr + 0xFB0 + 0x50 = hdr_vaddr + 0x1000.
410        const DEV_H_VADDR: u64 = 0xFFFF_8000_0042_0000;
411        const DEV_H_PADDR: u64 = 0x0042_0000;
412        const DEV_B_VADDR: u64 = 0xFFFF_8000_0042_1000; // = DEV_H_VADDR + 0x1000 ✓
413        const DEV_B_PADDR: u64 = 0x0042_1000;
414        const DEV_E_VADDR: u64 = 0xFFFF_8000_0043_0000;
415        const DEV_E_PADDR: u64 = 0x0043_0000;
416        const PIPE_H_VADDR: u64 = 0xFFFF_8000_0044_0000;
417        const PIPE_H_PADDR: u64 = 0x0044_0000;
418        const PIPE_B_VADDR: u64 = 0xFFFF_8000_0044_1000; // = PIPE_H_VADDR + 0x1000 ✓
419        const PIPE_B_PADDR: u64 = 0x0044_1000;
420        const PIPE_E_VADDR: u64 = 0xFFFF_8000_0045_0000;
421        const PIPE_E_PADDR: u64 = 0x0045_0000;
422        const MS_H_VADDR: u64 = 0xFFFF_8000_0046_0000;
423        const MS_H_PADDR: u64 = 0x0046_0000;
424        const MS_B_VADDR: u64 = 0xFFFF_8000_0046_1000; // = MS_H_VADDR + 0x1000 ✓
425        const MS_B_PADDR: u64 = 0x0046_1000;
426        // Entry inside NamedPipe dir pointing to msagent is stored at PIPE_E_VADDR.
427
428        let isf = make_isf_with_obp_root(SYM_VADDR);
429        let resolver = IsfResolver::from_value(&isf).unwrap();
430
431        // Header offset within header page: 0xFB0
432        // body_vaddr = hdr_page_vaddr + 0xFB0 + 0x50 = hdr_page_vaddr + 0x1000 = body_page_vaddr
433        const HDR_OFF: usize = 0xFB0;
434
435        // Build a header page for a named object.
436        // name_info at HDR_OFF, string at HDR_OFF - 0x200 (so no overlap).
437        let make_hdr_page = |hdr_page_vaddr: u64, body_page_vaddr: u64, name: &str| -> Vec<u8> {
438            let encoded = utf16le(name);
439            let len = encoded.len() as u16;
440            // Place string at 0x800 within the header page
441            let str_vaddr = hdr_page_vaddr + 0x800;
442            let mut page = vec![0u8; 4096];
443            // name_info at HDR_OFF: _UNICODE_STRING for the name
444            page[HDR_OFF + 0x10..HDR_OFF + 0x12].copy_from_slice(&len.to_le_bytes());
445            page[HDR_OFF + 0x12..HDR_OFF + 0x14].copy_from_slice(&len.to_le_bytes());
446            page[HDR_OFF + 0x18..HDR_OFF + 0x20].copy_from_slice(&str_vaddr.to_le_bytes());
447            // _OBJECT_HEADER at HDR_OFF + 0x20: InfoMask = 0x02
448            page[HDR_OFF + 0x20 + 0x1a] = 0x02;
449            // Inline string at 0x800
450            page[0x800..0x800 + encoded.len()].copy_from_slice(&encoded);
451            // Verify body vaddr = hdr_page_vaddr + HDR_OFF + 0x50 = hdr_page_vaddr + 0x1000
452            assert_eq!(
453                hdr_page_vaddr + HDR_OFF as u64 + 0x50,
454                body_page_vaddr,
455                "body must be at start of next page"
456            );
457            page
458        };
459
460        // Body of each intermediate object doubles as a directory page (bucket[0] = entry_vaddr).
461        let make_dir_page = |entry_vaddr: u64| -> Vec<u8> {
462            let mut page = vec![0u8; 4096];
463            page[0..8].copy_from_slice(&entry_vaddr.to_le_bytes());
464            page
465        };
466
467        // Entry page: { ChainLink=0, Object=obj_body_vaddr }
468        let make_entry_page = |obj_body: u64| -> Vec<u8> {
469            let mut page = vec![0u8; 4096];
470            page[8..16].copy_from_slice(&obj_body.to_le_bytes());
471            page
472        };
473
474        // Object body vaddrs (= start of body page)
475        let dev_body = DEV_B_VADDR;
476        let pipe_body = PIPE_B_VADDR;
477        let ms_body = MS_B_VADDR;
478
479        // Build pages
480        let mut sym_page = vec![0u8; 4096];
481        sym_page[0..8].copy_from_slice(&ROOT_VADDR.to_le_bytes());
482
483        let root_page = make_dir_page(ROOT_E_VADDR);
484        let root_e_page = make_entry_page(dev_body);
485        let dev_hdr_page = make_hdr_page(DEV_H_VADDR, DEV_B_VADDR, "Device");
486        let dev_dir_page = make_dir_page(DEV_E_VADDR);
487        let dev_e_page = make_entry_page(pipe_body);
488        let pipe_hdr_page = make_hdr_page(PIPE_H_VADDR, PIPE_B_VADDR, "NamedPipe");
489        let pipe_dir_page = make_dir_page(PIPE_E_VADDR);
490        let pipe_e_page = make_entry_page(ms_body);
491        let ms_hdr_page = make_hdr_page(MS_H_VADDR, MS_B_VADDR, "msagent_test");
492        let ms_body_page = vec![0u8; 4096]; // leaf — not used as directory
493
494        let (cr3, mem) = PageTableBuilder::new()
495            .map_4k(SYM_VADDR, SYM_PADDR, flags::WRITABLE)
496            .write_phys(SYM_PADDR, &sym_page)
497            .map_4k(ROOT_VADDR, ROOT_PADDR, flags::WRITABLE)
498            .write_phys(ROOT_PADDR, &root_page)
499            .map_4k(ROOT_E_VADDR, ROOT_E_PADDR, flags::WRITABLE)
500            .write_phys(ROOT_E_PADDR, &root_e_page)
501            .map_4k(DEV_H_VADDR, DEV_H_PADDR, flags::WRITABLE)
502            .write_phys(DEV_H_PADDR, &dev_hdr_page)
503            .map_4k(DEV_B_VADDR, DEV_B_PADDR, flags::WRITABLE)
504            .write_phys(DEV_B_PADDR, &dev_dir_page)
505            .map_4k(DEV_E_VADDR, DEV_E_PADDR, flags::WRITABLE)
