1//! LSA secrets extraction from Windows memory dumps.
2//!
3//! The SECURITY registry hive (`\REGISTRY\MACHINE\SECURITY`) stores LSA
4//! (Local Security Authority) secrets under `Policy\Secrets`. These secrets
5//! include service account passwords, VPN credentials, auto-logon passwords,
6//! DPAPI system master keys, and cached domain key material.
7//!
8//! Extracting LSA secrets from memory enables:
9//!
10//! - Recovering service account passwords (`_SC_*` secrets)
11//! - Detecting auto-logon credentials (`DefaultPassword`)
12//! - Extracting DPAPI system keys for offline decryption
13//! - Identifying VPN credentials stored in memory
14//! - Discovering cached domain key material (`NL$KM`)
15//!
16//! The SECURITY hive is structured as:
17//! `SECURITY\Policy\Secrets\<name>\CurrVal` — current secret value
18//! `SECURITY\Policy\Secrets\<name>\OldVal` — previous secret value
19
20use memf_core::object_reader::ObjectReader;
21use memf_format::PhysicalMemoryProvider;
22
23/// Maximum number of LSA secrets to enumerate (safety limit).
24const MAX_SECRETS: usize = 4096;
25
26/// Information about an LSA secret recovered from the SECURITY hive.
27#[derive(Debug, Clone, serde::Serialize)]
28pub struct LsaSecretInfo {
29    /// Secret name (e.g., `NL$KM`, `DPAPI_SYSTEM`, `_SC_servicename`).
30    pub name: String,
31    /// Classified secret type (e.g., `"service_password"`, `"dpapi_key"`).
32    pub secret_type: String,
33    /// Length of the secret data in bytes.
34    pub length: u32,
35    /// Whether this secret is suspicious based on heuristics.
36    pub is_suspicious: bool,
37}
38
39/// Classify an LSA secret by name.
40///
41/// Returns `(secret_type, is_suspicious)` based on the secret name pattern:
42/// - `_SC_*` — service account password (normal)
43/// - `NL$KM` — cached domain key material (normal)
44/// - `DPAPI_SYSTEM` — DPAPI system master key (normal)
45/// - `DefaultPassword` — auto-logon password (risky)
46/// - `$MACHINE.ACC` — machine account password (normal)
47/// - `L$_RasConn*` / `L$_RasDial*` — VPN credentials (suspicious)
48/// - Other `L$*` — generic LSA data (normal)
49/// - Anything else — unknown (suspicious if name > 30 chars)
50pub fn classify_lsa_secret(name: &str) -> (String, bool) {
51    if name.starts_with("_SC_") {
52        return ("service_password".to_string(), false);
53    }
54    if name == "NL$KM" {
55        return ("cached_domain_key".to_string(), false);
56    }
57    if name == "DPAPI_SYSTEM" {
58        return ("dpapi_key".to_string(), false);
59    }
60    if name == "DefaultPassword" {
61        return ("default_password".to_string(), true);
62    }
63    if name == "$MACHINE.ACC" {
64        return ("machine_password".to_string(), false);
65    }
66    if name.starts_with("L$_RasConn") || name.starts_with("L$_RasDial") {
67        return ("vpn_credential".to_string(), true);
68    }
69    if name.starts_with("L$") {
70        return ("lsa_data".to_string(), false);
71    }
72    ("unknown".to_string(), name.len() > 30)
73}

75/// Extract LSA secrets from the SECURITY registry hive in memory.
76///
77/// Navigates `SECURITY\Policy\Secrets` in the registry hive at
78/// `security_hive_addr`, enumerates subkeys (each representing a secret),
79/// reads the `CurrVal` subkey's default value for the secret length,
80/// classifies each secret, and returns the results.
81///
82/// Returns an empty `Vec` if the hive address is zero or navigation fails.
83pub fn walk_lsa_secrets<P: PhysicalMemoryProvider>(
84    reader: &ObjectReader<P>,
85    security_hive_addr: u64,
86) -> crate::Result<Vec<LsaSecretInfo>> {
87    if security_hive_addr == 0 {
88        return Ok(Vec::new());
89    }
90
91    // Read _HHIVE.BaseBlock pointer to get _HBASE_BLOCK address.
92    let base_block_off = reader
93        .symbols()
94        .field_offset("_HHIVE", "BaseBlock")
95        .unwrap_or(0x10);
96
97    let base_block_addr = match reader.read_bytes(security_hive_addr + base_block_off, 8) {
98        Ok(bytes) if bytes.len() == 8 => u64::from_le_bytes(bytes[..8].try_into().unwrap()),
99        _ => return Ok(Vec::new()),
100    };
101
102    if base_block_addr == 0 {
103        return Ok(Vec::new());
104    }
105
106    // Read root cell offset from _HBASE_BLOCK (at offset 0x24, u32).
107    let root_cell_off = match reader.read_bytes(base_block_addr + 0x24, 4) {
108        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
109        _ => return Ok(Vec::new()),
110    };
111
112    if root_cell_off == 0 || root_cell_off == u32::MAX {
113        return Ok(Vec::new());
114    }
115
116    // Compute flat storage base for cell address resolution.
117    let storage_off = reader
118        .symbols()
119        .field_offset("_HHIVE", "Storage")
120        .unwrap_or(0x30);
121
122    let flat_base = match reader.read_bytes(security_hive_addr + storage_off, 8) {
123        Ok(bytes) if bytes.len() == 8 => {
124            let addr = u64::from_le_bytes(bytes[..8].try_into().unwrap());
125            if addr != 0 {
126                addr
127            } else {
128                base_block_addr + 0x1000
129            }
130        }
131        _ => base_block_addr + 0x1000,
132    };
133
134    // Navigate: root → Policy → Secrets
135    let root_addr = read_cell_addr(reader, flat_base, root_cell_off);
136    if root_addr == 0 {
137        return Ok(Vec::new());
138    }
139
140    let policy_key = find_subkey_by_name(reader, flat_base, root_addr, "Policy");
141    if policy_key == 0 {
142        return Ok(Vec::new());
143    }
144
145    let secrets_key = find_subkey_by_name(reader, flat_base, policy_key, "Secrets");
146    if secrets_key == 0 {
147        return Ok(Vec::new());
148    }
149
150    // Enumerate subkeys under Secrets — each is a secret name.
151    let subkey_count: u32 = match reader.read_bytes(secrets_key + 0x18, 4) {
152        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
153        _ => 0,
154    };
155
156    if subkey_count == 0 || subkey_count > MAX_SECRETS as u32 {
157        return Ok(Vec::new());
158    }
159
160    let subkey_list_off: u32 = match reader.read_bytes(secrets_key + 0x20, 4) {
161        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
162        _ => return Ok(Vec::new()),
163    };
164
165    let list_addr = read_cell_addr(reader, flat_base, subkey_list_off);
166    if list_addr == 0 {
167        return Ok(Vec::new());
168    }
169
170    // Read list signature (lf/lh/li).
171    let list_sig = match reader.read_bytes(list_addr, 2) {
172        Ok(bytes) if bytes.len() == 2 => [bytes[0], bytes[1]],
173        _ => return Ok(Vec::new()),
174    };
175
176    let count: u16 = match reader.read_bytes(list_addr + 2, 2) {
177        Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
178        _ => return Ok(Vec::new()),
179    };
180
181    let mut secrets = Vec::new();
182
183    for i in 0..count.min(MAX_SECRETS as u16) {
184        let entry_off = match list_sig {
185            [b'l', b'f'] | [b'l', b'h'] => {
186                // lf/lh: 8-byte entries (offset + hash) starting at +4
187                match reader.read_bytes(list_addr + 4 + (i as u64) * 8, 4) {
188                    Ok(bytes) if bytes.len() == 4 => {
189                        u32::from_le_bytes(bytes[..4].try_into().unwrap())
190                    }
191                    _ => continue,
192                }
193            }
194            [b'l', b'i'] => {
195                // li: 4-byte entries (offset only) starting at +4
196                match reader.read_bytes(list_addr + 4 + (i as u64) * 4, 4) {
197                    Ok(bytes) if bytes.len() == 4 => {
198                        u32::from_le_bytes(bytes[..4].try_into().unwrap())
199                    }
200                    _ => continue,
201                }
202            }
203            _ => continue,
204        };
205
206        let key_addr = read_cell_addr(reader, flat_base, entry_off);
207        if key_addr == 0 {
208            continue;
209        }
210
211        // Read key name (at offset 0x4C in _CM_KEY_NODE, length at 0x4A).
212        let name_len: u16 = match reader.read_bytes(key_addr + 0x4A, 2) {
213            Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
214            _ => continue,
215        };
216
217        if name_len == 0 || name_len > 256 {
218            continue;
219        }
220
221        let secret_name = match reader.read_bytes(key_addr + 0x4C, name_len as usize) {
222            Ok(bytes) => String::from_utf8_lossy(&bytes).to_string(),
223            _ => continue,
224        };
225
226        // Read the CurrVal subkey's default value length.
227        let length = read_currval_length(reader, flat_base, key_addr);
228
229        let (secret_type, is_suspicious) = classify_lsa_secret(&secret_name);
230
231        secrets.push(LsaSecretInfo {
232            name: secret_name,
233            secret_type,
234            length,
235            is_suspicious,
236        });
237    }
238
239    Ok(secrets)
240}

242/// Read a cell address from the flat storage base + cell offset.
243fn read_cell_addr<P: PhysicalMemoryProvider>(
244    reader: &ObjectReader<P>,
245    flat_base: u64,
246    cell_off: u32,
247) -> u64 {
248    // Cell data starts 4 bytes after the cell offset (cell size header).
249    let addr = flat_base + (cell_off as u64) + 4;
250    // Verify we can read from this address.
251    match reader.read_bytes(addr, 2) {
252        Ok(bytes) if bytes.len() == 2 => addr,
253        _ => 0,
254    }
255}

257/// Find a subkey by name under a parent `_CM_KEY_NODE`.
258fn find_subkey_by_name<P: PhysicalMemoryProvider>(
259    reader: &ObjectReader<P>,
260    flat_base: u64,
261    parent_addr: u64,
262    target_name: &str,
263) -> u64 {
264    let subkey_count: u32 = match reader.read_bytes(parent_addr + 0x18, 4) {
265        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
266        _ => return 0,
267    };
268
269    if subkey_count == 0 || subkey_count > 4096 {
270        return 0;
271    }
272
273    let list_off: u32 = match reader.read_bytes(parent_addr + 0x20, 4) {
274        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
275        _ => return 0,
276    };
277
278    let list_addr = read_cell_addr(reader, flat_base, list_off);
279    if list_addr == 0 {
280        return 0;
281    }
282
283    let list_sig = match reader.read_bytes(list_addr, 2) {
284        Ok(bytes) if bytes.len() == 2 => [bytes[0], bytes[1]],
285        _ => return 0,
286    };
287
288    let count: u16 = match reader.read_bytes(list_addr + 2, 2) {
289        Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
290        _ => return 0,
291    };
292
293    for i in 0..count.min(4096) {
294        let entry_off = match list_sig {
295            [b'l', b'f'] | [b'l', b'h'] => {
296                match reader.read_bytes(list_addr + 4 + (i as u64) * 8, 4) {
297                    Ok(bytes) if bytes.len() == 4 => {
298                        u32::from_le_bytes(bytes[..4].try_into().unwrap())
299                    }
300                    _ => continue,
301                }
302            }
303            [b'l', b'i'] => match reader.read_bytes(list_addr + 4 + (i as u64) * 4, 4) {
304                Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
305                _ => continue,
306            },
307            _ => return 0,
308        };
309
310        let key_addr = read_cell_addr(reader, flat_base, entry_off);
311        if key_addr == 0 {
312            continue;
313        }
314
315        let name_len: u16 = match reader.read_bytes(key_addr + 0x4A, 2) {
316            Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
317            _ => continue,
318        };
319
320        if name_len == 0 || name_len > 256 {
321            continue;
322        }
323
324        let name = match reader.read_bytes(key_addr + 0x4C, name_len as usize) {
325            Ok(bytes) => String::from_utf8_lossy(&bytes).to_string(),
326            _ => continue,
327        };
328
329        if name.eq_ignore_ascii_case(target_name) {
330            return key_addr;
331        }
332    }
333
334    0
335}

337/// Read the data length from a secret's `CurrVal` subkey's default value.
338///
339/// Navigates `<secret_key>\CurrVal` and reads the `(Default)` value's
340/// `DataLength` field to determine the secret size.
341fn read_currval_length<P: PhysicalMemoryProvider>(
342    reader: &ObjectReader<P>,
343    flat_base: u64,
344    secret_key_addr: u64,
345) -> u32 {
346    // Find the CurrVal subkey.
347    let currval_addr = find_subkey_by_name(reader, flat_base, secret_key_addr, "CurrVal");
348    if currval_addr == 0 {
349        return 0;
350    }
351
352    // Read value count from CurrVal key node (offset 0x28).
353    let val_count: u32 = match reader.read_bytes(currval_addr + 0x28, 4) {
354        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
355        _ => return 0,
356    };
357
358    if val_count == 0 {
359        return 0;
360    }
361
362    // Read value list cell offset (0x2C in _CM_KEY_NODE).
363    let val_list_off: u32 = match reader.read_bytes(currval_addr + 0x2C, 4) {
364        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
365        _ => return 0,
366    };
367
368    let val_list_addr = read_cell_addr(reader, flat_base, val_list_off);
369    if val_list_addr == 0 {
370        return 0;
371    }
372
373    // Read the first value offset (the default value).
374    let val_off: u32 = match reader.read_bytes(val_list_addr, 4) {
375        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
376        _ => return 0,
377    };
378
379    let val_addr = read_cell_addr(reader, flat_base, val_off);
380    if val_addr == 0 {
381        return 0;
382    }
383
384    // _CM_KEY_VALUE: DataLength at offset 0x08 (u32). MSB indicates inline data.
385    match reader.read_bytes(val_addr + 0x08, 4) {
386        Ok(bytes) if bytes.len() == 4 => {
387            u32::from_le_bytes(bytes[..4].try_into().unwrap()) & 0x7FFF_FFFF
388        }
389        _ => 0,
390    }
391}

1061    fn walk_lsa_secrets_full_traversal_finds_service_password() {
1062        let hive_vaddr: u64 = 0x0074_0000;
1063        let hive_paddr: u64 = 0x0074_0000;
1064        let bb_vaddr: u64 = 0x0075_0000;
1065        let bb_paddr: u64 = 0x0075_0000;
1066        let flat_vaddr: u64 = 0x0076_0000; // explicit Storage pointer target
1067        let flat_paddr: u64 = 0x0076_0000;
1068
1069        // hive page: BaseBlock ptr at +0x10, Storage ptr at +0x30
1070        let mut hive_page = vec![0u8; 0x1000];
1071        hive_page[0x10..0x18].copy_from_slice(&bb_vaddr.to_le_bytes());
1072        hive_page[0x30..0x38].copy_from_slice(&flat_vaddr.to_le_bytes()); // explicit flat_base
1073
1074        // base_block page: root_cell_off at +0x24
1075        let mut bb_page = vec![0u8; 0x1000];
1076        bb_page[0x24..0x28].copy_from_slice(&0x100u32.to_le_bytes());
1077
1078        // flat_page: all nk/lf cells
1079        let mut flat_page = vec![0u8; 0x1000];
1080
1081        fn w32(page: &mut Vec<u8>, off: usize, val: u32) {
1082            page[off..off + 4].copy_from_slice(&val.to_le_bytes());
1083        }
1084        fn w16(page: &mut Vec<u8>, off: usize, val: u16) {
1085            page[off..off + 2].copy_from_slice(&val.to_le_bytes());
1086        }
1087
1088        // root nk data at flat_page offset 0x104 (cell_off=0x100, +4 skip header)
1089        let ro = 0x104usize;
1090        w32(&mut flat_page, ro + 0x18, 1); // subkey_count=1
1091        w32(&mut flat_page, ro + 0x20, 0x200); // list_cell_off=0x200
1092
1093        // lf1 list at 0x204
1094        let l1 = 0x204usize;
1095        flat_page[l1] = b'l';
1096        flat_page[l1 + 1] = b'f';
1097        w16(&mut flat_page, l1 + 2, 1); // count=1
1098        w32(&mut flat_page, l1 + 4, 0x300); // entry for Policy nk
1099        w32(&mut flat_page, l1 + 8, 0); // hash
1100
1101        // Policy nk at 0x304
1102        let po = 0x304usize;
1103        w32(&mut flat_page, po + 0x18, 1);
1104        w32(&mut flat_page, po + 0x20, 0x400);
1105        w16(&mut flat_page, po + 0x4A, 6);
1106        flat_page[po + 0x4C..po + 0x52].copy_from_slice(b"Policy");
1107
1108        // lf2 list at 0x404
1109        let l2 = 0x404usize;
1110        flat_page[l2] = b'l';
1111        flat_page[l2 + 1] = b'f';
1112        w16(&mut flat_page, l2 + 2, 1);
1113        w32(&mut flat_page, l2 + 4, 0x500);
1114        w32(&mut flat_page, l2 + 8, 0);
1115
1116        // Secrets nk at 0x504
1117        let se = 0x504usize;
1118        w32(&mut flat_page, se + 0x18, 1);
1119        w32(&mut flat_page, se + 0x20, 0x600);
1120        w16(&mut flat_page, se + 0x4A, 7);
1121        flat_page[se + 0x4C..se + 0x53].copy_from_slice(b"Secrets");
1122
1123        // lf3 list at 0x604
1124        let l3 = 0x604usize;
1125        flat_page[l3] = b'l';
1126        flat_page[l3 + 1] = b'f';
1127        w16(&mut flat_page, l3 + 2, 1);
1128        w32(&mut flat_page, l3 + 4, 0x700);
1129        w32(&mut flat_page, l3 + 8, 0);
1130
1131        // _SC_test nk at 0x704
1132        let sc = 0x704usize;
1133        w32(&mut flat_page, sc + 0x18, 0); // no CurrVal subkeys
1134        w16(&mut flat_page, sc + 0x4A, 8);
1135        flat_page[sc + 0x4C..sc + 0x54].copy_from_slice(b"_SC_test");
1136
1137        let isf = make_lsa_isf();
1138        let resolver = IsfResolver::from_value(&isf).unwrap();
1139        let (cr3, mem) = PageTableBuilder::new()
1140            .map_4k(hive_vaddr, hive_paddr, flags::WRITABLE)
1141            .write_phys(hive_paddr, &hive_page)
1142            .map_4k(bb_vaddr, bb_paddr, flags::WRITABLE)
1143            .write_phys(bb_paddr, &bb_page)
1144            .map_4k(flat_vaddr, flat_paddr, flags::WRITABLE)
1145            .write_phys(flat_paddr, &flat_page)
1146            .build();
1147        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
1148        let reader: ObjectReader<SyntheticPhysMem> = ObjectReader::new(vas, Box::new(resolver));
1149
1150        let result = walk_lsa_secrets(&reader, hive_vaddr).unwrap();
1151        assert!(!result.is_empty(), "should find at least one LSA secret");
1152        let secret = &result[0];
1153        assert_eq!(secret.name, "_SC_test");
1154        assert_eq!(secret.secret_type, "service_password");
1155        assert!(!secret.is_suspicious, "_SC_ secrets are not suspicious");
1156    }
