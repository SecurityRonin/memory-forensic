1//! Process hollowing detection.
2//!
3//! Detects process hollowing by reading the PE header at each process's
4//! `PEB.ImageBaseAddress` and checking for:
5//! 1. Missing MZ magic (`0x4D5A`) — image unmapped or overwritten
6//! 2. Missing PE signature (`PE\0\0`) — corrupt or replaced header
7//! 3. `SizeOfImage` mismatch between PE header and LDR module entry
8//!
9//! These indicators reveal when malware creates a legitimate process in a
10//! suspended state, unmaps its image, and replaces it with malicious code.
11
12use memf_core::object_reader::ObjectReader;
13use memf_format::PhysicalMemoryProvider;
14
15use crate::{process, Result, WinHollowingInfo};

17/// Check all running processes for signs of process hollowing.
18///
19/// For each process with a non-null PEB, switches to the process's CR3
20/// and reads the PE header at `PEB.ImageBaseAddress`. Compares the PE
21/// `SizeOfImage` against the first entry in `InLoadOrderModuleList`.
22///
23/// Returns one `WinHollowingInfo` per process (including clean ones).
24/// Check the `suspicious` field to filter findings.
25pub fn check_hollowing<P: PhysicalMemoryProvider + Clone>(
26    reader: &ObjectReader<P>,
27    ps_head_vaddr: u64,
28) -> Result<Vec<WinHollowingInfo>> {
29    let procs = process::walk_processes(reader, ps_head_vaddr)?;
30    let mut results = Vec::new();
31
32    for proc in &procs {
33        // Skip processes with no PEB (System, Idle, etc.)
34        if proc.peb_addr == 0 {
35            continue;
36        }
37
38        // Switch to process address space
39        let proc_reader = reader.with_cr3(proc.cr3);
40
41        // Read PEB.ImageBaseAddress
42        let image_base: u64 =
43            match proc_reader.read_field(proc.peb_addr, "_PEB", "ImageBaseAddress") {
44                Ok(v) => v,
45                Err(_) => continue, // PEB not readable (paged out)
46            };
47
48        if image_base == 0 {
49            continue;
50        }
51
52        // Read and validate PE header at ImageBaseAddress
53        let (has_mz, has_pe, pe_size_of_image) = read_pe_header(&proc_reader, image_base);
54
55        // Get LDR first module size for comparison
56        let ldr_size = ldr_first_image_size(&proc_reader, proc.peb_addr);
57
58        // Determine if suspicious
59        let mut suspicious = false;
60        let mut reasons = Vec::new();
61
62        if !has_mz {
63            suspicious = true;
64            reasons.push("no MZ header at ImageBaseAddress".to_string());
65        } else if !has_pe {
66            suspicious = true;
67            reasons.push("MZ present but PE signature invalid".to_string());
68        }
69
70        if has_mz && has_pe && ldr_size > 0 && u64::from(pe_size_of_image) != ldr_size {
71            suspicious = true;
72            reasons.push(format!(
73                "SizeOfImage mismatch: PE={pe_size_of_image:#x} vs LDR={ldr_size:#x}"
74            ));
75        }
76
77        results.push(WinHollowingInfo {
78            pid: proc.pid,
79            image_name: proc.image_name.clone(),
80            image_base,
81            has_mz,
82            has_pe,
83            pe_size_of_image,
84            ldr_size_of_image: ldr_size,
85            suspicious,
86            reason: if reasons.is_empty() {
87                String::new()
88            } else {
89                reasons.join("; ")
90            },
91        });
92    }
93
94    Ok(results)
95}

97/// Read and validate the PE header at the given virtual address.
98///
99/// Returns `(has_mz, has_pe, size_of_image)`.
100/// If the memory is unreadable, returns `(false, false, 0)`.
101fn read_pe_header<P: PhysicalMemoryProvider>(
102    reader: &ObjectReader<P>,
103    image_base: u64,
104) -> (bool, bool, u32) {
105    // Read first 512 bytes (enough for DOS header + PE header + optional header)
106    let Ok(header) = reader.read_bytes(image_base, 512) else {
107        return (false, false, 0);
108    };
109
110    // Check MZ magic (0x4D5A)
111    if header.len() < 2 || header[0] != 0x4D || header[1] != 0x5A {
112        return (false, false, 0);
113    }
114
115    // Read e_lfanew (PE header offset) at DOS header offset 0x3C
116    if header.len() < 0x40 {
117        return (true, false, 0);
118    }
119    let e_lfanew = u32::from_le_bytes([header[0x3C], header[0x3D], header[0x3E], header[0x3F]]);
120    let pe_off = e_lfanew as usize;
121
122    // Validate PE signature "PE\0\0"
123    if header.len() < pe_off + 4 {
124        // PE header is beyond our 512-byte read — try reading more
125        let Ok(extended) = reader.read_bytes(image_base + u64::from(e_lfanew), 128) else {
126            return (true, false, 0);
127        };
128        if extended.len() < 4
129            || extended[0] != b'P'
130            || extended[1] != b'E'
131            || extended[2] != 0
132            || extended[3] != 0
133        {
134            return (true, false, 0);
135        }
136        // Optional header starts at +24 from PE sig, SizeOfImage at +56 within optional
137        if extended.len() < 24 + 56 + 4 {
138            return (true, true, 0);
139        }
140        let soi = u32::from_le_bytes([extended[80], extended[81], extended[82], extended[83]]);
141        return (true, true, soi);
142    }
143
144    if header[pe_off] != b'P'
145        || header[pe_off + 1] != b'E'
146        || header[pe_off + 2] != 0
147        || header[pe_off + 3] != 0
148    {
149        return (true, false, 0);
150    }
151
152    // Optional header starts at pe_off + 24, SizeOfImage at optional + 56
153    let soi_off = pe_off + 24 + 56;
154    if header.len() < soi_off + 4 {
155        return (true, true, 0);
156    }
157    let size_of_image = u32::from_le_bytes([
158        header[soi_off],
159        header[soi_off + 1],
160        header[soi_off + 2],
161        header[soi_off + 3],
162    ]);
163
164    (true, true, size_of_image)
165}

167/// Get the image size from the first entry in `InLoadOrderModuleList`.
168///
169/// The first entry typically represents the process's main executable.
170/// Returns 0 if the LDR data is unreadable.
171fn ldr_first_image_size<P: PhysicalMemoryProvider>(reader: &ObjectReader<P>, peb_addr: u64) -> u64 {
172    // PEB.Ldr → _PEB_LDR_DATA
173    let ldr_addr: u64 = match reader.read_field(peb_addr, "_PEB", "Ldr") {
174        Ok(v) => v,
175        Err(_) => return 0,
176    };
177    if ldr_addr == 0 {
178        return 0;
179    }
180
181    // _PEB_LDR_DATA.InLoadOrderModuleList.Flink → first _LDR_DATA_TABLE_ENTRY
182    let Some(in_load_off) = reader
183        .symbols()
184        .field_offset("_PEB_LDR_DATA", "InLoadOrderModuleList")
185    else {
186        return 0;
187    };
188
189    let first_entry: u64 =
190        match reader.read_field(ldr_addr, "_PEB_LDR_DATA", "InLoadOrderModuleList") {
191            Ok(v) => v,
192            Err(_) => return 0,
193        };
194
195    // Check if the list is empty (Flink points back to list head)
196    let list_head = ldr_addr + in_load_off;
197    if first_entry == 0 || first_entry == list_head {
198        return 0;
199    }
200
201    // The Flink points to the InLoadOrderLinks of the first entry,
202    // which IS the start of _LDR_DATA_TABLE_ENTRY (offset 0).
203    // _LDR_DATA_TABLE_ENTRY.SizeOfImage is at offset 64 (from the preset).
204    // Read it as u32 (it's an unsigned long).
205    let size_off = first_entry + 64; // SizeOfImage offset within _LDR_DATA_TABLE_ENTRY
206    match reader.read_bytes(size_off, 4) {
207        Ok(bytes) if bytes.len() == 4 => {
208            u64::from(u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]))
209        }
210        _ => 0,
211    }
212}

226    /// Build a minimal valid PE header at a given offset in a buffer.
227    /// Returns the expected SizeOfImage value.
228    fn write_pe_header(buf: &mut [u8], offset: usize, size_of_image: u32) -> u32 {
229        // DOS header: MZ magic + e_lfanew at offset 0x3C
230        buf[offset] = 0x4D; // 'M'
231        buf[offset + 1] = 0x5A; // 'Z'
232        let pe_offset: u32 = 0x80; // PE header at relative offset 0x80
233        buf[offset + 0x3C..offset + 0x40].copy_from_slice(&pe_offset.to_le_bytes());
234        // PE signature
235        let pe_abs = offset + pe_offset as usize;
236        buf[pe_abs] = b'P';
237        buf[pe_abs + 1] = b'E';
238        buf[pe_abs + 2] = 0;
239        buf[pe_abs + 3] = 0;
240        // COFF header: Machine (2 bytes) + ... SizeOfOptionalHeader at +20
241        // Optional header starts at pe_abs + 24
242        // PE32+ magic
243        let opt_abs = pe_abs + 24;
244        buf[opt_abs..opt_abs + 2].copy_from_slice(&0x020Bu16.to_le_bytes()); // PE32+
245                                                                             // SizeOfImage is at optional header offset 56
246        buf[opt_abs + 56..opt_abs + 60].copy_from_slice(&size_of_image.to_le_bytes());
247        size_of_image
248    }

250    /// Set up a single-process memory layout for hollowing tests.
251    /// Returns (cr3, mem, ps_head_vaddr).
252    fn build_single_process_memory(
253        pid: u64,
254        name: &str,
255        peb_paddr: u64,
256        image_base_vaddr: u64,
257        image_base_paddr: u64,
258        image_data: &[u8],
259        ldr_size: u64,
260    ) -> (u64, memf_core::test_builders::SyntheticPhysMem, u64) {
261        let head_vaddr: u64 = 0xFFFF_8000_0010_0000;
262        let eproc_vaddr: u64 = 0xFFFF_8000_0020_0000;
263        let peb_vaddr: u64 = 0x0000_0000_7FFE_0000; // user-mode PEB
264
265        let head_paddr: u64 = 0x0080_0000;
266        let eproc_paddr: u64 = 0x0090_0000;
267
268        // LDR data addresses
269        let ldr_vaddr: u64 = 0x0000_0000_7FFD_0000;
270        let ldr_paddr: u64 = 0x00C0_0000;
271        let ldr_entry_vaddr: u64 = 0x0000_0000_7FFD_1000;
272        let ldr_entry_paddr: u64 = 0x00C1_0000;
273        let ldr_strings_vaddr: u64 = 0x0000_0000_7FFD_2000;
274        let ldr_strings_paddr: u64 = 0x00C2_0000;
275
276        let mut head_data = vec![0u8; 4096];
277        let mut eproc_data = vec![0u8; 8192];
278        let mut peb_data = vec![0u8; 4096];
279        let mut ldr_data = vec![0u8; 4096];
280        let mut ldr_entry_data = vec![0u8; 4096];
281        let mut ldr_string_data = vec![0u8; 4096];
282
283        let active_links_off: u64 = 0x448;
284
285        // PsActiveProcessHead: Flink → eproc.ActiveProcessLinks
286        let eproc_links = eproc_vaddr + active_links_off;
287        head_data[0..8].copy_from_slice(&eproc_links.to_le_bytes());
288        head_data[8..16].copy_from_slice(&eproc_links.to_le_bytes());
289
290        // _EPROCESS
291        // Pcb@0x0 (_KPROCESS): DirectoryTableBase@0x28
292        // We'll set CR3 later from PageTableBuilder
293        // CreateTime@0x430, ExitTime@0x438
294        eproc_data[0x430..0x438].copy_from_slice(&132800000000000000u64.to_le_bytes());
295        eproc_data[0x438..0x440].copy_from_slice(&0u64.to_le_bytes());
296        // UniqueProcessId@0x440
297        eproc_data[0x440..0x448].copy_from_slice(&pid.to_le_bytes());
298        // ActiveProcessLinks@0x448: circular back to head
299        eproc_data[0x448..0x450].copy_from_slice(&head_vaddr.to_le_bytes());
300        eproc_data[0x450..0x458].copy_from_slice(&head_vaddr.to_le_bytes());
301        // InheritedFromUniqueProcessId@0x540
302        eproc_data[0x540..0x548].copy_from_slice(&0u64.to_le_bytes());
303        // Peb@0x550
304        eproc_data[0x550..0x558].copy_from_slice(&peb_vaddr.to_le_bytes());
305        // ImageFileName@0x5A8
306        let name_bytes = name.as_bytes();
307        let copy_len = name_bytes.len().min(15);
308        eproc_data[0x5A8..0x5A8 + copy_len].copy_from_slice(&name_bytes[..copy_len]);
309        // ThreadListHead@0x30 (empty: points to itself)
310        let thread_list_head = eproc_vaddr + 0x30;
311        eproc_data[0x30..0x38].copy_from_slice(&thread_list_head.to_le_bytes());
312        eproc_data[0x38..0x40].copy_from_slice(&thread_list_head.to_le_bytes());
313
314        // _PEB
315        // ImageBaseAddress@0x10
316        peb_data[0x10..0x18].copy_from_slice(&image_base_vaddr.to_le_bytes());
317        // Ldr@0x18 → ldr_vaddr
318        peb_data[0x18..0x20].copy_from_slice(&ldr_vaddr.to_le_bytes());
319        // ProcessParameters@0x20 (null — not needed for hollowing)
320        peb_data[0x20..0x28].copy_from_slice(&0u64.to_le_bytes());
321
322        // _PEB_LDR_DATA at ldr_vaddr
323        // InLoadOrderModuleList@16: Flink → ldr_entry.InLoadOrderLinks
324        let ldr_entry_in_load = ldr_entry_vaddr; // InLoadOrderLinks is at offset 0
325        ldr_data[16..24].copy_from_slice(&ldr_entry_in_load.to_le_bytes()); // Flink
326        let ldr_in_load_head = ldr_vaddr + 16;
327        ldr_data[24..32].copy_from_slice(&ldr_entry_in_load.to_le_bytes()); // Blink
328
329        // _LDR_DATA_TABLE_ENTRY at ldr_entry_vaddr
330        // InLoadOrderLinks@0: circular back to PEB_LDR_DATA.InLoadOrderModuleList
331        ldr_entry_data[0..8].copy_from_slice(&ldr_in_load_head.to_le_bytes()); // Flink
332        ldr_entry_data[8..16].copy_from_slice(&ldr_in_load_head.to_le_bytes()); // Blink
333                                                                                // DllBase@48
334        ldr_entry_data[48..56].copy_from_slice(&image_base_vaddr.to_le_bytes());
335        // SizeOfImage@64 (as u32, but stored in u64 slot effectively — read 4 bytes)
336        ldr_entry_data[64..68].copy_from_slice(&(ldr_size as u32).to_le_bytes());
337        // FullDllName@72 (_UNICODE_STRING)
338        let full_name = utf16le(name);
339        let full_len = full_name.len() as u16;
340        ldr_entry_data[72..74].copy_from_slice(&full_len.to_le_bytes());
341        ldr_entry_data[74..76].copy_from_slice(&(full_len + 2).to_le_bytes());
342        ldr_entry_data[80..88].copy_from_slice(&ldr_strings_vaddr.to_le_bytes());
343        // BaseDllName@88
344        ldr_entry_data[88..90].copy_from_slice(&full_len.to_le_bytes());
345        ldr_entry_data[90..92].copy_from_slice(&(full_len + 2).to_le_bytes());
346        let base_str_vaddr = ldr_strings_vaddr + 0x100;
347        ldr_entry_data[96..104].copy_from_slice(&base_str_vaddr.to_le_bytes());
348
349        // String buffers
350        ldr_string_data[0..full_name.len()].copy_from_slice(&full_name);
351        ldr_string_data[0x100..0x100 + full_name.len()].copy_from_slice(&full_name);
352
353        // Image data at image_base_paddr
354        let mut image_page = vec![0u8; 4096];
355        let copy_len = image_data.len().min(4096);
356        image_page[..copy_len].copy_from_slice(&image_data[..copy_len]);
357
358        let (cr3, mut mem) = PageTableBuilder::new()
359            // Kernel space mappings (identity-like, high addresses)
360            .map_4k(head_vaddr, head_paddr, flags::WRITABLE)
361            .map_4k(eproc_vaddr, eproc_paddr, flags::WRITABLE)
362            .map_4k(eproc_vaddr + 0x1000, eproc_paddr + 0x1000, flags::WRITABLE)
363            // User space mappings (process-specific via shared CR3 for simplicity)
364            .map_4k(peb_vaddr, peb_paddr, flags::WRITABLE | flags::USER)
365            .map_4k(
366                image_base_vaddr,
367                image_base_paddr,
368                flags::WRITABLE | flags::USER,
369            )
370            .map_4k(ldr_vaddr, ldr_paddr, flags::WRITABLE | flags::USER)
371            .map_4k(
372                ldr_entry_vaddr,
373                ldr_entry_paddr,
374                flags::WRITABLE | flags::USER,
375            )
376            .map_4k(
377                ldr_strings_vaddr,
378                ldr_strings_paddr,
379                flags::WRITABLE | flags::USER,
380            )
381            .write_phys(head_paddr, &head_data)
382            .write_phys(eproc_paddr, &eproc_data[..4096])
383            .write_phys(eproc_paddr + 0x1000, &eproc_data[4096..])
384            .write_phys(peb_paddr, &peb_data)
385            .write_phys(image_base_paddr, &image_page)
386            .write_phys(ldr_paddr, &ldr_data)
387            .write_phys(ldr_entry_paddr, &ldr_entry_data)
388            .write_phys(ldr_strings_paddr, &ldr_string_data)
389            .build();
390
391        // Write CR3 into _EPROCESS.Pcb.DirectoryTableBase
392        // CR3 is at physical eproc_paddr + 0x28
393        mem.write_u64(eproc_paddr + 0x28, cr3);
394
395        (cr3, mem, head_vaddr)
396    }

526        let eproc_links = eproc_vaddr + active_links_off;

528        head_data[8..16].copy_from_slice(&eproc_links.to_le_bytes());
529
530        eproc_data[0x28..0x30].copy_from_slice(&0x1AB000u64.to_le_bytes()); // CR3
531        eproc_data[0x430..0x438].copy_from_slice(&132800000000000000u64.to_le_bytes());
532        eproc_data[0x440..0x448].copy_from_slice(&4u64.to_le_bytes()); // pid=4
533        eproc_data[0x448..0x450].copy_from_slice(&head_vaddr.to_le_bytes());
534        eproc_data[0x450..0x458].copy_from_slice(&head_vaddr.to_le_bytes());
535        eproc_data[0x550..0x558].copy_from_slice(&0u64.to_le_bytes()); // PEB=0
536        eproc_data[0x5A8..0x5AE].copy_from_slice(b"System");
537        let tl = eproc_vaddr + 0x30;
538        eproc_data[0x30..0x38].copy_from_slice(&tl.to_le_bytes());
539        eproc_data[0x38..0x40].copy_from_slice(&tl.to_le_bytes());
540
541        let (cr3, mem) = PageTableBuilder::new()
542            .map_4k(head_vaddr, head_paddr, flags::WRITABLE)
543            .map_4k(eproc_vaddr, eproc_paddr, flags::WRITABLE)
544            .map_4k(eproc_vaddr + 0x1000, eproc_paddr + 0x1000, flags::WRITABLE)
545            .write_phys(head_paddr, &head_data)
546            .write_phys(eproc_paddr, &eproc_data[..4096])
547            .write_phys(eproc_paddr + 0x1000, &eproc_data[4096..])
548            .build();
549
550        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
551        let reader = ObjectReader::new(vas, Box::new(resolver));
552
553        let results = check_hollowing(&reader, head_vaddr).unwrap();
554        // System has PEB=0, should be skipped entirely
555        assert!(results.is_empty(), "System (PEB=0) should be skipped");
556    }

598    /// read_pe_header: MZ present but header is exactly 0x40 bytes —
599    /// e_lfanew could be within the buffer but the PE sig is at the boundary.
600    /// Exercises the `header.len() < pe_off + 4` extended-read branch.
601    #[test]
602    fn read_pe_header_pe_beyond_512_bytes_extended_read() {
603        let isf = IsfBuilder::windows_kernel_preset().build_json();
604        let resolver = IsfResolver::from_value(&isf).unwrap();
605
606        let peb_paddr: u64 = 0x00B0_0000;
607        let image_base_vaddr: u64 = 0x0000_0000_0040_0000;
608        let image_base_paddr: u64 = 0x00A0_0000;
609
610        // Put PE signature beyond offset 512 so the extended read path is taken.
611        // e_lfanew = 0x200 (512), so pe_off = 0x200 which equals header.len() (512)
612        // → the condition `header.len() < pe_off + 4` is true.
613        let pe_offset: u32 = 0x200; // exactly at 512
614        let mut image_data = vec![0u8; 4096];
615        image_data[0] = 0x4D; // 'M'
616        image_data[1] = 0x5A; // 'Z'
617        image_data[0x3C..0x40].copy_from_slice(&pe_offset.to_le_bytes());
618        // PE signature at offset 0x200
619        image_data[0x200] = b'P';
620        image_data[0x201] = b'E';
621        image_data[0x202] = 0;
622        image_data[0x203] = 0;
623        // SizeOfImage at pe_off + 24 + 56 = 0x200 + 80 = 0x250
624        let soi: u32 = 0x2_0000;
625        image_data[0x250..0x254].copy_from_slice(&soi.to_le_bytes());
626
627        let ldr_size: u64 = u64::from(soi);
628
629        let (_cr3, mem, head_vaddr) = build_single_process_memory(
630            777,
631            "legit.exe",
632            peb_paddr,
633            image_base_vaddr,
634            image_base_paddr,
635            &image_data,
636            ldr_size,
637        );
638
639        let vas = VirtualAddressSpace::new(mem, _cr3, TranslationMode::X86_64FourLevel);
640        let reader = ObjectReader::new(vas, Box::new(resolver));
641
642        let results = check_hollowing(&reader, head_vaddr).unwrap();
643        assert_eq!(results.len(), 1);
644        assert!(results[0].has_mz, "MZ should be detected");
645        assert!(
646            results[0].has_pe,
647            "PE signature beyond 512 bytes should be read via extended path"
648        );
649        assert!(
650            !results[0].suspicious,
651            "matching sizes should not be flagged"
652        );
653    }

656    /// Exercises the `first_entry == list_head` guard in ldr_first_image_size.
657    #[test]

659        let isf = IsfBuilder::windows_kernel_preset().build_json();
