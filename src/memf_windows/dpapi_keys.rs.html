1//! DPAPI master key extraction from LSASS memory.
2//!
3//! The Data Protection API (DPAPI) uses master keys to encrypt/decrypt
4//! user secrets (browser credentials, WiFi passwords, private keys, etc.).
5//! Master keys are loaded into the LSASS process from disk
6//! (`%APPDATA%\Microsoft\Protect\<SID>\`) and cached in memory.
7//!
8//! This module provides:
9//! - `DpapiMasterKeyInfo` struct for recovered master key metadata
10//! - `walk_dpapi_master_keys` — graceful-degradation stub (RED phase)
11//!   returning empty when lsasrv.dll symbols are absent
12//!
13//! A full implementation requires parsing `lsasrv!g_MasterKeyCache` linked
14//! list entries, decrypting key blobs using the DPAPI internal session key,
15//! and validating key GUIDs against on-disk master key files.
16
17use memf_core::object_reader::ObjectReader;
18use memf_format::PhysicalMemoryProvider;
19
20use crate::Result;
21
22/// A DPAPI master key recovered from LSASS memory.
23#[derive(Debug, Clone, serde::Serialize)]
24pub struct DpapiMasterKeyInfo {
25    /// Master key GUID (matches filename under %APPDATA%\Microsoft\Protect\<SID>\).
26    pub guid: String,
27    /// Master key blob version.
28    pub version: u32,
29    /// Master key flags.
30    pub flags: u32,
31    /// Human-readable description from the key blob.
32    pub description: String,
33    /// Decrypted master key bytes (if recoverable from memory).
34    pub master_key: Vec<u8>,
35    /// True if the master key belongs to a non-standard or unexpected user.
36    pub is_suspicious: bool,
37}
38
39/// Classify a DPAPI master key entry as suspicious.
40///
41/// A master key is suspicious when it is found cached for an unexpected
42/// user account (e.g. a service account or machine account GUID not
43/// associated with any interactive logon session).
44pub fn classify_dpapi_master_key(guid: &str, description: &str) -> bool {
45    // Suspicious: empty GUID (memory corruption or uninitialized entry)
46    // or description containing unusual markers
47    guid.is_empty()
48        || guid == "{00000000-0000-0000-0000-000000000000}"
49        || description.to_ascii_lowercase().contains("backdoor")
50}
51
52/// Format a 16-byte Windows GUID (little-endian) as `{XXXXXXXX-XXXX-XXXX-XXXX-XXXXXXXXXXXX}`.
53///
54/// Windows stores GUIDs with the first three components in little-endian byte order:
55/// - Data1: 4 bytes LE → u32
56/// - Data2: 2 bytes LE → u16
57/// - Data3: 2 bytes LE → u16
58/// - Data4: 8 bytes as-is
59fn format_guid(raw: &[u8; 16]) -> String {
60    let d1 = u32::from_le_bytes([raw[0], raw[1], raw[2], raw[3]]);
61    let d2 = u16::from_le_bytes([raw[4], raw[5]]);
62    let d3 = u16::from_le_bytes([raw[6], raw[7]]);
63    format!(
64        "{{{:08X}-{:04X}-{:04X}-{:02X}{:02X}-{:02X}{:02X}{:02X}{:02X}{:02X}{:02X}}}",
65        d1, d2, d3, raw[8], raw[9], raw[10], raw[11], raw[12], raw[13], raw[14], raw[15]
66    )
67}

69/// Walk LSASS memory for cached DPAPI master keys via `lsasrv!g_MasterKeyCache`.
70///
71/// Returns `Ok(Vec::new())` when `lsasrv`-related symbols (e.g.
72/// `g_MasterKeyCache`) are absent from the symbol table (graceful degradation).
73///
74/// Walks the doubly-linked `LIST_ENTRY` chain rooted at `g_MasterKeyCache`.
75/// For each entry it reads the GUID (offset +0x18), blob pointer (+0x28),
76/// and blob length (+0x30), then reads up to 512 bytes of blob data.
77pub fn walk_dpapi_master_keys<P: PhysicalMemoryProvider>(
78    reader: &ObjectReader<P>,
79) -> Result<Vec<DpapiMasterKeyInfo>> {
80    // Graceful degradation: require g_MasterKeyCache symbol from lsasrv.dll
81    let list_head_va = match reader.symbols().symbol_address("g_MasterKeyCache") {
82        None => return Ok(Vec::new()),
83        Some(va) => va,
84    };
85
86    let mut results = Vec::new();
87    let mut seen = std::collections::HashSet::new();
88
89    // Read the first Flink from the list head
90    let flink_bytes = match reader.read_bytes(list_head_va, 8) {
91        Ok(b) => b,
92        Err(_) => return Ok(results),
93    };
94    let mut flink_va = u64::from_le_bytes(flink_bytes.try_into().unwrap_or([0u8; 8]));
95
96    for _ in 0..1000 {
97        // Stop conditions: full circle or null pointer
98        if flink_va == list_head_va || flink_va == 0 {
99            break;
100        }
101        // Cycle detection
102        if !seen.insert(flink_va) {
103            break;
104        }
105
106        let entry_va = flink_va;
107
108        // Read GUID: 16 bytes at entry_va + 0x18
109        if let Ok(guid_bytes) = reader.read_bytes(entry_va + 0x18, 16) {
110            let mut guid_raw = [0u8; 16];
111            guid_raw.copy_from_slice(&guid_bytes);
112
113            // Read blob pointer: 8 bytes at entry_va + 0x28
114            // Read blob length: 4 bytes at entry_va + 0x30
115            let blob = match (
116                reader.read_bytes(entry_va + 0x28, 8),
117                reader.read_bytes(entry_va + 0x30, 4),
118            ) {
119                (Ok(ptr_b), Ok(len_b)) => {
120                    let blob_ptr = u64::from_le_bytes(ptr_b.try_into().unwrap_or([0u8; 8]));
121                    let blob_len =
122                        u32::from_le_bytes(len_b.try_into().unwrap_or([0u8; 4])) as usize;
123                    if blob_len > 0 && blob_len <= 512 && blob_ptr != 0 {
124                        reader.read_bytes(blob_ptr, blob_len).unwrap_or_default()
125                    } else {
126                        Vec::new()
127                    }
128                }
129                _ => Vec::new(),
130            };
131
132            let guid = format_guid(&guid_raw);
133            results.push(DpapiMasterKeyInfo {
134                guid,
135                version: 1,
136                flags: 0,
137                description: String::new(),
138                master_key: blob,
139                is_suspicious: false,
140            });
141        }
142
143        // Advance: read Flink at entry_va + 0x00
144        match reader.read_bytes(entry_va, 8) {
145            Ok(b) => flink_va = u64::from_le_bytes(b.try_into().unwrap_or([0u8; 8])),
146            Err(_) => break,
147        }
148    }
149
150    Ok(results)
151}

213    /// Walk g_MasterKeyCache linked list with a synthetic one-entry cache.
214    #[test]
215    fn walk_master_key_cache_with_synthetic_entry() {
216        use memf_core::object_reader::ObjectReader;
217        use memf_core::test_builders::{flags, PageTableBuilder};
218        use memf_core::vas::{TranslationMode, VirtualAddressSpace};
219        use memf_symbols::isf::IsfResolver;
220        use memf_symbols::test_builders::IsfBuilder;
221
222        // VAs for our synthetic structures
223        const LIST_HEAD_VA: u64 = 0xFFFF_8800_0001_0000;
224        const ENTRY_VA: u64 = 0xFFFF_8800_0002_0000;
225        const BLOB_VA: u64 = 0xFFFF_8800_0003_0000;
226        // Corresponding PAs
227        const LIST_HEAD_PA: u64 = 0x0010_0000;
228        const ENTRY_PA: u64 = 0x0020_0000;
229        const BLOB_PA: u64 = 0x0030_0000;
230
231        let guid_bytes = [
232            0xAAu8, 0xBB, 0xCC, 0xDD, 0xEE, 0xFF, 0x11, 0x22, 0x33, 0x44, 0x55, 0x66, 0x77, 0x88,
233            0x99, 0x00,
234        ];
235        let blob_data = [0x42u8; 64];
236        const BLOB_LEN: u32 = 64;
237
238        // List head: Flink → ENTRY_VA
239        let mut list_head = [0u8; 16];
240        list_head[0..8].copy_from_slice(&ENTRY_VA.to_le_bytes()); // Flink
241        list_head[8..16].copy_from_slice(&ENTRY_VA.to_le_bytes()); // Blink
242
243        // Cache entry at ENTRY_VA
244        let mut entry = [0u8; 0x40];
245        // [0x00] Flink → LIST_HEAD_VA (one-entry list, circles back)
246        entry[0x00..0x08].copy_from_slice(&LIST_HEAD_VA.to_le_bytes());
247        // [0x08] Blink
248        entry[0x08..0x10].copy_from_slice(&LIST_HEAD_VA.to_le_bytes());
249        // [0x18] GUID (16 bytes)
250        entry[0x18..0x28].copy_from_slice(&guid_bytes);
251        // [0x28] blob pointer
252        entry[0x28..0x30].copy_from_slice(&BLOB_VA.to_le_bytes());
253        // [0x30] blob length
254        entry[0x30..0x34].copy_from_slice(&BLOB_LEN.to_le_bytes());
255
256        let (cr3, mem) = PageTableBuilder::new()
257            .map_4k(LIST_HEAD_VA, LIST_HEAD_PA, flags::WRITABLE)
258            .write_phys(LIST_HEAD_PA, &list_head)
259            .map_4k(ENTRY_VA, ENTRY_PA, flags::WRITABLE)
260            .write_phys(ENTRY_PA, &entry)
261            .map_4k(BLOB_VA, BLOB_PA, flags::WRITABLE)
262            .write_phys(BLOB_PA, &blob_data)
263            .build();
264
265        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
266
267        // ISF: g_MasterKeyCache symbol points to the list head VA
268        let isf = IsfBuilder::new()
269            .add_symbol("g_MasterKeyCache", LIST_HEAD_VA)
270            .build_json();
271        let resolver = IsfResolver::from_value(&isf).unwrap();
272        let reader = ObjectReader::new(vas, Box::new(resolver));
273
274        let results = walk_dpapi_master_keys(&reader).expect("walk must succeed");
275        assert_eq!(results.len(), 1, "should find exactly one cache entry");
276        assert_eq!(results[0].master_key.len(), 64, "blob should be 64 bytes");
277        assert_eq!(results[0].master_key, blob_data, "blob bytes must match");
278        // GUID formatting: first 4 bytes LE → 0xDDCCBBAA
279        assert!(
280            results[0].guid.starts_with("{DDCCBBAA"),
281            "GUID prefix must match LE-decoded bytes"
282        );
283    }

298        let page_paddr: u64 = 0x0080_0000;
299        let ptb = PageTableBuilder::new()
300            .map_4k(page_vaddr, page_paddr, flags::WRITABLE)
