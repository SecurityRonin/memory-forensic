1//! PEB image path masquerading detection.
2//!
3//! Detects processes that have modified their PEB `ImagePathName` or
4//! `CommandLine` to masquerade as legitimate system processes. Malware
5//! commonly overwrites PEB fields to appear as `svchost.exe`, `csrss.exe`,
6//! or other system processes. This compares the PEB image path against the
7//! `_EPROCESS.ImageFileName`.
8//!
9//! MITRE ATT&CK T1036.005 (Masquerading: Match Legitimate Name or Location).
10
11use forensicnomicon::processes::is_masquerade_target;
12use memf_core::object_reader::ObjectReader;
13use memf_format::PhysicalMemoryProvider;
14
15use crate::unicode::read_unicode_string;
16use crate::{Error, Result};
17
18/// Information about a potential PEB masquerade for a single process.
19#[derive(Debug, Clone, serde::Serialize)]
20pub struct PebMasqueradeInfo {
21    /// Process ID.
22    pub pid: u32,
23    /// Image file name from `_EPROCESS.ImageFileName`.
24    pub eprocess_name: String,
25    /// Image path from `_RTL_USER_PROCESS_PARAMETERS.ImagePathName`.
26    pub peb_image_path: String,
27    /// Command line from `_RTL_USER_PROCESS_PARAMETERS.CommandLine`.
28    pub peb_command_line: String,
29    /// Whether this process is likely masquerading.
30    pub is_masquerading: bool,
31}
32
33/// Pure classifier: determine if a process is masquerading based on its
34/// EPROCESS name and PEB image path.
35///
36/// Returns `true` (masquerading) when:
37/// - The PEB image path filename differs from the EPROCESS name AND the
38///   EPROCESS name matches a high-value target (svchost.exe, csrss.exe, etc.)
39/// - The PEB image path is empty but the EPROCESS name is not (PEB wiped)
40///
41/// Returns `false` when:
42/// - Both names match (case-insensitive)
43/// - Both are empty
44/// - Names differ but the EPROCESS name is not a high-value target
45pub fn classify_peb_masquerade(eprocess_name: &str, peb_image_path: &str) -> bool {
46    // Both empty — nothing to compare.
47    if eprocess_name.is_empty() && peb_image_path.is_empty() {
48        return false;
49    }
50
51    // PEB wiped: path is empty but EPROCESS has a name.
52    if peb_image_path.is_empty() && !eprocess_name.is_empty() {
53        return true;
54    }
55
56    // Extract filename from the PEB image path (last path component).
57    let peb_filename = peb_image_path
58        .rsplit('\\')
59        .next()
60        .or_else(|| peb_image_path.rsplit('/').next())
61        .unwrap_or(peb_image_path);
62
63    // Case-insensitive comparison.
64    if eprocess_name.eq_ignore_ascii_case(peb_filename) {
65        return false;
66    }
67
68    // Names differ — only flag as masquerading if the EPROCESS name is a
69    // high-value target that attackers commonly impersonate.
70    is_masquerade_target(eprocess_name)
71}

73/// Walk a single process's PEB to detect image path masquerading.
74///
75/// Reads the PEB `ImagePathName` and `CommandLine` from
76/// `_RTL_USER_PROCESS_PARAMETERS`, then classifies the result against the
77/// EPROCESS `ImageFileName`.
78///
79/// Returns `Ok(None)` for kernel processes (PEB address is 0/null).
80pub fn walk_peb_masquerade<P: PhysicalMemoryProvider>(
81    reader: &ObjectReader<P>,
82    eprocess_addr: u64,
83    pid: u32,
84    eprocess_name: &str,
85) -> Result<Option<PebMasqueradeInfo>> {
86    // Read PEB address from _EPROCESS.Peb
87    let peb_addr: u64 = reader.read_field(eprocess_addr, "_EPROCESS", "Peb")?;
88
89    // Kernel processes have no PEB (address is 0/null).
90    if peb_addr == 0 {
91        return Ok(None);
92    }
93
94    // Read PEB.ProcessParameters pointer.
95    let params_ptr: u64 = reader.read_field(peb_addr, "_PEB", "ProcessParameters")?;
96    if params_ptr == 0 {
97        return Ok(None);
98    }
99
100    // Resolve ImagePathName offset within _RTL_USER_PROCESS_PARAMETERS.
101    let image_path_offset = reader
102        .symbols()
103        .field_offset("_RTL_USER_PROCESS_PARAMETERS", "ImagePathName")
104        .ok_or_else(|| Error::MissingField {
105            struct_name: "_RTL_USER_PROCESS_PARAMETERS".into(),
106            field_name: "ImagePathName".into(),
107        })?;
108    let image_path_ustr_addr = params_ptr.wrapping_add(image_path_offset);
109    let peb_image_path = read_unicode_string(reader, image_path_ustr_addr)?;
110
111    // Resolve CommandLine offset within _RTL_USER_PROCESS_PARAMETERS.
112    let cmdline_offset = reader
113        .symbols()
114        .field_offset("_RTL_USER_PROCESS_PARAMETERS", "CommandLine")
115        .ok_or_else(|| Error::MissingField {
116            struct_name: "_RTL_USER_PROCESS_PARAMETERS".into(),
117            field_name: "CommandLine".into(),
118        })?;
119    let cmdline_ustr_addr = params_ptr.wrapping_add(cmdline_offset);
120    let peb_command_line = read_unicode_string(reader, cmdline_ustr_addr)?;
121
122    // Classify the result.
123    let is_masquerading = classify_peb_masquerade(eprocess_name, &peb_image_path);
124
125    Ok(Some(PebMasqueradeInfo {
126        pid,
127        eprocess_name: eprocess_name.to_string(),
128        peb_image_path,
129        peb_command_line,
130        is_masquerading,
131    }))
132}

225    /// Encode a Rust &str as UTF-16LE bytes.
226    fn utf16le_bytes(s: &str) -> Vec<u8> {
227        s.encode_utf16().flat_map(u16::to_le_bytes).collect()
228    }

230    /// Build a _UNICODE_STRING in a byte buffer at the given offset.
231    /// Layout: [offset..+2] Length, [offset+2..+4] MaximumLength,
232    ///         [offset+8..+16] Buffer pointer.
233    fn build_unicode_string_at(buf: &mut [u8], offset: usize, length: u16, buffer_ptr: u64) {
234        buf[offset..offset + 2].copy_from_slice(&length.to_le_bytes());
235        buf[offset + 2..offset + 4].copy_from_slice(&length.to_le_bytes());
236        buf[offset + 8..offset + 16].copy_from_slice(&buffer_ptr.to_le_bytes());
237    }

239    fn make_win_reader(ptb: PageTableBuilder) -> ObjectReader<SyntheticPhysMem> {
240        let isf = IsfBuilder::windows_kernel_preset().build_json();
241        let resolver = IsfResolver::from_value(&isf).unwrap();
242        let (cr3, mem) = ptb.build();
243        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
244        ObjectReader::new(vas, Box::new(resolver))
245    }

271    #[test]
272    fn walk_detects_masquerade() {
273        // Layout:
274        //   Page 1 (eproc): _EPROCESS with Peb pointing to page 2
275        //   Page 2 (peb):   PEB + _RTL_USER_PROCESS_PARAMETERS + string data
276        //
277        // EPROCESS name is "svchost.exe" but PEB ImagePathName points to
278        // "C:\Temp\evil.exe" → masquerade detected.
279
280        let eproc_vaddr: u64 = 0xFFFF_8000_0010_0000;
281        let eproc_paddr: u64 = 0x0010_0000;
282        let peb_vaddr: u64 = 0xFFFF_8000_0020_0000;
283        let peb_paddr: u64 = 0x0020_0000;
284
285        // Build EPROCESS page: write PEB pointer at offset 0x550.
286        let mut eproc_data = vec![0u8; 4096];
287        eproc_data[EPROCESS_PEB as usize..EPROCESS_PEB as usize + 8]
288            .copy_from_slice(&peb_vaddr.to_le_bytes());
289
290        // Build PEB page:
291        //   [0x20..0x28]: ProcessParameters pointer → same page at offset 0x100
292        //   [0x100 + 0x60]: ImagePathName UNICODE_STRING
293        //   [0x100 + 0x70]: CommandLine UNICODE_STRING
294        //   String data at offset 0x400 and 0x500
295        let params_offset: usize = 0x100;
296        let params_vaddr = peb_vaddr + params_offset as u64;
297
298        let image_path = r"C:\Temp\evil.exe";
299        let image_path_utf16 = utf16le_bytes(image_path);
300        let image_path_len = image_path_utf16.len() as u16;
301        let image_path_str_offset: usize = 0x400;
302        let image_path_str_vaddr = peb_vaddr + image_path_str_offset as u64;
303
304        let cmdline = r"C:\Temp\evil.exe --payload";
305        let cmdline_utf16 = utf16le_bytes(cmdline);
306        let cmdline_len = cmdline_utf16.len() as u16;
307        let cmdline_str_offset: usize = 0x500;
308        let cmdline_str_vaddr = peb_vaddr + cmdline_str_offset as u64;
309
310        let mut peb_data = vec![0u8; 4096];
311
312        // PEB.ProcessParameters pointer
313        peb_data[PEB_PROCESS_PARAMETERS as usize..PEB_PROCESS_PARAMETERS as usize + 8]
314            .copy_from_slice(&params_vaddr.to_le_bytes());
315
316        // ImagePathName UNICODE_STRING at ProcessParameters + 0x60
317        build_unicode_string_at(
318            &mut peb_data,
319            params_offset + PARAMS_IMAGE_PATH_NAME as usize,
320            image_path_len,
321            image_path_str_vaddr,
322        );
323
324        // CommandLine UNICODE_STRING at ProcessParameters + 0x70
325        build_unicode_string_at(
326            &mut peb_data,
327            params_offset + PARAMS_COMMAND_LINE as usize,
328            cmdline_len,
329            cmdline_str_vaddr,
330        );
331
332        // String data
333        peb_data[image_path_str_offset..image_path_str_offset + image_path_utf16.len()]
334            .copy_from_slice(&image_path_utf16);
335        peb_data[cmdline_str_offset..cmdline_str_offset + cmdline_utf16.len()]
336            .copy_from_slice(&cmdline_utf16);
337
338        let ptb = PageTableBuilder::new()
339            .map_4k(eproc_vaddr, eproc_paddr, flags::WRITABLE)
340            .write_phys(eproc_paddr, &eproc_data)
341            .map_4k(peb_vaddr, peb_paddr, flags::WRITABLE)
342            .write_phys(peb_paddr, &peb_data);
343
344        let reader = make_win_reader(ptb);
345        let result = walk_peb_masquerade(&reader, eproc_vaddr, 1234, "svchost.exe")
346            .unwrap()
347            .expect("should return Some for process with PEB");
348
349        assert_eq!(result.pid, 1234);
350        assert_eq!(result.eprocess_name, "svchost.exe");
351        assert_eq!(result.peb_image_path, image_path);
352        assert_eq!(result.peb_command_line, cmdline);
353        assert!(result.is_masquerading);
354    }
