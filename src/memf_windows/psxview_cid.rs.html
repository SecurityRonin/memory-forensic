1//! `PspCidTable` process enumeration — the 5th psxview column.
2//!
3//! Enumerates processes by walking `PspCidTable`, the kernel handle table that
4//! maps process/thread IDs to their `_EPROCESS` / `_ETHREAD` objects.  Because
5//! this table is separate from the `ActiveProcessLinks` doubly-linked list,
6//! cross-referencing the two views reveals DKOM-hidden processes.
7//!
8//! `_EX_HANDLE_TABLE.TableCode` encodes the table level in its low 2 bits:
9//! - `0` — direct: array of `_EX_HANDLE_TABLE_ENTRY` at 16-byte strides.
10//! - `1` — one level of indirection.
11//! - `2` — two levels of indirection.
12//!
13//! For simplicity this walker currently handles only the direct (level 0) case.
14
15use memf_core::object_reader::ObjectReader;
16use memf_format::PhysicalMemoryProvider;
17
18use crate::Result;

20/// A process entry found by walking `PspCidTable`.
21#[derive(Debug, Clone, serde::Serialize)]
22pub struct CidTableEntry {
23    /// Process ID.
24    pub pid: u32,
25    /// Virtual address of the corresponding `_EPROCESS` structure.
26    pub eprocess_addr: u64,
27    /// Image name read from `_EPROCESS.ImageFileName`.
28    pub image_name: String,
29    /// `true` when this PID was also found in `ActiveProcessLinks`.
30    pub in_active_list: bool,
31    /// `true` when the PID is in `PspCidTable` but absent from `ActiveProcessLinks`
32    /// (potential DKOM hiding).
33    pub is_hidden: bool,
34}
35
36/// Maximum number of CID table entries to scan (safety limit).
37pub(crate) const MAX_CID_ENTRIES: u64 = 16384;
38
39/// Walk `PspCidTable` and return one [`CidTableEntry`] per process found.
40///
41/// Returns an empty `Vec` when the `PspCidTable` symbol is absent
42/// (graceful degradation).
43pub fn walk_psp_cid_table<P: PhysicalMemoryProvider>(
44    reader: &ObjectReader<P>,
45) -> Result<Vec<CidTableEntry>> {
46    // Graceful degradation: require PspCidTable symbol.
47    let cid_table_ptr = match reader.symbols().symbol_address("PspCidTable") {
48        Some(addr) => addr,
49        None => return Ok(Vec::new()),
50    };
51
52    // PspCidTable stores a pointer to _HANDLE_TABLE; dereference it.
53    let ht_addr: u64 = {
54        let bytes = reader.read_bytes(cid_table_ptr, 8)?;
55        u64::from_le_bytes(bytes.try_into().map_err(|_| crate::Error::WalkFailed {
56            walker: "psxview_cid",
57            reason: "failed to read PspCidTable pointer".into(),
58        })?)
59    };
60
61    if ht_addr == 0 {
62        return Ok(Vec::new());
63    }
64
65    // Read TableCode from the _HANDLE_TABLE
66    let table_code: u64 = reader.read_field(ht_addr, "_HANDLE_TABLE", "TableCode")?;
67
68    // Level = low 2 bits of TableCode
69    let level = table_code & 0x3;
70    let base_addr = table_code & !0x3;
71
72    if base_addr == 0 {
73        return Ok(Vec::new());
74    }
75
76    // Only support level-0 (flat) tables for now
77    if level != 0 {
78        return Err(crate::Error::WalkFailed {
79            walker: "psxview_cid",
80            reason: format!("PspCidTable level-{level} not yet supported; results incomplete"),
81        });
82    }
83
84    let entry_size = reader
85        .symbols()
86        .struct_size("_HANDLE_TABLE_ENTRY")
87        .ok_or_else(|| crate::Error::WalkFailed {
88            walker: "psxview_cid",
89            reason: "missing _HANDLE_TABLE_ENTRY size".into(),
90        })?;
91
92    // Read NextHandleNeedingPool to determine entry count
93    let next_handle: u32 = reader.read_field(ht_addr, "_HANDLE_TABLE", "NextHandleNeedingPool")?;
94
95    // Number of entries = next_handle / 4 (handle values are index * 4)
96    let num_entries = u64::from(next_handle) / 4;
97    let num_entries = num_entries.min(MAX_CID_ENTRIES);
98
99    let mut entries = Vec::new();
100
101    // In PspCidTable, handle value = index * 4 = PID for processes.
102    // Index 0 is reserved.
103    for idx in 1..num_entries {
104        let entry_addr = base_addr + idx * entry_size;
105
106        let obj_ptr: u64 =
107            match reader.read_field(entry_addr, "_HANDLE_TABLE_ENTRY", "ObjectPointerBits") {
108                Ok(v) => v,
109                Err(_) => continue,
110            };
111
112        if obj_ptr == 0 {
113            continue;
114        }
115
116        // ObjectPointerBits is shifted right by 4; reconstruct the pointer
117        // with kernel canonical high bits set.
118        let object_addr = (obj_ptr << 4) | 0xFFFF_0000_0000_0000;
119
120        // object_addr points to _OBJECT_HEADER; the body (_EPROCESS) follows
121        // at the Body field offset (typically 0x30).
122        let body_offset = reader
123            .symbols()
124            .field_offset("_OBJECT_HEADER", "Body")
125            .unwrap_or(0x30);
126        let eprocess_addr = object_addr.wrapping_add(body_offset);
127
128        // Verify this is a process by reading PID and checking it matches
129        let pid: u64 = match reader.read_field(eprocess_addr, "_EPROCESS", "UniqueProcessId") {
130            Ok(v) => v,
131            Err(_) => continue,
132        };
133
134        // In PspCidTable, handle value = pid = idx * 4
135        let expected_pid = idx * 4;
136        if pid != expected_pid {
137            // Not a process entry (could be a thread), or corrupted
138            continue;
139        }
140
141        let image_name = reader
142            .read_field_string(eprocess_addr, "_EPROCESS", "ImageFileName", 15)
143            .unwrap_or_default();
144
145        entries.push(CidTableEntry {
146            pid: pid as u32,
147            eprocess_addr,
148            image_name,
149            in_active_list: false, // caller must cross-reference
150            is_hidden: false,      // caller must compute
151        });
152    }
153
154    Ok(entries)
155}

157/// Returns `true` when a PID found in `PspCidTable` is absent from
158/// the `ActiveProcessLinks` list, indicating potential DKOM hiding.
159pub fn classify_hidden_process(in_active_list: bool) -> bool {
160    !in_active_list
161}

172    fn make_reader_no_symbols() -> ObjectReader<memf_core::test_builders::SyntheticPhysMem> {
173        let isf = IsfBuilder::new().build_json();
174        let resolver = IsfResolver::from_value(&isf).unwrap();
175
176        let page_vaddr: u64 = 0xFFFF_8000_0010_0000;
177        let page_paddr: u64 = 0x0080_0000;
178        let ptb = PageTableBuilder::new()
179            .map_4k(page_vaddr, page_paddr, flags::WRITABLE)
180            .write_phys(page_paddr, &[0u8; 16]);
181        let (cr3, mem) = ptb.build();
182
183        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
184        ObjectReader::new(vas, Box::new(resolver))
185    }

187    /// When `PspCidTable` symbol is absent the walker returns empty.
188    #[test]
189    fn walk_psp_cid_table_no_symbol_returns_empty() {
190        let reader = make_reader_no_symbols();
191        let results = walk_psp_cid_table(&reader).unwrap();
192        assert!(results.is_empty());
193    }

202    // RED: missing _HANDLE_TABLE_ENTRY size → WalkFailed
203    #[test]
204    fn walk_psp_cid_table_missing_handle_table_entry_size_returns_walk_failed() {
205        use memf_core::test_builders::{flags, PageTableBuilder};
206        use memf_core::vas::{TranslationMode, VirtualAddressSpace};
207        use memf_symbols::test_builders::IsfBuilder;
208
209        // Use windows_kernel_preset (has _HANDLE_TABLE.TableCode etc.) but remove
210        // _HANDLE_TABLE_ENTRY so struct_size returns None.
211        // Set up: PspCidTable → ht_vaddr; ht.TableCode = entries_vaddr (level 0);
212        // ht.NextHandleNeedingPool = 8 so num_entries = 2.
213        let psp_cid_paddr: u64 = 0x0010_0000;
214        let ht_paddr: u64 = 0x0020_0000;
215        let entries_paddr: u64 = 0x0030_0000;
216        let psp_cid_vaddr: u64 = 0xFFFFF805_5A500000;
217        let ht_vaddr: u64 = 0xFFFFF805_5A600000;
218        let entries_vaddr: u64 = 0xFFFFF805_5A700000;
219
220        let ptb = PageTableBuilder::new()
221            .map_4k(psp_cid_vaddr, psp_cid_paddr, flags::WRITABLE)
222            .map_4k(ht_vaddr, ht_paddr, flags::WRITABLE)
223            .map_4k(entries_vaddr, entries_paddr, flags::WRITABLE)
224            // PspCidTable pointer → ht_vaddr
225            .write_phys_u64(psp_cid_paddr, ht_vaddr)
226            // _HANDLE_TABLE.TableCode at preset offset 0x08 = entries_vaddr (level 0)
227            .write_phys_u64(ht_paddr + 0x08, entries_vaddr)
228            // _HANDLE_TABLE.NextHandleNeedingPool at preset offset 0x3C = 8 (num_entries=2)
229            .write_phys(ht_paddr + 0x3C, &8u32.to_le_bytes());
230
231        let mut isf_json = IsfBuilder::windows_kernel_preset()
232            .add_symbol("PspCidTable", psp_cid_vaddr)
233            .build_json();
234        // Remove _HANDLE_TABLE_ENTRY so struct_size returns None
235        if let Some(user_types) = isf_json["user_types"].as_object_mut() {
236            user_types.remove("_HANDLE_TABLE_ENTRY");
237        }
238        let resolver = memf_symbols::isf::IsfResolver::from_value(&isf_json).unwrap();
239        let (cr3, mem) = ptb.build();
240        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
241        let reader = memf_core::object_reader::ObjectReader::new(vas, Box::new(resolver));
242
243        let result = walk_psp_cid_table(&reader);
244        assert!(
245            matches!(
246                result,
247                Err(crate::Error::WalkFailed { walker, .. }) if walker == "psxview_cid"
248            ),
249            "expected WalkFailed(psxview_cid), got {result:?}"
250        );
251    }
