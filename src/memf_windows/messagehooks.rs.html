1//! Windows message hook detection (`SetWindowsHookEx`).
2//!
3//! `SetWindowsHookEx()` installs message hooks that intercept keyboard/mouse
4//! events — used by keyloggers and credential stealers. The win32k subsystem
5//! maintains hook chains via `_HOOK` structures linked from `_DESKTOP` objects.
6//!
7//! This module enumerates installed message hooks from kernel memory and
8//! classifies them as suspicious based on hook type and owning module.
9
10use memf_core::object_reader::ObjectReader;
11use memf_format::PhysicalMemoryProvider;
12
13use crate::unicode::read_unicode_string;
14
15/// Maximum number of hooks to enumerate (safety limit).
16const MAX_HOOKS: usize = 4096;
17
18/// Number of hook types in the aphkStart array (WH_MSGFILTER=-1 through WH_MOUSE_LL=14).
19const HOOK_TYPE_COUNT: usize = 16;
20
21/// Information about a Windows message hook recovered from kernel memory.
22#[derive(Debug, Clone, serde::Serialize)]
23pub struct MessageHookInfo {
24    /// Virtual address of the hook object.
25    pub address: u64,
26    /// Hook type name (e.g., `WH_KEYBOARD_LL`, `WH_MOUSE_LL`).
27    pub hook_type: String,
28    /// Process ID that installed the hook.
29    pub owner_pid: u32,
30    /// Address of the hook procedure.
31    pub hook_proc_addr: u64,
32    /// DLL containing the hook procedure.
33    pub module_name: String,
34    /// Whether this hook looks suspicious (heuristic flag).
35    pub is_suspicious: bool,
36}
37
38/// Map a raw hook type value to its symbolic name.
39///
40/// Hook type values range from -1 (`WH_MSGFILTER`) to 14 (`WH_MOUSE_LL`).
41/// The kernel stores these as `u32` where -1 is `0xFFFF_FFFF`.
42pub fn hook_type_name(raw: u32) -> String {
43    // WH_MSGFILTER = -1 (0xFFFFFFFF), WH_MIN = 0, ..., WH_MOUSE_LL = 14
44    // No entry for type 8 (reserved/not defined in standard headers)
45    match raw {
46        0xFFFF_FFFF => "WH_MSGFILTER".to_string(),
47        0 => "WH_JOURNALRECORD".to_string(),
48        1 => "WH_JOURNALPLAYBACK".to_string(),
49        2 => "WH_KEYBOARD".to_string(),
50        3 => "WH_GETMESSAGE".to_string(),
51        4 => "WH_CALLWNDPROC".to_string(),
52        5 => "WH_CBT".to_string(),
53        6 => "WH_SYSMSGFILTER".to_string(),
54        7 => "WH_MOUSE".to_string(),
55        9 => "WH_DEBUG".to_string(),
56        10 => "WH_SHELL".to_string(),
57        11 => "WH_FOREGROUNDIDLE".to_string(),
58        12 => "WH_CALLWNDPROCRET".to_string(),
59        13 => "WH_KEYBOARD_LL".to_string(),
60        14 => "WH_MOUSE_LL".to_string(),
61        n => format!("WH_UNKNOWN({n})"),
62    }
63}

65/// Classify a message hook as suspicious based on hook type and module.
66///
67/// Returns `true` when the hook matches common keylogger/stealer patterns:
68/// - `WH_KEYBOARD_LL` or `WH_MOUSE_LL` from non-system modules
69/// - Any hook from temp, appdata, or downloads directories
70/// - Empty module name (injected or unknown origin)
71///
72/// Known benign Windows modules: `user32.dll`, `imm32.dll`, `msctf.dll`.
73pub fn classify_message_hook(hook_type: &str, module: &str) -> bool {
74    const BENIGN_MODULES: &[&str] = &["user32.dll", "imm32.dll", "msctf.dll"];
75
76    if module.is_empty() {
77        return true;
78    }
79
80    let lower_module = module.to_lowercase();
81
82    // Any hook from temp/appdata/downloads is suspicious regardless of type
83    if lower_module.contains("\\temp\\")
84        || lower_module.contains("\\appdata\\")
85        || lower_module.contains("\\downloads\\")
86    {
87        return true;
88    }
89
90    // Check if module ends with a known benign module name
91    let is_benign_module = BENIGN_MODULES.iter().any(|&m| lower_module.ends_with(m));
92
93    if is_benign_module {
94        return false;
95    }
96
97    // WH_KEYBOARD_LL or WH_MOUSE_LL from non-system modules is suspicious
98    if hook_type == "WH_KEYBOARD_LL" || hook_type == "WH_MOUSE_LL" {
99        return true;
100    }
101
102    false
103}

105/// Enumerate Windows message hooks from kernel memory.
106///
107/// Walks the win32k desktop hook chains starting from the `grpWinStaList`
108/// symbol. For each window station, iterates desktops and reads the
109/// `aphkStart` array of hook chain heads.
110///
111/// Returns `Ok(Vec::new())` if the required symbols are not present.
112pub fn walk_message_hooks<P: PhysicalMemoryProvider>(
113    reader: &ObjectReader<P>,
114) -> crate::Result<Vec<MessageHookInfo>> {
115    let list_sym = reader.symbols().symbol_address("grpWinStaList");
116    let Some(list_sym) = list_sym else {
117        return Ok(Vec::new());
118    };
119
120    // Read the first winsta pointer from the symbol
121    let first_ws: u64 = match reader.read_bytes(list_sym, 8) {
122        Ok(b) => u64::from_le_bytes(b[..8].try_into().unwrap()),
123        Err(_) => return Ok(Vec::new()),
124    };
125    if first_ws == 0 {
126        return Ok(Vec::new());
127    }
128
129    // Field offsets for _WINSTATION_OBJECT
130    let ws_next_off = reader
131        .symbols()
132        .field_offset("_WINSTATION_OBJECT", "rpwinstaNext")
133        .unwrap_or(0x28) as u64;
134    let ws_desk_off = reader
135        .symbols()
136        .field_offset("_WINSTATION_OBJECT", "rpdeskList")
137        .unwrap_or(0x30) as u64;
138
139    // Field offsets for tagDESKTOP
140    let desk_next_off = reader
141        .symbols()
142        .field_offset("tagDESKTOP", "rpdeskNext")
143        .unwrap_or(0x28) as u64;
144    let desk_aphk_off = reader
145        .symbols()
146        .field_offset("tagDESKTOP", "aphkStart")
147        .unwrap_or(0x38) as u64;
148
149    // Field offsets for _HOOK
150    let hook_next_off = reader
151        .symbols()
152        .field_offset("_HOOK", "phkNext")
153        .unwrap_or(0x00) as u64;
154    let hook_type_off = reader
155        .symbols()
156        .field_offset("_HOOK", "iHook")
157        .unwrap_or(0x08) as u64;
158    let hook_proc_off = reader
159        .symbols()
160        .field_offset("_HOOK", "pfn")
161        .unwrap_or(0x10) as u64;
162    let hook_ihmod_off = reader
163        .symbols()
164        .field_offset("_HOOK", "ihmod")
165        .unwrap_or(0x18) as u64;
166    let hook_pti_off = reader
167        .symbols()
168        .field_offset("_HOOK", "pti")
169        .unwrap_or(0x20) as u64;
170
171    // Field offsets for PID chain from tagTHREADINFO
172    let threadinfo_eprocess_off = reader
173        .symbols()
174        .field_offset("tagTHREADINFO", "pEThread")
175        .unwrap_or(0x00) as u64;
176    let ethread_process_off = reader
177        .symbols()
178        .field_offset("_ETHREAD", "ThreadsProcess")
179        .unwrap_or(0x220) as u64;
180    let pid_off = reader
181        .symbols()
182        .field_offset("_EPROCESS", "UniqueProcessId")
183        .unwrap_or(0x440) as u64;
184
185    let mut results = Vec::new();
186    let mut hook_count = 0usize;
187
188    let mut ws_addr = first_ws;
189    while ws_addr != 0 {
190        // Walk desktops of this station
191        let first_desk: u64 = match reader.read_bytes(ws_addr + ws_desk_off, 8) {
192            Ok(b) => u64::from_le_bytes(b[..8].try_into().unwrap()),
193            Err(_) => 0,
194        };
195
196        let mut desk_addr = first_desk;
197        while desk_addr != 0 {
198            // Walk aphkStart array
199            for i in 0..HOOK_TYPE_COUNT {
200                let slot_addr = desk_addr + desk_aphk_off + i as u64 * 8;
201                let head_hook: u64 = match reader.read_bytes(slot_addr, 8) {
202                    Ok(b) => u64::from_le_bytes(b[..8].try_into().unwrap()),
203                    Err(_) => 0,
204                };
205
206                let mut hook_addr = head_hook;
207                let mut seen = std::collections::HashSet::new();
208                while hook_addr != 0 && hook_count < MAX_HOOKS {
209                    if !seen.insert(hook_addr) {
210                        break;
211                    }
212                    hook_count += 1;
213
214                    // Read hook type
215                    let i_hook_raw: u32 = reader
216                        .read_bytes(hook_addr + hook_type_off, 4)
217                        .map(|b| u32::from_le_bytes(b[..4].try_into().unwrap()))
218                        .unwrap_or(0xFFFF_FFFF);
219                    let hook_type = hook_type_name(i_hook_raw);
220
221                    // Read proc address
222                    let hook_proc_addr: u64 = reader
223                        .read_bytes(hook_addr + hook_proc_off, 8)
224                        .map(|b| u64::from_le_bytes(b[..8].try_into().unwrap()))
225                        .unwrap_or(0);
226
227                    // Read ihmod to determine module origin (<=0xFFFF → injected/unknown)
228                    let ihmod: i32 = reader
229                        .read_bytes(hook_addr + hook_ihmod_off, 4)
230                        .map(|b| i32::from_le_bytes(b[..4].try_into().unwrap()))
231                        .unwrap_or(-1);
232                    let module_name = if ihmod > 0 {
233                        // Try to read module name via _UNICODE_STRING at ihmod offset
234                        let uni_addr = hook_addr + hook_ihmod_off + 8;
235                        read_unicode_string(reader, uni_addr).unwrap_or_default()
236                    } else {
237                        String::new()
238                    };
239
240                    // Extract owner PID via tagTHREADINFO chain
241                    let pti: u64 = reader
242                        .read_bytes(hook_addr + hook_pti_off, 8)
243                        .map(|b| u64::from_le_bytes(b[..8].try_into().unwrap()))
244                        .unwrap_or(0);
245                    let owner_pid = extract_pid_from_threadinfo(
246                        reader,
247                        pti,
248                        threadinfo_eprocess_off,
249                        ethread_process_off,
250                        pid_off,
251                    );
252
253                    let is_suspicious = classify_message_hook(&hook_type, &module_name);
254                    results.push(MessageHookInfo {
255                        address: hook_addr,
256                        hook_type,
257                        owner_pid,
258                        hook_proc_addr,
259                        module_name,
260                        is_suspicious,
261                    });
262
263                    // Follow phkNext
264                    hook_addr = match reader.read_bytes(hook_addr + hook_next_off, 8) {
265                        Ok(b) => u64::from_le_bytes(b[..8].try_into().unwrap()),
266                        Err(_) => 0,
267                    };
268                }
269            }
270
271            // Next desktop
272            desk_addr = match reader.read_bytes(desk_addr + desk_next_off, 8) {
273                Ok(b) => u64::from_le_bytes(b[..8].try_into().unwrap()),
274                Err(_) => 0,
275            };
276        }
277
278        // Next window station
279        ws_addr = match reader.read_bytes(ws_addr + ws_next_off, 8) {
280            Ok(b) => u64::from_le_bytes(b[..8].try_into().unwrap()),
281            Err(_) => 0,
282        };
283    }
284
285    Ok(results)
286}

288/// Extract a PID from a `tagTHREADINFO` pointer by chasing through
289/// `pEThread` → `ThreadsProcess` → `UniqueProcessId`.
290fn extract_pid_from_threadinfo<P: PhysicalMemoryProvider>(
291    reader: &ObjectReader<P>,
292    threadinfo_addr: u64,
293    threadinfo_eprocess_off: u64,
294    ethread_process_off: u64,
295    pid_off: u64,
296) -> u32 {
297    if threadinfo_addr == 0 {
298        return 0;
299    }
300    let ethread: u64 = reader
301        .read_bytes(threadinfo_addr + threadinfo_eprocess_off, 8)
302        .map(|b| u64::from_le_bytes(b[..8].try_into().unwrap()))
303        .unwrap_or(0);
304    if ethread == 0 {
305        return 0;
306    }
307    let eprocess: u64 = reader
308        .read_bytes(ethread + ethread_process_off, 8)
309        .map(|b| u64::from_le_bytes(b[..8].try_into().unwrap()))
310        .unwrap_or(0);
311    if eprocess == 0 {
312        return 0;
313    }
314    reader
315        .read_bytes(eprocess + pid_off, 8)
316        .map(|b| u64::from_le_bytes(b[..8].try_into().unwrap()) as u32)
317        .unwrap_or(0)
318}

637    /// walker: winsta → desktop → deskinfo → one hook in aphkStart[0] → one hook returned.
638    ///
639    /// This exercises the inner aphk loop, hook chain traversal, and hook field reads.
640    /// The hook has ihmod <= 0xFFFF so module_name = "" → classify_message_hook → suspicious.
641    #[test]
642    fn walk_message_hooks_one_hook_inner_loop() {
643        const SYM_VADDR: u64 = 0xFFFF_8000_0090_0000;
644        const SYM_PADDR: u64 = 0x0090_0000;
645        const WS_VADDR: u64 = 0xFFFF_8000_0091_0000;
646        const WS_PADDR: u64 = 0x0091_0000;
647        const DESK_VADDR: u64 = 0xFFFF_8000_0092_0000;
648        const DESK_PADDR: u64 = 0x0092_0000;
649        const HOOK_VADDR: u64 = 0xFFFF_8000_0093_0000;
650        const HOOK_PADDR: u64 = 0x0093_0000;
651
652        // aphkStart offset = 0x38, phkNext=0, iHook=0x08, pfn=0x10, ihmod=0x18, pti=0x20
653        let isf = IsfBuilder::new()
654            .add_struct("_WINSTATION_OBJECT", 64)
655            .add_field("_WINSTATION_OBJECT", "rpwinstaNext", 0x28, "pointer")
656            .add_field("_WINSTATION_OBJECT", "rpdeskList", 0x30, "pointer")
657            .add_struct("tagDESKTOP", 256)
658            .add_field("tagDESKTOP", "rpdeskNext", 0x28, "pointer")
659            .add_field("tagDESKTOP", "aphkStart", 0x38, "pointer")
660            .add_struct("_HOOK", 64)
661            .add_field("_HOOK", "phkNext", 0x00, "pointer")
662            .add_field("_HOOK", "iHook", 0x08, "unsigned long")
663            .add_field("_HOOK", "pfn", 0x10, "pointer")
664            .add_field("_HOOK", "ihmod", 0x18, "long")
665            .add_field("_HOOK", "pti", 0x20, "pointer")
666            .add_symbol("grpWinStaList", SYM_VADDR)
667            .build_json();
668        let resolver = IsfResolver::from_value(&isf).unwrap();
669
670        let mut sym_page = vec![0u8; 4096];
671        sym_page[0..8].copy_from_slice(&WS_VADDR.to_le_bytes());
672
673        let mut ws_page = vec![0u8; 4096];
674        ws_page[0x28..0x30].copy_from_slice(&0u64.to_le_bytes()); // rpwinstaNext = 0
675        ws_page[0x30..0x38].copy_from_slice(&DESK_VADDR.to_le_bytes()); // rpdeskList
676
677        let mut desk_page = vec![0u8; 4096];
678        desk_page[0x28..0x30].copy_from_slice(&0u64.to_le_bytes()); // rpdeskNext = 0
679                                                                    // aphkStart[0] = HOOK_VADDR
680        desk_page[0x38..0x40].copy_from_slice(&HOOK_VADDR.to_le_bytes());
681
682        let mut hook_page = vec![0u8; 4096];
683        hook_page[0x00..0x08].copy_from_slice(&0u64.to_le_bytes()); // phkNext = 0 (end)
684        hook_page[0x08..0x0c].copy_from_slice(&13u32.to_le_bytes()); // iHook = WH_KEYBOARD_LL
685        hook_page[0x10..0x18].copy_from_slice(&0xDEAD_BEEFu64.to_le_bytes()); // pfn
686        hook_page[0x18..0x1c].copy_from_slice(&(-1i32).to_le_bytes()); // ihmod = -1 → no module
687        hook_page[0x20..0x28].copy_from_slice(&0u64.to_le_bytes()); // pti = 0
688
689        let (cr3, mem) = PageTableBuilder::new()
690            .map_4k(SYM_VADDR, SYM_PADDR, flags::WRITABLE)
691            .write_phys(SYM_PADDR, &sym_page)
692            .map_4k(WS_VADDR, WS_PADDR, flags::WRITABLE)
693            .write_phys(WS_PADDR, &ws_page)
694            .map_4k(DESK_VADDR, DESK_PADDR, flags::WRITABLE)
695            .write_phys(DESK_PADDR, &desk_page)
696            .map_4k(HOOK_VADDR, HOOK_PADDR, flags::WRITABLE)
697            .write_phys(HOOK_PADDR, &hook_page)
698            .build();
699        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
700        let reader = ObjectReader::new(vas, Box::new(resolver));
701        let hooks = walk_message_hooks(&reader).unwrap();
702        assert_eq!(hooks.len(), 1);
703        assert_eq!(hooks[0].hook_type, "WH_KEYBOARD_LL");
704        assert_eq!(hooks[0].hook_proc_addr, 0xDEAD_BEEF);
705        assert!(hooks[0].is_suspicious); // empty module → suspicious
706    }
