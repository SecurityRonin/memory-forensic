1//! COM object hijacking detection.
2//!
3//! Detects when a CLSID has a `HKCU\Software\Classes\CLSID\...\InprocServer32`
4//! value that overrides the trusted `HKCR` path, a technique used by malware
5//! to load arbitrary DLLs into COM clients without admin privileges.
6
7use memf_core::object_reader::ObjectReader;
8use memf_format::PhysicalMemoryProvider;
9
10use crate::Result;
11
12// ── Registry cell layout constants ──────────────────────────────────────
13
14const HBIN_START: u64 = 0x1000;
15const ROOT_CELL_OFFSET: u64 = 0x24;
16const NK_SIG: u16 = 0x6B6E;
17
18/// Stable subkey count offset within a key node cell payload.
19const NK_STABLE_COUNT: usize = 0x14;
20/// Stable subkeys list cell index offset.
21const NK_STABLE_LIST: usize = 0x1C;
22/// Value count offset.
23const NK_VALUE_COUNT: usize = 0x24;
24/// Values list cell index offset.
25const NK_VALUES_LIST: usize = 0x28;
26/// Name length offset.
27const NK_NAME_LEN: usize = 0x48;
28/// Name data offset.
29const NK_NAME_DATA: usize = 0x4C;
30
31const VK_SIG: u16 = 0x6B76;
32const VK_DATA_LEN_OFF: usize = 0x04;
33const VK_DATA_OFF_OFF: usize = 0x08;
34
35/// Maximum CLSID subkeys to enumerate (safety cap).
36const MAX_CLSIDS: usize = 50_000;
37
38// ── Output type ──────────────────────────────────────────────────────────
39
40/// A COM class registration where HKCU overrides HKCR (potential hijack).
41#[derive(Debug, Clone, serde::Serialize)]
42pub struct ComHijackInfo {
43    /// The CLSID string, e.g. `{xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx}`.
44    pub clsid: String,
45    /// `HKCR\CLSID\<clsid>\InprocServer32` path (system-wide, trusted).
46    pub hkcr_path: String,
47    /// `HKCU\Software\Classes\CLSID\<clsid>\InprocServer32` path (override).
48    pub hkcu_path: String,
49    /// DLL path registered under HKCR (empty if not present).
50    pub hkcr_server: String,
51    /// DLL path registered under HKCU (the hijacked value).
52    pub hkcu_server: String,
53    /// `true` when the HKCU server path is in an unusual/writable location.
54    pub is_suspicious: bool,
55}
56
57// ── Classification ───────────────────────────────────────────────────────
58
59/// Returns `true` when the HKCU COM server path looks like a hijack.
60///
61/// A path is suspicious when it resides in a user-writable directory
62/// (`%TEMP%`, `%APPDATA%`, `%DOWNLOADS%`, `%PUBLIC%`, `%PROGRAMDATA%`)
63/// **or** when it overrides a non-empty HKCR registration with a different path.
64pub fn classify_com_hijack(hkcr_server: &str, hkcu_server: &str) -> bool {
65    if hkcu_server.is_empty() {
66        return false;
67    }
68    let lower = hkcu_server.to_ascii_lowercase();
69    lower.contains("\\temp\\")
70        || lower.contains("\\appdata\\")
71        || lower.contains("\\downloads\\")
72        || lower.contains("\\public\\")
73        || lower.contains("\\programdata\\")
74        // Any HKCU override of a different HKCR registration is a hijack.
75        || (!hkcr_server.is_empty() && !hkcu_server.eq_ignore_ascii_case(hkcr_server))
76}

78// ── Walker ───────────────────────────────────────────────────────────────

80/// Walk the in-memory registry hives for COM hijacking candidates.
81///
82/// Enumerates all CLSIDs under `HKCU\Software\Classes\CLSID` (in the hive at
83/// `hku_hive_addr`), reads each `InprocServer32` default value, looks up the
84/// same CLSID in `HKCR\CLSID` (in the hive at `hkcr_hive_addr`), and compares
85/// the DLL paths.
86///
87/// Either address may be `0` to skip that hive (graceful degradation).
88pub fn walk_com_hijacking<P: PhysicalMemoryProvider>(
89    reader: &ObjectReader<P>,
90    hku_hive_addr: u64,
91    hkcr_hive_addr: u64,
92) -> Result<Vec<ComHijackInfo>> {
93    if hku_hive_addr == 0 {
94        return Ok(Vec::new());
95    }
96
97    // Navigate to `Software\Classes\CLSID` in the HKCU hive.
98    let clsid_cell = match find_key_cell(reader, hku_hive_addr, &["Software", "Classes", "CLSID"]) {
99        Some(c) => c,
100        None => return Ok(Vec::new()),
101    };
102
103    // Enumerate CLSID subkeys.
104    let clsid_children = match list_subkeys(reader, hku_hive_addr, clsid_cell) {
105        Some(v) => v,
106        None => return Ok(Vec::new()),
107    };
108
109    let mut results = Vec::new();
110
111    for (guid_name, guid_cell) in clsid_children.iter().take(MAX_CLSIDS) {
112        // Find InprocServer32 under this GUID key in HKCU.
113        let hkcu_inproc_cell =
114            match find_subkey_named(reader, hku_hive_addr, *guid_cell, "InprocServer32") {
115                Some(c) => c,
116                None => continue,
117            };
118
119        // Read the default value of InprocServer32 in HKCU.
120        let hkcu_server = match read_default_value_string(reader, hku_hive_addr, hkcu_inproc_cell) {
121            Some(s) if !s.is_empty() => s,
122            _ => continue,
123        };
124
125        let hkcu_path = format!(r"HKCU\Software\Classes\CLSID\{guid_name}\InprocServer32");
126
127        // Look up the same CLSID in HKCR.
128        let (hkcr_server, hkcr_path) = if hkcr_hive_addr != 0 {
129            let hkcr_guid_cell =
130                find_key_cell(reader, hkcr_hive_addr, &["CLSID", guid_name.as_str()]);
131            match hkcr_guid_cell {
132                Some(gc) => {
133                    let inproc_cell =
134                        find_subkey_named(reader, hkcr_hive_addr, gc, "InprocServer32");
135                    match inproc_cell {
136                        Some(ic) => {
137                            let srv = read_default_value_string(reader, hkcr_hive_addr, ic)
138                                .unwrap_or_default();
139                            let path = format!(r"HKCR\CLSID\{guid_name}\InprocServer32");
140                            (srv, path)
141                        }
142                        None => (String::new(), String::new()),
143                    }
144                }
145                None => (String::new(), String::new()),
146            }
147        } else {
148            (String::new(), String::new())
149        };
150
151        let is_suspicious = classify_com_hijack(&hkcr_server, &hkcu_server);
152        if is_suspicious {
153            results.push(ComHijackInfo {
154                clsid: guid_name.clone(),
155                hkcr_path,
156                hkcu_path,
157                hkcr_server,
158                hkcu_server,
159                is_suspicious,
160            });
161        }
162    }
163
164    Ok(results)
165}

169/// Compute the virtual address of a cell from its cell index.
170fn cell_vaddr(hive_addr: u64, cell_index: u32) -> u64 {
171    hive_addr
172        .wrapping_add(HBIN_START)
173        .wrapping_add(cell_index as u64)
174}

176/// Read raw cell payload (skipping the 4-byte size header).
177fn read_cell<P: PhysicalMemoryProvider>(reader: &ObjectReader<P>, vaddr: u64) -> Option<Vec<u8>> {
178    reader.read_bytes(vaddr + 4, 4096).ok()
179}

181/// Read the ASCII name from a key node cell payload buffer.
182fn key_node_name(data: &[u8]) -> String {
183    if data.len() < NK_NAME_DATA {
184        return String::new();
185    }
186    let len = u16::from_le_bytes(
187        data[NK_NAME_LEN..NK_NAME_LEN + 2]
188            .try_into()
189            .unwrap_or([0; 2]),
190    ) as usize;
191    let available = data.len().saturating_sub(NK_NAME_DATA);
192    let end = NK_NAME_DATA + len.min(available);
193    String::from_utf8_lossy(&data[NK_NAME_DATA..end]).into_owned()
194}

196/// Navigate a key path (slice of components) from the hive root.
197/// Returns the cell index of the target key, or `None` if not found.
198fn find_key_cell<P: PhysicalMemoryProvider>(
199    reader: &ObjectReader<P>,
200    hive_addr: u64,
201    path: &[&str],
202) -> Option<u32> {
203    let root_bytes = reader.read_bytes(hive_addr + ROOT_CELL_OFFSET, 4).ok()?;
204    let mut current = u32::from_le_bytes(root_bytes[..4].try_into().ok()?);
205
206    for &component in path {
207        current = find_subkey_named(reader, hive_addr, current, component)?;
208    }
209    Some(current)
210}

212/// Find a direct child key by name under `parent_cell`.
213fn find_subkey_named<P: PhysicalMemoryProvider>(
214    reader: &ObjectReader<P>,
215    hive_addr: u64,
216    parent_cell: u32,
217    name: &str,
218) -> Option<u32> {
219    let data = read_cell(reader, cell_vaddr(hive_addr, parent_cell))?;
220    if data.len() < 4 {
221        return None;
222    }
223    let sig = u16::from_le_bytes(data[0..2].try_into().ok()?);
224    if sig != NK_SIG {
225        return None;
226    }
227
228    let count =
229        u32::from_le_bytes(data[NK_STABLE_COUNT..NK_STABLE_COUNT + 4].try_into().ok()?) as usize;
230    if count == 0 {
231        return None;
232    }
233
234    let list_cell = u32::from_le_bytes(data[NK_STABLE_LIST..NK_STABLE_LIST + 4].try_into().ok()?);
235    let list_data = read_cell(reader, cell_vaddr(hive_addr, list_cell))?;
236    if list_data.len() < 4 {
237        return None;
238    }
239
240    let list_sig = u16::from_le_bytes(list_data[0..2].try_into().ok()?);
241    let list_count = u16::from_le_bytes(list_data[2..4].try_into().ok()?) as usize;
242
243    let (entry_size, base) = match list_sig {
244        0x666C | 0x686C => (8usize, 4usize),
245        0x696C => (4usize, 4usize),
246        _ => return None,
247    };
248
249    for i in 0..list_count {
250        let off = base + i * entry_size;
251        if off + 4 > list_data.len() {
252            break;
253        }
254        let child_cell = u32::from_le_bytes(list_data[off..off + 4].try_into().ok()?);
255        let child_data = read_cell(reader, cell_vaddr(hive_addr, child_cell))?;
256        let child_name = key_node_name(&child_data);
257        if child_name.eq_ignore_ascii_case(name) {
258            return Some(child_cell);
259        }
260    }
261    None
262}

264/// List all direct subkeys of `parent_cell` as `(name, cell_index)` pairs.
265fn list_subkeys<P: PhysicalMemoryProvider>(
266    reader: &ObjectReader<P>,
267    hive_addr: u64,
268    parent_cell: u32,
269) -> Option<Vec<(String, u32)>> {
270    let data = read_cell(reader, cell_vaddr(hive_addr, parent_cell))?;
271    if data.len() < 4 {
272        return None;
273    }
274    let sig = u16::from_le_bytes(data[0..2].try_into().ok()?);
275    if sig != NK_SIG {
276        return None;
277    }
278
279    let count =
280        u32::from_le_bytes(data[NK_STABLE_COUNT..NK_STABLE_COUNT + 4].try_into().ok()?) as usize;
281    if count == 0 {
282        return Some(Vec::new());
283    }
284
285    let list_cell = u32::from_le_bytes(data[NK_STABLE_LIST..NK_STABLE_LIST + 4].try_into().ok()?);
286    let list_data = read_cell(reader, cell_vaddr(hive_addr, list_cell))?;
287    if list_data.len() < 4 {
288        return None;
289    }
290
291    let list_sig = u16::from_le_bytes(list_data[0..2].try_into().ok()?);
292    let list_count = u16::from_le_bytes(list_data[2..4].try_into().ok()?) as usize;
293
294    let (entry_size, base) = match list_sig {
295        0x666C | 0x686C => (8usize, 4usize),
296        0x696C => (4usize, 4usize),
297        _ => return Some(Vec::new()),
298    };
299
300    let mut children = Vec::with_capacity(list_count.min(MAX_CLSIDS));
301    for i in 0..list_count.min(MAX_CLSIDS) {
302        let off = base + i * entry_size;
303        if off + 4 > list_data.len() {
304            break;
305        }
306        let child_cell = u32::from_le_bytes(list_data[off..off + 4].try_into().ok()?);
307        let child_data = match read_cell(reader, cell_vaddr(hive_addr, child_cell)) {
308            Some(d) => d,
309            None => continue,
310        };
311        let child_name = key_node_name(&child_data);
312        if !child_name.is_empty() {
313            children.push((child_name, child_cell));
314        }
315    }
316    Some(children)
317}

319/// Read the default value (empty name) of a key as a UTF-16LE string.
320///
321/// Returns `None` if the key has no values, the default value is absent, or
322/// any read fails.
323fn read_default_value_string<P: PhysicalMemoryProvider>(
324    reader: &ObjectReader<P>,
325    hive_addr: u64,
326    key_cell: u32,
327) -> Option<String> {
328    let nk_data = read_cell(reader, cell_vaddr(hive_addr, key_cell))?;
329    if nk_data.len() < NK_VALUES_LIST + 4 {
330        return None;
331    }
332    let sig = u16::from_le_bytes(nk_data[0..2].try_into().ok()?);
333    if sig != NK_SIG {
334        return None;
335    }
336
337    let val_count = u32::from_le_bytes(
338        nk_data[NK_VALUE_COUNT..NK_VALUE_COUNT + 4]
339            .try_into()
340            .ok()?,
341    ) as usize;
342    if val_count == 0 {
343        return None;
344    }
345
346    let vl_cell = u32::from_le_bytes(
347        nk_data[NK_VALUES_LIST..NK_VALUES_LIST + 4]
348            .try_into()
349            .ok()?,
350    );
351    let vl_data = read_cell(reader, cell_vaddr(hive_addr, vl_cell))?;
352
353    // Scan values for the default (empty-name) entry.
354    for i in 0..val_count.min(1000) {
355        let off = i * 4;
356        if off + 4 > vl_data.len() {
357            break;
358        }
359        let vk_cell_idx = u32::from_le_bytes(vl_data[off..off + 4].try_into().ok()?);
360        let vk_data = read_cell(reader, cell_vaddr(hive_addr, vk_cell_idx))?;
361        if vk_data.len() < 0x14 {
362            continue;
363        }
364        let vk_sig = u16::from_le_bytes(vk_data[0..2].try_into().ok()?);
365        if vk_sig != VK_SIG {
366            continue;
367        }
368        let name_len = u16::from_le_bytes(vk_data[0x02..0x04].try_into().ok()?) as usize;
369        // Default value has name_len == 0.
370        if name_len != 0 {
371            continue;
372        }
373
374        let data_length_raw = u32::from_le_bytes(
375            vk_data[VK_DATA_LEN_OFF..VK_DATA_LEN_OFF + 4]
376                .try_into()
377                .ok()?,
378        );
379        let data_offset = u32::from_le_bytes(
380            vk_data[VK_DATA_OFF_OFF..VK_DATA_OFF_OFF + 4]
381                .try_into()
382                .ok()?,
383        );
384
385        let raw_bytes: Vec<u8> = if data_length_raw & 0x8000_0000 != 0 {
386            // Inline data (up to 4 bytes) stored in the DataOffset field.
387            let inline_len = ((data_length_raw & 0x7FFF_FFFF) as usize).min(4);
388            data_offset.to_le_bytes()[..inline_len].to_vec()
389        } else if data_length_raw > 0 {
390            let data_vaddr = cell_vaddr(hive_addr, data_offset);
391            let data_cell = read_cell(reader, data_vaddr)?;
392            let len = (data_length_raw as usize).min(data_cell.len());
393            data_cell[..len].to_vec()
394        } else {
395            return None;
396        };
397
398        // Decode UTF-16LE.
399        if raw_bytes.len() < 2 {
400            return None;
401        }
402        let words: Vec<u16> = raw_bytes
403            .chunks_exact(2)
404            .map(|c| u16::from_le_bytes([c[0], c[1]]))
405            .take_while(|&w| w != 0)
406            .collect();
407        return Some(String::from_utf16_lossy(&words).into());
408    }
409    None
410}

444    /// Wrap `data` in a cell: i32 size (negative = allocated) + data + alignment padding.
445    fn build_cell(data: &[u8]) -> Vec<u8> {
446        let total = ((4 + data.len() + 7) & !7) as i32;
447        let mut cell = Vec::with_capacity(total as usize);
448        cell.extend_from_slice(&(-total).to_le_bytes());
449        cell.extend_from_slice(data);
450        cell.resize(total as usize, 0);
451        cell
452    }

454    /// Build an nk cell data buffer (the payload after the 4-byte size header).
455    fn build_nk(
456        name: &str,
457        stable_subkey_count: u32,
458        stable_subkeys_list: u32,
459        value_count: u32,
460        values_list: u32,
461    ) -> Vec<u8> {
462        let name_bytes = name.as_bytes();
463        let mut data = vec![0u8; NK_NAME_DATA + name_bytes.len()];
464        data[0..2].copy_from_slice(&NK_SIG.to_le_bytes());
465        // KEY_COMP_NAME flag (0x0020) → ASCII name
466        data[2..4].copy_from_slice(&0x0020u16.to_le_bytes());
467        data[NK_STABLE_COUNT..NK_STABLE_COUNT + 4]
468            .copy_from_slice(&stable_subkey_count.to_le_bytes());
469        data[NK_STABLE_LIST..NK_STABLE_LIST + 4]
470            .copy_from_slice(&stable_subkeys_list.to_le_bytes());
471        data[NK_VALUE_COUNT..NK_VALUE_COUNT + 4].copy_from_slice(&value_count.to_le_bytes());
472        data[NK_VALUES_LIST..NK_VALUES_LIST + 4].copy_from_slice(&values_list.to_le_bytes());
473        data[NK_NAME_LEN..NK_NAME_LEN + 2]
474            .copy_from_slice(&(name_bytes.len() as u16).to_le_bytes());
475        data[NK_NAME_DATA..NK_NAME_DATA + name_bytes.len()].copy_from_slice(name_bytes);
476        data
477    }

479    /// Build an lf subkey list cell data buffer.
480    /// Each entry is 8 bytes: cell_index (u32) + hash (u32, zeroed).
481    fn build_lf(children: &[u32]) -> Vec<u8> {
482        let mut data = vec![0u8; 4 + children.len() * 8];
483        data[0..2].copy_from_slice(&0x666Cu16.to_le_bytes()); // "lf"
484        data[2..4].copy_from_slice(&(children.len() as u16).to_le_bytes());
485        for (i, &cell) in children.iter().enumerate() {
486            let off = 4 + i * 8;
487            data[off..off + 4].copy_from_slice(&cell.to_le_bytes());
488        }
489        data
490    }
491
492    /// Build a value-list cell: array of u32 cell indices.
493    fn build_value_list(val_cells: &[u32]) -> Vec<u8> {
494        let mut data = vec![0u8; val_cells.len() * 4];
495        for (i, &c) in val_cells.iter().enumerate() {
496            data[i * 4..i * 4 + 4].copy_from_slice(&c.to_le_bytes());
497        }
498        data
499    }

501    /// Build a vk cell data buffer with inline string data (MSB set, data in
502    /// DataOffset field as UTF-16LE first 4 bytes) — only works for short strings.
503    ///
504    /// For longer strings we use an external data cell; see `build_vk_external`.
505    fn build_vk_external(name: &str, data_cell_index: u32, data_len: u32) -> Vec<u8> {
506        let name_bytes = name.as_bytes();
507        let mut data = vec![0u8; VK_NAME_DATA + name_bytes.len()];
508        data[0..2].copy_from_slice(&VK_SIG.to_le_bytes());
509        data[VK_NAME_LEN..VK_NAME_LEN + 2]
510            .copy_from_slice(&(name_bytes.len() as u16).to_le_bytes());
511        // REG_SZ = 1
512        data[0x0C..0x10].copy_from_slice(&1u32.to_le_bytes());
513        data[VK_DATA_LEN..VK_DATA_LEN + 4].copy_from_slice(&data_len.to_le_bytes());
514        data[VK_DATA_OFF..VK_DATA_OFF + 4].copy_from_slice(&data_cell_index.to_le_bytes());
515        data[VK_NAME_DATA..VK_NAME_DATA + name_bytes.len()].copy_from_slice(name_bytes);
516        data
517    }

519    /// Build a data cell containing a UTF-16LE string.
520    fn build_utf16_data_cell(s: &str) -> Vec<u8> {
521        let utf16: Vec<u8> = s
522            .encode_utf16()
523            .flat_map(u16::to_le_bytes)
524            .chain([0u8, 0u8]) // null terminator
525            .collect();
526        build_cell(&utf16)
527    }

529    /// Build a minimal `ObjectReader` from a `PageTableBuilder`.
530    fn make_reader(
531        ptb: PageTableBuilder,
532    ) -> ObjectReader<memf_core::test_builders::SyntheticPhysMem> {
533        let isf = IsfBuilder::new().build_json();
534        let resolver = IsfResolver::from_value(&isf).unwrap();
535        let (cr3, mem) = ptb.build();
536        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
537        ObjectReader::new(vas, Box::new(resolver))
538    }

540    // ── Hive layout builder ───────────────────────────────────────────────
541    //
542    // For HKCU the registry path under which CLSIDs live is:
543    //   Software\Classes\CLSID\{guid}\InprocServer32
544    //
545    // For HKCR the path is:
546    //   CLSID\{guid}\InprocServer32
547    //
548    // Both hives are built flat in a single 4-KB HBIN page.
549    // We use a simple bump-allocator: cell offsets start at 0x20 and advance.
550    //
551    // Hive layout (all offsets within the HBIN page, i.e. cell indices):
552    //
553    //   HKCU hive (hive_paddr = 0x010000):
554    //     0x0020  root nk  ("ROOT", 1 subkey → lf@0x0100)
555    //     0x0100  lf list  [0x0140]
556    //     0x0140  nk "Software" (1 subkey → lf@0x01C0)
557    //     0x01C0  lf [0x0200]
558    //     0x0200  nk "Classes" (1 subkey → lf@0x0280)
559    //     0x0280  lf [0x02C0]
560    //     0x02C0  nk "CLSID"   (1 subkey → lf@0x0340)
561    //     0x0340  lf [0x0380]
562    //     0x0380  nk "{GUID}"  (1 subkey → lf@0x0400)
563    //     0x0400  lf [0x0440]
564    //     0x0440  nk "InprocServer32" (0 subkeys, 1 value → vl@0x04C0)
565    //     0x04C0  value-list [0x0500]
566    //     0x0500  vk "(default)"  data→0x0580  len=<string length>
567    //     0x0580  data cell: UTF-16LE DLL path
568    //
569    //   HKCR hive (hive_paddr = 0x020000):
570    //     0x0020  root nk  ("ROOT", 1 subkey → lf@0x0100)
571    //     0x0100  lf [0x0140]
572    //     0x0140  nk "CLSID" (1 subkey → lf@0x01C0)
573    //     0x01C0  lf [0x0200]
574    //     0x0200  nk "{GUID}" (1 subkey → lf@0x0280)
575    //     0x0280  lf [0x02C0]
576    //     0x02C0  nk "InprocServer32" (0 subkeys, 1 value → vl@0x0340)
577    //     0x0340  value-list [0x0380]
578    //     0x0380  vk "(default)"  data→0x03C0  len=<string length>
579    //     0x03C0  data cell: UTF-16LE DLL path
580
581    struct HiveLayout {
582        /// Physical base address of the _HBASE_BLOCK page.
583        hive_paddr: u64,
584        /// Virtual base address of the _HBASE_BLOCK page.
585        hive_vaddr: u64,
586        /// Physical base address of the HBIN page (hive + 0x1000).
587        hbin_paddr: u64,
588        /// Virtual base address of the HBIN page.
589        hbin_vaddr: u64,
590        /// 4 KB HBASE_BLOCK page bytes.
591        hbase: Vec<u8>,
592        /// 4 KB HBIN page bytes.
593        hbin: Vec<u8>,
594    }
595
596    /// Build an HKCU hive page-pair.
597    ///
598    /// The CLSID path is `Software\Classes\CLSID\{clsid}\InprocServer32`
599    /// and the default value is `hkcu_dll`.
600    fn build_hkcu_hive(
601        hive_vaddr: u64,
602        hive_paddr: u64,
603        clsid: &str,
604        hkcu_dll: &str,
605    ) -> HiveLayout {
606        let hbin_vaddr = hive_vaddr + HBIN_START;
607        let hbin_paddr = hive_paddr + HBIN_START;
608
609        let root_cell: u32 = 0x0020;
610        let lf_root: u32 = 0x0100;
611        let nk_software: u32 = 0x0140;
612        let lf_software: u32 = 0x01C0;
613        let nk_classes: u32 = 0x0200;
614        let lf_classes: u32 = 0x0280;
615        let nk_clsid: u32 = 0x02C0;
616        let lf_clsid: u32 = 0x0340;
617        let nk_guid: u32 = 0x0380;
618        let lf_guid: u32 = 0x0400;
619        let nk_inproc: u32 = 0x0440;
620        let vl_inproc: u32 = 0x04C0;
621        let vk_default: u32 = 0x0500;
622        let data_cell: u32 = 0x0580;
623
624        let dll_utf16: Vec<u8> = hkcu_dll
625            .encode_utf16()
626            .flat_map(u16::to_le_bytes)
627            .chain([0u8, 0u8])
628            .collect();
629        let dll_data_cell = build_cell(&dll_utf16);
630
631        let mut hbase = vec![0u8; 4096];
632        hbase[0x24..0x28].copy_from_slice(&root_cell.to_le_bytes());
633
634        let mut hbin = vec![0u8; 4096];
635
636        macro_rules! place {
637            ($off:expr, $cell:expr) => {
638                let c = $cell;
639                hbin[$off as usize..$off as usize + c.len()].copy_from_slice(&c);
640            };
641        }
642
643        place!(root_cell, build_cell(&build_nk("ROOT", 1, lf_root, 0, 0)));
644        place!(lf_root, build_cell(&build_lf(&[nk_software])));
645        place!(
646            nk_software,
647            build_cell(&build_nk("Software", 1, lf_software, 0, 0))
648        );
649        place!(lf_software, build_cell(&build_lf(&[nk_classes])));
650        place!(
651            nk_classes,
652            build_cell(&build_nk("Classes", 1, lf_classes, 0, 0))
653        );
654        place!(lf_classes, build_cell(&build_lf(&[nk_clsid])));
655        place!(nk_clsid, build_cell(&build_nk("CLSID", 1, lf_clsid, 0, 0)));
656        place!(lf_clsid, build_cell(&build_lf(&[nk_guid])));
657        place!(nk_guid, build_cell(&build_nk(clsid, 1, lf_guid, 0, 0)));
658        place!(lf_guid, build_cell(&build_lf(&[nk_inproc])));
659        place!(
660            nk_inproc,
661            build_cell(&build_nk("InprocServer32", 0, 0, 1, vl_inproc))
662        );
663        place!(vl_inproc, build_cell(&build_value_list(&[vk_default])));
664        place!(
665            vk_default,
666            build_cell(&build_vk_external("", data_cell, dll_utf16.len() as u32))
667        );
668        place!(data_cell, dll_data_cell);
669
670        HiveLayout {
671            hive_paddr,
672            hive_vaddr,
673            hbin_paddr,
674            hbin_vaddr,
675            hbase,
676            hbin,
677        }
678    }
679
680    /// Build an HKCR hive page-pair.
681    ///
682    /// The CLSID path is `CLSID\{clsid}\InprocServer32` and the default value
683    /// is `hkcr_dll`.  Pass `hkcr_dll = ""` to omit the InprocServer32 key
684    /// entirely (simulates "no HKCR entry").
685    fn build_hkcr_hive(
686        hive_vaddr: u64,
687        hive_paddr: u64,
688        clsid: &str,
689        hkcr_dll: &str,
690    ) -> HiveLayout {
691        let hbin_vaddr = hive_vaddr + HBIN_START;
692        let hbin_paddr = hive_paddr + HBIN_START;
693
694        let root_cell: u32 = 0x0020;
695        let lf_root: u32 = 0x0100;
696        let nk_clsid: u32 = 0x0140;
697        let lf_clsid: u32 = 0x01C0;
698        let nk_guid: u32 = 0x0200;
699
700        let mut hbase = vec![0u8; 4096];
701        hbase[0x24..0x28].copy_from_slice(&root_cell.to_le_bytes());
702
703        let mut hbin = vec![0u8; 4096];
704
705        macro_rules! place {
706            ($off:expr, $cell:expr) => {
707                let c = $cell;
708                hbin[$off as usize..$off as usize + c.len()].copy_from_slice(&c);
709            };
710        }
711
712        if hkcr_dll.is_empty() {
713            // No InprocServer32 — GUID key has no subkeys
714            place!(root_cell, build_cell(&build_nk("ROOT", 1, lf_root, 0, 0)));
715            place!(lf_root, build_cell(&build_lf(&[nk_clsid])));
716            place!(nk_clsid, build_cell(&build_nk("CLSID", 1, lf_clsid, 0, 0)));
717            place!(lf_clsid, build_cell(&build_lf(&[nk_guid])));
718            place!(nk_guid, build_cell(&build_nk(clsid, 0, 0, 0, 0)));
719        } else {
720            let lf_guid: u32 = 0x0280;
721            let nk_inproc: u32 = 0x02C0;
722            let vl_inproc: u32 = 0x0340;
723            let vk_default: u32 = 0x0380;
724            let data_cell: u32 = 0x03C0;
725
726            let dll_utf16: Vec<u8> = hkcr_dll
727                .encode_utf16()
728                .flat_map(u16::to_le_bytes)
729                .chain([0u8, 0u8])
730                .collect();
731            let dll_data_cell = build_cell(&dll_utf16);
732
733            place!(root_cell, build_cell(&build_nk("ROOT", 1, lf_root, 0, 0)));
734            place!(lf_root, build_cell(&build_lf(&[nk_clsid])));
735            place!(nk_clsid, build_cell(&build_nk("CLSID", 1, lf_clsid, 0, 0)));
736            place!(lf_clsid, build_cell(&build_lf(&[nk_guid])));
737            place!(nk_guid, build_cell(&build_nk(clsid, 1, lf_guid, 0, 0)));
738            place!(lf_guid, build_cell(&build_lf(&[nk_inproc])));
739            place!(
740                nk_inproc,
741                build_cell(&build_nk("InprocServer32", 0, 0, 1, vl_inproc))
742            );
743            place!(vl_inproc, build_cell(&build_value_list(&[vk_default])));
744            place!(
745                vk_default,
746                build_cell(&build_vk_external("", data_cell, dll_utf16.len() as u32))
747            );
748            place!(data_cell, dll_data_cell);
749        }
750
751        HiveLayout {
752            hive_paddr,
753            hive_vaddr,
754            hbin_paddr,
755            hbin_vaddr,
756            hbase,
757            hbin,
758        }
759    }
760
761    /// Map both hive page-pairs into a `PageTableBuilder`.
762    fn map_hive(ptb: PageTableBuilder, h: &HiveLayout) -> PageTableBuilder {
763        ptb.map_4k(h.hive_vaddr, h.hive_paddr, flags::WRITABLE)
764            .map_4k(h.hbin_vaddr, h.hbin_paddr, flags::WRITABLE)
765            .write_phys(h.hive_paddr, &h.hbase)
766            .write_phys(h.hbin_paddr, &h.hbin)
767    }

802    /// HKCU has a CLSID pointing to an evil DLL, HKCR has the same CLSID
803    /// pointing to the legitimate shell32.dll → should detect one suspicious entry.
804    #[test]
805    fn walk_com_hijacking_detects_hkcu_override() {
806        let clsid = "{11111111-1111-1111-1111-111111111111}";
807        let hkcu_dll = r"C:\Users\evil\payload.dll";
808        let hkcr_dll = r"C:\Windows\System32\shell32.dll";
809
810        // HKCU hive: vaddr/paddr pair A (must be < 0x00FF_FFFF physical)
811        let hkcu_vaddr: u64 = 0xFFFF_8000_0010_0000;
812        let hkcu_paddr: u64 = 0x0010_0000;
813
814        // HKCR hive: vaddr/paddr pair B
815        let hkcr_vaddr: u64 = 0xFFFF_8000_0030_0000;
816        let hkcr_paddr: u64 = 0x0030_0000;
817
818        let hkcu = build_hkcu_hive(hkcu_vaddr, hkcu_paddr, clsid, hkcu_dll);
819        let hkcr = build_hkcr_hive(hkcr_vaddr, hkcr_paddr, clsid, hkcr_dll);
820
821        let ptb = map_hive(map_hive(PageTableBuilder::new(), &hkcu), &hkcr);
822        let reader = make_reader(ptb);
823
824        let results = walk_com_hijacking(&reader, hkcu_vaddr, hkcr_vaddr).unwrap();
825
826        assert_eq!(results.len(), 1, "expected exactly one COM hijack entry");
827        let entry = &results[0];
828        assert_eq!(entry.clsid, clsid);
829        assert!(
830            entry.hkcu_server.eq_ignore_ascii_case(hkcu_dll),
831            "hkcu_server mismatch: {}",
832            entry.hkcu_server
833        );
834        assert!(
835            entry.is_suspicious,
836            "override of HKCR entry must be suspicious"
837        );
838    }
