1//! Windows per-process handle table walking.
2//!
3//! Walks `_EPROCESS.ObjectTable` → `_HANDLE_TABLE.TableCode` →
4//! `_HANDLE_TABLE_ENTRY` array to enumerate open handles per process.
5//! Each entry's `ObjectPointerBits` field (shifted left 4, with low bits
6//! masked) yields an `_OBJECT_HEADER`, whose `TypeIndex` indexes into
7//! the `ObTypeIndexTable` to resolve the object type name.
8
9use memf_core::object_reader::ObjectReader;
10use memf_format::PhysicalMemoryProvider;
11
12use crate::{Result, WinHandleInfo};
13
14/// Maximum number of handle entries to scan per process.
15/// Prevents runaway iteration on corrupted handle tables.
16const MAX_HANDLE_ENTRIES: u64 = 16384;
17
18/// Walk all processes and enumerate their open handles.
19///
20/// For each process, reads `_EPROCESS.ObjectTable` → `_HANDLE_TABLE`,
21/// then iterates the level-0 handle entry array. Returns a flat list
22/// of all handles across all processes.
23pub fn walk_handles<P: PhysicalMemoryProvider>(
24    reader: &ObjectReader<P>,
25    ps_head_vaddr: u64,
26) -> Result<Vec<WinHandleInfo>> {
27    let procs = crate::process::walk_processes(reader, ps_head_vaddr)?;
28
29    let entry_size = reader
30        .symbols()
31        .struct_size("_HANDLE_TABLE_ENTRY")
32        .ok_or_else(|| crate::Error::WalkFailed {
33            walker: "handles",
34            reason: "missing _HANDLE_TABLE_ENTRY size".into(),
35        })?;
36
37    // Resolve ObTypeIndexTable symbol for type name lookup
38    let ob_type_table_addr = reader
39        .symbols()
40        .symbol_address("ObTypeIndexTable")
41        .ok_or_else(|| crate::Error::MissingKernelSymbol {
42            name: "ObTypeIndexTable".into(),
43        })?;
44
45    let mut results = Vec::new();
46
47    for proc in &procs {
48        // Read _EPROCESS.ObjectTable pointer
49        let obj_table: u64 = reader.read_field(proc.vaddr, "_EPROCESS", "ObjectTable")?;
50        if obj_table == 0 {
51            continue;
52        }
53
54        // Read _HANDLE_TABLE.TableCode
55        let table_code: u64 = match reader.read_field(obj_table, "_HANDLE_TABLE", "TableCode") {
56            Ok(v) => v,
57            Err(_) => continue,
58        };
59
60        // Level = low 2 bits of TableCode
61        let level = table_code & 0x3;
62        let base_addr = table_code & !0x3;
63
64        if level != 0 || base_addr == 0 {
65            // Only support level-0 (flat) tables for now; skip multi-level
66            continue;
67        }
68
69        // Read NextHandleNeedingPool to determine entry count
70        let next_handle: u32 =
71            match reader.read_field(obj_table, "_HANDLE_TABLE", "NextHandleNeedingPool") {
72                Ok(v) => v,
73                Err(_) => continue,
74            };
75
76        // Number of entries = next_handle / 4 (handle values are index × 4)
77        let num_entries = u64::from(next_handle) / 4;
78        let num_entries = num_entries.min(MAX_HANDLE_ENTRIES);
79
80        // Iterate entries starting at index 1 (index 0 is reserved)
81        for idx in 1..num_entries {
82            let entry_addr = base_addr + idx * entry_size;
83
84            // Read ObjectPointerBits (first u64 of entry)
85            let obj_ptr: u64 =
86                match reader.read_field(entry_addr, "_HANDLE_TABLE_ENTRY", "ObjectPointerBits") {
87                    Ok(v) => v,
88                    Err(_) => continue,
89                };
90
91            // Skip free/empty entries
92            if obj_ptr == 0 {
93                continue;
94            }
95
96            // ObjectPointerBits is shifted right by 4 in the kernel; reconstruct
97            // the full pointer by shifting left 4 and ORing in canonical high bits.
98            let object_addr = (obj_ptr << 4) | 0xFFFF_0000_0000_0000;
99
100            // Read GrantedAccessBits
101            let granted_access: u32 = reader
102                .read_field(entry_addr, "_HANDLE_TABLE_ENTRY", "GrantedAccessBits")
103                .unwrap_or(0);
104
105            // Read _OBJECT_HEADER.TypeIndex
106            let type_index: u8 = match reader.read_bytes(
107                object_addr.wrapping_add(
108                    reader
109                        .symbols()
110                        .field_offset("_OBJECT_HEADER", "TypeIndex")
111                        .unwrap_or(0x18),
112                ),
113                1,
114            ) {
115                Ok(bytes) => bytes[0],
116                Err(_) => continue,
117            };
118
119            // Look up type name via ObTypeIndexTable[TypeIndex]
120            let object_type = resolve_type_name(reader, ob_type_table_addr, type_index);
121
122            let handle_value = u32::try_from(idx).unwrap_or(u32::MAX).saturating_mul(4);
123
124            results.push(WinHandleInfo {
125                pid: proc.pid,
126                image_name: proc.image_name.clone(),
127                handle_value,
128                object_addr,
129                object_type,
130                granted_access,
131            });
132        }
133    }
134
135    Ok(results)
136}

138/// Resolve the object type name from `ObTypeIndexTable[type_index]`.
139///
140/// Reads the pointer at `ob_table_addr + type_index * 8`, which yields
141/// an `_OBJECT_TYPE` address, then reads `_OBJECT_TYPE.Name`.
142fn resolve_type_name<P: PhysicalMemoryProvider>(
143    reader: &ObjectReader<P>,
144    ob_table_addr: u64,
145    type_index: u8,
146) -> String {
147    let slot_addr = ob_table_addr.wrapping_add(u64::from(type_index) * 8);
148    let obj_type_addr: u64 = match reader.read_bytes(slot_addr, 8) {
149        Ok(bytes) => bytes.try_into().map_or(0, u64::from_le_bytes),
150        Err(_) => return String::from("<unknown>"),
151    };
152
153    if obj_type_addr == 0 {
154        return String::from("<unknown>");
155    }
156
157    // Read _OBJECT_TYPE.Name (_UNICODE_STRING at offset 0x10)
158    let name_off = reader
159        .symbols()
160        .field_offset("_OBJECT_TYPE", "Name")
161        .unwrap_or(0x10);
162
163    match crate::unicode::read_unicode_string(reader, obj_type_addr.wrapping_add(name_off)) {
164        Ok(name) if !name.is_empty() => name,
165        _ => String::from("<unknown>"),
166    }
167}

193    // _EPROCESS offsets
194    const EPROCESS_PID: u64 = 0x440;
195    const EPROCESS_LINKS: u64 = 0x448;
196    const EPROCESS_PPID: u64 = 0x540;
197    const EPROCESS_PEB: u64 = 0x550;
198    const EPROCESS_OBJECT_TABLE: u64 = 0x570;
199    const EPROCESS_IMAGE_NAME: u64 = 0x5A8;
200    const EPROCESS_CREATE_TIME: u64 = 0x430;
201    const EPROCESS_EXIT_TIME: u64 = 0x438;
202    const KPROCESS_DTB: u64 = 0x28;
203
204    // _HANDLE_TABLE offsets
205    const HANDLE_TABLE_CODE: u64 = 0x08;
206
207    // _HANDLE_TABLE_ENTRY: 16 bytes each
208    // ObjectPointerBits@0x0, GrantedAccessBits@0x8
209    const ENTRY_OBJECT_PTR: u64 = 0x0;
210    const ENTRY_GRANTED_ACCESS: u64 = 0x8;
211    const ENTRY_SIZE: u64 = 16;
212
213    // _OBJECT_HEADER offsets
214    const OBJ_HEADER_TYPE_INDEX: u64 = 0x18;
215
216    // _OBJECT_TYPE offsets
217    const OBJ_TYPE_NAME: u64 = 0x10;
218
219    // ISF preset defines ObTypeIndexTable at this address
220    const OB_TYPE_INDEX_TABLE_VADDR: u64 = 0xFFFFF805_5A490000;
221
222    /// Build a minimal process + handle table layout in synthetic memory.
223    ///
224    /// Returns the head_vaddr for PsActiveProcessHead.
225    /// `handles` is a slice of (object_header_vaddr, granted_access, type_index).
226    fn build_process_with_handles(
227        pid: u64,
228        image_name: &[u8],
229        handles: &[(u64, u32, u8)],
230        // Object type table: (type_index, type_name, obj_type_vaddr)
231        type_table: &[(u8, &str, u64)],
232    ) -> (u64, PageTableBuilder) {
233        let head_vaddr: u64 = 0xFFFF_8000_0010_0000;
234        let eproc_vaddr: u64 = 0xFFFF_8000_0010_1000;
235        let handle_table_vaddr: u64 = 0xFFFF_8000_0010_2000;
236        let entry_page_vaddr: u64 = 0xFFFF_8000_0010_3000;
237        let strings_vaddr: u64 = 0xFFFF_8000_0010_4000;
238
239        let head_paddr: u64 = 0x0080_0000;
240        let eproc_paddr: u64 = 0x0080_1000;
241        let handle_table_paddr: u64 = 0x0080_2000;
242        let entry_page_paddr: u64 = 0x0080_3000;
243        let strings_paddr: u64 = 0x0080_4000;
244
245        // Number of handle entries: we need handles.len() entries + 1 for the
246        // reserved slot at index 0 (handle value 0 is not used)
247        let num_entries = handles.len() + 1;
248        // Level 0: TableCode points directly to the entry page (low 2 bits = 0)
249        let table_code = entry_page_vaddr; // level 0
250
251        let mut ptb = PageTableBuilder::new()
252            .map_4k(head_vaddr, head_paddr, flags::WRITABLE)
253            .map_4k(eproc_vaddr, eproc_paddr, flags::WRITABLE)
254            .map_4k(handle_table_vaddr, handle_table_paddr, flags::WRITABLE)
255            .map_4k(entry_page_vaddr, entry_page_paddr, flags::WRITABLE)
256            .map_4k(strings_vaddr, strings_paddr, flags::WRITABLE)
257            // PsActiveProcessHead sentinel
258            .write_phys_u64(head_paddr, eproc_vaddr + EPROCESS_LINKS)
259            .write_phys_u64(head_paddr + 8, eproc_vaddr + EPROCESS_LINKS)
260            // _EPROCESS
261            .write_phys_u64(eproc_paddr + KPROCESS_DTB, 0x1AB000)
262            .write_phys_u64(eproc_paddr + EPROCESS_CREATE_TIME, 132800000000000000)
263            .write_phys_u64(eproc_paddr + EPROCESS_EXIT_TIME, 0)
264            .write_phys_u64(eproc_paddr + EPROCESS_PID, pid)
265            .write_phys_u64(eproc_paddr + EPROCESS_LINKS, head_vaddr)
266            .write_phys_u64(eproc_paddr + EPROCESS_LINKS + 8, head_vaddr)
267            .write_phys_u64(eproc_paddr + EPROCESS_PPID, 0)
268            .write_phys_u64(eproc_paddr + EPROCESS_PEB, 0)
269            .write_phys_u64(eproc_paddr + EPROCESS_OBJECT_TABLE, handle_table_vaddr)
270            .write_phys(eproc_paddr + EPROCESS_IMAGE_NAME, image_name)
271            // _HANDLE_TABLE
272            .write_phys_u64(handle_table_paddr + HANDLE_TABLE_CODE, table_code);
273
274        // Write NextHandleNeedingPool = (num_entries * 4) so the walker knows
275        // how many entries to scan. Handle values are index * 4.
276        let next_handle = (num_entries as u32) * 4;
277        ptb = ptb.write_phys(handle_table_paddr + 0x3C, &next_handle.to_le_bytes());
278
279        // Entry at index 0 is reserved (handle value 0), leave zeroed.
280        // Write actual handle entries starting at index 1.
281        for (i, &(obj_header_vaddr, granted_access, _type_index)) in handles.iter().enumerate() {
282            let entry_offset = ((i + 1) as u64) * ENTRY_SIZE;
283            let entry_paddr = entry_page_paddr + entry_offset;
284
285            // ObjectPointerBits: the kernel stores (addr >> 4) with the top
286            // canonical bits stripped.  The walker reconstructs the address as
287            // (obj_ptr << 4) | 0xFFFF_0000_0000_0000.
288            let obj_ptr_bits = (obj_header_vaddr & 0x0000_FFFF_FFFF_FFFF) >> 4;
289            ptb = ptb
290                .write_phys_u64(entry_paddr + ENTRY_OBJECT_PTR, obj_ptr_bits)
291                .write_phys(
292                    entry_paddr + ENTRY_GRANTED_ACCESS,
293                    &granted_access.to_le_bytes(),
294                );
295        }
296
297        // Map object header pages and write _OBJECT_HEADER data
298        for &(obj_header_vaddr, _granted_access, type_index) in handles {
299            let obj_header_paddr = obj_header_vaddr & 0x00FF_FFFF; // simple mapping
300            ptb = ptb
301                .map_4k(obj_header_vaddr, obj_header_paddr, flags::WRITABLE)
302                .write_phys(obj_header_paddr + OBJ_HEADER_TYPE_INDEX, &[type_index]);
303        }
304
305        // ObTypeIndexTable: array of pointers indexed by TypeIndex
306        let ob_table_paddr: u64 = 0x0090_0000;
307        ptb = ptb.map_4k(OB_TYPE_INDEX_TABLE_VADDR, ob_table_paddr, flags::WRITABLE);
308
309        // Write type object pointers and type name strings
310        let mut string_offset: u64 = 0;
311        for &(type_index, type_name, obj_type_vaddr) in type_table {
312            let slot_offset = u64::from(type_index) * 8;
313            ptb = ptb.write_phys_u64(ob_table_paddr + slot_offset, obj_type_vaddr);
314
315            // Map and write _OBJECT_TYPE with Name (_UNICODE_STRING)
316            let obj_type_paddr = obj_type_vaddr & 0x00FF_FFFF;
317            let name_utf16 = utf16le(type_name);
318            let name_len = name_utf16.len() as u16;
319            let name_buf_vaddr = strings_vaddr + string_offset;
320
321            ptb = ptb
322                .map_4k(obj_type_vaddr, obj_type_paddr, flags::WRITABLE)
323                // _UNICODE_STRING at _OBJECT_TYPE + 0x10: Length, MaxLen, Buffer
324                .write_phys(obj_type_paddr + OBJ_TYPE_NAME, &name_len.to_le_bytes())
325                .write_phys(
326                    obj_type_paddr + OBJ_TYPE_NAME + 2,
327                    &(name_len + 2).to_le_bytes(),
328                )
329                .write_phys_u64(obj_type_paddr + OBJ_TYPE_NAME + 8, name_buf_vaddr)
330                // Write the UTF-16LE string data
331                .write_phys(strings_paddr + string_offset, &name_utf16);
332
333            string_offset += (name_utf16.len() as u64) + 16; // pad between strings
334        }
335
336        (head_vaddr, ptb)
337    }

340    fn walks_single_process_handles() {
341        let obj_header1_vaddr: u64 = 0xFFFF_8000_0020_0000;
342        let obj_header2_vaddr: u64 = 0xFFFF_8000_0020_1000;
343        let obj_type_file_vaddr: u64 = 0xFFFF_8000_0030_0000;
344        let obj_type_key_vaddr: u64 = 0xFFFF_8000_0030_1000;
345
346        let type_index_file: u8 = 37;
347        let type_index_key: u8 = 38;
348
349        let (head_vaddr, ptb) = build_process_with_handles(
350            1234,
351            b"explorer.exe\0",
352            &[
353                (obj_header1_vaddr, 0x001F_0003, type_index_file),
354                (obj_header2_vaddr, 0x000F_003F, type_index_key),
355            ],
356            &[
357                (type_index_file, "File", obj_type_file_vaddr),
358                (type_index_key, "Key", obj_type_key_vaddr),
359            ],
360        );
361
362        let reader = make_win_reader(ptb);
363        let handles = walk_handles(&reader, head_vaddr).unwrap();
364
365        assert_eq!(handles.len(), 2);
366
367        // Handle values: index 1 → handle 4, index 2 → handle 8
368        assert_eq!(handles[0].handle_value, 4);
369        assert_eq!(handles[0].pid, 1234);
370        assert_eq!(handles[0].image_name, "explorer.exe");
371        assert_eq!(handles[0].object_type, "File");
372        assert_eq!(handles[0].granted_access, 0x001F_0003);
373        assert_eq!(handles[0].object_addr, obj_header1_vaddr);
374
375        assert_eq!(handles[1].handle_value, 8);
376        assert_eq!(handles[1].object_type, "Key");
377        assert_eq!(handles[1].granted_access, 0x000F_003F);
378    }

461            // _HANDLE_TABLE
462            .write_phys_u64(handle_table_paddr + HANDLE_TABLE_CODE, entry_page_vaddr)
463            .write_phys(handle_table_paddr + 0x3C, &next_handle.to_le_bytes())
464            // Entry index 1: zeroed (free slot) — entry_page_paddr + 16 stays 0
465            // Entry index 2: real handle — encode as kernel does: (addr >> 4) with top bits stripped
466            .write_phys_u64(

486

488        let handles = walk_handles(&reader, head_vaddr).unwrap();
489
490        // Only index 2 should be returned (index 0 reserved, index 1 free)
491        assert_eq!(handles.len(), 1);
492        assert_eq!(handles[0].handle_value, 8); // index 2 × 4 = 8
