15pub fn parse_dpapi_blob(data: &[u8]) -> Result<DpapiBlob, DpapiError> {
16    // Minimum fixed header: 4 (version) + 16 (provider GUID) + 16 (mk GUID)
17    //                      + 4 (flags) + 4 (desc len) = 44 bytes
18    if data.len() < 44 {
19        return Err(DpapiError::TooShort { needed: 44, got: data.len() });
20    }
21
22    let mut pos = 0usize;
23
24    let version = read_u32(data, &mut pos);
25    if version != 1 && version != 2 {
26        return Err(DpapiError::UnsupportedVersion(version));
27    }
28    pos += 16; // skip provider GUID
29    // Guarded by the `data.len() < 44` check above (here pos == 20, pos+16 == 36);
30    // the fallback keeps construction panic-free if that invariant ever changes.
31    let master_key_guid: [u8; 16] = data
32        .get(pos..pos + 16)
33        .and_then(|s| s.try_into().ok())
34        .unwrap_or([0u8; 16]);
35    pos += 16;
36    let _flags = read_u32(data, &mut pos);
37    let desc_len = read_u32(data, &mut pos) as usize;
38
39    if pos + desc_len > data.len() {
40        return Err(DpapiError::TooShort { needed: pos + desc_len, got: data.len() });
41    }
42    let desc_bytes = &data[pos..pos + desc_len];
43    pos += desc_len;
44
45    let description = if desc_len >= 2 {
46        let words: Vec<u16> = desc_bytes
47            .chunks_exact(2)
48            .map(|c| u16::from_le_bytes([c[0], c[1]]))
49            .collect();
50        String::from_utf16_lossy(&words)
51            .trim_end_matches('\0')
52            .to_string()
53    } else {
54        String::new()
55    };
56
57    // Need 20 more bytes: algIdEncrypt(4) + algIdHash(4) + dataLen(4) + hmacKeyLen(4) + algIdHash2(4)
58    if pos + 20 > data.len() {
59        return Err(DpapiError::TooShort { needed: pos + 20, got: data.len() });
60    }
61    let alg_id_encrypt = read_u32(data, &mut pos);
62    let alg_id_hash = read_u32(data, &mut pos);
63    let _data_len = read_u32(data, &mut pos);
64    let hmac_key_len = read_u32(data, &mut pos) as usize;
65    pos += 4; // skip repeated algIdHash
66
67    if pos + hmac_key_len > data.len() {
68        return Err(DpapiError::TooShort { needed: pos + hmac_key_len, got: data.len() });
69    }
70    let hmac_key = data[pos..pos + hmac_key_len].to_vec();
71    pos += hmac_key_len;
72
73    // HMAC digest size depends on hash algorithm
74    let digest_len: usize = match alg_id_hash {
75        0x800C => 64, // SHA512
76        _ => 20,      // SHA1 (0x8004) and default
77    };
78
79    if data.len() < pos + digest_len {
80        return Err(DpapiError::TooShort { needed: pos + digest_len, got: data.len() });
81    }
82    let hmac = data[data.len() - digest_len..].to_vec();
83    let ciphertext = data[pos..data.len() - digest_len].to_vec();
84
85    Ok(DpapiBlob {
86        version,
87        master_key_guid,
88        description,
89        alg_id_encrypt,
90        alg_id_hash,
91        hmac_key,
92        ciphertext,
93        hmac,
94    })
95}

97/// Read a little-endian u32 at `*pos`, advancing `pos` by 4.
98/// Out-of-range yields 0 (never panics); callers range-check before relying on
99/// the value, so a 0 here is a defensive fallback, not a silent success.
100#[inline]
101fn read_u32(data: &[u8], pos: &mut usize) -> u32 {
102    let v = data
103        .get(*pos..*pos + 4)
104        .and_then(|s| s.try_into().ok())
105        .map_or(0, u32::from_le_bytes);
106    *pos += 4;
107    v
108}

114    fn make_minimal_blob(desc_bytes: &[u8], hmac_key: &[u8], ciphertext: &[u8], hmac: &[u8]) -> Vec<u8> {
115        let mut v = Vec::new();
116        v.extend_from_slice(&2u32.to_le_bytes()); // version
117        v.extend_from_slice(&[0u8; 16]);           // provider GUID
118        v.extend_from_slice(&[0xAAu8; 16]);        // master key GUID
119        v.extend_from_slice(&0u32.to_le_bytes());  // flags
120        v.extend_from_slice(&(desc_bytes.len() as u32).to_le_bytes()); // desc length
121        v.extend_from_slice(desc_bytes);
122        v.extend_from_slice(&0x6610u32.to_le_bytes()); // AES-256
123        v.extend_from_slice(&0x8004u32.to_le_bytes()); // SHA1
124        let data_len = 4 + 4 + hmac_key.len() + ciphertext.len() + hmac.len();
125        v.extend_from_slice(&(data_len as u32).to_le_bytes());
126        v.extend_from_slice(&(hmac_key.len() as u32).to_le_bytes());
127        v.extend_from_slice(&0x8004u32.to_le_bytes()); // algIdHash repeated
128        v.extend_from_slice(hmac_key);
129        v.extend_from_slice(ciphertext);
130        v.extend_from_slice(hmac);
131        v
132    }
