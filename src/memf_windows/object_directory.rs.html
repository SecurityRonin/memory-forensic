1//! Windows kernel Object Directory walker.
2//!
3//! Enumerates objects in the kernel namespace by walking `_OBJECT_DIRECTORY`
4//! hash tables. The kernel Object Manager stores objects in a tree of
5//! directories, each using a 37-bucket hash table of `_OBJECT_DIRECTORY_ENTRY`
6//! chains. This module finds `_DRIVER_OBJECT` instances under `\Driver`
7//! for IRP dispatch table checking.
8
9use memf_core::object_reader::ObjectReader;
10use memf_format::PhysicalMemoryProvider;
11
12use crate::unicode::read_unicode_string;
13use crate::Result;
14
15/// Number of hash buckets in an `_OBJECT_DIRECTORY`.
16const HASH_BUCKET_COUNT: usize = 37;
17
18/// Maximum entries per hash bucket chain (safety limit against corruption).
19const MAX_CHAIN_LENGTH: usize = 1024;
20
21/// Size of `_OBJECT_HEADER_CREATOR_INFO` (Windows 10+, InfoMask bit 0x1).
22const CREATOR_INFO_SIZE: u64 = 0x20;
23
24/// Read the name of a kernel object from its `_OBJECT_HEADER_NAME_INFO`.
25///
26/// Given the object body address, walks backwards to `_OBJECT_HEADER`,
27/// checks `InfoMask` for the presence of `_OBJECT_HEADER_NAME_INFO`
28/// (bit 0x2), and reads the `Name` UNICODE_STRING if present.
29pub fn read_object_name<P: PhysicalMemoryProvider>(
30    reader: &ObjectReader<P>,
31    object_body_addr: u64,
32) -> Result<String> {
33    let body_offset = reader
34        .symbols()
35        .field_offset("_OBJECT_HEADER", "Body")
36        .ok_or_else(|| crate::Error::Walker("missing _OBJECT_HEADER.Body offset".into()))?;
37    let header_addr = object_body_addr.wrapping_sub(body_offset);
38
39    let info_mask: u8 = reader.read_field(header_addr, "_OBJECT_HEADER", "InfoMask")?;
40    if info_mask & 0x02 == 0 {
41        return Ok(String::new());
42    }
43
44    let mut name_info_dist = reader
45        .symbols()
46        .struct_size("_OBJECT_HEADER_NAME_INFO")
47        .ok_or_else(|| crate::Error::Walker("missing _OBJECT_HEADER_NAME_INFO size".into()))?;
48    if info_mask & 0x01 != 0 {
49        name_info_dist += CREATOR_INFO_SIZE;
50    }
51
52    let name_info_addr = header_addr.wrapping_sub(name_info_dist);
53    let name_offset = reader
54        .symbols()
55        .field_offset("_OBJECT_HEADER_NAME_INFO", "Name")
56        .ok_or_else(|| {
57            crate::Error::Walker("missing _OBJECT_HEADER_NAME_INFO.Name offset".into())
58        })?;
59
60    Ok(read_unicode_string(reader, name_info_addr.wrapping_add(name_offset)).unwrap_or_default())
61}

63/// Walk an `_OBJECT_DIRECTORY` hash table and return all entries.
64///
65/// Iterates all 37 hash buckets, following `_OBJECT_DIRECTORY_ENTRY`
66/// chains in each bucket. Returns `(name, object_body_addr)` pairs.
67pub fn walk_directory<P: PhysicalMemoryProvider>(
68    reader: &ObjectReader<P>,
69    dir_addr: u64,
70) -> Result<Vec<(String, u64)>> {
71    let bucket_bytes = reader.read_bytes(dir_addr, HASH_BUCKET_COUNT * 8)?;
72    let mut entries = Vec::new();
73
74    for bucket_idx in 0..HASH_BUCKET_COUNT {
75        let off = bucket_idx * 8;
76        let mut entry_ptr =
77            u64::from_le_bytes(bucket_bytes[off..off + 8].try_into().expect("8 bytes"));
78
79        let mut chain_len = 0;
80        while entry_ptr != 0 && chain_len < MAX_CHAIN_LENGTH {
81            let chain_link: u64 =
82                reader.read_field(entry_ptr, "_OBJECT_DIRECTORY_ENTRY", "ChainLink")?;
83            let object_body: u64 =
84                reader.read_field(entry_ptr, "_OBJECT_DIRECTORY_ENTRY", "Object")?;
85
86            if object_body != 0 {
87                let name = read_object_name(reader, object_body).unwrap_or_default();
88                entries.push((name, object_body));
89            }
90
91            entry_ptr = chain_link;
92            chain_len += 1;
93        }
94    }
95
96    Ok(entries)
97}

99/// Find `\Driver` within the root object directory and return all
100/// `_DRIVER_OBJECT` body addresses.
101///
102/// Walks the root `_OBJECT_DIRECTORY` looking for an entry named "Driver".
103/// If found, walks that subdirectory and returns the body addresses of
104/// all objects within it (each is a `_DRIVER_OBJECT`).
105pub fn walk_driver_objects<P: PhysicalMemoryProvider>(
106    reader: &ObjectReader<P>,
107    root_dir_addr: u64,
108) -> Result<Vec<u64>> {
109    let root_entries = walk_directory(reader, root_dir_addr)?;
110
111    for (name, body_addr) in &root_entries {
112        if name == "Driver" {
113            let driver_entries = walk_directory(reader, *body_addr)?;
114            return Ok(driver_entries.into_iter().map(|(_, addr)| addr).collect());
115        }
116    }
117
118    Ok(Vec::new())
119}

129    /// Encode a Rust &str as UTF-16LE bytes.
130    fn utf16le_bytes(s: &str) -> Vec<u8> {
131        s.encode_utf16().flat_map(u16::to_le_bytes).collect()
132    }

134    fn write_named_object(
135        buf: &mut [u8],
136        obj_offset: usize,
137        vaddr_base: u64,
138        name: &str,
139        str_offset: usize,
140    ) -> u64 {
141        let utf16 = utf16le_bytes(name);
142        let str_len = utf16.len() as u16;
143        buf[str_offset..str_offset + utf16.len()].copy_from_slice(&utf16);
144
145        let ni = obj_offset;
146        buf[ni + 0x10..ni + 0x12].copy_from_slice(&str_len.to_le_bytes());
147        buf[ni + 0x12..ni + 0x14].copy_from_slice(&str_len.to_le_bytes());
148        let str_vaddr = vaddr_base + str_offset as u64;
149        buf[ni + 0x18..ni + 0x20].copy_from_slice(&str_vaddr.to_le_bytes());
150
151        let hdr = obj_offset + 0x20;
152        buf[hdr + 0x1a] = 0x02; // InfoMask = NAME_INFO present (bit 1)
153
154        vaddr_base + (obj_offset + 0x50) as u64
155    }

157    fn write_dir_entry(
158        buf: &mut [u8],
159        entry_offset: usize,
160        chain_link: u64,
161        object_body: u64,
162        hash_value: u32,
163    ) {
164        buf[entry_offset..entry_offset + 8].copy_from_slice(&chain_link.to_le_bytes());
165        buf[entry_offset + 8..entry_offset + 16].copy_from_slice(&object_body.to_le_bytes());
166        buf[entry_offset + 0x10..entry_offset + 0x14].copy_from_slice(&hash_value.to_le_bytes());
167    }
168
169    fn set_bucket(buf: &mut [u8], dir_offset: usize, bucket_idx: usize, entry_vaddr: u64) {
170        let off = dir_offset + bucket_idx * 8;
171        buf[off..off + 8].copy_from_slice(&entry_vaddr.to_le_bytes());
172    }

174    fn make_test_reader(pages: &[(u64, u64, &[u8])]) -> ObjectReader<SyntheticPhysMem> {
175        let isf = IsfBuilder::windows_kernel_preset().build_json();
176        let resolver = IsfResolver::from_value(&isf).unwrap();
177        let mut builder = PageTableBuilder::new();
178        for &(vaddr, paddr, data) in pages {
179            builder = builder
180                .map_4k(vaddr, paddr, flags::WRITABLE)
181                .write_phys(paddr, data);
182        }
183        let (cr3, mem) = builder.build();
184        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
185        ObjectReader::new(vas, Box::new(resolver))
186    }

276    #[test]
277    fn walk_driver_objects_finds_drivers() {
278        let vaddr1: u64 = 0xFFFF_8000_0010_0000;
279        let paddr1: u64 = 0x0080_0000;
280        let vaddr2: u64 = 0xFFFF_8000_0020_0000;
281        let paddr2: u64 = 0x0090_0000;
282
283        let mut page1 = vec![0u8; 4096];
284        let mut page2 = vec![0u8; 4096];
285
286        // "Driver" directory object on page2
287        let drv_str = utf16le_bytes("Driver");
288        let drv_len = drv_str.len() as u16;
289        page2[0x800..0x800 + drv_str.len()].copy_from_slice(&drv_str);
290        page2[0x10..0x12].copy_from_slice(&drv_len.to_le_bytes());
291        page2[0x12..0x14].copy_from_slice(&drv_len.to_le_bytes());
292        page2[0x18..0x20].copy_from_slice(&(vaddr2 + 0x800).to_le_bytes());
293        page2[0x3a] = 0x02;
294        let driver_dir_body = vaddr2 + 0x050;
295
296        // Driver A: "ACPI"
297        let acpi_str = utf16le_bytes("ACPI");
298        let acpi_len = acpi_str.len() as u16;
299        page2[0x880..0x880 + acpi_str.len()].copy_from_slice(&acpi_str);
300        page2[0x200 + 0x10..0x200 + 0x12].copy_from_slice(&acpi_len.to_le_bytes());
301        page2[0x200 + 0x12..0x200 + 0x14].copy_from_slice(&acpi_len.to_le_bytes());
302        page2[0x200 + 0x18..0x200 + 0x20].copy_from_slice(&(vaddr2 + 0x880).to_le_bytes());
303        page2[0x220 + 0x1a] = 0x02;
304        let drv_a_body = vaddr2 + 0x250;
305
306        // Driver B: "Null"
307        let null_str = utf16le_bytes("Null");
308        let null_len = null_str.len() as u16;
309        page2[0x8C0..0x8C0 + null_str.len()].copy_from_slice(&null_str);
310        page2[0x300 + 0x10..0x300 + 0x12].copy_from_slice(&null_len.to_le_bytes());
311        page2[0x300 + 0x12..0x300 + 0x14].copy_from_slice(&null_len.to_le_bytes());
312        page2[0x300 + 0x18..0x300 + 0x20].copy_from_slice(&(vaddr2 + 0x8C0).to_le_bytes());
313        page2[0x320 + 0x1a] = 0x02;
314        let drv_b_body = vaddr2 + 0x350;
315
316        write_dir_entry(&mut page2, 0x178, 0, drv_a_body, 0);
317        set_bucket(&mut page2, 0x050, 0, vaddr2 + 0x178);
318        write_dir_entry(&mut page2, 0x190, 0, drv_b_body, 0);
319        set_bucket(&mut page2, 0x050, 3, vaddr2 + 0x190);
320
321        write_dir_entry(&mut page1, 0x200, 0, driver_dir_body, 0);
322        set_bucket(&mut page1, 0, 2, vaddr1 + 0x200);
323
324        let reader = make_test_reader(&[(vaddr1, paddr1, &page1), (vaddr2, paddr2, &page2)]);
325
326        let driver_addrs = walk_driver_objects(&reader, vaddr1).unwrap();
327        assert_eq!(driver_addrs.len(), 2);
328        assert!(driver_addrs.contains(&drv_a_body));
329        assert!(driver_addrs.contains(&drv_b_body));
330    }
