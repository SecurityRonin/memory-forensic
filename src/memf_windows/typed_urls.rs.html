32/// A URL typed directly into the IE/Edge address bar.
33#[derive(Debug, Clone, serde::Serialize)]
34pub struct TypedUrlEntry {
35    /// Windows username whose hive contained this entry.
36    pub username: String,
37    /// The typed URL string.
38    pub url: String,
39    /// Last-visited timestamp stored in the registry (FILETIME).
40    pub timestamp: u64,
41    /// `true` when the URL matches a known suspicious domain or pattern.
42    pub is_suspicious: bool,
43}

46    "pastebin.com",
47    "paste.ee",
48    "hastebin.com",

55/// Return `true` when the URL matches a known suspicious domain or pattern.
56pub fn classify_typed_url(url: &str) -> bool {
57    let lower = url.to_ascii_lowercase();
58    for domain in SUSPICIOUS_DOMAINS {
59        if lower.contains(domain) {
60            return true;
61        }
62    }
63    if lower.starts_with("file://") {
64        let path_part = &lower[7..];
65        if path_part.starts_with("\\\\") || path_part.starts_with("//") {
66            return true;
67        }
68    }
69    if let Some(scheme_end) = lower.find("://") {
70        let after_scheme = &lower[scheme_end + 3..];
71        let authority = match after_scheme.find('/') {
72            Some(pos) => &after_scheme[..pos],
73            None => after_scheme,
74        };
75        if authority.contains('@') && authority.contains(':') {
76            return true;
77        }
78    }
79    false
80}

82fn cell_address(hive_addr: u64, cell_index: u32) -> u64 {
83    hive_addr
84        .wrapping_add(HBIN_START_OFFSET)
85        .wrapping_add(cell_index as u64)

88fn read_cell_data<P: PhysicalMemoryProvider>(
89    reader: &ObjectReader<P>,
90    cell_vaddr: u64,
91) -> crate::Result<Vec<u8>> {
92    let size_bytes = reader.read_bytes(cell_vaddr, 4)?;
93    let raw_size = i32::from_le_bytes(size_bytes[..4].try_into().unwrap());
94    let abs_size = raw_size.unsigned_abs() as usize;
95    if abs_size <= 4 {
96        return Ok(Vec::new());
97    }
98    let data_len = (abs_size - 4).min(0x10000);
99    reader
100        .read_bytes(cell_vaddr.wrapping_add(4), data_len)
101        .map_err(Into::into)
102}

104fn read_key_name(nk_data: &[u8]) -> String {
105    if nk_data.len() < NK_NAME_OFFSET + 1 {
106        return String::new();
107    }
108    let name_len = u16::from_le_bytes(
109        nk_data[NK_NAME_LENGTH_OFFSET..NK_NAME_LENGTH_OFFSET + 2]
110            .try_into()
111            .unwrap(),
112    ) as usize;
113    let end = NK_NAME_OFFSET + name_len;
114    if end > nk_data.len() {
115        return String::new();
116    }
117    String::from_utf8_lossy(&nk_data[NK_NAME_OFFSET..end]).into_owned()
118}

120fn read_value_name(vk_data: &[u8]) -> String {
121    if vk_data.len() < VK_NAME_OFFSET + 1 {
122        return String::new();
123    }
124    let name_len = u16::from_le_bytes(
125        vk_data[VK_NAME_LENGTH_OFFSET..VK_NAME_LENGTH_OFFSET + 2]
126            .try_into()
127            .unwrap(),
128    ) as usize;
129    let end = VK_NAME_OFFSET + name_len;
130    if end > vk_data.len() {
131        return String::new();
132    }
133    String::from_utf8_lossy(&vk_data[VK_NAME_OFFSET..end]).into_owned()
134}

136fn find_subkey<P: PhysicalMemoryProvider>(
137    reader: &ObjectReader<P>,
138    hive_addr: u64,
139    nk_data: &[u8],
140    target_name: &str,
141) -> crate::Result<Option<u32>> {
142    if nk_data.len() < NK_STABLE_SUBKEYS_LIST_OFFSET + 4 {
143        return Ok(None);
144    }
145    let subkey_count = u32::from_le_bytes(
146        nk_data[NK_STABLE_SUBKEY_COUNT_OFFSET..NK_STABLE_SUBKEY_COUNT_OFFSET + 4]
147            .try_into()
148            .unwrap(),
149    ) as usize;
150    if subkey_count == 0 {
151        return Ok(None);
152    }
153    let subkeys_list_cell = u32::from_le_bytes(
154        nk_data[NK_STABLE_SUBKEYS_LIST_OFFSET..NK_STABLE_SUBKEYS_LIST_OFFSET + 4]
155            .try_into()
156            .unwrap(),
157    );
158    let list_vaddr = cell_address(hive_addr, subkeys_list_cell);
159    let list_data = read_cell_data(reader, list_vaddr)?;
160    if list_data.len() < 4 {
161        return Ok(None);
162    }
163    let sig = u16::from_le_bytes(list_data[0..2].try_into().unwrap());
164    let count = u16::from_le_bytes(list_data[2..4].try_into().unwrap()) as usize;
165    let count = count.min(MAX_SUBKEYS);
166
167    match sig {
168        0x666C | 0x686C => {
169            for i in 0..count {
170                let off = 4 + i * 8;
171                if off + 4 > list_data.len() {
172                    break;
173                }
174                let child_cell = u32::from_le_bytes(list_data[off..off + 4].try_into().unwrap());
175                let child_vaddr = cell_address(hive_addr, child_cell);
176                if let Ok(child_nk) = read_cell_data(reader, child_vaddr) {
177                    if child_nk.len() >= NK_NAME_OFFSET {
178                        let child_sig = u16::from_le_bytes(child_nk[0..2].try_into().unwrap());
179                        if child_sig == NK_SIGNATURE {
180                            let name = read_key_name(&child_nk);
181                            if name.eq_ignore_ascii_case(target_name) {
182                                return Ok(Some(child_cell));
183                            }
184                        }
185                    }
186                }
187            }
188        }
189        0x696C => {
190            for i in 0..count {
191                let off = 4 + i * 4;
192                if off + 4 > list_data.len() {
193                    break;
194                }
195                let child_cell = u32::from_le_bytes(list_data[off..off + 4].try_into().unwrap());
196                let child_vaddr = cell_address(hive_addr, child_cell);
197                if let Ok(child_nk) = read_cell_data(reader, child_vaddr) {
198                    if child_nk.len() >= NK_NAME_OFFSET {
199                        let child_sig = u16::from_le_bytes(child_nk[0..2].try_into().unwrap());
200                        if child_sig == NK_SIGNATURE {
201                            let name = read_key_name(&child_nk);
202                            if name.eq_ignore_ascii_case(target_name) {
203                                return Ok(Some(child_cell));
204                            }
205                        }
206                    }
207                }
208            }
209        }
210        0x6972 => {
211            for i in 0..count {
212                let off = 4 + i * 4;
213                if off + 4 > list_data.len() {
214                    break;
215                }
216                let sub_list_cell = u32::from_le_bytes(list_data[off..off + 4].try_into().unwrap());
217                let sub_vaddr = cell_address(hive_addr, sub_list_cell);
218                let sub_data = read_cell_data(reader, sub_vaddr)?;
219                if sub_data.len() < 4 {
220                    continue;
221                }
222                let sub_sig = u16::from_le_bytes(sub_data[0..2].try_into().unwrap());
223                let sub_count = u16::from_le_bytes(sub_data[2..4].try_into().unwrap()) as usize;
224                let sub_count = sub_count.min(MAX_SUBKEYS);
225                let entry_size: usize = match sub_sig {
226                    0x666C | 0x686C => 8,
227                    0x696C => 4,
228                    _ => continue,
229                };
230                for j in 0..sub_count {
231                    let soff = 4 + j * entry_size;
232                    if soff + 4 > sub_data.len() {
233                        break;
234                    }
235                    let child_cell =
236                        u32::from_le_bytes(sub_data[soff..soff + 4].try_into().unwrap());
237                    let child_vaddr = cell_address(hive_addr, child_cell);
238                    if let Ok(child_nk) = read_cell_data(reader, child_vaddr) {
239                        if child_nk.len() >= NK_NAME_OFFSET {
240                            let child_sig = u16::from_le_bytes(child_nk[0..2].try_into().unwrap());
241                            if child_sig == NK_SIGNATURE {
242                                let name = read_key_name(&child_nk);
243                                if name.eq_ignore_ascii_case(target_name) {
244                                    return Ok(Some(child_cell));
245                                }
246                            }
247                        }
248                    }
249                }
250            }
251        }
252        _ => {}
253    }
254    Ok(None)
255}

257/// Walk typed URL entries from a registry hive in memory.
258pub fn walk_typed_urls<P: PhysicalMemoryProvider>(
259    reader: &ObjectReader<P>,
260    hive_addr: u64,
261    username: &str,
262) -> crate::Result<Vec<TypedUrlEntry>> {
263    if hive_addr == 0 {
264        return Ok(Vec::new());
265    }
266    let root_cell_bytes =
267        match reader.read_bytes(hive_addr.wrapping_add(HBASE_BLOCK_ROOT_CELL_OFFSET), 4) {
268            Ok(b) => b,
269            Err(_) => return Ok(Vec::new()),
270        };
271    let root_cell_index = u32::from_le_bytes(root_cell_bytes[..4].try_into().unwrap());
272    let root_vaddr = cell_address(hive_addr, root_cell_index);
273    let root_nk = match read_cell_data(reader, root_vaddr) {
274        Ok(d) => d,
275        Err(_) => return Ok(Vec::new()),
276    };
277    if root_nk.len() < 2 {
278        return Ok(Vec::new());
279    }
280    let sig = u16::from_le_bytes(root_nk[0..2].try_into().unwrap());
281    if sig != NK_SIGNATURE {
282        return Ok(Vec::new());
283    }
284
285    let mut current_nk = root_nk;
286    for component in TYPED_URLS_PATH {
287        match find_subkey(reader, hive_addr, &current_nk, component)? {
288            Some(cell_idx) => {
289                let vaddr = cell_address(hive_addr, cell_idx);
290                match read_cell_data(reader, vaddr) {
291                    Ok(d) => current_nk = d,
292                    Err(_) => return Ok(Vec::new()),
293                }
294            }
295            None => return Ok(Vec::new()),
296        }
297    }
298
299    if current_nk.len() < NK_VALUES_LIST_OFFSET + 4 {
300        return Ok(Vec::new());
301    }
302    let value_count = u32::from_le_bytes(
303        current_nk[NK_VALUE_COUNT_OFFSET..NK_VALUE_COUNT_OFFSET + 4]
304            .try_into()
305            .unwrap(),
306    ) as usize;
307    if value_count == 0 {
308        return Ok(Vec::new());
309    }
310    let values_list_cell = u32::from_le_bytes(
311        current_nk[NK_VALUES_LIST_OFFSET..NK_VALUES_LIST_OFFSET + 4]
312            .try_into()
313            .unwrap(),
314    );
315    let vlist_vaddr = cell_address(hive_addr, values_list_cell);
316    let vlist_data = match read_cell_data(reader, vlist_vaddr) {
317        Ok(d) => d,
318        Err(_) => return Ok(Vec::new()),
319    };
320
321    // Try to navigate TypedURLsTime for timestamps.
322    let mut time_nk_opt: Option<Vec<u8>> = None;
323    {
324        let root_cell_bytes2 = reader
325            .read_bytes(hive_addr.wrapping_add(HBASE_BLOCK_ROOT_CELL_OFFSET), 4)
326            .ok();
327        if let Some(b) = root_cell_bytes2 {
328            let root_cell_index2 = u32::from_le_bytes(b[..4].try_into().unwrap());
329            let root_vaddr2 = cell_address(hive_addr, root_cell_index2);
330            if let Ok(root_nk2) = read_cell_data(reader, root_vaddr2) {
331                let mut cur = root_nk2;
332                let mut ok = true;
333                for component in TYPED_URLS_TIME_PATH {
334                    match find_subkey(reader, hive_addr, &cur, component) {
335                        Ok(Some(cell_idx)) => {
336                            let va = cell_address(hive_addr, cell_idx);
337                            match read_cell_data(reader, va) {
338                                Ok(d) => cur = d,
339                                Err(_) => {
340                                    ok = false;
341                                    break;
342                                }
343                            }
344                        }
345                        _ => {
346                            ok = false;
347                            break;
348                        }
349                    }
350                }
351                if ok {
352                    time_nk_opt = Some(cur);
353                }
354            }
355        }
356    }
357
358    let mut time_map: std::collections::HashMap<String, u64> = std::collections::HashMap::new();
359    if let Some(time_nk) = time_nk_opt {
360        if time_nk.len() >= NK_VALUES_LIST_OFFSET + 4 {
361            let tc = u32::from_le_bytes(
362                time_nk[NK_VALUE_COUNT_OFFSET..NK_VALUE_COUNT_OFFSET + 4]
363                    .try_into()
364                    .unwrap(),
365            ) as usize;
366            if tc > 0 {
367                let tvc = u32::from_le_bytes(
368                    time_nk[NK_VALUES_LIST_OFFSET..NK_VALUES_LIST_OFFSET + 4]
369                        .try_into()
370                        .unwrap(),
371                );
372                let tvl_vaddr = cell_address(hive_addr, tvc);
373                if let Ok(tvl_data) = read_cell_data(reader, tvl_vaddr) {
374                    let tc = tc.min(MAX_TYPED_URLS);
375                    for i in 0..tc {
376                        let off = i * 4;
377                        if off + 4 > tvl_data.len() {
378                            break;
379                        }
380                        let vk_cell =
381                            u32::from_le_bytes(tvl_data[off..off + 4].try_into().unwrap());
382                        let vk_vaddr = cell_address(hive_addr, vk_cell);
383                        if let Ok(vk_data) = read_cell_data(reader, vk_vaddr) {
384                            if vk_data.len() >= 2 {
385                                let vsig = u16::from_le_bytes(vk_data[0..2].try_into().unwrap());
386                                if vsig == VK_SIGNATURE {
387                                    let vname = read_value_name(&vk_data);
388                                    if vk_data.len() >= VK_DATA_OFFSET_OFFSET + 4 {
389                                        let data_len = u32::from_le_bytes(
390                                            vk_data
391                                                [VK_DATA_LENGTH_OFFSET..VK_DATA_LENGTH_OFFSET + 4]
392                                                .try_into()
393                                                .unwrap(),
394                                        );
395                                        let data_cell = u32::from_le_bytes(
396                                            vk_data
397                                                [VK_DATA_OFFSET_OFFSET..VK_DATA_OFFSET_OFFSET + 4]
398                                                .try_into()
399                                                .unwrap(),
400                                        );
401                                        if data_len >= 8 {
402                                            let dc_vaddr = cell_address(hive_addr, data_cell);
403                                            if let Ok(dc_data) = read_cell_data(reader, dc_vaddr) {
404                                                if dc_data.len() >= 8 {
405                                                    let ts = u64::from_le_bytes(
406                                                        dc_data[0..8].try_into().unwrap(),
407                                                    );
408                                                    time_map.insert(vname, ts);
409                                                }
410                                            }
411                                        }
412                                    }
413                                }
414                            }
415                        }
416                    }
417                }
418            }
419        }
420    }
421
422    let mut results = Vec::new();
423    let count = value_count.min(MAX_TYPED_URLS);
424    for i in 0..count {
425        let off = i * 4;
426        if off + 4 > vlist_data.len() {
427            break;
428        }
429        let vk_cell = u32::from_le_bytes(vlist_data[off..off + 4].try_into().unwrap());
430        let vk_vaddr = cell_address(hive_addr, vk_cell);
431        let vk_data = match read_cell_data(reader, vk_vaddr) {
432            Ok(d) => d,
433            Err(_) => continue,
434        };
435        if vk_data.len() < 2 {
436            continue;
437        }
438        let vsig = u16::from_le_bytes(vk_data[0..2].try_into().unwrap());
439        if vsig != VK_SIGNATURE {
440            continue;
441        }
442        let value_name = read_value_name(&vk_data);
443        if vk_data.len() < VK_DATA_OFFSET_OFFSET + 4 {
444            continue;
445        }
446        let data_len_raw = u32::from_le_bytes(
447            vk_data[VK_DATA_LENGTH_OFFSET..VK_DATA_LENGTH_OFFSET + 4]
448                .try_into()
449                .unwrap(),
450        );
451        let data_len = (data_len_raw & 0x7FFF_FFFF) as usize;
452        if data_len < 2 {
453            continue;
454        }
455        let data_cell = u32::from_le_bytes(
456            vk_data[VK_DATA_OFFSET_OFFSET..VK_DATA_OFFSET_OFFSET + 4]
457                .try_into()
458                .unwrap(),
459        );
460        let dc_vaddr = cell_address(hive_addr, data_cell);
461        let dc_data = match read_cell_data(reader, dc_vaddr) {
462            Ok(d) => d,
463            Err(_) => continue,
464        };
465        if dc_data.len() < 2 {
466            continue;
467        }
468        let str_len = data_len.min(dc_data.len()) & !1;
469        let utf16_units: Vec<u16> = dc_data[..str_len]
470            .chunks_exact(2)
471            .map(|c| u16::from_le_bytes([c[0], c[1]]))
472            .take_while(|&c| c != 0)
473            .collect();
474        let url = String::from_utf16_lossy(&utf16_units);
475        if url.is_empty() {
476            continue;
477        }
478        let timestamp = time_map.get(&value_name).copied().unwrap_or(0);
479        let is_suspicious = classify_typed_url(&url);
480        results.push(TypedUrlEntry {
481            username: username.to_string(),
482            url,
483            timestamp,
484            is_suspicious,
485        });
486    }
487    Ok(results)
488}

1558        write_nk(&mut pg, 0x800, 0, 0, 1, 0xA00, b"TypedURLs");

1560        write_nk(&mut pg, 0x900, 0, 0, 1, 0xB00, b"TypedURLsTime");

1563        pg[0xA00..0xA04].copy_from_slice(&(-64i32).to_le_bytes());

1572        write_vk(&mut pg, 0xD00, b"url1", 8, 0xF00);

1588        let mut hive_hdr2 = vec![0u8; 0x1000];

1595            builder = builder.write_phys(hbin_paddr + i as u64 * 0x1000, &pg[src..src + 0x1000]);
