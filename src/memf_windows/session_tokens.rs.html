1//! Authentication session token forensics walker.
2//!
3//! Scans process heap memory for authentication tokens that applications hold
4//! in memory during active sessions. Covers JWTs, OAuth Bearer tokens, and
5//! provider-specific token formats.
6//!
7//! This capability has no equivalent in Volatility 3, pypykatz, or MemProcFS
8//! — those tools focus on Windows credential structures; this walker targets
9//! modern application-layer authentication material.
10//!
11//! # Token types detected
12//!
13//! - **JWT** (`eyJ*.*.*`) — JSON Web Tokens used by OAuth2/OIDC, API auth
14//! - **Bearer** — OAuth2 access tokens in `Authorization: Bearer` headers
15//! - **GitHub PAT** (`ghp_*`, `github_pat_*`) — GitHub personal access tokens
16//! - **GitLab PAT** (`glpat-*`) — GitLab personal access tokens
17//! - **Slack** (`xoxb-*`, `xoxp-*`) — Slack bot/user tokens
18//! - **AWS STS** — Temporary AWS session tokens (AQOD/AQIA prefix)
19//! - **Generic** — `access_token`/`refresh_token` JSON field values
20//!
21//! # Forensic guarantee
22//!
23//! Read-only. No live process access, no Win32 API calls, no state modification.
24
25use std::collections::HashSet;
26use std::sync::OnceLock;
27
28use memf_core::object_reader::ObjectReader;
29use memf_format::PhysicalMemoryProvider;
30
31use crate::{types::SessionTokenInfo, Result};
32
33struct TokenPattern {
34    /// Human-readable token type label emitted in [`SessionTokenInfo::token_type`].
35    label: &'static str,
36    /// Compiled regex for this token format.
37    re: regex::Regex,
38    /// When `true`, capture group 1 is used as the token value;
39    /// when `false`, the full match is used.
40    use_capture: bool,
41}
42
43static TOKEN_PATTERNS: OnceLock<Vec<TokenPattern>> = OnceLock::new();
44
45/// (label, pattern, use_capture) spec table. Patterns are compile-time
46/// constants; any that fail to compile are dropped (defence in depth) rather
47/// than panicking, so a bad edit degrades to a missing detector, not an abort.
48const TOKEN_PATTERN_SPECS: &[(&str, &str, bool)] = &[
49    ("Jwt", JWT_PATTERN, false),
50    ("Bearer", BEARER_PATTERN, true),
51    ("GitHub", GITHUB_PATTERN, false),
52    ("GitLab", GITLAB_PATTERN, false),
53    ("Slack", SLACK_PATTERN, false),
54    ("AwsSts", AWS_STS_PATTERN, false),
55    ("GenericAccessToken", GENERIC_TOKEN_PATTERN, true),
56];
57
58fn token_patterns() -> &'static [TokenPattern] {
59    TOKEN_PATTERNS.get_or_init(|| {
60        TOKEN_PATTERN_SPECS
61            .iter()
62            .filter_map(|&(label, pat, use_capture)| {
63                regex::Regex::new(pat).ok().map(|re| TokenPattern {
64                    label,
65                    re,
66                    use_capture,
67                })
68            })
69            .collect()
70    })
71}

73/// Walk committed, writable heap regions of ALL processes in the dump and
74/// extract any authentication tokens found as strings.
75///
76/// # Arguments
77///
78/// * `reader` — kernel-space `ObjectReader` (uses kernel CR3 / symbol table).
79/// * `ps_head_vaddr` — virtual address of `PsActiveProcessHead`.
80pub fn walk_session_tokens<P: PhysicalMemoryProvider + Clone>(
81    reader: &ObjectReader<P>,
82    ps_head_vaddr: u64,
83) -> Result<Vec<SessionTokenInfo>> {
84    let wr = crate::heap_walker::for_each_heap_region(
85        reader,
86        ps_head_vaddr,
87        |_proc| true,
88        |bytes, proc| scan_for_tokens(bytes, proc.pid, &proc.image_name),
89        |info: &SessionTokenInfo| (info.pid, info.token_value.clone()),
90    )?;
91    Ok(wr.items)
92}

94// JWT: three base64url segments separated by dots, header starts with eyJ ({"  in base64)
95const JWT_PATTERN: &str = r"eyJ[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{10,}";
96
97// Bearer token in Authorization header or JSON
98const BEARER_PATTERN: &str = r"[Bb]earer ([A-Za-z0-9\-._~+/]{20,})";
99
100// GitHub PAT (classic: ghp_, fine-grained: github_pat_)
101const GITHUB_PATTERN: &str = r"(?:ghp_[A-Za-z0-9]{36}|github_pat_[A-Za-z0-9_]{82})";
102
103// GitLab PAT
104const GITLAB_PATTERN: &str = r"glpat-[A-Za-z0-9_-]{20}";
105
106// Slack tokens
107const SLACK_PATTERN: &str = r"xox[bpoas]-[0-9A-Za-z-]{10,}";
108
109// AWS STS temporary token (longer than AKIA access key, starts with AQOD/AQIA/IQoJ etc.)
110const AWS_STS_PATTERN: &str = r"(?:AQOD|AQIA|IQoJ|IQoI)[A-Za-z0-9+/=]{100,}";
111
112// Generic access_token / refresh_token in JSON
113const GENERIC_TOKEN_PATTERN: &str =
114    r#"["'](?:access_token|refresh_token|id_token)["']\s*:\s*["']([A-Za-z0-9\-._~+/=]{20,})["']"#;
115
116/// Scan raw bytes for authentication token patterns.
117///
118/// Returns one [`SessionTokenInfo`] per unique token value found in `data`.
119/// For patterns with capture groups (Bearer, Generic), the captured group
120/// value is used; for patterns without (JWT, GitHub, GitLab, Slack, AWS STS),
121/// the full match is used.
122pub(crate) fn scan_for_tokens(data: &[u8], pid: u64, process_name: &str) -> Vec<SessionTokenInfo> {
123    let text = String::from_utf8_lossy(data);
124
125    let mut seen: HashSet<String> = HashSet::new();
126    let mut out = Vec::new();
127
128    for tp in token_patterns() {
129        for caps in tp.re.captures_iter(&text) {
130            let value = if tp.use_capture {
131                caps.get(1).map(|m| m.as_str().to_string())
132            } else {
133                caps.get(0).map(|m| m.as_str().to_string())
134            };
135
136            let Some(token_value) = value else { continue };
137
138            if seen.insert(token_value.clone()) {
139                out.push(SessionTokenInfo {
140                    pid,
141                    image_name: process_name.to_string(),
142                    token_type: tp.label.to_string(),
143                    token_value,
144                });
145            }
146        }
147    }
148
149    out
150}

152#[cfg(test)]
153mod tests {
154    use super::*;
155
156    const DUMMY_PID: u64 = 1234;
157    const DUMMY_PROC: &str = "test.exe";
158
159    #[test]
160    fn scan_empty_returns_nothing() {
161        assert!(scan_for_tokens(b"", DUMMY_PID, DUMMY_PROC).is_empty());
162    }
163
164    #[test]
165    fn scan_no_tokens_returns_nothing() {
166        assert!(scan_for_tokens(b"hello world no auth here", DUMMY_PID, DUMMY_PROC).is_empty());
167    }
168
169    #[test]
170    fn scan_detects_jwt() {
171        // Real JWT structure: three base64url segments
172        let data = b"Authorization: Bearer eyJhbGciOiJSUzI1NiIsInR5cCI6IkpXVCJ9.eyJzdWIiOiJ1c2VyMTIzIiwiaWF0IjoxNzAwMDAwMDAwfQ.SflKxwRJSMeKKF2QT4fwpMeJf36POk6yJV_adQssw5c";
173        let results = scan_for_tokens(data, DUMMY_PID, DUMMY_PROC);
174        assert!(!results.is_empty(), "should detect JWT");
175        assert!(results
176            .iter()
177            .any(|t| t.token_type == "Jwt" || t.token_type == "Bearer"));
178    }
179
180    #[test]
181    fn scan_detects_github_pat() {
182        let data = b"git remote set-url origin https://ghp_ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghij@github.com/org/repo";
183        let results = scan_for_tokens(data, DUMMY_PID, DUMMY_PROC);
184        assert!(!results.is_empty(), "should detect GitHub PAT");
185        assert!(results.iter().any(|t| t.token_type == "GitHub"));
186        assert!(results.iter().any(|t| t.token_value.starts_with("ghp_")));
187    }
188
189    #[test]
190    fn scan_detects_slack_token() {
191        let data = b"token=xoxb-123456789012-123456789012-AbCdEfGhIjKlMnOpQrStUvWx";
192        let results = scan_for_tokens(data, DUMMY_PID, DUMMY_PROC);
193        assert!(!results.is_empty(), "should detect Slack token");
194        assert!(results.iter().any(|t| t.token_type == "Slack"));
195    }

200        let results = scan_for_tokens(data, DUMMY_PID, DUMMY_PROC);

202    }

204    #[test]
205    fn scan_deduplicates_same_token_multiple_occurrences() {
206        // Same JWT appears twice in the buffer
207        let jwt = b"eyJhbGciOiJSUzI1NiIsInR5cCI6IkpXVCJ9.eyJzdWIiOiJ1c2VyMTIzIn0.SflKxwRJSMeKKF2QT4fwpMeJf36POk6yJV_adQssw5c";
208        let mut data = Vec::new();
209        data.extend_from_slice(jwt);
210        data.extend_from_slice(b" some data ");
211        data.extend_from_slice(jwt);
212        let results = scan_for_tokens(&data, DUMMY_PID, DUMMY_PROC);
213        let jwt_results: Vec<_> = results.iter().filter(|t| t.token_type == "Jwt").collect();
214        assert_eq!(jwt_results.len(), 1, "duplicate JWT should be deduplicated");
215    }
216
217    #[test]
218    fn scan_for_tokens_is_deterministic() {
219        // Regression guard: sequential calls must return identical results.
220        let data = b"Authorization: Bearer eyJhbGciOiJSUzI1NiIsInR5cCI6IkpXVCJ9.eyJzdWIiOiJ1c2VyMTIzIiwiaWF0IjoxNzAwMDAwMDAwfQ.SflKxwRJSMeKKF2QT4fwpMeJf36POk6yJV_adQssw5c";
221        let r1 = scan_for_tokens(data, DUMMY_PID, DUMMY_PROC);
222        let r2 = scan_for_tokens(data, DUMMY_PID, DUMMY_PROC);
223        assert_eq!(
224            r1.len(),
225            r2.len(),
226            "second call must return same number of results"
227        );
228        for (t1, t2) in r1.iter().zip(r2.iter()) {
229            assert_eq!(
230                t1.token_type, t2.token_type,
231                "token_type must be identical across calls"
232            );
233            assert_eq!(
234                t1.token_value, t2.token_value,
235                "token_value must be identical across calls"
236            );
237        }
238    }
