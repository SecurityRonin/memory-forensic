1//! Token impersonation chain detection.
2//!
3//! Detects threads that are impersonating a higher-privileged security
4//! context than their owning process. The most suspicious pattern is a
5//! thread in a user-mode process impersonating the SYSTEM account
6//! (S-1-5-18) at Impersonation or Delegation level — a technique used
7//! by privilege escalation exploits and pass-the-token attacks.
8//!
9//! Detection walks ETHREAD entries and checks `CrossThreadFlags` bit
10//! `ActiveImpersonationInfo`. When set, the thread's `ImpersonationInfo`
11//! pointer leads to an impersonation token whose user SID can be compared
12//! against the process's primary token SID.
13
14use memf_core::object_reader::ObjectReader;
15use memf_format::PhysicalMemoryProvider;
16
17use crate::Result;
18
19/// Information about a thread performing token impersonation.
20#[derive(Debug, Clone, serde::Serialize)]
21pub struct TokenImpersonationInfo {
22    /// Process ID of the owning process.
23    pub pid: u32,
24    /// Thread ID.
25    pub tid: u32,
26    /// Process image name.
27    pub process_name: String,
28    /// SID string of the process primary token user.
29    pub primary_token_user: String,
30    /// SID string of the thread impersonation token user.
31    pub impersonation_token_user: String,
32    /// Impersonation level: 0=Anonymous, 1=Identification, 2=Impersonation, 3=Delegation.
33    pub impersonation_level: u32,
34    /// Human-readable impersonation level name.
35    pub impersonation_level_name: String,
36    /// True if the thread is impersonating SYSTEM from a non-SYSTEM process at level ≥2.
37    pub is_suspicious: bool,
38}
39
40/// Returns the human-readable name for a Windows impersonation level.
41pub fn impersonation_level_name(level: u32) -> &'static str {
42    match level {
43        0 => "Anonymous",
44        1 => "Identification",
45        2 => "Impersonation",
46        3 => "Delegation",
47        _ => "Unknown",
48    }
49}
50
51/// Classify a thread's impersonation as suspicious.
52///
53/// Suspicious criteria: the thread is impersonating SYSTEM (S-1-5-18)
54/// from a process that is not running as SYSTEM, at impersonation level
55/// ≥ 2 (actual impersonation or delegation — not just identification).
56pub fn classify_token_impersonation(
57    primary_user: &str,
58    impersonation_user: &str,
59    level: u32,
60) -> bool {
61    impersonation_user.contains("S-1-5-18") && !primary_user.contains("S-1-5-18") && level >= 2
62}
63
64/// Walk all threads and detect suspicious token impersonation.
65///
66/// For each thread with `ActiveImpersonationInfo` set in `CrossThreadFlags`,
67/// compares the impersonation token SID against the process primary token SID.
68///
69/// Returns `Ok(Vec::new())` if `PsActiveProcessHead` symbol is absent
70/// (graceful degradation).
71pub fn walk_token_impersonation<P: PhysicalMemoryProvider>(
72    reader: &ObjectReader<P>,
73) -> Result<Vec<TokenImpersonationInfo>> {
74    // Graceful degradation: require PsActiveProcessHead to walk processes/threads
75    if reader
76        .symbols()
77        .symbol_address("PsActiveProcessHead")
78        .is_none()
79    {
80        return Ok(Vec::new());
81    }
82
83    // In a full implementation we would:
84    // 1. Walk process list via PsActiveProcessHead
85    // 2. For each process, walk its thread list via _KPROCESS.ThreadListHead
86    // 3. For each _ETHREAD, check CrossThreadFlags.ActiveImpersonationInfo bit
87    // 4. If set, read ImpersonationInfo pointer → _PS_IMPERSONATION_INFORMATION
88    // 5. Read SecurityContext.Token from ETHREAD and process EPROCESS
89    // 6. Extract user SIDs from both tokens
90    // 7. Call classify_token_impersonation and build entries
91    //
92    // Returning empty pending full token/SID walker integration.
93    Ok(Vec::new())
94}

100    /// A non-SYSTEM process thread impersonating SYSTEM at level 2 is suspicious.
101    #[test]
102    fn classify_system_impersonation_from_user_process_suspicious() {
103        assert!(classify_token_impersonation(
104            "S-1-5-21-1234567890-1234567890-1234567890-1001", // regular user
105            "S-1-5-18",                                       // SYSTEM
106            2,                                                // Impersonation level
107        ));
108    }

110    /// A SYSTEM process thread also impersonating SYSTEM is benign (same user).
111    #[test]
112    fn classify_same_user_impersonation_benign() {
113        assert!(!classify_token_impersonation(
114            "S-1-5-18", // process is SYSTEM
115            "S-1-5-18", // impersonating SYSTEM too — same user, not escalation
116            2,
117        ));
118    }

191    /// walk_token_impersonation with PsActiveProcessHead present returns empty
192    /// (walker body is a stub pending integration).
193    #[test]
194    fn walk_token_impersonation_with_symbol_returns_empty() {
195        use memf_core::test_builders::{flags, PageTableBuilder};
196        use memf_core::vas::{TranslationMode, VirtualAddressSpace};
197        use memf_symbols::isf::IsfResolver;
198        use memf_symbols::test_builders::IsfBuilder;
199
200        let ps_head_vaddr: u64 = 0xFFFF_8001_0000_0000;
201        let ps_head_paddr: u64 = 0x0090_0000;
202
203        let isf = IsfBuilder::windows_kernel_preset()
204            .add_symbol("PsActiveProcessHead", ps_head_vaddr)
205            .build_json();
206        let resolver = IsfResolver::from_value(&isf).unwrap();
207        let mut page = vec![0u8; 4096];
208        // Circular empty list: Flink → self
209        page[0..8].copy_from_slice(&ps_head_vaddr.to_le_bytes());
210        let (cr3, mem) = PageTableBuilder::new()
211            .map_4k(ps_head_vaddr, ps_head_paddr, flags::WRITABLE)
212            .write_phys(ps_head_paddr, &page)
213            .build();
214        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
215        let reader = ObjectReader::new(vas, Box::new(resolver));
216
217        let results = walk_token_impersonation(&reader).unwrap();
218        assert!(results.is_empty());
219    }
