1//! UserAssist evidence-of-execution walker.
2//!
3//! Windows stores program launch counts and last-run timestamps in the
4//! NTUSER.DAT registry hive under
5//! `Software\Microsoft\Windows\CurrentVersion\Explorer\UserAssist\{GUID}\Count`.
6//! Values are ROT13-encoded file paths with a fixed-size binary data
7//! structure containing run count and FILETIME.
8//!
9//! The binary value data (72 bytes on Vista+) has the following layout:
10//!   - Offset  4: Run count (u32)
11//!   - Offset  8: Focus count (u32)
12//!   - Offset 12: Focus time in milliseconds (u32)
13//!   - Offset 60: Last run time (u64, FILETIME)
14

84/// A single UserAssist entry recovered from the registry.

104/// Decode a ROT13-encoded string.
105///
106/// ROT13 rotates ASCII letters by 13 positions, wrapping around.
107/// Non-alphabetic characters pass through unchanged. This is used by
108/// Windows to obfuscate UserAssist value names in the registry.
109pub fn rot13_decode(s: &str) -> String {
110    s.chars()
111        .map(|c| match c {
112            'A'..='M' | 'a'..='m' => (c as u8 + 13) as char,
113            'N'..='Z' | 'n'..='z' => (c as u8 - 13) as char,
114            other => other,
115        })
116        .collect()
117}

151    "wscript.exe",

160/// Classify a decoded UserAssist name as suspicious.
161///
162/// Returns `true` if the name matches patterns commonly associated with
163/// post-exploitation tools, living-off-the-land abuse, or programs run
164/// from unusual locations:
165///
166/// - Known offensive tools: mimikatz, psexec, cobalt strike, etc.
167/// - Shell interpreters from unusual paths (not `\Windows\System32\`)
168/// - Script engines (wscript, cscript, mshta) -- frequently abused
169/// - Encoded command-line launchers
170pub fn classify_userassist(name: &str) -> bool {
171    let lower = name.to_ascii_lowercase();
172
173    // Check for known offensive tool names anywhere in the path.
174    for tool in SUSPICIOUS_TOOLS {
175        if lower.contains(tool) {
176            return true;
177        }
178    }
179
180    // Check for LOLBins — these are suspicious when they appear in
181    // UserAssist because it means a user interactively launched them.
182    for lolbin in SUSPICIOUS_LOLBINS {
183        if lower.ends_with(lolbin)
184            || lower.contains(&format!("\\{lolbin}"))
185            || lower.contains(&format!("/{lolbin}"))
186        {
187            return true;
188        }
189    }
190
191    // cmd.exe or powershell.exe from outside System32 is suspicious.
192    let is_cmd_or_ps = lower.contains("cmd.exe") || lower.contains("powershell.exe");
193    if is_cmd_or_ps && !lower.contains("\\windows\\system32\\") {
194        return true;
195    }
196
197    false
198}

202/// Compute the virtual address of a cell given its cell index.
203fn cell_address(hive_addr: u64, cell_index: u32) -> u64 {
204    hive_addr
205        .wrapping_add(HBIN_START_OFFSET)
206        .wrapping_add(cell_index as u64)
207}

209/// Read cell data from a cell at `cell_vaddr`, skipping the 4-byte size header.
210fn read_cell_data<P: PhysicalMemoryProvider>(
211    reader: &ObjectReader<P>,
212    cell_vaddr: u64,
213) -> crate::Result<Vec<u8>> {
214    let size_bytes = reader.read_bytes(cell_vaddr, 4)?;
215    let raw_size = i32::from_le_bytes(size_bytes[..4].try_into().unwrap());
216    let abs_size = raw_size.unsigned_abs() as usize;
217    if abs_size <= 4 {
218        return Ok(Vec::new());
219    }
220    let data_len = (abs_size - 4).min(0x10000);
221    reader
222        .read_bytes(cell_vaddr.wrapping_add(4), data_len)
223        .map_err(Into::into)
224}

226/// Extract the key name from an nk cell's data bytes (ASCII, compressed).
227fn read_key_name(nk_data: &[u8]) -> String {
228    if nk_data.len() < NK_NAME_OFFSET + 1 {
229        return String::new();
230    }
231    let name_len = u16::from_le_bytes(
232        nk_data[NK_NAME_LENGTH_OFFSET..NK_NAME_LENGTH_OFFSET + 2]
233            .try_into()
234            .unwrap(),
235    ) as usize;
236    let end = NK_NAME_OFFSET + name_len;
237    if end > nk_data.len() {
238        return String::new();
239    }
240    String::from_utf8_lossy(&nk_data[NK_NAME_OFFSET..end]).into_owned()
241}

260/// Find a subkey by name (case-insensitive) under a given nk cell.
261///
262/// Returns the cell index of the matching subkey, or `None`.
263fn find_subkey<P: PhysicalMemoryProvider>(
264    reader: &ObjectReader<P>,
265    hive_addr: u64,
266    nk_data: &[u8],
267    target_name: &str,
268) -> crate::Result<Option<u32>> {
269    if nk_data.len() < NK_STABLE_SUBKEYS_LIST_OFFSET + 4 {
270        return Ok(None);
271    }
272
273    let subkey_count = u32::from_le_bytes(
274        nk_data[NK_STABLE_SUBKEY_COUNT_OFFSET..NK_STABLE_SUBKEY_COUNT_OFFSET + 4]
275            .try_into()
276            .unwrap(),
277    ) as usize;
278
279    if subkey_count == 0 {
280        return Ok(None);
281    }
282
283    let subkeys_list_cell = u32::from_le_bytes(
284        nk_data[NK_STABLE_SUBKEYS_LIST_OFFSET..NK_STABLE_SUBKEYS_LIST_OFFSET + 4]
285            .try_into()
286            .unwrap(),
287    );
288
289    // The subkeys list cell may be an index node (lf, lh, ri, li).
290    // Read its data and iterate child cell indices.
291    let list_vaddr = cell_address(hive_addr, subkeys_list_cell);
292    let list_data = read_cell_data(reader, list_vaddr)?;
293
294    if list_data.len() < 4 {
295        return Ok(None);
296    }
297
298    let sig = u16::from_le_bytes(list_data[0..2].try_into().unwrap());
299    let count = u16::from_le_bytes(list_data[2..4].try_into().unwrap()) as usize;
300    let count = count.min(MAX_SUBKEYS);
301
302    match sig {
303        // "lf" (0x666C) or "lh" (0x686C): each entry is 8 bytes (cell index + hash).
304        0x666C | 0x686C => {
305            for i in 0..count {
306                let off = 4 + i * 8;
307                if off + 4 > list_data.len() {
308                    break;
309                }
310                let child_cell = u32::from_le_bytes(list_data[off..off + 4].try_into().unwrap());
311                let child_vaddr = cell_address(hive_addr, child_cell);
312                if let Ok(child_nk) = read_cell_data(reader, child_vaddr) {
313                    if child_nk.len() >= NK_NAME_OFFSET {
314                        let child_sig = u16::from_le_bytes(child_nk[0..2].try_into().unwrap());
315                        if child_sig == NK_SIGNATURE {
316                            let name = read_key_name(&child_nk);
317                            if name.eq_ignore_ascii_case(target_name) {
318                                return Ok(Some(child_cell));
319                            }
320                        }
321                    }
322                }
323            }
324        }
325        // "li" (0x696C): each entry is 4 bytes (cell index only).
326        0x696C => {
327            for i in 0..count {
328                let off = 4 + i * 4;
329                if off + 4 > list_data.len() {
330                    break;
331                }
332                let child_cell = u32::from_le_bytes(list_data[off..off + 4].try_into().unwrap());
333                let child_vaddr = cell_address(hive_addr, child_cell);
334                if let Ok(child_nk) = read_cell_data(reader, child_vaddr) {
335                    if child_nk.len() >= NK_NAME_OFFSET {
336                        let child_sig = u16::from_le_bytes(child_nk[0..2].try_into().unwrap());
337                        if child_sig == NK_SIGNATURE {
338                            let name = read_key_name(&child_nk);
339                            if name.eq_ignore_ascii_case(target_name) {
340                                return Ok(Some(child_cell));
341                            }
342                        }
343                    }
344                }
345            }
346        }
347        // "ri" (0x6972): index of indices — each entry is a cell index to
348        // another lf/lh/li list. Recurse one level.
349        0x6972 => {
350            for i in 0..count {
351                let off = 4 + i * 4;
352                if off + 4 > list_data.len() {
353                    break;
354                }
355                let sub_list_cell = u32::from_le_bytes(list_data[off..off + 4].try_into().unwrap());
356                // Build a synthetic nk_data-like slice so we can call ourselves
357                // with the sub-list. Instead, just read the sub-list directly.
358                let sub_vaddr = cell_address(hive_addr, sub_list_cell);
359                let sub_data = read_cell_data(reader, sub_vaddr)?;
360                if sub_data.len() < 4 {
361                    continue;
362                }
363                let sub_sig = u16::from_le_bytes(sub_data[0..2].try_into().unwrap());
364                let sub_count = u16::from_le_bytes(sub_data[2..4].try_into().unwrap()) as usize;
365                let sub_count = sub_count.min(MAX_SUBKEYS);
366                let entry_size: usize = match sub_sig {
367                    0x666C | 0x686C => 8,
368                    0x696C => 4,
369                    _ => continue,
370                };
371                for j in 0..sub_count {
372                    let soff = 4 + j * entry_size;
373                    if soff + 4 > sub_data.len() {
374                        break;
375                    }
376                    let child_cell =
377                        u32::from_le_bytes(sub_data[soff..soff + 4].try_into().unwrap());
378                    let child_vaddr = cell_address(hive_addr, child_cell);
379                    if let Ok(child_nk) = read_cell_data(reader, child_vaddr) {
380                        if child_nk.len() >= NK_NAME_OFFSET {
381                            let child_sig = u16::from_le_bytes(child_nk[0..2].try_into().unwrap());
382                            if child_sig == NK_SIGNATURE {
383                                let name = read_key_name(&child_nk);
384                                if name.eq_ignore_ascii_case(target_name) {
385                                    return Ok(Some(child_cell));
386                                }
387                            }
388                        }
389                    }
390                }
391            }
392        }
393        _ => {}
394    }
395
396    Ok(None)
397}

399/// List all subkey cell indices under a given nk cell.
400fn list_subkeys<P: PhysicalMemoryProvider>(
401    reader: &ObjectReader<P>,
402    hive_addr: u64,
403    nk_data: &[u8],
404) -> crate::Result<Vec<u32>> {
405    if nk_data.len() < NK_STABLE_SUBKEYS_LIST_OFFSET + 4 {
406        return Ok(Vec::new());
407    }
408
409    let subkey_count = u32::from_le_bytes(
410        nk_data[NK_STABLE_SUBKEY_COUNT_OFFSET..NK_STABLE_SUBKEY_COUNT_OFFSET + 4]
411            .try_into()
412            .unwrap(),
413    ) as usize;
414
415    if subkey_count == 0 {
416        return Ok(Vec::new());
417    }
418
419    let subkeys_list_cell = u32::from_le_bytes(
420        nk_data[NK_STABLE_SUBKEYS_LIST_OFFSET..NK_STABLE_SUBKEYS_LIST_OFFSET + 4]
421            .try_into()
422            .unwrap(),
423    );
424
425    let list_vaddr = cell_address(hive_addr, subkeys_list_cell);
426    let list_data = read_cell_data(reader, list_vaddr)?;
427
428    if list_data.len() < 4 {
429        return Ok(Vec::new());
430    }
431
432    let sig = u16::from_le_bytes(list_data[0..2].try_into().unwrap());
433    let count = u16::from_le_bytes(list_data[2..4].try_into().unwrap()) as usize;
434    let count = count.min(MAX_SUBKEYS);
435    let mut cells = Vec::with_capacity(count);
436
437    match sig {
438        0x666C | 0x686C => {
439            for i in 0..count {
440                let off = 4 + i * 8;
441                if off + 4 > list_data.len() {
442                    break;
443                }
444                cells.push(u32::from_le_bytes(
445                    list_data[off..off + 4].try_into().unwrap(),
446                ));
447            }
448        }
449        0x696C => {
450            for i in 0..count {
451                let off = 4 + i * 4;
452                if off + 4 > list_data.len() {
453                    break;
454                }
455                cells.push(u32::from_le_bytes(
456                    list_data[off..off + 4].try_into().unwrap(),
457                ));
458            }
459        }
460        0x6972 => {
461            for i in 0..count {
462                let off = 4 + i * 4;
463                if off + 4 > list_data.len() {
464                    break;
465                }
466                let sub_cell = u32::from_le_bytes(list_data[off..off + 4].try_into().unwrap());
467                let sub_vaddr = cell_address(hive_addr, sub_cell);
468                if let Ok(sub_data) = read_cell_data(reader, sub_vaddr) {
469                    if sub_data.len() >= 4 {
470                        let sub_sig = u16::from_le_bytes(sub_data[0..2].try_into().unwrap());
471                        let sub_count =
472                            u16::from_le_bytes(sub_data[2..4].try_into().unwrap()) as usize;
473                        let sub_count = sub_count.min(MAX_SUBKEYS);
474                        let entry_size: usize = match sub_sig {
475                            0x666C | 0x686C => 8,
476                            0x696C => 4,
477                            _ => continue,
478                        };
479                        for j in 0..sub_count {
480                            let soff = 4 + j * entry_size;
481                            if soff + 4 > sub_data.len() {
482                                break;
483                            }
484                            cells.push(u32::from_le_bytes(
485                                sub_data[soff..soff + 4].try_into().unwrap(),
486                            ));
487                        }
488                    }
489                }
490            }
491        }
492        _ => {}
493    }
494
495    Ok(cells)
496}

500/// Walk UserAssist entries from a registry hive loaded in memory.
501///
502/// `hive_addr` is the virtual address of the `_HBASE_BLOCK` for an
503/// NTUSER.DAT hive. The walker navigates:
504///   1. Root key -> `Software\Microsoft\Windows\CurrentVersion\Explorer\UserAssist`
505///   2. Each GUID subkey -> `Count` subkey
506///   3. Each value in `Count`: ROT13-decode the name, parse the 72-byte
507///      binary data for run count, focus count, focus time, and last-run FILETIME.
508///
509/// Returns `Ok(Vec::new())` if required symbols are missing or the path
510/// does not exist (graceful degradation).
511pub fn walk_userassist<P: PhysicalMemoryProvider>(
512    reader: &ObjectReader<P>,
513    hive_addr: u64,
514) -> crate::Result<Vec<UserAssistEntry>> {
515    // Read root cell index from _HBASE_BLOCK.
516    let root_cell_bytes =
517        match reader.read_bytes(hive_addr.wrapping_add(HBASE_BLOCK_ROOT_CELL_OFFSET), 4) {
518            Ok(b) => b,
519            Err(_) => return Ok(Vec::new()),
520        };
521    let root_cell = u32::from_le_bytes(root_cell_bytes[..4].try_into().unwrap());
522    if root_cell == 0 {
523        return Ok(Vec::new());
524    }
525
526    // Navigate from root to the UserAssist key.
527    let mut current_cell = root_cell;
528    for &component in USERASSIST_PATH {
529        let cell_vaddr = cell_address(hive_addr, current_cell);
530        let nk_data = match read_cell_data(reader, cell_vaddr) {
531            Ok(d) => d,
532            Err(_) => return Ok(Vec::new()),
533        };
534        if nk_data.len() < NK_NAME_OFFSET {
535            return Ok(Vec::new());
536        }
537        let sig = u16::from_le_bytes(nk_data[0..2].try_into().unwrap());
538        if sig != NK_SIGNATURE {
539            return Ok(Vec::new());
540        }
541        match find_subkey(reader, hive_addr, &nk_data, component)? {
542            Some(child) => current_cell = child,
543            None => return Ok(Vec::new()),
544        }
545    }
546
547    // `current_cell` now points to the UserAssist key.
548    // Enumerate GUID subkeys.
549    let ua_vaddr = cell_address(hive_addr, current_cell);
550    let ua_nk = read_cell_data(reader, ua_vaddr)?;
551    let guid_cells = list_subkeys(reader, hive_addr, &ua_nk)?;
552
553    let mut entries = Vec::new();
554
555    for guid_cell in guid_cells {
556        if entries.len() >= MAX_USERASSIST_ENTRIES {
557            break;
558        }
559
560        let guid_vaddr = cell_address(hive_addr, guid_cell);
561        let guid_nk = match read_cell_data(reader, guid_vaddr) {
562            Ok(d) => d,
563            Err(_) => continue,
564        };
565        if guid_nk.len() < NK_NAME_OFFSET {
566            continue;
567        }
568        let sig = u16::from_le_bytes(guid_nk[0..2].try_into().unwrap());
569        if sig != NK_SIGNATURE {
570            continue;
571        }
572
573        // Find the "Count" subkey under the GUID key.
574        let count_cell = match find_subkey(reader, hive_addr, &guid_nk, "Count")? {
575            Some(c) => c,
576            None => continue,
577        };
578
579        // Read values from the Count key.
580        let count_vaddr = cell_address(hive_addr, count_cell);
581        let count_nk = match read_cell_data(reader, count_vaddr) {
582            Ok(d) => d,
583            Err(_) => continue,
584        };
585        if count_nk.len() < NK_VALUES_LIST_OFFSET + 4 {
586            continue;
587        }
588
589        let value_count = u32::from_le_bytes(
590            count_nk[NK_VALUE_COUNT_OFFSET..NK_VALUE_COUNT_OFFSET + 4]
591                .try_into()
592                .unwrap(),
593        ) as usize;
594
595        if value_count == 0 {
596            continue;
597        }
598
599        let values_list_cell = u32::from_le_bytes(
600            count_nk[NK_VALUES_LIST_OFFSET..NK_VALUES_LIST_OFFSET + 4]
601                .try_into()
602                .unwrap(),
603        );
604
605        let vl_vaddr = cell_address(hive_addr, values_list_cell);
606        let vl_data = match read_cell_data(reader, vl_vaddr) {
607            Ok(d) => d,
608            Err(_) => continue,
609        };
610
611        let count = value_count.min(MAX_VALUES);
612        for i in 0..count {
613            if entries.len() >= MAX_USERASSIST_ENTRIES {
614                break;
615            }
616            let off = i * 4;
617            if off + 4 > vl_data.len() {
618                break;
619            }
620            let val_cell = u32::from_le_bytes(vl_data[off..off + 4].try_into().unwrap());
621
622            match parse_userassist_value(reader, hive_addr, val_cell) {
623                Ok(Some(entry)) => entries.push(entry),
624                _ => continue,
625            }
626        }
627    }
628
629    Ok(entries)
630}

632/// Parse a single UserAssist value cell into a `UserAssistEntry`.
633///
634/// Returns `Ok(None)` if the value data is too small (not Vista+ format)
635/// or the value cell is invalid.
636fn parse_userassist_value<P: PhysicalMemoryProvider>(
637    reader: &ObjectReader<P>,
638    hive_addr: u64,
639    val_cell: u32,
640) -> crate::Result<Option<UserAssistEntry>> {
641    let val_vaddr = cell_address(hive_addr, val_cell);
642    let vk_data = read_cell_data(reader, val_vaddr)?;
643
644    if vk_data.len() < VK_NAME_OFFSET {
645        return Ok(None);
646    }
647
648    let sig = u16::from_le_bytes(vk_data[0..2].try_into().unwrap());
649    if sig != VK_SIGNATURE {
650        return Ok(None);
651    }
652
653    // Read the ROT13-encoded value name.
654    let rot13_name = read_value_name(&vk_data);
655    let decoded_name = rot13_decode(&rot13_name);
656
657    // Read the binary value data.
658    let data_length = u32::from_le_bytes(
659        vk_data[VK_DATA_LENGTH_OFFSET..VK_DATA_LENGTH_OFFSET + 4]
660            .try_into()
661            .unwrap(),
662    );
663
664    // Strip MSB (inline data flag) for size check.
665    let actual_length = (data_length & 0x7FFF_FFFF) as usize;
666
667    if actual_length < USERASSIST_DATA_SIZE {
668        return Ok(None);
669    }
670
671    // Read the value data from the data cell.
672    let data_cell = u32::from_le_bytes(
673        vk_data[VK_DATA_OFFSET_OFFSET..VK_DATA_OFFSET_OFFSET + 4]
674            .try_into()
675            .unwrap(),
676    );
677
678    let data_vaddr = cell_address(hive_addr, data_cell);
679    let raw_data = read_cell_data(reader, data_vaddr)?;
680
681    if raw_data.len() < USERASSIST_DATA_SIZE {
682        return Ok(None);
683    }
684
685    // Parse the 72-byte UserAssist data structure (Vista+ format).
686    let run_count = u32::from_le_bytes(raw_data[4..8].try_into().unwrap());
687    let focus_count = u32::from_le_bytes(raw_data[8..12].try_into().unwrap());
688    let focus_time_ms = u32::from_le_bytes(raw_data[12..16].try_into().unwrap());
689    let last_run_time = u64::from_le_bytes(raw_data[60..68].try_into().unwrap());
690
691    let is_suspicious = classify_userassist(&decoded_name);
692
693    Ok(Some(UserAssistEntry {
694        name: decoded_name,
695        run_count,
696        focus_count,
697        last_run_time,
698        focus_time_ms,
699        is_suspicious,
700    }))
701}

1968    //
1969    // Builds a minimal synthetic NTUSER.DAT hive with the full path:
1970    //   root -> Software -> Microsoft -> Windows -> CurrentVersion ->
1971    //   Explorer -> UserAssist -> {GUID} -> Count -> value "zvzvxngm.rkr"
1972    //
1973    // Layout (all cells on a single 4K HBIN page):
1974    //   Slot size: 0x80 bytes each
1975    //   nk cells:    [0] root  [1] Software  [2] Microsoft  [3] Windows
1976    //                [4] CurrentVersion  [5] Explorer  [6] UserAssist
1977    //                [7] {GUID}  [8] Count
1978    //   lf lists:    [0x480..0x800] — 8 lf lists, one per parent nk
1979    //   values_list: 0x880
1980    //   vk cell:     0x900
1981    //   data cell:   0x980
1982    //
1983    // Each nk cell at slot N has:
1984    //   - 4-byte size header (negative = allocated)
1985    //   - NK_SIGNATURE at [0..2]
1986    //   - subkey_count=1 at [0x14]
1987    //   - subkeys_list_cell = lf_list_cell_for_N at [0x1C]
1988    //   - name at [0x48..], length at [0x48]  (wait: NK_NAME_LENGTH_OFFSET=0x48, NK_NAME_OFFSET=0x4C)
1989    //
1990    // The lf list for slot N:
1991    //   - points to the nk cell at slot N+1 (child)
1992    //   - sig = "lf" (0x666C), count = 1, entry = (child_cell, hash=0)
1993    //
1994    // The {GUID} nk (slot 7) has subkey_count=1, lf→Count nk (slot 8).
1995    // The Count nk (slot 8) has value_count=1, values_list_cell pointing to values_list.
1996    // The values_list has one entry: vk_cell.
1997    // The vk cell: name="zvzvxngm.rkr" (ROT13 of "mimikatz.exe"), data_length=72, data_cell.
1998    // The data cell: 72 bytes of UA data (run_count=5 at [4], focus_count=2 at [8], etc.)

2017        // ── Cell index plan (offsets within HBIN page) ──────────────────
2018        // Each nk slot is 0x80 bytes; cell index = slot_base within hbin page.
2019        // Since cell_address = hive_addr + 0x1000 + cell_index,
2020        // cell index 0 → vaddr = hbin_vaddr, which is on hbin_paddr.
2021        //
2022        // nk cells at indices: root=0x000, sw=0x080, ms=0x100, win=0x180,
2023        //   cv=0x200, exp=0x280, ua=0x300, guid=0x380, count=0x400
2024        // lf lists at: lf0=0x480(root→sw), lf1=0x500(sw→ms), lf2=0x580(ms→win),
2025        //   lf3=0x600(win→cv), lf4=0x680(cv→exp), lf5=0x700(exp→ua),

2050        const LF_UA: u32 = 0x560;
2051        const LF_GUID: u32 = 0x580;

2053        const VK_IDX: u32 = 0x5C0;
2054        const DATA_IDX: u32 = 0x1000; // on second HBIN page

2058        // Helper: write an allocated cell at `off` in buf.
2059        // Writes: i32 size (negative), then `payload` bytes.
2060        let write_cell = |buf: &mut Vec<u8>, off: usize, payload: &[u8]| {

2062            buf[off..off + 4].copy_from_slice(&(-abs_size).to_le_bytes());

2066        // Helper: build a minimal nk cell data (payload, no size header).
2067        // name is ASCII, subkey_count=1 (unless is_leaf), lf_cell is the subkeys list.

2070                       subkey_count: u32,
2071                       lf_cell: u32,
2072                       value_count: u32,

2078            nk[NK_STABLE_SUBKEY_COUNT_OFFSET..NK_STABLE_SUBKEY_COUNT_OFFSET + 4]
2079                .copy_from_slice(&subkey_count.to_le_bytes());
2080            nk[NK_STABLE_SUBKEYS_LIST_OFFSET..NK_STABLE_SUBKEYS_LIST_OFFSET + 4]
2081                .copy_from_slice(&lf_cell.to_le_bytes());
2082            nk[NK_VALUE_COUNT_OFFSET..NK_VALUE_COUNT_OFFSET + 4]
2083                .copy_from_slice(&value_count.to_le_bytes());
2084            nk[NK_VALUES_LIST_OFFSET..NK_VALUES_LIST_OFFSET + 4]
2085                .copy_from_slice(&vlist_cell.to_le_bytes());
2086            let name_len = name.len() as u16;
2087            nk[NK_NAME_LENGTH_OFFSET..NK_NAME_LENGTH_OFFSET + 2]
2088                .copy_from_slice(&name_len.to_le_bytes());
2089            nk[NK_NAME_OFFSET..NK_NAME_OFFSET + name.len()].copy_from_slice(name);
2090            nk
2091        };
2092
2093        // Helper: build an lf list payload pointing to one child cell.
2094        let make_lf = |child_cell: u32| -> Vec<u8> {

2097            lf[2..4].copy_from_slice(&1u16.to_le_bytes());
2098            lf[4..8].copy_from_slice(&child_cell.to_le_bytes());
2099            lf[8..12].copy_from_slice(&0u32.to_le_bytes()); // hash
2100            lf
2101        };

2103        let mut hbin = vec![0u8; 0x1000];
2104

2106        write_cell(

2108            NK_ROOT as usize,
2109            &make_nk(b"$$$PROTO.HIV", 1, LF_ROOT, 0, 0),
2110        );
2111        write_cell(
2112            &mut hbin,
2113            NK_SW as usize,
2114            &make_nk(b"Software", 1, LF_SW, 0, 0),
2115        );
2116        write_cell(
2117            &mut hbin,

2123            NK_WIN as usize,

2127            &mut hbin,
2128            NK_CV as usize,
2129            &make_nk(b"CurrentVersion", 1, LF_CV, 0, 0),

2132            &mut hbin,

2137            &mut hbin,
2138            NK_UA as usize,
2139            &make_nk(b"UserAssist", 1, LF_UA, 0, 0),
2140        );
2141        write_cell(
2142            &mut hbin,
2143            NK_GUID as usize,
2144            &make_nk(b"{GUIDTEST}", 1, LF_GUID, 0, 0),
2145        );
2146        // Count nk: subkey_count=0, value_count=1, values_list=VLIST
2147        write_cell(
2148            &mut hbin,
2149            NK_COUNT as usize,
2150            &make_nk(b"Count", 0, 0, 1, VLIST),
2151        );
2152
2153        // lf lists
2154        write_cell(&mut hbin, LF_ROOT as usize, &make_lf(NK_SW));
2155        write_cell(&mut hbin, LF_SW as usize, &make_lf(NK_MS));
2156        write_cell(&mut hbin, LF_MS as usize, &make_lf(NK_WIN));
2157        write_cell(&mut hbin, LF_WIN as usize, &make_lf(NK_CV));
2158        write_cell(&mut hbin, LF_CV as usize, &make_lf(NK_EXP));
2159        write_cell(&mut hbin, LF_EXP as usize, &make_lf(NK_UA));
2160        write_cell(&mut hbin, LF_UA as usize, &make_lf(NK_GUID));
2161        write_cell(&mut hbin, LF_GUID as usize, &make_lf(NK_COUNT));
