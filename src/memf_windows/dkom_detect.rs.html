1//! DKOM (Direct Kernel Object Manipulation) cross-reference detection.
2//!
3//! Compares multiple kernel enumeration sources to detect objects that have
4//! been unlinked from one or more lists — the hallmark of DKOM rootkits.
5//! MITRE ATT&CK T1014.
6
7use memf_core::object_reader::ObjectReader;
8use memf_format::PhysicalMemoryProvider;
9
10use crate::{
11    types::{DkomDiscrepancy, DkomType},
12    Result,
13};

15/// Classify a kernel object's presence across multiple enumeration sources and
16/// return the type of DKOM hiding detected, if any.
17///
18/// - Returns `Some(DkomType::ProcessUnlinked)` if the object is in `CidTable`
19///   (`in_cid`) but absent from `PsActiveProcessHead` (`!in_active`).
20/// - Returns `Some(DkomType::DriverUnlinked)` if the driver is in `MmDriverList`
21///   (`in_driver`) but absent from `PsLoadedModuleList` (`!in_module`).
22/// - Returns `None` if no discrepancy is detected.
23///
24/// Process unlinking takes priority over driver unlinking when both conditions
25/// are simultaneously true.
26#[allow(clippy::fn_params_excessive_bools)]
27pub fn classify_dkom(
28    in_cid: bool,
29    in_active: bool,
30    in_driver: bool,
31    in_module: bool,
32) -> Option<DkomType> {
33    if in_cid && !in_active {
34        Some(DkomType::ProcessUnlinked)
35    } else if in_driver && !in_module {
36        Some(DkomType::DriverUnlinked)
37    } else {
38        None
39    }
40}

42/// Cross-reference kernel process/driver/thread lists to detect DKOM hiding.
43///
44/// Compares:
45/// - `PsActiveProcessHead` linked list vs `CidTable` handle table (processes)
46/// - `PsLoadedModuleList` vs `MmDriverList` (drivers)
47/// - Per-process `_EPROCESS.ThreadListHead` vs global thread scans (threads)
48///
49/// Any object present in one source but absent in another is reported as a
50/// `DkomDiscrepancy`.
51///
52/// # MITRE ATT&CK
53/// T1014 — Rootkit
54pub fn scan_dkom<P: PhysicalMemoryProvider>(
55    reader: &ObjectReader<P>,
56) -> Result<Vec<DkomDiscrepancy>> {
57    let _ = reader;
58    Ok(vec![])
59}

70    fn make_minimal_reader(
71    ) -> ObjectReader<memf_core::test_builders::SyntheticPhysMem> {
72        let isf = IsfBuilder::new().build_json();
73        let resolver = IsfResolver::from_value(&isf).unwrap();
74        let (cr3, mem) = PageTableBuilder::new().build();
75        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
76        ObjectReader::new(vas, Box::new(resolver))
77    }

79    #[test]
80    fn empty_memory_returns_ok_empty() {
81        let reader = make_minimal_reader();
82        let result = scan_dkom(&reader);
83        assert!(result.is_ok(), "should succeed with minimal reader");
84        assert!(result.unwrap().is_empty(), "empty memory → no DKOM discrepancies");
85    }

98            image_name: "hidden.exe".to_string(),
99            present_in: vec!["CidTable".to_string()],
100            missing_from: vec!["PsActiveProcessHead".to_string()],

103        assert_eq!(disc.pid, 1234);
