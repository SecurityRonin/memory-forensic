1//! Section object forensics walker — MITRE ATT&CK T1055.
2//!
3//! Enumerates Windows section objects from the object manager namespace
4//! and detects suspicious configurations: image sections without a disk
5//! backing file, RWX anonymous sections, and sections shared across many
6//! processes without a backing file.
7
8use memf_core::object_reader::ObjectReader;
9use memf_format::PhysicalMemoryProvider;
10
11use crate::{types::SectionObjectInfo, Result};
12
13/// Classify whether a section object is suspicious based on how many processes
14/// have it mapped.
15///
16/// Returns `true` if `mapped_count > threshold`, indicating the section is
17/// shared across an abnormal number of processes — consistent with a
18/// process-doppelgänging or shared shellcode staging pattern.
19pub fn is_suspicious_section(mapped_count: u32, threshold: u32) -> bool {
20    mapped_count > threshold
21}
22
23/// Enumerate section objects from the kernel object manager and analyse
24/// them for suspicious attributes.
25///
26/// Walks `ObpRootDirectoryObject` → `\BaseNamedObjects` and process handle
27/// tables to find `_SECTION` objects. For each section inspects:
28/// - Whether it is an image section (`SEC_IMAGE`) and its backing file
29///   exists on disk.
30/// - The page protection of its `_SEGMENT`.
31/// - How many processes have it mapped.
32///
33/// # MITRE ATT&CK
34/// T1055 — Process Injection (Section / Doppelgänging)
35pub fn scan_section_objects<P: PhysicalMemoryProvider>(
36    reader: &ObjectReader<P>,
37) -> Result<Vec<SectionObjectInfo>> {
38    let _ = reader;
39    Ok(vec![])
40}

50    fn make_minimal_reader(
51    ) -> ObjectReader<memf_core::test_builders::SyntheticPhysMem> {
52        let isf = IsfBuilder::new().build_json();
53        let resolver = IsfResolver::from_value(&isf).unwrap();
54        let (cr3, mem) = PageTableBuilder::new().build();
55        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
56        ObjectReader::new(vas, Box::new(resolver))
57    }

59    #[test]
60    fn empty_memory_returns_ok_empty() {
61        let reader = make_minimal_reader();
62        let result = scan_section_objects(&reader);
63        assert!(result.is_ok(), "should succeed with minimal reader");
64        assert!(
65            result.unwrap().is_empty(),
66            "empty memory → no section objects"
67        );
68    }

70    #[test]
71    fn result_is_vec_of_section_object_info() {
72        let reader = make_minimal_reader();
73        let result: Result<Vec<SectionObjectInfo>> = scan_section_objects(&reader);
74        assert!(result.is_ok());
75    }

77    #[test]
78    fn section_object_info_fields_constructible() {
79        let info = SectionObjectInfo {
80            pid: 777,
81            image_name: "ghost.exe".to_string(),
82            section_name: r"\BaseNamedObjects\evil".to_string(),
83            backing_file: String::new(),
84            protection: 0x40, // PAGE_EXECUTE_READWRITE
85            mapped_process_count: 3,
86            is_image_section: true,
87            file_on_disk: false,
88        };
89        assert_eq!(info.pid, 777);
90        assert_eq!(info.protection, 0x40);
91        assert!(info.is_image_section);
92        assert!(!info.file_on_disk);
93        assert_eq!(info.mapped_process_count, 3);
94    }
