1//! WoW64 / Heaven's Gate anomaly detection walker — MITRE ATT&CK T1055.
2//!
3//! Detects 32-bit processes that use the Heaven's Gate technique (switching
4//! to 64-bit mode via a far JMP to CS=0x33) or whose WoW64 syscall stub
5//! has been patched to bypass hooks.
6
7use memf_core::object_reader::ObjectReader;
8use memf_format::PhysicalMemoryProvider;
9
10use crate::{types::Wow64AnomalyInfo, Result};
11
12/// Classify whether a WoW64 process has an anomalous configuration.
13///
14/// Returns `true` if the process is marked as WoW64 (`is_wow64`) but does not
15/// have a 32-bit PEB (`!has_32bit_peb`). A genuine WoW64 process always has
16/// a 32-bit PEB; its absence while the WoW64 flag is set indicates that the
17/// WoW64 machinery has been tampered with (e.g. Heaven's Gate manipulation).
18pub fn is_wow64_suspicious(is_wow64: bool, has_32bit_peb: bool) -> bool {
19    is_wow64 && !has_32bit_peb
20}
21
22/// Scan all WoW64 (32-bit) processes for Heaven's Gate and syscall-stub
23/// tampering anomalies.
24///
25/// For each process where `_EPROCESS.Wow64Process` is non-null:
26/// - Checks for a CS=0x33 segment selector in any thread's context
27///   (Heaven's Gate indicator).
28/// - Reads the first bytes of the `KiFastSystemCall` stub in the 32-bit
29///   `ntdll.dll` mapping and compares with the expected sequence; any
30///   deviation sets `syscall_stub_tampered`.
31/// - Records whether `wow64.dll` is present in the process module list.
32///
33/// # MITRE ATT&CK
34/// T1055 — Process Injection (WoW64 / Heaven's Gate)
35pub fn scan_wow64_anomalies<P: PhysicalMemoryProvider>(
36    reader: &ObjectReader<P>,
37) -> Result<Vec<Wow64AnomalyInfo>> {
38    let _ = reader;
39    Ok(vec![])
40}

50    fn make_minimal_reader(
51    ) -> ObjectReader<memf_core::test_builders::SyntheticPhysMem> {
52        let isf = IsfBuilder::new().build_json();
53        let resolver = IsfResolver::from_value(&isf).unwrap();
54        let (cr3, mem) = PageTableBuilder::new().build();
55        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
56        ObjectReader::new(vas, Box::new(resolver))
57    }

59    #[test]
60    fn empty_memory_returns_ok_empty() {
61        let reader = make_minimal_reader();
62        let result = scan_wow64_anomalies(&reader);
63        assert!(result.is_ok(), "should succeed with minimal reader");
64        assert!(
65            result.unwrap().is_empty(),
66            "empty memory → no WoW64 anomaly hits"
67        );
68    }

72        let reader = make_minimal_reader();
73        let result: Result<Vec<Wow64AnomalyInfo>> = scan_wow64_anomalies(&reader);
74        assert!(result.is_ok());
75    }
