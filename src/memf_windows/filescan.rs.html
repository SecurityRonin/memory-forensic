1//! Windows file object scanning and extraction.
2//!
3//! Filters handle table entries for "File" object types and reads
4//! the underlying `_FILE_OBJECT` structures to extract file name,
5//! device name, flags, size, and sharing disposition.
6
7use memf_core::object_reader::ObjectReader;
8use memf_format::PhysicalMemoryProvider;
9
10use crate::unicode::read_unicode_string;
11use crate::{FileObjectInfo, Result, WinHandleInfo};
12
13/// Maximum number of file objects to return (safety limit).
14const MAX_FILE_OBJECTS: usize = 10_000;
15
16/// Walk a list of handles and extract file object information.
17///
18/// Filters `handles` for entries whose `object_type` is `"File"`,
19/// then reads each `_FILE_OBJECT` at the handle's `object_addr` to
20/// extract file name, flags, device name, and sharing disposition.
21///
22/// Returns file objects sorted by file name.
23pub fn walk_file_objects<P: PhysicalMemoryProvider>(
24    reader: &ObjectReader<P>,
25    handles: &[WinHandleInfo],
26) -> Result<Vec<FileObjectInfo>> {
27    // Resolve _OBJECT_HEADER.Body offset to find the _FILE_OBJECT body
28    let body_offset = reader
29        .symbols()
30        .field_offset("_OBJECT_HEADER", "Body")
31        .ok_or_else(|| crate::Error::Walker("missing _OBJECT_HEADER.Body offset".into()))?;
32
33    let mut results = Vec::new();
34
35    for handle in handles {
36        if handle.object_type != "File" {
37            continue;
38        }
39
40        if results.len() >= MAX_FILE_OBJECTS {
41            break;
42        }
43
44        // The handle's object_addr points to the _OBJECT_HEADER.
45        // The _FILE_OBJECT body starts at object_addr + Body offset.
46        let file_obj_addr = handle.object_addr.wrapping_add(body_offset);
47
48        if let Ok(info) = read_file_object(reader, file_obj_addr, handle.granted_access) {
49            results.push(info);
50        }
51    }
52
53    results.sort_by(|a, b| a.file_name.cmp(&b.file_name));
54    Ok(results)
55}

57/// Convenience function that combines handle walking with file object extraction.
58///
59/// Calls `walk_handles` to enumerate all process handles, then passes them
60/// to `walk_file_objects`.  Falls back to an empty `Vec` on handle walking failure.
61pub fn scan_file_objects<P: PhysicalMemoryProvider>(
62    reader: &ObjectReader<P>,
63) -> Result<Vec<FileObjectInfo>> {
64    let ps_head = reader
65        .symbols()
66        .symbol_address("PsActiveProcessHead")
67        .ok_or_else(|| crate::Error::Walker("missing PsActiveProcessHead symbol".into()))?;
68
69    let Ok(handles) = crate::handles::walk_handles(reader, ps_head) else {
70        return Ok(Vec::new());
71    };
72
73    walk_file_objects(reader, &handles)
74}

76/// Read a single `_FILE_OBJECT` at the given virtual address.
77fn read_file_object<P: PhysicalMemoryProvider>(
78    reader: &ObjectReader<P>,
79    file_obj_addr: u64,
80    access_mask: u32,
81) -> Result<FileObjectInfo> {
82    // Read FileName (_UNICODE_STRING)
83    let filename_off = reader
84        .symbols()
85        .field_offset("_FILE_OBJECT", "FileName")
86        .ok_or_else(|| crate::Error::Walker("missing _FILE_OBJECT.FileName offset".into()))?;
87    let file_name =
88        read_unicode_string(reader, file_obj_addr.wrapping_add(filename_off)).unwrap_or_default();
89
90    // Read Flags (u32)
91    let fo_flags: u32 = reader.read_field(file_obj_addr, "_FILE_OBJECT", "Flags")?;
92
93    // Read CurrentByteOffset as file size proxy
94    let size: u64 = reader
95        .read_field(file_obj_addr, "_FILE_OBJECT", "CurrentByteOffset")
96        .unwrap_or(0);
97
98    // Read boolean fields
99    let delete_pending: u8 = reader
100        .read_field(file_obj_addr, "_FILE_OBJECT", "DeletePending")
101        .unwrap_or(0);
102    let shared_read: u8 = reader
103        .read_field(file_obj_addr, "_FILE_OBJECT", "SharedRead")
104        .unwrap_or(0);
105    let shared_write: u8 = reader
106        .read_field(file_obj_addr, "_FILE_OBJECT", "SharedWrite")
107        .unwrap_or(0);
108    let shared_delete: u8 = reader
109        .read_field(file_obj_addr, "_FILE_OBJECT", "SharedDelete")
110        .unwrap_or(0);
111
112    // Read DeviceObject pointer and resolve device name
113    let device_ptr: u64 = reader
114        .read_field(file_obj_addr, "_FILE_OBJECT", "DeviceObject")
115        .unwrap_or(0);
116    let device_name = resolve_device_name(reader, device_ptr);
117
118    Ok(FileObjectInfo {
119        object_addr: file_obj_addr,
120        file_name,
121        device_name,
122        access_mask,
123        flags: fo_flags,
124        size,
125        delete_pending: delete_pending != 0,
126        shared_read: shared_read != 0,
127        shared_write: shared_write != 0,
128        shared_delete: shared_delete != 0,
129    })
130}

132/// Resolve the device name from a `_DEVICE_OBJECT` pointer.
133///
134/// Follows `DeviceObject` to `DriverObject`, then reads the driver name.
135/// Returns an empty string if the device pointer is null or unreadable.
136fn resolve_device_name<P: PhysicalMemoryProvider>(
137    reader: &ObjectReader<P>,
138    device_ptr: u64,
139) -> String {
140    if device_ptr == 0 {
141        return String::new();
142    }
143
144    // Read _DEVICE_OBJECT.DriverObject pointer
145    let driver_obj: u64 = match reader.read_field(device_ptr, "_DEVICE_OBJECT", "DriverObject") {
146        Ok(v) => v,
147        Err(_) => return String::new(),
148    };
149
150    if driver_obj == 0 {
151        return String::new();
152    }
153
154    // Read _DRIVER_OBJECT.DriverName (_UNICODE_STRING)
155    let Some(name_off) = reader
156        .symbols()
157        .field_offset("_DRIVER_OBJECT", "DriverName")
158    else {
159        return String::new();
160    };
161
162    read_unicode_string(reader, driver_obj.wrapping_add(name_off)).unwrap_or_default()
163}

174    fn make_win_reader(ptb: PageTableBuilder) -> ObjectReader<SyntheticPhysMem> {
175        let isf = IsfBuilder::windows_kernel_preset().build_json();
176        let resolver = IsfResolver::from_value(&isf).unwrap();
177        let (cr3, mem) = ptb.build();
178        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
179        ObjectReader::new(vas, Box::new(resolver))

239    /// A synthetic _FILE_OBJECT with a file handle should be extracted correctly.
240    #[test]
241    fn walk_file_objects_with_file() {
242        // Layout:
243        // _OBJECT_HEADER at 0xFFFF_8000_0020_0000 (paddr 0x0080_0000)
244        // _FILE_OBJECT body at _OBJECT_HEADER + 0x30 = 0xFFFF_8000_0020_0030
245        // Filename string data at 0xFFFF_8000_0020_1000 (paddr 0x0081_0000)
246        let obj_header_vaddr: u64 = 0xFFFF_8000_0020_0000;
247        let obj_header_paddr: u64 = 0x0080_0000;
248        let file_obj_vaddr = obj_header_vaddr + OBJ_HEADER_BODY;
249        let file_obj_paddr = obj_header_paddr + OBJ_HEADER_BODY;
250
251        let strings_vaddr: u64 = 0xFFFF_8000_0020_1000;
252        let strings_paddr: u64 = 0x0081_0000;
253
254        // Build the file name: \Windows\System32\config\SYSTEM
255        let file_name = "\\Windows\\System32\\config\\SYSTEM";
256        let file_name_utf16 = utf16le(file_name);
257        let file_name_len = file_name_utf16.len() as u16;
258
259        let ptb = PageTableBuilder::new()
260            .map_4k(obj_header_vaddr, obj_header_paddr, flags::WRITABLE)
261            .map_4k(strings_vaddr, strings_paddr, flags::WRITABLE)
262            // _FILE_OBJECT.FileName (_UNICODE_STRING at file_obj + 0x58)
263            .write_phys(file_obj_paddr + FO_FILENAME, &file_name_len.to_le_bytes())
264            .write_phys(
265                file_obj_paddr + FO_FILENAME + 2,
266                &(file_name_len + 2).to_le_bytes(),
267            )
268            .write_phys_u64(file_obj_paddr + FO_FILENAME + 8, strings_vaddr)
269            .write_phys(strings_paddr, &file_name_utf16)
270            // _FILE_OBJECT.Flags = 0x40 (FO_CACHE_SUPPORTED)
271            .write_phys(file_obj_paddr + FO_FLAGS, &0x40u32.to_le_bytes())
272            // _FILE_OBJECT.CurrentByteOffset = 4096
273            .write_phys_u64(file_obj_paddr + FO_CURRENT_BYTE_OFFSET, 4096)
274            // _FILE_OBJECT.DeletePending = 0
275            .write_phys(file_obj_paddr + FO_DELETE_PENDING, &[0u8])
276            // _FILE_OBJECT.SharedRead = 1
277            .write_phys(file_obj_paddr + FO_SHARED_READ, &[1u8])
278            // _FILE_OBJECT.SharedWrite = 0
279            .write_phys(file_obj_paddr + FO_SHARED_WRITE, &[0u8])
280            // _FILE_OBJECT.SharedDelete = 1
281            .write_phys(file_obj_paddr + FO_SHARED_DELETE, &[1u8])
282            // _FILE_OBJECT.DeviceObject = 0 (no device for simplicity)
283            .write_phys_u64(file_obj_paddr + FO_DEVICE_OBJECT, 0);
284
285        let reader = make_win_reader(ptb);
286
287        let handles = vec![WinHandleInfo {
288            pid: 4,
289            image_name: "System".into(),
290            handle_value: 4,
291            object_addr: obj_header_vaddr,
292            object_type: "File".into(),
293            granted_access: 0x0012_019F,
294        }];
295
296        let result = walk_file_objects(&reader, &handles).unwrap();
297        assert_eq!(result.len(), 1);
298
299        let fo = &result[0];
300        assert_eq!(fo.file_name, "\\Windows\\System32\\config\\SYSTEM");
301        assert_eq!(fo.flags, 0x40);
302        assert_eq!(fo.size, 4096);
303        assert!(!fo.delete_pending);
304        assert!(fo.shared_read);
305        assert!(!fo.shared_write);
306        assert!(fo.shared_delete);
307        assert_eq!(fo.access_mask, 0x0012_019F);
308        assert_eq!(fo.object_addr, file_obj_vaddr);
309        // Device name should be empty since DeviceObject is null
310        assert_eq!(fo.device_name, "");
311    }

414    /// walk_file_objects with DeviceObject != 0 triggers resolve_device_name.
415    ///
416    /// Covers resolve_device_name lines 140-162 (non-null device path).
417    /// Layout:
418    ///   _FILE_OBJECT.DeviceObject → _DEVICE_OBJECT.DriverObject → _DRIVER_OBJECT.DriverName
419    #[test]
420    fn walk_file_objects_with_device_object_resolves_driver_name() {
421        // _DEVICE_OBJECT.DriverObject is at offset 0x08 (from windows_kernel_preset).
422        // _DRIVER_OBJECT.DriverName (_UNICODE_STRING) is at offset 0x38.
423        // _UNICODE_STRING: Length(u16)+MaxLen(u16) at 0/2, Buffer(ptr) at 8.
424        //
425        // Addresses (paddr < 0x00FF_FFFF):
426        let obj_header_vaddr: u64 = 0xFFFF_8000_0030_0000;
427        let obj_header_paddr: u64 = 0x0050_0000;
428        let file_obj_paddr = obj_header_paddr + OBJ_HEADER_BODY;
429
430        let dev_vaddr: u64 = 0xFFFF_8000_0031_0000;
431        let dev_paddr: u64 = 0x0051_0000;
432        let drv_vaddr: u64 = 0xFFFF_8000_0032_0000;
433        let drv_paddr: u64 = 0x0052_0000;
434        let str_vaddr: u64 = 0xFFFF_8000_0033_0000;
435        let str_paddr: u64 = 0x0053_0000;
436
437        let driver_name = "\\Driver\\disk";
438        let name_utf16 = utf16le(driver_name);
439        let name_len = name_utf16.len() as u16;
440
441        let ptb = PageTableBuilder::new()
442            // _OBJECT_HEADER + _FILE_OBJECT
443            .map_4k(obj_header_vaddr, obj_header_paddr, flags::WRITABLE)
444            // _FILE_OBJECT.DeviceObject at fo_paddr + 0x08 = dev_vaddr
445            .write_phys_u64(file_obj_paddr + FO_DEVICE_OBJECT, dev_vaddr)
446            // _FILE_OBJECT.FileName = empty (length = 0 at FO_FILENAME)
447            .write_phys(file_obj_paddr + FO_FLAGS, &0u32.to_le_bytes())
448            .write_phys(file_obj_paddr + FO_DELETE_PENDING, &[0u8])
449            .write_phys(file_obj_paddr + FO_SHARED_READ, &[0u8])
450            .write_phys(file_obj_paddr + FO_SHARED_WRITE, &[0u8])
451            .write_phys(file_obj_paddr + FO_SHARED_DELETE, &[0u8])
452            // _DEVICE_OBJECT
453            .map_4k(dev_vaddr, dev_paddr, flags::WRITABLE)
454            // _DEVICE_OBJECT.DriverObject at dev_paddr + 0x08 = drv_vaddr
455            .write_phys_u64(dev_paddr + 0x08, drv_vaddr)
456            // _DRIVER_OBJECT
457            .map_4k(drv_vaddr, drv_paddr, flags::WRITABLE)
458            // _DRIVER_OBJECT.DriverName (_UNICODE_STRING) at drv_paddr + 0x38
459            .write_phys(drv_paddr + 0x38, &name_len.to_le_bytes()) // Length
460            .write_phys(drv_paddr + 0x3A, &(name_len + 2).to_le_bytes()) // MaximumLength
461            .write_phys_u64(drv_paddr + 0x40, str_vaddr) // Buffer ptr
462            // String data
463            .map_4k(str_vaddr, str_paddr, flags::WRITABLE)
464            .write_phys(str_paddr, &name_utf16);
465
466        let reader = make_win_reader(ptb);
467
468        let handles = vec![WinHandleInfo {
469            pid: 4,
470            image_name: "System".into(),
471            handle_value: 4,
472            object_addr: obj_header_vaddr,
473            object_type: "File".into(),
474            granted_access: 0x0012_019F,
475        }];
476
477        let result = walk_file_objects(&reader, &handles).unwrap();
478        assert_eq!(result.len(), 1, "should find one file object");
479        assert_eq!(
480            result[0].device_name, "\\Driver\\disk",
481            "device_name should be resolved from driver"
482        );
483    }
