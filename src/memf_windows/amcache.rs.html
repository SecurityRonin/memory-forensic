1//! Amcache evidence-of-execution walker.
2//!
3//! The Windows Amcache.hve registry hive stores program execution evidence
4//! including file paths, SHA1 hashes, timestamps, publisher info, and
5//! product names. In memory forensics, the Amcache data lives inside the
6//! registry CM structures. This walker reads from the in-memory registry
7//! hive structures pointed to by the `_CMHIVE` at the given address.
8//!
9//! The Amcache hive's `Root\InventoryApplicationFile` key contains child
10//! keys, one per tracked executable. Each child key has value cells for:
11//! - `LowerCaseLongPath` — full file path
12//! - `FileId` — SHA1 hash (prefixed with `0000`)
13//! - `Size` — file size in bytes
14//! - `LinkDate` — link/compile timestamp
15//! - `Publisher` — code-signing publisher
16//! - `ProductName` — application product name
17//!
18//! The `classify_amcache_entry` heuristic flags entries with no publisher,
19//! temp/download paths, or other suspicious indicators that may warrant
20//! further investigation.
21
22use memf_core::object_reader::ObjectReader;
23use memf_format::PhysicalMemoryProvider;

25/// Maximum number of Amcache entries to enumerate (safety limit).
26const MAX_AMCACHE_ENTRIES: usize = 8192;

28/// Maximum depth when navigating to `Root\InventoryApplicationFile`.
29#[allow(dead_code)]
30const MAX_NAV_DEPTH: usize = 8;

32/// A single Amcache program execution evidence entry.
33#[derive(Debug, Clone, serde::Serialize)]
34pub struct AmcacheEntry {
35    /// Full file path of the executable.
36    pub file_path: String,
37    /// SHA1 hash of the file (from the `FileId` value, stripped of `0000` prefix).
38    pub sha1_hash: String,
39    /// File size in bytes.
40    pub file_size: u64,
41    /// Link/compile timestamp as a Windows FILETIME (100-ns intervals since 1601-01-01).
42    pub link_timestamp: u64,
43    /// Code-signing publisher name.
44    pub publisher: String,
45    /// Application product name.
46    pub product_name: String,
47    /// Whether this entry looks suspicious based on heuristics.
48    pub is_suspicious: bool,
49}

51/// Classify an Amcache entry as suspicious based on path and publisher heuristics.
52///
53/// Returns `true` if any of the following conditions are met:
54/// - The publisher is empty (unsigned/unknown binary)
55/// - The file path contains temp directories (`\Temp\`, `\AppData\`, `\Downloads\`)
56/// - The file path is in a user-writable location with no known publisher
57///
58/// Well-known publishers (e.g., "Microsoft") in standard system paths are
59/// considered benign.
60pub fn classify_amcache_entry(path: &str, publisher: &str) -> bool {
61    // An empty publisher is always suspicious — unsigned or unknown binary.
62    if publisher.is_empty() {
63        return true;
64    }
65
66    // Case-insensitive path check for suspicious locations.
67    let path_lower = path.to_ascii_lowercase();
68
69    // Known suspicious directories where untrusted executables often land.
70    let suspicious_dirs = [
71        r"\temp\",
72        r"\appdata\",
73        r"\downloads\",
74        r"\users\public\",
75        r"\programdata\",
76        r"\recycle",
77    ];
78
79    // Even with a publisher, binaries in temp/download paths are worth flagging
80    // if the publisher is not a well-known trusted name.
81    let well_known_publishers = [
82        "microsoft",
83        "mozilla",
84        "google",
85        "apple",
86        "adobe",
87        "oracle",
88        "vmware",
89        "citrix",
90        "intel",
91    ];
92
93    let publisher_lower = publisher.to_ascii_lowercase();
94    let is_trusted_publisher = well_known_publishers
95        .iter()
96        .any(|known| publisher_lower.contains(known));
97
98    // If the path is in a suspicious directory AND the publisher is not
99    // a well-known trusted name, flag it.
100    if !is_trusted_publisher && suspicious_dirs.iter().any(|dir| path_lower.contains(dir)) {
101        return true;
102    }
103
104    false
105}

107/// Walk the Amcache registry hive from kernel memory.
108///
109/// Takes the virtual address of the Amcache hive's `_CMHIVE` structure.
110/// Reads the `_HHIVE.BaseBlock` to locate the `_HBASE_BLOCK`, then
111/// navigates to `Root\InventoryApplicationFile` and reads each child
112/// key's value cells.
113///
114/// Returns an empty `Vec` if the required symbols are not present
115/// (graceful degradation).
116///
117/// # Errors
118///
119/// Returns an error if memory reads fail after the hive has been
120/// located and validated.
121// ── Registry cell layout constants (same as registry_keys.rs) ────────
122
123/// Offset of `RootCell` (u32) within `_HBASE_BLOCK`.
124const HBASE_BLOCK_ROOT_CELL_OFFSET: u64 = 0x24;
125
126/// Offset from `_HBASE_BLOCK` to the first HBIN (cell storage start).
127const HBIN_START_OFFSET: u64 = 0x1000;
128
129/// `_CM_KEY_NODE` signature: "nk" (0x6B6E).
130const NK_SIGNATURE: u16 = 0x6B6E;
131
132/// `_CM_KEY_VALUE` signature: "vk" (0x6B76).
133const VK_SIGNATURE: u16 = 0x6B76;
134
135// CM_KEY_NODE offsets (relative to cell data start, after 4-byte cell size)
136#[allow(dead_code)]
137const NK_LAST_WRITE_TIME_OFFSET: usize = 4;
138const NK_STABLE_SUBKEY_COUNT_OFFSET: usize = 0x14;
139const NK_STABLE_SUBKEYS_LIST_OFFSET: usize = 0x1C;
140const NK_VALUE_COUNT_OFFSET: usize = 0x24;
141const NK_VALUES_LIST_OFFSET: usize = 0x28;
142const NK_NAME_LENGTH_OFFSET: usize = 0x48;
143const NK_NAME_OFFSET: usize = 0x4C;
144
145// CM_KEY_VALUE offsets (relative to cell data start)
146const VK_NAME_LENGTH_OFFSET: usize = 0x02;
147const VK_DATA_LENGTH_OFFSET: usize = 0x04;
148const VK_DATA_OFFSET: usize = 0x08;
149const VK_TYPE_OFFSET: usize = 0x0C;
150const VK_NAME_OFFSET: usize = 0x14;
151
152/// REG_SZ type for string values.
153#[allow(dead_code)]
154const REG_SZ: u32 = 1;
155/// REG_QWORD type for 64-bit integer values.
156const REG_QWORD: u32 = 11;
157/// REG_DWORD type for 32-bit integer values.
158const REG_DWORD: u32 = 4;
159
160/// Read cell data at a given cell index within the hive.
161///
162/// The cell index is relative to the start of cell storage (HBIN area).
163/// Each cell starts with an i32 size (negative = allocated). We skip the
164/// 4-byte size prefix and return the cell data.
165fn read_cell_data<P: PhysicalMemoryProvider>(
166    reader: &ObjectReader<P>,
167    hive_base: u64,
168    cell_index: u32,
169    max_len: usize,
170) -> Option<Vec<u8>> {
171    if cell_index == 0 || cell_index == 0xFFFF_FFFF {
172        return None;
173    }
174    let cell_vaddr = hive_base
175        .wrapping_add(HBIN_START_OFFSET)
176        .wrapping_add(cell_index as u64);
177
178    // Read the 4-byte size field first.
179    let size_bytes = reader.read_bytes(cell_vaddr, 4).ok()?;
180    let raw_size = i32::from_le_bytes(size_bytes[..4].try_into().ok()?);
181    // Negative = allocated cell; absolute value is total cell size including header.
182    let data_size = (raw_size.unsigned_abs() as usize).saturating_sub(4);
183    let read_len = data_size.min(max_len);
184    if read_len == 0 {
185        return None;
186    }
187
188    reader.read_bytes(cell_vaddr.wrapping_add(4), read_len).ok()
189}

191/// Read an ASCII key/value name from cell data at the given offset.
192fn read_ascii_name(data: &[u8], name_length_offset: usize, name_offset: usize) -> String {
193    if data.len() < name_length_offset + 2 {
194        return String::new();
195    }
196    let name_len =
197        u16::from_le_bytes([data[name_length_offset], data[name_length_offset + 1]]) as usize;
198    if name_offset + name_len > data.len() || name_len == 0 {
199        return String::new();
200    }
201    String::from_utf8_lossy(&data[name_offset..name_offset + name_len]).to_string()
202}

204/// Navigate from a key node to a named subkey.
205///
206/// Reads the stable subkeys list and searches for a child whose name
207/// matches `target_name` (case-insensitive). Returns the cell index
208/// of the matching child key node, or `None`.
209fn find_subkey<P: PhysicalMemoryProvider>(
210    reader: &ObjectReader<P>,
211    hive_base: u64,
212    parent_data: &[u8],
213    target_name: &str,
214) -> Option<u32> {
215    if parent_data.len() < NK_STABLE_SUBKEYS_LIST_OFFSET + 4 {
216        return None;
217    }
218    let subkey_count = u32::from_le_bytes(
219        parent_data[NK_STABLE_SUBKEY_COUNT_OFFSET..NK_STABLE_SUBKEY_COUNT_OFFSET + 4]
220            .try_into()
221            .ok()?,
222    );
223    if subkey_count == 0 {
224        return None;
225    }
226    let subkeys_list_index = u32::from_le_bytes(
227        parent_data[NK_STABLE_SUBKEYS_LIST_OFFSET..NK_STABLE_SUBKEYS_LIST_OFFSET + 4]
228            .try_into()
229            .ok()?,
230    );
231
232    // The subkeys list cell contains a signature (lf/lh/ri/li, 2 bytes),
233    // a count (u16), then pairs of (cell_index: u32, hash: u32).
234    let list_data = read_cell_data(reader, hive_base, subkeys_list_index, 0x2000)?;
235    if list_data.len() < 4 {
236        return None;
237    }
238    let list_count = u16::from_le_bytes([list_data[2], list_data[3]]) as usize;
239    let target_lower = target_name.to_ascii_lowercase();
240
241    // Dispatch on list signature: li (0x696C) uses 4-byte entries (raw cell offset
242    // only); lf (0x666C) and lh (0x686C) use 8-byte entries (cell offset + hash).
243    let list_sig = u16::from_le_bytes([list_data[0], list_data[1]]);
244    let stride = if list_sig == 0x696C { 4 } else { 8 };
245
246    for i in 0..list_count {
247        let offset = 4 + i * stride;
248        if offset + 4 > list_data.len() {
249            break;
250        }
251        let child_index = u32::from_le_bytes(list_data[offset..offset + 4].try_into().ok()?);
252
253        // Read the child key node to check its name.
254        let child_data = read_cell_data(reader, hive_base, child_index, 0x200)?;
255        if child_data.len() < 4 {
256            continue;
257        }
258        let sig = u16::from_le_bytes([child_data[0], child_data[1]]);
259        if sig != NK_SIGNATURE {
260            continue;
261        }
262        let name = read_ascii_name(&child_data, NK_NAME_LENGTH_OFFSET, NK_NAME_OFFSET);
263        if name.to_ascii_lowercase() == target_lower {
264            return Some(child_index);
265        }
266    }
267
268    None
269}

271/// Read a REG_SZ string value from a value cell's data.
272fn read_value_string<P: PhysicalMemoryProvider>(
273    reader: &ObjectReader<P>,
274    hive_base: u64,
275    vk_data: &[u8],
276) -> String {
277    if vk_data.len() < VK_NAME_OFFSET {
278        return String::new();
279    }
280    let data_length = u32::from_le_bytes(
281        vk_data[VK_DATA_LENGTH_OFFSET..VK_DATA_LENGTH_OFFSET + 4]
282            .try_into()
283            .unwrap_or([0; 4]),
284    );
285    // Bit 31 set means data is stored inline in the DataOffset field itself.
286    let is_inline = data_length & 0x8000_0000 != 0;
287    let real_len = (data_length & 0x7FFF_FFFF) as usize;
288
289    if real_len == 0 {
290        return String::new();
291    }
292
293    let raw = if is_inline {
294        // Inline data: stored in the 4 bytes of the DataOffset field.
295        vk_data[VK_DATA_OFFSET..VK_DATA_OFFSET + real_len.min(4)].to_vec()
296    } else {
297        let data_cell_index = u32::from_le_bytes(
298            vk_data[VK_DATA_OFFSET..VK_DATA_OFFSET + 4]
299                .try_into()
300                .unwrap_or([0; 4]),
301        );
302        match read_cell_data(reader, hive_base, data_cell_index, real_len) {
303            Some(d) => d,
304            None => return String::new(),
305        }
306    };
307
308    // Interpret as UTF-16LE.
309    let u16_vec: Vec<u16> = raw
310        .chunks_exact(2)
311        .map(|pair| u16::from_le_bytes([pair[0], pair[1]]))
312        .collect();
313    String::from_utf16_lossy(&u16_vec)
314        .trim_end_matches('\0')
315        .to_string()
316}

318/// Read a QWORD (u64) or DWORD (u32, widened) value from a value cell.
319fn read_value_u64<P: PhysicalMemoryProvider>(
320    reader: &ObjectReader<P>,
321    hive_base: u64,
322    vk_data: &[u8],
323) -> u64 {
324    if vk_data.len() < VK_NAME_OFFSET {
325        return 0;
326    }
327    let value_type = u32::from_le_bytes(
328        vk_data[VK_TYPE_OFFSET..VK_TYPE_OFFSET + 4]
329            .try_into()
330            .unwrap_or([0; 4]),
331    );
332    let data_length = u32::from_le_bytes(
333        vk_data[VK_DATA_LENGTH_OFFSET..VK_DATA_LENGTH_OFFSET + 4]
334            .try_into()
335            .unwrap_or([0; 4]),
336    );
337    let is_inline = data_length & 0x8000_0000 != 0;
338    let real_len = (data_length & 0x7FFF_FFFF) as usize;
339
340    let raw = if is_inline {
341        vk_data[VK_DATA_OFFSET..VK_DATA_OFFSET + real_len.min(4)].to_vec()
342    } else {
343        let data_cell_index = u32::from_le_bytes(
344            vk_data[VK_DATA_OFFSET..VK_DATA_OFFSET + 4]
345                .try_into()
346                .unwrap_or([0; 4]),
347        );
348        match read_cell_data(reader, hive_base, data_cell_index, real_len) {
349            Some(d) => d,
350            None => return 0,
351        }
352    };
353
354    match value_type {
355        REG_QWORD if raw.len() >= 8 => u64::from_le_bytes(raw[..8].try_into().unwrap_or([0; 8])),
356        REG_DWORD if raw.len() >= 4 => {
357            u32::from_le_bytes(raw[..4].try_into().unwrap_or([0; 4])) as u64
358        }
359        _ => 0,
360    }
361}

363/// Look up a named value from a key node's value list.
364///
365/// Returns the raw vk cell data for the matching value, or `None`.
366fn find_value<P: PhysicalMemoryProvider>(
367    reader: &ObjectReader<P>,
368    hive_base: u64,
369    key_data: &[u8],
370    target_name: &str,
371) -> Option<Vec<u8>> {
372    if key_data.len() < NK_VALUES_LIST_OFFSET + 4 {
373        return None;
374    }
375    let value_count = u32::from_le_bytes(
376        key_data[NK_VALUE_COUNT_OFFSET..NK_VALUE_COUNT_OFFSET + 4]
377            .try_into()
378            .ok()?,
379    ) as usize;
380    if value_count == 0 {
381        return None;
382    }
383    let values_list_index = u32::from_le_bytes(
384        key_data[NK_VALUES_LIST_OFFSET..NK_VALUES_LIST_OFFSET + 4]
385            .try_into()
386            .ok()?,
387    );
388
389    // The values list cell is an array of u32 cell indices.
390    let list_data = read_cell_data(reader, hive_base, values_list_index, value_count * 4)?;
391    let target_lower = target_name.to_ascii_lowercase();
392
393    for i in 0..value_count {
394        let offset = i * 4;
395        if offset + 4 > list_data.len() {
396            break;
397        }
398        let vk_index = u32::from_le_bytes(list_data[offset..offset + 4].try_into().ok()?);
399        let vk_data = read_cell_data(reader, hive_base, vk_index, 0x200)?;
400        if vk_data.len() < 4 {
401            continue;
402        }
403        let sig = u16::from_le_bytes([vk_data[0], vk_data[1]]);
404        if sig != VK_SIGNATURE {
405            continue;
406        }
407        let name = read_ascii_name(&vk_data, VK_NAME_LENGTH_OFFSET, VK_NAME_OFFSET);
408        if name.to_ascii_lowercase() == target_lower {
409            return Some(vk_data);
410        }
411    }
412
413    None
414}

416/// Walk the Amcache registry hive from kernel memory.
417///
418/// Takes the virtual address of the Amcache hive's `_CMHIVE` structure.
419/// Reads the `_HHIVE.BaseBlock` to locate the `_HBASE_BLOCK`, then
420/// navigates to `Root\InventoryApplicationFile` and reads each child
421/// key's value cells.
422///
423/// Returns an empty `Vec` if the required symbols are not present
424/// (graceful degradation).
425///
426/// # Errors
427///
428/// Returns an error if memory reads fail after the hive has been
429/// located and validated.
430pub fn walk_amcache<P: PhysicalMemoryProvider>(
431    reader: &ObjectReader<P>,
432    amcache_hive_addr: u64,
433) -> crate::Result<Vec<AmcacheEntry>> {
434    // Zero address means no hive — graceful degradation.
435    if amcache_hive_addr == 0 {
436        return Ok(Vec::new());
437    }
438
439    // Read the BaseBlock pointer from _CMHIVE._HHIVE.BaseBlock.
440    // _CMHIVE starts with _HHIVE, and _HHIVE.BaseBlock is at offset 0x10
441    // (after Signature u32 @ 0x00, pad, etc.).
442    // Try to read it via ISF first; fall back to raw offset.
443    let base_block_addr: u64 = match reader.read_field(amcache_hive_addr, "_HHIVE", "BaseBlock") {
444        Ok(addr) => addr,
445        Err(_) => {
446            // Fallback: read pointer at offset 0x10 from hive addr.
447            match reader.read_bytes(amcache_hive_addr.wrapping_add(0x10), 8) {
448                Ok(bytes) if bytes.len() == 8 => u64::from_le_bytes(bytes[..8].try_into().unwrap()),
449                _ => return Ok(Vec::new()),
450            }
451        }
452    };
453
454    if base_block_addr == 0 {
455        return Ok(Vec::new());
456    }
457
458    // Read root cell index from _HBASE_BLOCK.
459    let root_cell_bytes = match reader.read_bytes(
460        base_block_addr.wrapping_add(HBASE_BLOCK_ROOT_CELL_OFFSET),
461        4,
462    ) {
463        Ok(b) if b.len() == 4 => b,
464        _ => return Ok(Vec::new()),
465    };
466    let root_cell = u32::from_le_bytes(root_cell_bytes[..4].try_into().unwrap());
467    if root_cell == 0 {
468        return Ok(Vec::new());
469    }
470
471    let hive_base = base_block_addr;
472
473    // Read root key node.
474    let root_data = match read_cell_data(reader, hive_base, root_cell, 0x200) {
475        Some(d) if d.len() >= 4 && u16::from_le_bytes([d[0], d[1]]) == NK_SIGNATURE => d,
476        _ => return Ok(Vec::new()),
477    };
478
479    // Navigate: Root → InventoryApplicationFile
480    // First find "Root" if the root key is the hive root, then
481    // find "InventoryApplicationFile" under it. In Amcache.hve the
482    // root key is typically named "Root" and InventoryApplicationFile
483    // is a direct child.
484    let iaf_cell =
485        if let Some(iaf) = find_subkey(reader, hive_base, &root_data, "InventoryApplicationFile") {
486            iaf
487        } else if let Some(root_child) = find_subkey(reader, hive_base, &root_data, "Root") {
488            // Try one level deeper: Root → InventoryApplicationFile
489            let root_child_data = match read_cell_data(reader, hive_base, root_child, 0x200) {
490                Some(d) => d,
491                None => return Ok(Vec::new()),
492            };
493            match find_subkey(
494                reader,
495                hive_base,
496                &root_child_data,
497                "InventoryApplicationFile",
498            ) {
499                Some(idx) => idx,
500                None => return Ok(Vec::new()),
501            }
502        } else {
503            return Ok(Vec::new());
504        };
505
506    // Read the InventoryApplicationFile key node.
507    let iaf_data = match read_cell_data(reader, hive_base, iaf_cell, 0x200) {
508        Some(d) if d.len() >= NK_STABLE_SUBKEYS_LIST_OFFSET + 4 => d,
509        _ => return Ok(Vec::new()),
510    };
511
512    // Enumerate child keys (each represents one tracked executable).
513    let subkey_count = u32::from_le_bytes(
514        iaf_data[NK_STABLE_SUBKEY_COUNT_OFFSET..NK_STABLE_SUBKEY_COUNT_OFFSET + 4]
515            .try_into()
516            .unwrap(),
517    ) as usize;
518    if subkey_count == 0 {
519        return Ok(Vec::new());
520    }
521
522    let subkeys_list_index = u32::from_le_bytes(
523        iaf_data[NK_STABLE_SUBKEYS_LIST_OFFSET..NK_STABLE_SUBKEYS_LIST_OFFSET + 4]
524            .try_into()
525            .unwrap(),
526    );
527    let list_data = match read_cell_data(reader, hive_base, subkeys_list_index, 0x4000) {
528        Some(d) if d.len() >= 4 => d,
529        _ => return Ok(Vec::new()),
530    };
531
532    let list_count =
533        (u16::from_le_bytes([list_data[2], list_data[3]]) as usize).min(MAX_AMCACHE_ENTRIES);
534
535    // Dispatch on list signature: li (0x696C) → 4-byte entries; lf/lh → 8-byte.
536    let list_sig = u16::from_le_bytes([list_data[0], list_data[1]]);
537    let stride = if list_sig == 0x696C { 4 } else { 8 };
538
539    let mut entries = Vec::with_capacity(list_count);
540
541    for i in 0..list_count {
542        let offset = 4 + i * stride;
543        if offset + 4 > list_data.len() {
544            break;
545        }
546        let child_index = match list_data[offset..offset + 4].try_into() {
547            Ok(b) => u32::from_le_bytes(b),
548            Err(_) => continue,
549        };
550
551        let child_data = match read_cell_data(reader, hive_base, child_index, 0x400) {
552            Some(d) if d.len() >= NK_NAME_OFFSET => d,
553            _ => continue,
554        };
555
556        let sig = u16::from_le_bytes([child_data[0], child_data[1]]);
557        if sig != NK_SIGNATURE {
558            continue;
559        }
560
561        // Extract values from this child key.
562        let file_path = find_value(reader, hive_base, &child_data, "LowerCaseLongPath")
563            .map(|vk| read_value_string(reader, hive_base, &vk))
564            .unwrap_or_default();
565
566        let sha1_raw = find_value(reader, hive_base, &child_data, "FileId")
567            .map(|vk| read_value_string(reader, hive_base, &vk))
568            .unwrap_or_default();
569        // Strip the "0000" prefix that Amcache prepends to SHA1 hashes.
570        let sha1_hash = sha1_raw
571            .strip_prefix("0000")
572            .unwrap_or(&sha1_raw)
573            .to_string();
574
575        let file_size = find_value(reader, hive_base, &child_data, "Size")
576            .map(|vk| read_value_u64(reader, hive_base, &vk))
577            .unwrap_or(0);
578
579        let link_timestamp = find_value(reader, hive_base, &child_data, "LinkDate")
580            .map(|vk| read_value_u64(reader, hive_base, &vk))
581            .unwrap_or(0);
582
583        let publisher = find_value(reader, hive_base, &child_data, "Publisher")
584            .map(|vk| read_value_string(reader, hive_base, &vk))
585            .unwrap_or_default();
586
587        let product_name = find_value(reader, hive_base, &child_data, "ProductName")
588            .map(|vk| read_value_string(reader, hive_base, &vk))
589            .unwrap_or_default();
590
591        let is_suspicious = classify_amcache_entry(&file_path, &publisher);
592
593        entries.push(AmcacheEntry {
594            file_path,

1022
1023        let list_cell_index: u32 = 0x100;
1024        let child_cell_index: u32 = 0x200;
1025
1026        let hbin_page_vaddr = hive_base + HBIN_START_OFFSET; // 0x00B1_0000
1027        let hbin_page_paddr: u64 = 0x00B1_0000;
1028
1029        // parent nk_data: subkey_count=1, subkeys_list_index=list_cell_index
1030        let mut parent_data = vec![0u8; 0x60];
1031        parent_data[NK_STABLE_SUBKEY_COUNT_OFFSET..NK_STABLE_SUBKEY_COUNT_OFFSET + 4]
1032            .copy_from_slice(&1u32.to_le_bytes());
1033        parent_data[NK_STABLE_SUBKEYS_LIST_OFFSET..NK_STABLE_SUBKEYS_LIST_OFFSET + 4]
1034            .copy_from_slice(&list_cell_index.to_le_bytes());
1035
1036        // Build the hbin page.
1037        // li list cell at offset 0x100:
1038        //   [i32 size=-12][sig=li 0x696C][count=1][child_cell_index u32]
1039        //   (no hash word — li entries are 4 bytes each)
1040        // child NK cell at offset 0x200:
1041        //   [i32 size=-0x80][nk_data with NK_SIGNATURE, name="InventoryApplicationFile"]
1042        let mut hbin_page = vec![0u8; 0x1000];
1043
1044        let list_off = list_cell_index as usize;
1045        hbin_page[list_off..list_off + 4].copy_from_slice(&(-12i32).to_le_bytes());
1046        // li sig = 0x696C → LE bytes [0x6C, 0x69]
1047        hbin_page[list_off + 4] = 0x6C;
1048        hbin_page[list_off + 5] = 0x69;
1049        // count = 1
1050        hbin_page[list_off + 6..list_off + 8].copy_from_slice(&1u16.to_le_bytes());

1731    // This exercises lines 532-549 (the enumeration loop with sig check).
1732
1733    fn build_iaf_hive(flat_page: &mut Vec<u8>, child_nk_sig: u16) -> (u64, u32) {
1734        // Layout within flat_page:
1735        //   root NK: cell_off = 0x020 → addr = 0x024
1736        //   root list: cell_off = 0x060 → addr = 0x064
1737        //   IAF NK: cell_off = 0x0A0 → addr = 0x0A4
1738        //   IAF list: cell_off = 0x0E0 → addr = 0x0E4
1739        //   child NK: cell_off = 0x120 → addr = 0x124
1740        let root_cell_off: u32 = 0x020;
1741        let root_list_off: u32 = 0x060;
1742        let iaf_cell_off: u32 = 0x0A0;
1743        let iaf_list_off: u32 = 0x0E0;
1744        let child_cell_off: u32 = 0x120;
1745
1746        let a = |off: u32| (off + 4) as usize; // addr offset in page
1747
1748        // Root NK: subkey_count=1, list=root_list_off, name len=0
1749        flat_page[a(root_cell_off) + NK_STABLE_SUBKEY_COUNT_OFFSET
1750            ..a(root_cell_off) + NK_STABLE_SUBKEY_COUNT_OFFSET + 4]
1751            .copy_from_slice(&1u32.to_le_bytes());
1752        flat_page[a(root_cell_off) + NK_STABLE_SUBKEYS_LIST_OFFSET
1753            ..a(root_cell_off) + NK_STABLE_SUBKEYS_LIST_OFFSET + 4]
1754            .copy_from_slice(&root_list_off.to_le_bytes());
1755        // NK_SIGNATURE at root
1756        flat_page[a(root_cell_off)..a(root_cell_off) + 2]
1757            .copy_from_slice(&NK_SIGNATURE.to_le_bytes());
1758
1759        // Root list: lf, count=1, entry[0]=iaf_cell_off
1760        flat_page[a(root_list_off)] = 0x6C; // 'l'
1761        flat_page[a(root_list_off) + 1] = 0x66; // 'f'
1762        flat_page[a(root_list_off) + 2..a(root_list_off) + 4].copy_from_slice(&1u16.to_le_bytes());
1763        flat_page[a(root_list_off) + 4..a(root_list_off) + 8]
1764            .copy_from_slice(&iaf_cell_off.to_le_bytes());
1765
1766        // IAF NK: NK_SIGNATURE, name="InventoryApplicationFile", subkey_count=1, list=iaf_list_off
1767        flat_page[a(iaf_cell_off)..a(iaf_cell_off) + 2]
1768            .copy_from_slice(&NK_SIGNATURE.to_le_bytes());
1769        flat_page[a(iaf_cell_off) + NK_STABLE_SUBKEY_COUNT_OFFSET
1770            ..a(iaf_cell_off) + NK_STABLE_SUBKEY_COUNT_OFFSET + 4]
1771            .copy_from_slice(&1u32.to_le_bytes());
1772        flat_page[a(iaf_cell_off) + NK_STABLE_SUBKEYS_LIST_OFFSET
1773            ..a(iaf_cell_off) + NK_STABLE_SUBKEYS_LIST_OFFSET + 4]
1774            .copy_from_slice(&iaf_list_off.to_le_bytes());
1775        let iaf_name = b"InventoryApplicationFile";
1776        flat_page
1777            [a(iaf_cell_off) + NK_NAME_LENGTH_OFFSET..a(iaf_cell_off) + NK_NAME_LENGTH_OFFSET + 2]
1778            .copy_from_slice(&(iaf_name.len() as u16).to_le_bytes());
1779        flat_page
1780            [a(iaf_cell_off) + NK_NAME_OFFSET..a(iaf_cell_off) + NK_NAME_OFFSET + iaf_name.len()]
1781            .copy_from_slice(iaf_name);
1782
1783        // IAF list: lf, count=1, entry[0]=child_cell_off
1784        flat_page[a(iaf_list_off)] = 0x6C;
1785        flat_page[a(iaf_list_off) + 1] = 0x66;
1786        flat_page[a(iaf_list_off) + 2..a(iaf_list_off) + 4].copy_from_slice(&1u16.to_le_bytes());
1787        flat_page[a(iaf_list_off) + 4..a(iaf_list_off) + 8]
1788            .copy_from_slice(&child_cell_off.to_le_bytes());
1789
1790        // Child NK: given signature (to control NK_SIGNATURE check)
1791        flat_page[a(child_cell_off)..a(child_cell_off) + 2]
1792            .copy_from_slice(&child_nk_sig.to_le_bytes());
1793        // Make cell big enough: NK_NAME_OFFSET + 4 bytes of name
1794        // child NK has no values (value_count=0 at NK_VALUE_COUNT_OFFSET)
1795
1796        (root_cell_off as u64, root_cell_off)
1797    }
