1//! Cloud provider credential forensics walker.
2//!
3//! Scans process heap memory for cloud service credentials that applications
4//! cache in memory during operation. Covers AWS IAM access keys, GCP API keys,
5//! Azure storage and SAS tokens, Stripe secret keys, Twilio SIDs, and generic
6//! high-entropy API key patterns.
7//!
8//! This capability is absent from all major memory forensics frameworks
9//! (Volatility 3, pypykatz, MemProcFS) as of 2024.
10//!
11//! # Forensic guarantee
12//!
13//! Read-only. No live process access, no Win32 API calls, no state modification.
14
15use std::collections::HashSet;
16
17use memf_core::object_reader::ObjectReader;
18use memf_format::PhysicalMemoryProvider;
19
20use crate::{types::CloudCredentialInfo, Result};

22/// Walk committed, writable heap regions of ALL processes in the dump and
23/// extract any cloud provider credentials found as strings.
24///
25/// # Arguments
26///
27/// * `reader` — kernel-space `ObjectReader` (uses kernel CR3 / symbol table).
28/// * `ps_head_vaddr` — virtual address of `PsActiveProcessHead`.
29pub fn walk_cloud_credentials<P: PhysicalMemoryProvider + Clone>(
30    reader: &ObjectReader<P>,
31    ps_head_vaddr: u64,
32) -> Result<Vec<CloudCredentialInfo>> {
33    let wr = crate::heap_walker::for_each_heap_region(
34        reader,
35        ps_head_vaddr,
36        |_proc| true,
37        |bytes, proc| scan_cloud_region(bytes, proc.pid, &proc.image_name),
38        |info: &CloudCredentialInfo| (info.pid, info.value.clone()),
39    )?;
40    Ok(wr.items)
41}
42
43// AWS IAM Access Key ID (permanent: AKIA, temporary STS: ASIA)
44const AWS_KEY_ID: &str = r"(?:AKIA|ASIA)[A-Z0-9]{16}";
45
46// GCP API key
47const GCP_API_KEY: &str = r"AIza[0-9A-Za-z\-_]{35}";
48
49// Azure Storage Account key (base64, 88 chars ending in ==)
50const AZURE_STORAGE_KEY: &str = r"AccountKey=([A-Za-z0-9+/]{86}==)";
51
52// Azure SAS token
53const AZURE_SAS: &str = r"SharedAccessSignature=([A-Za-z0-9%&=+]{30,})";
54
55// Stripe secret key (live and test)
56const STRIPE_SECRET: &str = r"sk_(?:live|test)_[A-Za-z0-9]{24,}";
57
58// Stripe publishable key
59const STRIPE_PK: &str = r"pk_(?:live|test)_[A-Za-z0-9]{24,}";
60
61// Twilio Account SID
62const TWILIO_SID: &str = r"AC[a-f0-9]{32}";
63
64// Generic high-entropy API key pattern near "api_key", "apikey", "api-key" labels
65const GENERIC_API_KEY: &str = r#"(?i)api[_\-]?key["\s:=]+([A-Za-z0-9\-_]{32,64})"#;
66
67/// Scan raw bytes from a memory region for cloud provider credentials.
68///
69/// Run all patterns, label each match with provider + credential_type.
70/// For patterns with capture groups (Azure, generic API key), use capture
71/// group 1. For others, use full match. Deduplicates by value within the
72/// returned slice.
73pub(crate) fn scan_cloud_region(
74    data: &[u8],
75    pid: u64,
76    process_name: &str,
77) -> Vec<CloudCredentialInfo> {
78    let text = String::from_utf8_lossy(data);
79    let mut out: Vec<CloudCredentialInfo> = Vec::new();
80    let mut seen_values: HashSet<String> = HashSet::new();
81
82    macro_rules! scan_pattern {
83        ($pat:expr, $provider:expr, $cred_type:expr, capture: false) => {{
84            if let Ok(re) = regex::Regex::new($pat) {
85                for m in re.find_iter(&text) {
86                    let value = m.as_str().to_owned();
87                    if seen_values.insert(value.clone()) {
88                        out.push(CloudCredentialInfo {
89                            pid,
90                            image_name: process_name.to_owned(),
91                            provider: $provider.to_owned(),
92                            credential_type: $cred_type.to_owned(),
93                            value,
94                        });
95                    }
96                }
97            }
98        }};
99        ($pat:expr, $provider:expr, $cred_type:expr, capture: true) => {{
100            if let Ok(re) = regex::Regex::new($pat) {
101                for caps in re.captures_iter(&text) {
102                    if let Some(m) = caps.get(1) {
103                        let value = m.as_str().to_owned();
104                        if seen_values.insert(value.clone()) {
105                            out.push(CloudCredentialInfo {
106                                pid,
107                                image_name: process_name.to_owned(),
108                                provider: $provider.to_owned(),
109                                credential_type: $cred_type.to_owned(),
110                                value,
111                            });
112                        }
113                    }
114                }
115            }
116        }};
117    }
118
119    // AWS — both AKIA (permanent) and ASIA (temporary STS) are AccessKeyId credentials.
120    if let Ok(re) = regex::Regex::new(AWS_KEY_ID) {
121        for m in re.find_iter(&text) {
122            let value = m.as_str().to_owned();
123            if seen_values.insert(value.clone()) {
124                out.push(CloudCredentialInfo {
125                    pid,
126                    image_name: process_name.to_owned(),
127                    provider: "AWS".to_owned(),
128                    credential_type: "AccessKeyId".to_owned(),
129                    value,
130                });
131            }
132        }
133    }
134
135    scan_pattern!(GCP_API_KEY, "GCP", "ApiKey", capture: false);
136    scan_pattern!(AZURE_STORAGE_KEY, "Azure", "StorageKey", capture: true);
137    scan_pattern!(AZURE_SAS, "Azure", "SasToken", capture: true);
138    scan_pattern!(STRIPE_SECRET, "Stripe", "SecretKey", capture: false);
139    scan_pattern!(STRIPE_PK, "Stripe", "PublishableKey", capture: false);
140    scan_pattern!(TWILIO_SID, "Twilio", "AccountSid", capture: false);
141    scan_pattern!(GENERIC_API_KEY, "Generic", "ApiKey", capture: true);
142
143    out
144}

146#[cfg(test)]
147mod tests {
148    use super::*;
149
150    #[test]
151    fn scan_empty_returns_nothing() {
152        assert!(scan_cloud_region(b"", 1, "x").is_empty());
153    }
154
155    #[test]
156    fn scan_detects_aws_access_key_id() {
157        let data = b"AWS_ACCESS_KEY_ID=AKIAIOSFODNN7EXAMPLE123";
158        let results = scan_cloud_region(data, 1, "aws.exe");
159        assert!(!results.is_empty());
160        assert!(results
161            .iter()
162            .any(|c| c.provider == "AWS" && c.credential_type == "AccessKeyId"));
163        assert!(results.iter().any(|c| c.value.starts_with("AKIA")));
164    }
165
166    #[test]
167    fn scan_detects_aws_sts_temp_key() {
168        let data = b"AWS_ACCESS_KEY_ID=ASIAIOSFODNN7EXAMPLE123";
169        let results = scan_cloud_region(data, 1, "app.exe");
170        assert!(results
171            .iter()
172            .any(|c| c.provider == "AWS" && c.value.starts_with("ASIA")));
173    }
174
175    #[test]
176    fn scan_detects_gcp_api_key() {
177        let data = b"api_key = AIzaSyD-9tSrke72I6e672jd9dgkajhd8273hd91";
178        let results = scan_cloud_region(data, 1, "app.exe");
179        assert!(!results.is_empty());
180        assert!(results.iter().any(|c| c.provider == "GCP"));
181    }
182
183    #[test]
184    fn scan_detects_stripe_secret_key() {
185        let data = b"STRIPE_SECRET_KEY=sk_live_ABCDEFGHIJKLMNOPQRSTUVWXabcdefgh";
186        let results = scan_cloud_region(data, 1, "app.exe");
187        assert!(!results.is_empty());
188        assert!(results
189            .iter()
190            .any(|c| c.provider == "Stripe" && c.credential_type == "SecretKey"));
191    }
192
193    #[test]
194    fn scan_deduplicates_repeated_key() {
195        let key = b"AKIAIOSFODNN7EXAMPLE123";
196        let mut data = Vec::new();
197        data.extend_from_slice(b"key1=");
198        data.extend_from_slice(key);
199        data.extend_from_slice(b" key2=");
200        data.extend_from_slice(key);
201        let results = scan_cloud_region(&data, 1, "app.exe");
202        let aws: Vec<_> = results.iter().filter(|c| c.provider == "AWS").collect();
203        assert_eq!(aws.len(), 1, "same key should deduplicate");
204    }
205
206    #[test]
207    fn output_type_has_image_name_field() {
208        // Compile-time check: field must be named image_name, not process_name.
209        let info = CloudCredentialInfo {
210            pid: 1,
211            image_name: "aws-cli.exe".to_string(),
212            provider: "AWS".to_string(),
213            credential_type: "AccessKeyId".to_string(),
214            value: "AKIAIOSFODNN7EXAMPLE".to_string(),
215        };
216        assert_eq!(info.image_name, "aws-cli.exe");
217    }
