1//! Pool-tag process scanning (`psscan`) — finds `_EPROCESS` objects by their
2//! `Proc` pool tag in **physical** memory, independent of `PsActiveProcessHead`
3//! and the kernel image being page-resident.
4//!
5//! This is the robust fallback the reference tools (Volatility 2 `psscan`,
6//! Volatility 3 `PoolScanner`) use when the active-process linked list cannot be
7//! walked — e.g. a dump whose kernel data pages are paged out. Reimplemented
8//! clean-room: scan for the tag, then read `UniqueProcessId` / `ImageFileName`
9//! at their ISF-derived offsets from candidate object positions within the pool
10//! block, accepting only entries that pass strict structural validation.
11
12use std::collections::HashSet;
13
14use memf_core::object_reader::ObjectReader;
15use memf_format::PhysicalMemoryProvider;

17/// A process recovered by physical pool-tag scanning.
18#[derive(Debug, Clone, PartialEq, Eq)]
19pub struct ScannedProcess {
20    /// Physical address of the `_EPROCESS` object.
21    pub physical_addr: u64,
22    /// `UniqueProcessId`.
23    pub pid: u64,
24    /// `ImageFileName` (≤ 15 chars).
25    pub name: String,
26    /// `_KPROCESS.DirectoryTableBase` — the process page-table root (CR3).
27    ///
28    /// Only present when the caller supplies a `dtb_off`; `None` when the
29    /// offset is not available (e.g. the resolver has no `_KPROCESS` type).
30    pub dtb: Option<u64>,
31}
32
33/// `Proc` pool tag — standard non-paged pool tag for `_EPROCESS` objects.
34///
35/// Source: volatility3/framework/plugins/windows/poolscanner.py `builtin_constraints()`,
36/// which registers `b"Proc"` (standard) and `b"Pro\xe3"` (protected) as the two
37/// EPROCESS pool tag variants.
38const PROC_TAG: [u8; 4] = *b"Proc";
39/// `Pro\xe3` pool tag — protected non-paged pool allocation (Windows 10+).
40///
41/// The MSB of the 4-byte pool tag carries protection flags in the Windows 10
42/// pool manager; `\xe3` signals a non-paged, protected allocation. The first
43/// three bytes remain `Pro`, identifying this as an `_EPROCESS` pool block.
44///
45/// Source: volatility3/framework/plugins/windows/psscan.py:
46/// `constraints = poolscanner.PoolScanner.builtin_constraints(…, [b"Pro\xe3", b"Proc"])`
47const PROC_PROTECTED_TAG: [u8; 4] = [b'P', b'r', b'o', 0xe3];
48/// Pool tags to scan: the standard `Proc` and the Win10+ protected-pool variant `Pro\xe3`.
49const POOL_TAGS: &[[u8; 4]] = &[PROC_TAG, PROC_PROTECTED_TAG];
50/// Candidate `_EPROCESS` start offsets relative to the pool block base
51/// (`pool_header`), covering the usual `_POOL_HEADER` + optional/`_OBJECT_HEADER`
52/// span on x64 Windows. Validation rejects wrong guesses.
53const EPROCESS_DELTAS: &[u64] = &[
54    0x10, 0x20, 0x30, 0x40, 0x50, 0x60, 0x70, 0x80, 0x90, 0xA0, 0xB0, 0xC0,
55];
56
57/// Read `n` bytes at physical `pa`, returning `None` on a short read.
58fn read_phys_exact<P: PhysicalMemoryProvider + ?Sized>(prov: &P, pa: u64, n: usize) -> Option<Vec<u8>> {
59    let mut buf = vec![0u8; n];
60    if prov.read_phys(pa, &mut buf).unwrap_or(0) < n {
61        return None;
62    }
63    Some(buf)
64}

66/// True if `pid` looks like a real Windows process id: a small, non-zero
67/// multiple of four (the kernel allocates client ids in steps of four).
68fn plausible_pid(pid: u64) -> bool {
69    (4..=0x4_0000).contains(&pid) && pid % 4 == 0
70}

72/// Decode a 15-byte `ImageFileName` to a validated process name, or `None`.
73/// Requires printable ASCII, at least one letter, and NUL-or-end termination.
74fn decode_image_name(raw: &[u8]) -> Option<String> {
75    let end = raw.iter().position(|&b| b == 0).unwrap_or(raw.len());
76    // A real `ImageFileName` is an executable base name — never a single
77    // character; require at least two to reject lone-byte coincidences.
78    if !(2..=15).contains(&end) {
79        return None;
80    }
81    let name = &raw[..end];
82    if !name.iter().all(|&b| (0x20..=0x7E).contains(&b)) {
83        return None;
84    }
85    if !name.iter().any(|&b| b.is_ascii_alphabetic()) {
86        return None;
87    }
88    Some(String::from_utf8_lossy(name).into_owned())
89}

91/// Scan `prov`'s physical memory for `_EPROCESS` objects by their `Proc` pool
92/// tag, reading `UniqueProcessId` at `pid_off` and `ImageFileName` at `name_off`
93/// (both `_EPROCESS`-relative, from the ISF). Deduplicated by `(pid, name)`.
94///
95/// Panic-free and bounded; only structurally valid entries are returned.
96///
97/// This is a convenience wrapper over [`scan_processes_dtb`] with `dtb_off = None`
98/// (no DTB validation). Prefer [`scan_processes_dtb`] when the `_KPROCESS`
99/// `DirectoryTableBase` offset is available from the ISF.
100pub fn scan_processes<P: PhysicalMemoryProvider + ?Sized>(
101    prov: &P,
102    pid_off: u64,
103    name_off: u64,
104) -> Vec<ScannedProcess> {
105    scan_processes_dtb(prov, pid_off, name_off, None)
106}

108/// Scan `prov`'s physical memory for `_EPROCESS` objects by their `Proc` /
109/// `Pro\xe3` pool tag, reading `UniqueProcessId` at `pid_off`, `ImageFileName`
110/// at `name_off`, and optionally `_KPROCESS.DirectoryTableBase` at `dtb_off`
111/// (all offsets are `_EPROCESS`-relative, sourced from the ISF).
112///
113/// When `dtb_off` is `Some`, the DTB is read and validated:
114/// - Non-zero (a zero DTB has no valid page-table root)
115/// - 4 KiB-aligned (x86-64 CR3 values are always page-aligned)
116///
117/// Candidates failing either check are silently skipped (false-positive
118/// suppression without a panic). Results are deduplicated by `(pid, name)`.
119///
120/// Panic-free and bounded.
121pub fn scan_processes_dtb<P: PhysicalMemoryProvider + ?Sized>(
122    prov: &P,
123    pid_off: u64,
124    name_off: u64,
125    dtb_off: Option<u64>,
126) -> Vec<ScannedProcess> {
127    const CHUNK: usize = 1 << 20; // 1 MiB
128    const OVERLAP: u64 = 4;
129    let mut out = Vec::new();
130    let mut seen: HashSet<(u64, String)> = HashSet::new();
131
132    // Determine scan extents: the provider's ranges, or [0,total) if none.
133    let ranges: Vec<(u64, u64)> = {
134        let r = prov.ranges();
135        if r.is_empty() {
136            vec![(0, prov.total_size())]
137        } else {
138            r.iter().map(|x| (x.start, x.end)).collect()
139        }
140    };
141
142    let mut buf = vec![0u8; CHUNK + OVERLAP as usize];
143    for (start, end) in ranges {
144        let mut addr = start;
145        while addr < end {
146            let n = prov.read_phys(addr, &mut buf).unwrap_or(0);
147            if n < 4 {
148                addr = addr.saturating_add(CHUNK as u64);
149                continue;
150            }
151            // Find every `Proc` or `Pro\xe3` tag in this window.
152            // Both are valid EPROCESS pool tags; see POOL_TAGS for references.
153            let mut i = 0usize;
154            while i + 4 <= n {
155                let tag_match = POOL_TAGS.iter().any(|t| &buf[i..i + 4] == t.as_slice());
156                if !tag_match {
157                    i += 1;
158                    continue;
159                }
160                // Pool tag sits at +4 in the _POOL_HEADER, so the block base is
161                // 4 bytes before the tag's physical address.
162                let tag_pa = addr + i as u64;
163                let pool_base = tag_pa.saturating_sub(4);
164                if let Some(p) = try_eprocess(prov, pool_base, pid_off, name_off, dtb_off) {
165                    if seen.insert((p.pid, p.name.to_lowercase())) {
166                        out.push(p);
167                    }
168                }
169                i += 1;
170            }
171            addr = addr.saturating_add(CHUNK as u64 - OVERLAP);
172        }
173    }
174    out
175}

177/// Try each candidate `_EPROCESS` offset within the pool block; return the first
178/// that yields a plausible pid and a valid image name.
179/// `_DISPATCHER_HEADER.Type` for a process object (`KOBJECTS::ProcessObject`).
180/// `_EPROCESS` begins with `_KPROCESS` → `_DISPATCHER_HEADER`, so byte 0 of a
181/// genuine `_EPROCESS` is this value — the discriminator that separates real
182/// process objects from the many coincidental `Proc` byte sequences in a dump.
183const DISPATCHER_TYPE_PROCESS: u8 = 0x03;
184
185fn try_eprocess<P: PhysicalMemoryProvider + ?Sized>(
186    prov: &P,
187    pool_base: u64,
188    pid_off: u64,
189    name_off: u64,
190    dtb_off: Option<u64>,
191) -> Option<ScannedProcess> {
192    for &delta in EPROCESS_DELTAS {
193        let eproc = pool_base + delta;
194        // Strong gate first: the object must start with a process
195        // _DISPATCHER_HEADER. This rejects the bulk of coincidental tag matches.
196        match read_phys_exact(prov, eproc, 1) {
197            Some(b) if b[0] == DISPATCHER_TYPE_PROCESS => {}
198            _ => continue,
199        }
200        let pid_bytes = read_phys_exact(prov, eproc + pid_off, 8)?;
201        let pid = u64::from_le_bytes(pid_bytes.try_into().ok()?);
202        if !plausible_pid(pid) {
203            continue;
204        }
205        // DTB validation: when the offset is provided, read DirectoryTableBase
206        // from _KPROCESS (at _EPROCESS+0, since Pcb is the first field) and
207        // reject candidates with a zero or non-page-aligned value.
208        let dtb = if let Some(off) = dtb_off {
209            let Some(dtb_raw) = read_phys_exact(prov, eproc + off, 8) else {
210                continue;
211            };
212            let v = u64::from_le_bytes(dtb_raw.try_into().ok()?);
213            if !plausible_dtb(v) {
214                continue;
215            }
216            Some(v)
217        } else {
218            None
219        };
220        let Some(name_raw) = read_phys_exact(prov, eproc + name_off, 15) else {
221            continue;
222        };
223        if let Some(name) = decode_image_name(&name_raw) {
224            return Some(ScannedProcess {
225                physical_addr: eproc,
226                pid,
227                name,
228                dtb,
229            });
230        }
231    }
232    None
233}

235/// True if `dtb` looks like a valid page-table root.
236///
237/// `_KPROCESS.DirectoryTableBase` holds the raw CR3 value. On Windows 10+
238/// with KPTI (Kernel Page-Table Isolation) and PCID (Process-Context IDs)
239/// enabled, the low 12 bits carry the PCID tag (typically 0x001 for kernel,
240/// 0x003 for user) rather than being zero. The physical PML4 base is
241/// therefore `dtb & !0xFFF`; the PCID bits are not an error.
242///
243/// Validation: the PML4 physical base (bits 63:12) must be non-zero —
244/// a zero page-frame address has no valid page table regardless of PCID.
245///
246/// Source: x86-64 architecture manual §4.5 (CR3 bits 63:12 = PML4 base,
247/// bits 11:0 = PCID when CR4.PCIDE=1); Windows Internals 7th ed. §5
248/// (KPTI DirectoryTableBase includes PCID bits).
249fn plausible_dtb(dtb: u64) -> bool {
250    // Physical page frame address (bits 63:12) must be non-zero.
251    dtb & !0xFFF != 0
252}

254/// Convenience wrapper: pull the `_EPROCESS` / `_KPROCESS` field offsets from
255/// the reader's resolver and run `scan_processes_dtb`.
256///
257/// `UniqueProcessId` and `ImageFileName` are required; if absent, returns empty.
258/// `_KPROCESS.DirectoryTableBase` is optional — when present, DTB validation is
259/// enabled (recommended for zero false-positives on real dumps).
260#[must_use]
261pub fn psscan<P: PhysicalMemoryProvider>(reader: &ObjectReader<P>) -> Vec<ScannedProcess> {
262    let syms = reader.symbols();
263    let (Some(pid_off), Some(name_off)) = (
264        syms.field_offset("_EPROCESS", "UniqueProcessId"),
265        syms.field_offset("_EPROCESS", "ImageFileName"),
266    ) else {
267        return Vec::new();
268    };
269    // _KPROCESS.DirectoryTableBase — available in real ISFs; optional for robustness.
270    let dtb_off = syms.field_offset("_KPROCESS", "DirectoryTableBase");
271    scan_processes_dtb(reader.vas().physical(), pid_off, name_off, dtb_off)
272}

280    /// A flat physical memory provider over an owned byte buffer.
281    struct VecMem {
282        data: Vec<u8>,
283        ranges: Vec<PhysicalRange>,
284    }
285    impl VecMem {
286        fn new(data: Vec<u8>) -> Self {
287            let ranges = vec![PhysicalRange {
288                start: 0,
289                end: data.len() as u64,
290            }];
291            Self { data, ranges }
292        }

294    impl PhysicalMemoryProvider for VecMem {
295        fn read_phys(&self, addr: u64, buf: &mut [u8]) -> FmtResult<usize> {
296            let off = addr as usize;
297            if off >= self.data.len() {
298                return Ok(0);
299            }
300            let n = buf.len().min(self.data.len() - off);
301            buf[..n].copy_from_slice(&self.data[off..off + n]);
302            Ok(n)
303        }
304        fn ranges(&self) -> &[PhysicalRange] {
305            &self.ranges
306        }
307        fn total_size(&self) -> u64 {
308            self.data.len() as u64
309        }
310        fn format_name(&self) -> &str {
311            "vec"
312        }
313    }

315    const PID_OFF: u64 = 0x2e0;

318    /// A pool block whose `_EPROCESS` (at pool_base + 0x30) carries `pid` and
319    /// `name`, embedded at `at` within a larger buffer.
320    fn place_proc(buf: &mut [u8], at: usize, pid: u64, name: &str) {
321        // _POOL_HEADER: tag "Proc" at +4.
322        buf[at + 4..at + 8].copy_from_slice(b"Proc");
323        let eproc = at + 0x30;
324        // _DISPATCHER_HEADER.Type = ProcessObject at _EPROCESS+0.
325        buf[eproc] = 0x03;
326        buf[eproc + PID_OFF as usize..eproc + PID_OFF as usize + 8]
327            .copy_from_slice(&pid.to_le_bytes());
328        let nb = name.as_bytes();
329        buf[eproc + NAME_OFF as usize..eproc + NAME_OFF as usize + nb.len()].copy_from_slice(nb);
330    }

491    #[test]
492    #[ignore = "requires /tmp/vol3_test/DESKTOP-SDN1RPT.mem and oracle_psscan.json"]
493    fn psscan_reconcile_vs_vol3() {
494        use memf_format::open_dump_with_raw_fallback as open_dump;
495        use memf_symbols::isf::IsfResolver;
496        use memf_symbols::SymbolResolver as _;
497        use std::collections::HashSet;

499        let dump_path = "/tmp/vol3_test/DESKTOP-SDN1RPT.mem";
500        let isf_path = "/tmp/vol3_test/ntkrnlmp_81BC5C37.json";
501        let oracle_path = "/tmp/vol3_test/oracle_psscan.json";
502
503        // Load the ISF (real offsets for this dump).
504        let isf_bytes = std::fs::read(isf_path).expect("ISF must exist at {isf_path}");
505        let isf_json: serde_json::Value =
506            serde_json::from_slice(&isf_bytes).expect("ISF must be valid JSON");
507        let resolver =
508            IsfResolver::from_value(&isf_json).expect("IsfResolver must parse the ISF");
509
510        // Offsets from the real ISF (ntkrnlmp_81BC5C37.json):
511        //   _EPROCESS.UniqueProcessId  offset=1088 (0x440)
512        //   _EPROCESS.ImageFileName    offset=1448 (0x5A8)
513        //   _KPROCESS.DirectoryTableBase offset=40 (0x28) — Pcb is at _EPROCESS+0
514        let pid_off = resolver
515            .field_offset("_EPROCESS", "UniqueProcessId")
516            .expect("UniqueProcessId offset must be in ISF");
517        let name_off = resolver
518            .field_offset("_EPROCESS", "ImageFileName")
519            .expect("ImageFileName offset must be in ISF");
520        let dtb_off = resolver.field_offset("_KPROCESS", "DirectoryTableBase");
521
522        // Open the raw dump and run our psscan.
523        let dump = open_dump(std::path::Path::new(dump_path)).expect("dump must open");
524        // scan_processes_dtb accepts &P where P: ?Sized, so Box<dyn …>.as_ref() works.
525        let our_procs = scan_processes_dtb(dump.as_ref(), pid_off, name_off, dtb_off);
526
527        let our_pids: HashSet<u64> = our_procs.iter().map(|p| p.pid).collect();
528
529        // Load and parse the vol3 oracle.
530        let oracle_bytes = std::fs::read(oracle_path).expect("oracle must exist");
531        let oracle: Vec<serde_json::Value> =
532            serde_json::from_slice(&oracle_bytes).expect("oracle must be valid JSON");
533        let vol3_pids: HashSet<u64> = oracle
534            .iter()
535            .filter_map(|e| e.get("PID").and_then(serde_json::Value::as_u64))
536            .collect();
537
538        let overlap: Vec<u64> = {
539            let mut v: Vec<u64> = our_pids.intersection(&vol3_pids).copied().collect();
540            v.sort_unstable();
541            v
542        };
543        let memf_only: Vec<u64> = {
544            let mut v: Vec<u64> = our_pids.difference(&vol3_pids).copied().collect();
545            v.sort_unstable();
546            v
547        };
548        let vol3_only: Vec<u64> = {
549            let mut v: Vec<u64> = vol3_pids.difference(&our_pids).copied().collect();
550            v.sort_unstable();
551            v
552        };
553
554        println!("=== psscan reconciliation vs vol3 ===");
555        println!("  Our unique PIDs:  {}", our_pids.len());
556        println!("  Vol3 unique PIDs: {}", vol3_pids.len());
557        println!("  Overlap:          {} PIDs", overlap.len());
558        println!("  memf-only (FP?):  {memf_only:?}");
559        println!("  vol3-only (miss): {vol3_only:?}");
560
561        // memf-only entries are potential false positives — target 0.
562        assert!(
563            memf_only.is_empty(),
564            "psscan found {} PID(s) not in vol3 oracle (false positives): {:?}",
565            memf_only.len(),
566            memf_only,
567        );
568
569        // We must find at least as many unique PIDs as vol3.
570        assert!(
571            our_pids.len() >= vol3_pids.len(),
572            "psscan found {} unique PIDs, vol3 found {} — we should be >= vol3",
573            our_pids.len(),
574            vol3_pids.len(),
575        );
576    }

578    /// Vol3's `psscan` scans for BOTH `b"Proc"` AND `b"Pro\xe3"` pool tags.
579    /// The `\xe3` variant is the "protected" (high-MSB) non-paged pool allocation
580    /// that Windows 10+ uses for kernel objects — the pool tag bytes 0..3 are `Pro`
581    /// and byte 3 is set to `\xe3` (non-paged + protection flag).
582    ///
583    /// Reference: volatility3/framework/plugins/windows/psscan.py line
584    /// `constraints = poolscanner.PoolScanner.builtin_constraints(…, [b"Pro\xe3", b"Proc"])`
585    ///
586    /// This test places an EPROCESS under the `Pro\xe3` tag and expects it to be
587    /// recovered. It will FAIL until the scanner is extended to match both tags.
588    #[test]
589    fn psscan_finds_eprocess_with_protected_pool_tag() {
590        let mut data = vec![0u8; 0x4000];
591        // Place an EPROCESS with the PROTECTED pool tag (Pro\xe3) at offset 0x200.
592        // _POOL_HEADER: tag at +4, using the protected variant.
593        data[0x204..0x208].copy_from_slice(b"Pro\xe3");
594        let eproc = 0x200 + 0x30;
595        data[eproc] = 0x03; // _DISPATCHER_HEADER.Type = ProcessObject
596        // pid=1236 is a valid Windows process id: non-zero, <= 0x40000, multiple of 4.
597        data[eproc + PID_OFF as usize..eproc + PID_OFF as usize + 8]
598            .copy_from_slice(&1236u64.to_le_bytes());
599        let name = b"hidden.exe";
600        data[eproc + NAME_OFF as usize..eproc + NAME_OFF as usize + name.len()]
601            .copy_from_slice(name);
602        let mem = VecMem::new(data);
603        let found = scan_processes(&mem, PID_OFF, NAME_OFF);
604        assert!(
605            found.iter().any(|p| p.pid == 1236 && p.name == "hidden.exe"),
606            "must recover process under Pro\\xe3 (protected) pool tag: {found:?}"
607        );
608    }
