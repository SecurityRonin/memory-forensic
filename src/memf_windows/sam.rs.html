1//! Windows SAM (Security Account Manager) user account extraction.
2//!
3//! The SAM registry hive (`\REGISTRY\MACHINE\SAM`) stores local user
4//! account metadata: usernames, RIDs, account flags, and (encrypted)
5//! password hashes. Extracting SAM data from memory enables:
6//!
7//! - Identifying all local accounts (including hidden/disabled ones)
8//! - Detecting recently created accounts (persistence via new user)
9//! - Recovering account metadata when disk SAM is locked/encrypted
10//!
11//! The SAM hive is structured as:
12//! `SAM\Domains\Account\Users\<RID>\V` — per-user binary data
13//! `SAM\Domains\Account\Users\Names\<username>` — name→RID mapping
14
15use memf_core::object_reader::ObjectReader;
16use memf_format::PhysicalMemoryProvider;
17
18/// Maximum number of SAM user entries to walk (safety limit).
19const MAX_USERS: usize = 4096;
20
21/// Information about a local user account recovered from the SAM hive.
22#[derive(Debug, Clone, serde::Serialize)]
23pub struct SamUserInfo {
24    /// User account name.
25    pub username: String,
26    /// Relative Identifier (RID) — unique per-user on the local machine.
27    pub rid: u32,
28    /// Account flags (USER_ACCOUNT_CONTROL): disabled, locked out, etc.
29    pub account_flags: u32,
30    /// Whether the account is currently disabled.
31    pub is_disabled: bool,
32    /// Whether the account has an empty (blank) password hint.
33    pub has_empty_password: bool,
34    /// Last login time (FILETIME).
35    pub last_login_time: u64,
36    /// Last password change time (FILETIME).
37    pub last_password_change: u64,
38    /// Account creation time (FILETIME, from F value).
39    pub account_created: u64,
40    /// Login count.
41    pub login_count: u32,
42    /// Whether this account looks suspicious based on heuristics.
43    pub is_suspicious: bool,
44}
45
46/// Account flag constants from USER_ACCOUNT_CONTROL.
47pub const UAC_ACCOUNT_DISABLED: u32 = 0x0001;
48/// Account lockout flag (ADS_UF_LOCKOUT).
49pub const UAC_LOCKOUT: u32 = 0x0010;
50/// Password not required flag (ADS_UF_PASSWD_NOTREQD).
51pub const UAC_PASSWORD_NOT_REQUIRED: u32 = 0x0020;
52/// Normal user account flag (ADS_UF_NORMAL_ACCOUNT).
53pub const UAC_NORMAL_ACCOUNT: u32 = 0x0200;
54
55/// Classify a SAM user account as suspicious.
56///
57/// Returns `true` for accounts that match patterns of attacker-created
58/// persistence accounts:
59/// - Username ends with '$' (hidden account convention)
60/// - Account has admin-like RID (500) but unusual name
61/// - Recently created account with password-not-required flag
62/// - Username matches known attack tool default accounts
63pub fn classify_sam_user(username: &str, rid: u32, flags: u32) -> bool {
64    const SUSPICIOUS_NAMES: &[&str] = &[
65        "defaultaccount0",
66        "support_388945a0",
67        "svc_admin",
68        "backdoor",
69        "hacker",
70        "test123",
71    ];
72
73    if username.is_empty() {
74        return false;
75    }
76
77    let lower = username.to_ascii_lowercase();
78
79    // Hidden accounts end with '$' (a Windows convention)
80    if lower.ends_with('$') && !lower.ends_with("machine$") {
81        return true;
82    }
83
84    // RID 500 is the built-in Administrator — if renamed to something unusual
85    if rid == 500 && lower != "administrator" && lower != "admin" {
86        return true;
87    }
88
89    // Password not required is suspicious for normal accounts
90    if (flags & UAC_PASSWORD_NOT_REQUIRED) != 0 && (flags & UAC_NORMAL_ACCOUNT) != 0 {
91        return true;
92    }
93
94    // Known attack tool default account names
95    if SUSPICIOUS_NAMES.iter().any(|&s| lower == s) {
96        return true;
97    }
98
99    false
100}

102/// Extract local user accounts from the SAM registry hive in memory.
103///
104/// Reads the SAM hive at `hive_addr` and walks
105/// `SAM\Domains\Account\Users` to enumerate user accounts.
106/// Returns an empty `Vec` if the hive is unreadable or the path is missing.
107pub fn walk_sam_users<P: PhysicalMemoryProvider>(
108    reader: &ObjectReader<P>,
109    hive_addr: u64,
110) -> crate::Result<Vec<SamUserInfo>> {
111    if hive_addr == 0 {
112        return Ok(Vec::new());
113    }
114
115    // Read _HHIVE.BaseBlock pointer (at offset 0x10 typically) to get _HBASE_BLOCK.
116    let base_block_off = reader
117        .symbols()
118        .field_offset("_HHIVE", "BaseBlock")
119        .unwrap_or(0x10);
120
121    let base_block_addr = match reader.read_bytes(hive_addr + base_block_off, 8) {
122        Ok(bytes) if bytes.len() == 8 => u64::from_le_bytes(bytes[..8].try_into().unwrap()),
123        _ => return Ok(Vec::new()),
124    };
125
126    if base_block_addr == 0 {
127        return Ok(Vec::new());
128    }
129
130    // Read root cell offset from _HBASE_BLOCK (at offset 0x24, u32).
131    let root_cell_off = match reader.read_bytes(base_block_addr + 0x24, 4) {
132        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
133        _ => return Ok(Vec::new()),
134    };
135
136    if root_cell_off == 0 || root_cell_off == u32::MAX {
137        return Ok(Vec::new());
138    }
139
140    // The storage base is at _HHIVE + Hive.Storage[Stable].Map or we can compute
141    // cell addresses as hive_addr + 0x1000 (hbin start) + cell_offset + 4 (cell header).
142    // For simplicity, we use the dual-mapping approach: read from base_block + 0x1000 + offset.
143    let storage_base = reader
144        .symbols()
145        .field_offset("_HHIVE", "Storage")
146        .unwrap_or(0x30);
147
148    // Try to read the flat storage base pointer (Stable storage BlockList).
149    let flat_base = match reader.read_bytes(hive_addr + storage_base, 8) {
150        Ok(bytes) if bytes.len() == 8 => {
151            let addr = u64::from_le_bytes(bytes[..8].try_into().unwrap());
152            if addr != 0 {
153                addr
154            } else {
155                base_block_addr + 0x1000
156            }
157        }
158        _ => base_block_addr + 0x1000,
159    };
160
161    // Navigate: root → SAM → Domains → Account → Users → Names
162    // Each _CM_KEY_NODE has Signature at 0x0 (should be "nk" = 0x6B6E),
163    // SubKeyCount at 0x18 (u32), SubKeyLists at 0x20 (u32 offset).
164    // We navigate by reading subkey lists and matching key names.
165    let root_addr = read_cell_addr(reader, flat_base, root_cell_off);
166    if root_addr == 0 {
167        return Ok(Vec::new());
168    }
169
170    // Walk SAM\Domains\Account\Users\Names to get username→RID mappings,
171    // then read the F/V values from SAM\Domains\Account\Users\<RID>.
172    // For the simplified walker, we enumerate child keys under the
173    // Users\Names path and extract metadata from each user's RID key.
174
175    // Navigate: root → SAM → Domains → Account → Users
176    let sam_key = find_subkey_by_name(reader, flat_base, root_addr, "SAM");
177    if sam_key == 0 {
178        return Ok(Vec::new());
179    }
180    let domains_key = find_subkey_by_name(reader, flat_base, sam_key, "Domains");
181    if domains_key == 0 {
182        return Ok(Vec::new());
183    }
184    let account_key = find_subkey_by_name(reader, flat_base, domains_key, "Account");
185    if account_key == 0 {
186        return Ok(Vec::new());
187    }
188    let users_key = find_subkey_by_name(reader, flat_base, account_key, "Users");
189    if users_key == 0 {
190        return Ok(Vec::new());
191    }
192
193    // Read the Names subkey to get username→RID mappings.
194    let names_key = find_subkey_by_name(reader, flat_base, users_key, "Names");
195
196    let mut users = Vec::new();
197
198    // Enumerate RID subkeys under Users (hex RID strings like "000001F4").
199    let subkey_count: u32 = match reader.read_bytes(users_key + 0x18, 4) {
200        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
201        _ => 0,
202    };
203
204    if subkey_count == 0 || subkey_count > MAX_USERS as u32 {
205        return Ok(users);
206    }
207
208    let subkey_list_off: u32 = match reader.read_bytes(users_key + 0x20, 4) {
209        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
210        _ => return Ok(users),
211    };
212
213    let list_addr = read_cell_addr(reader, flat_base, subkey_list_off);
214    if list_addr == 0 {
215        return Ok(users);
216    }
217
218    // Read list signature (lf/lh/li).
219    let list_sig = match reader.read_bytes(list_addr, 2) {
220        Ok(bytes) if bytes.len() == 2 => [bytes[0], bytes[1]],
221        _ => return Ok(users),
222    };
223
224    let count: u16 = match reader.read_bytes(list_addr + 2, 2) {
225        Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
226        _ => return Ok(users),
227    };
228
229    for i in 0..count.min(MAX_USERS as u16) {
230        let entry_off = match list_sig {
231            [b'l', b'f' | b'h'] => {
232                // lf/lh: 8-byte entries (offset + hash) starting at +4
233                match reader.read_bytes(list_addr + 4 + u64::from(i) * 8, 4) {
234                    Ok(bytes) if bytes.len() == 4 => {
235                        u32::from_le_bytes(bytes[..4].try_into().unwrap())
236                    }
237                    _ => continue,
238                }
239            }
240            [b'l', b'i'] => {
241                // li: 4-byte entries (offset only) starting at +4
242                match reader.read_bytes(list_addr + 4 + u64::from(i) * 4, 4) {
243                    Ok(bytes) if bytes.len() == 4 => {
244                        u32::from_le_bytes(bytes[..4].try_into().unwrap())
245                    }
246                    _ => continue,
247                }
248            }
249            _ => continue,
250        };
251
252        let key_addr = read_cell_addr(reader, flat_base, entry_off);
253        if key_addr == 0 {
254            continue;
255        }
256
257        // Read key name (at offset 0x4C in _CM_KEY_NODE, length at 0x4A).
258        let name_len: u16 = match reader.read_bytes(key_addr + 0x4A, 2) {
259            Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
260            _ => continue,
261        };
262
263        if name_len == 0 || name_len > 256 {
264            continue;
265        }
266
267        let key_name = match reader.read_bytes(key_addr + 0x4C, name_len as usize) {
268            Ok(bytes) => String::from_utf8_lossy(&bytes).to_string(),
269            _ => continue,
270        };
271
272        // Skip the "Names" subkey — we only want RID keys (hex strings).
273        if key_name.eq_ignore_ascii_case("Names") {
274            continue;
275        }
276
277        // Parse RID from hex key name (e.g., "000001F4" → 500).
278        let rid = match u32::from_str_radix(&key_name, 16) {
279            Ok(r) => r,
280            Err(_) => continue,
281        };
282
283        // Try to find the username from the Names subkey.
284        let username = if names_key != 0 {
285            find_name_for_rid(reader, flat_base, names_key, rid)
286        } else {
287            format!("RID-{rid}")
288        };
289
290        // Read the F value for account metadata.
291        // F value is at the Values list of this key.
292        let (account_flags, last_login_time, last_password_change, account_created, login_count) =
293            read_f_value(reader, flat_base, key_addr);
294
295        let is_disabled = (account_flags & UAC_ACCOUNT_DISABLED) != 0;
296        let has_empty_password = (account_flags & UAC_PASSWORD_NOT_REQUIRED) != 0;
297        let is_suspicious = classify_sam_user(&username, rid, account_flags);
298
299        users.push(SamUserInfo {
300            username,
301            rid,
302            account_flags,
303            is_disabled,
304            has_empty_password,
305            last_login_time,
306            last_password_change,
307            account_created,
308            login_count,
309            is_suspicious,
310        });
311    }
312
313    Ok(users)
314}

316/// Read a cell address from the flat storage base + cell offset.
317fn read_cell_addr<P: PhysicalMemoryProvider>(
318    reader: &ObjectReader<P>,
319    flat_base: u64,
320    cell_off: u32,
321) -> u64 {
322    // Cell data starts 4 bytes after the cell offset (cell size header).
323    let addr = flat_base + u64::from(cell_off) + 4;
324    // Verify we can read from this address.
325    match reader.read_bytes(addr, 2) {
326        Ok(bytes) if bytes.len() == 2 => addr,
327        _ => 0,
328    }
329}

331/// Find a subkey by name under a parent _CM_KEY_NODE.
332fn find_subkey_by_name<P: PhysicalMemoryProvider>(
333    reader: &ObjectReader<P>,
334    flat_base: u64,
335    parent_addr: u64,
336    target_name: &str,
337) -> u64 {
338    let subkey_count: u32 = match reader.read_bytes(parent_addr + 0x18, 4) {
339        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
340        _ => return 0,
341    };
342
343    if subkey_count == 0 || subkey_count > 4096 {
344        return 0;
345    }
346
347    let list_off: u32 = match reader.read_bytes(parent_addr + 0x20, 4) {
348        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
349        _ => return 0,
350    };
351
352    let list_addr = read_cell_addr(reader, flat_base, list_off);
353    if list_addr == 0 {
354        return 0;
355    }
356
357    let list_sig = match reader.read_bytes(list_addr, 2) {
358        Ok(bytes) if bytes.len() == 2 => [bytes[0], bytes[1]],
359        _ => return 0,
360    };
361
362    let count: u16 = match reader.read_bytes(list_addr + 2, 2) {
363        Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
364        _ => return 0,
365    };
366
367    for i in 0..count.min(4096) {
368        let entry_off = match list_sig {
369            [b'l', b'f' | b'h'] => {
370                match reader.read_bytes(list_addr + 4 + u64::from(i) * 8, 4) {
371                    Ok(bytes) if bytes.len() == 4 => {
372                        u32::from_le_bytes(bytes[..4].try_into().unwrap())
373                    }
374                    _ => continue,
375                }
376            }
377            [b'l', b'i'] => match reader.read_bytes(list_addr + 4 + u64::from(i) * 4, 4) {
378                Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
379                _ => continue,
380            },
381            _ => return 0,
382        };
383
384        let key_addr = read_cell_addr(reader, flat_base, entry_off);
385        if key_addr == 0 {
386            continue;
387        }
388
389        let name_len: u16 = match reader.read_bytes(key_addr + 0x4A, 2) {
390            Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
391            _ => continue,
392        };
393
394        if name_len == 0 || name_len > 256 {
395            continue;
396        }
397
398        let name = match reader.read_bytes(key_addr + 0x4C, name_len as usize) {
399            Ok(bytes) => String::from_utf8_lossy(&bytes).to_string(),
400            _ => continue,
401        };
402
403        if name.eq_ignore_ascii_case(target_name) {
404            return key_addr;
405        }
406    }
407
408    0
409}

411/// Find the username associated with a RID from the Names subkey.
412fn find_name_for_rid<P: PhysicalMemoryProvider>(
413    reader: &ObjectReader<P>,
414    flat_base: u64,
415    names_key: u64,
416    target_rid: u32,
417) -> String {
418    // Under Names, each subkey's name IS the username, and the default value
419    // type encodes the RID. We read each subkey name and check the value type.
420    let subkey_count: u32 = match reader.read_bytes(names_key + 0x18, 4) {
421        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
422        _ => return format!("RID-{target_rid}"),
423    };
424
425    if subkey_count == 0 || subkey_count > 4096 {
426        return format!("RID-{target_rid}");
427    }
428
429    let list_off: u32 = match reader.read_bytes(names_key + 0x20, 4) {
430        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
431        _ => return format!("RID-{target_rid}"),
432    };
433
434    let list_addr = read_cell_addr(reader, flat_base, list_off);
435    if list_addr == 0 {
436        return format!("RID-{target_rid}");
437    }
438
439    let list_sig = match reader.read_bytes(list_addr, 2) {
440        Ok(bytes) if bytes.len() == 2 => [bytes[0], bytes[1]],
441        _ => return format!("RID-{target_rid}"),
442    };
443
444    let count: u16 = match reader.read_bytes(list_addr + 2, 2) {
445        Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
446        _ => return format!("RID-{target_rid}"),
447    };
448
449    for i in 0..count.min(4096) {
450        let entry_off = match list_sig {
451            [b'l', b'f' | b'h'] => {
452                match reader.read_bytes(list_addr + 4 + u64::from(i) * 8, 4) {
453                    Ok(bytes) if bytes.len() == 4 => {
454                        u32::from_le_bytes(bytes[..4].try_into().unwrap())
455                    }
456                    _ => continue,
457                }
458            }
459            [b'l', b'i'] => match reader.read_bytes(list_addr + 4 + u64::from(i) * 4, 4) {
460                Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
461                _ => continue,
462            },
463            _ => break,
464        };
465
466        let key_addr = read_cell_addr(reader, flat_base, entry_off);
467        if key_addr == 0 {
468            continue;
469        }
470
471        // The default value's type field encodes the RID.
472        // In the Names key, each name key has a single default value
473        // whose data type is the RID (a Windows registry trick).
474        // The value list offset is at _CM_KEY_NODE + 0x2C, count at +0x28.
475        let val_count: u32 = match reader.read_bytes(key_addr + 0x28, 4) {
476            Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
477            _ => continue,
478        };
479
480        if val_count == 0 {
481            continue;
482        }
483
484        let val_list_off: u32 = match reader.read_bytes(key_addr + 0x2C, 4) {
485            Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
486            _ => continue,
487        };
488
489        let val_list_addr = read_cell_addr(reader, flat_base, val_list_off);
490        if val_list_addr == 0 {
491            continue;
492        }
493
494        // Read first value offset.
495        let val_off: u32 = match reader.read_bytes(val_list_addr, 4) {
496            Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
497            _ => continue,
498        };
499
500        let val_addr = read_cell_addr(reader, flat_base, val_off);
501        if val_addr == 0 {
502            continue;
503        }
504
505        // _CM_KEY_VALUE: Type at offset 0x10 (u32).
506        let val_type: u32 = match reader.read_bytes(val_addr + 0x10, 4) {
507            Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
508            _ => continue,
509        };
510
511        if val_type == target_rid {
512            // Read the key name as the username.
513            let name_len: u16 = match reader.read_bytes(key_addr + 0x4A, 2) {
514                Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
515                _ => continue,
516            };
517
518            if name_len > 0 && name_len <= 256 {
519                if let Ok(bytes) = reader.read_bytes(key_addr + 0x4C, name_len as usize) {
520                    return String::from_utf8_lossy(&bytes).to_string();
521                }
522            }
523        }
524    }
525
526    format!("RID-{target_rid}")
527}

529/// Read account metadata from the F value of a user's RID key.
530/// Returns (flags, last_login, last_pw_change, created, login_count).
531fn read_f_value<P: PhysicalMemoryProvider>(
532    reader: &ObjectReader<P>,
533    flat_base: u64,
534    key_addr: u64,
535) -> (u32, u64, u64, u64, u32) {
536    let default = (0u32, 0u64, 0u64, 0u64, 0u32);
537
538    // Read value count and list.
539    let val_count: u32 = match reader.read_bytes(key_addr + 0x28, 4) {
540        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
541        _ => return default,
542    };
543
544    if val_count == 0 {
545        return default;
546    }
547
548    let val_list_off: u32 = match reader.read_bytes(key_addr + 0x2C, 4) {
549        Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
550        _ => return default,
551    };
552
553    let val_list_addr = read_cell_addr(reader, flat_base, val_list_off);
554    if val_list_addr == 0 {
555        return default;
556    }
557
558    // Scan values for "F".
559    for v in 0..val_count.min(64) {
560        let val_off: u32 = match reader.read_bytes(val_list_addr + u64::from(v) * 4, 4) {
561            Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
562            _ => continue,
563        };
564
565        let val_addr = read_cell_addr(reader, flat_base, val_off);
566        if val_addr == 0 {
567            continue;
568        }
569
570        // _CM_KEY_VALUE: NameLength at 0x02 (u16), Name at 0x18.
571        let vname_len: u16 = match reader.read_bytes(val_addr + 0x02, 2) {
572            Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
573            _ => continue,
574        };
575
576        if vname_len != 1 {
577            continue;
578        }
579
580        let vname = match reader.read_bytes(val_addr + 0x18, 1) {
581            Ok(bytes) if !bytes.is_empty() => bytes[0],
582            _ => continue,
583        };
584
585        if vname != b'F' {
586            continue;
587        }
588
589        // F value data: DataLength at 0x08 (u32), DataOffset at 0x0C (u32).
590        let data_len: u32 = match reader.read_bytes(val_addr + 0x08, 4) {
591            Ok(bytes) if bytes.len() == 4 => {
592                u32::from_le_bytes(bytes[..4].try_into().unwrap()) & 0x7FFFFFFF
593            }
594            _ => return default,
595        };
596
597        if data_len < 0x38 {
598            return default;
599        }
600
601        let data_off: u32 = match reader.read_bytes(val_addr + 0x0C, 4) {
602            Ok(bytes) if bytes.len() == 4 => u32::from_le_bytes(bytes[..4].try_into().unwrap()),
603            _ => return default,
604        };
605
606        let data_addr = read_cell_addr(reader, flat_base, data_off);
607        if data_addr == 0 {
608            return default;
609        }
610
611        // F value layout:
612        // 0x08: last login time (FILETIME, 8 bytes)
613        // 0x18: last password change (FILETIME, 8 bytes)
614        // 0x20: account creation time (FILETIME, 8 bytes)
615        // 0x30: account flags (u16)
616        // 0x38: login count (u16)
617        let last_login = match reader.read_bytes(data_addr + 0x08, 8) {
618            Ok(bytes) if bytes.len() == 8 => u64::from_le_bytes(bytes[..8].try_into().unwrap()),
619            _ => 0,
620        };
621
622        let last_pw = match reader.read_bytes(data_addr + 0x18, 8) {
623            Ok(bytes) if bytes.len() == 8 => u64::from_le_bytes(bytes[..8].try_into().unwrap()),
624            _ => 0,
625        };
626
627        let created = match reader.read_bytes(data_addr + 0x20, 8) {
628            Ok(bytes) if bytes.len() == 8 => u64::from_le_bytes(bytes[..8].try_into().unwrap()),
629            _ => 0,
630        };
631
632        let flags_raw: u16 = match reader.read_bytes(data_addr + 0x30, 2) {
633            Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
634            _ => 0,
635        };
636
637        let login_cnt: u16 = match reader.read_bytes(data_addr + 0x38, 2) {
638            Ok(bytes) if bytes.len() == 2 => u16::from_le_bytes(bytes[..2].try_into().unwrap()),
639            _ => 0,
640        };
641
642        return (
643            u32::from(flags_raw),
644            last_login,
645            last_pw,
646            created,
647            u32::from(login_cnt),
648        );
649    }
650
651    default
652}

823    /// Exercises the full hive navigation path: builds a minimal hive with a valid
824    /// root NK cell that has subkey_count=0, so find_subkey_by_name("SAM") returns 0.
825    /// Covers lines 177-184 (sam_key == 0 branch).
826    #[test]
827    fn walk_sam_users_root_cell_no_sam_subkey() {
828        // Layout:
829        //   hive_vaddr  = 0x0070_0000  → maps to 0x0070_0000
830        //   base_block  = 0x0071_0000  → maps to 0x0071_0000
831        //   flat_base   = base_block + 0x1000 = 0x0072_0000  → maps to 0x0072_0000
832        //
833        // root_cell_off = 0x20 (from base_block+0x24)
834        // root cell address = flat_base + 0x20 + 4 = 0x0072_0024
835        // At root cell we write a valid-looking header (2 readable bytes) so
836        // read_cell_addr returns non-zero, then find_subkey_by_name reads
837        // subkey_count at +0x18 = 0 → returns 0 (sam_key == 0 branch).
838
839        let hive_vaddr: u64 = 0x0070_0000;
840        let hive_paddr: u64 = 0x0070_0000;
841        let base_block: u64 = 0x0071_0000;
842        let base_block_paddr: u64 = 0x0071_0000;
843        let flat_base_paddr: u64 = 0x0072_0000;
844        let root_cell_off: u32 = 0x20;
845        // flat_base = base_block + 0x1000 (storage ptr is 0 → fallback)
846        // root_addr = flat_base + root_cell_off + 4 = flat_base + 0x24
847        // = 0x0072_0000 + 0x24 = 0x0072_0024
848
849        let mut hive_page = vec![0u8; 0x1000];
850        hive_page[0x10..0x18].copy_from_slice(&base_block.to_le_bytes());
851        // storage at 0x30 = 0 → flat_base = base_block + 0x1000
852        hive_page[0x30..0x38].copy_from_slice(&0u64.to_le_bytes());
853
854        let mut bb_page = vec![0u8; 0x1000];
855        bb_page[0x24..0x28].copy_from_slice(&root_cell_off.to_le_bytes());
856
857        // flat_base page: root cell at offset root_cell_off within flat_base.
858        // read_cell_addr reads 2 bytes at flat_base + root_cell_off + 4
859        // For this to succeed we just need those 2 bytes to be readable (mapped).
860        // Then find_subkey_by_name reads subkey_count at root_addr + 0x18.
861        // root_addr = flat_base_paddr + root_cell_off + 4 = flat_base_paddr + 0x24
862        // subkey_count at root_addr + 0x18 = flat_base_paddr + 0x3C = offset 0x3C in flat_base page
863        // Write subkey_count = 0 so sam_key = 0 → returns empty.
864        let mut flat_page = vec![0u8; 0x1000];
865        // subkey_count at offset (root_cell_off + 4 + 0x18) = 0x3C
866        let subkey_count_offset = (root_cell_off + 4 + 0x18) as usize;
867        flat_page[subkey_count_offset..subkey_count_offset + 4]
868            .copy_from_slice(&0u32.to_le_bytes());
869
870        let isf = IsfBuilder::new()
871            .add_struct("_HHIVE", 0x600)
872            .add_field("_HHIVE", "BaseBlock", 0x10, "pointer")
873            .add_field("_HHIVE", "Storage", 0x30, "pointer")
874            .build_json();
875        let resolver = IsfResolver::from_value(&isf).unwrap();
876
877        let (cr3, mem) = PageTableBuilder::new()
878            .map_4k(hive_vaddr, hive_paddr, flags::WRITABLE)
879            .write_phys(hive_paddr, &hive_page)
880            .map_4k(base_block, base_block_paddr, flags::WRITABLE)
881            .write_phys(base_block_paddr, &bb_page)
882            .map_4k(base_block + 0x1000, flat_base_paddr, flags::WRITABLE)
883            .write_phys(flat_base_paddr, &flat_page)
884            .build();
885
886        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
887        let reader = ObjectReader::new(vas, Box::new(resolver));
888
889        let result = walk_sam_users(&reader, hive_vaddr).unwrap();
890        assert!(result.is_empty(), "no SAM subkey should return empty");
891    }

1640    //
1641    // Build a full 5-level NK chain so walk_sam_users traverses the inner
1642    // enumeration loop (lines 228-310), encounters a valid RID key name

1645    // SamUserInfo into the output Vec.

1648    // addresses stay well below the 16 MB SyntheticPhysMem limit.

1651        // Cell offsets (within flat_base page):
1652        let root_cell_off: u32 = 0x020;
1653        let root_list_off: u32 = 0x060;
1654        let sam_cell_off: u32 = 0x0A0;
1655        let sam_list_off: u32 = 0x100;
1656        let domains_cell_off: u32 = 0x140;
1657        let dom_list_off: u32 = 0x1A0;
1658        let account_cell_off: u32 = 0x1E0;
1659        let acc_list_off: u32 = 0x240;
1660        let users_cell_off: u32 = 0x280;
1661        let users_list_off: u32 = 0x2E0;
1662        let rid_cell_off: u32 = 0x320;
1663
1664        // Helper: page offset of cell_addr = cell_off + 4
1665        let addr_off = |cell_off: u32| (cell_off + 4) as usize;
1666
1667        // ── Root NK: subkey_count=1, list→root_list_off ──
1668        let ro = addr_off(root_cell_off);
1669        flat_page[ro + 0x18..ro + 0x1C].copy_from_slice(&1u32.to_le_bytes());
1670        flat_page[ro + 0x20..ro + 0x24].copy_from_slice(&root_list_off.to_le_bytes());

1954    //   names_cell_off = 0x380   names NK (name="Names", 1 subkey)
1955    //   names_list_off = 0x3C0   lf list for Names subkey
1956    //   uname_cell_off = 0x400   username NK (name matches RID=500)
1957    //   uname_vlist_off= 0x440   value list cell for username NK (1 entry)
1958    //   uname_vk_off   = 0x480   VK cell (default value, type=RID=500=0x1F4)
1959    //   rid_vlist_off  = 0x4C0   value list cell for RID key (1 entry → F vk)
1960    //   f_vk_off       = 0x500   VK cell (NameLength=1, Name[0]='F', data_len=0x38)
1961    //   f_data_off     = 0x540   data cell (0x38 bytes of FILETIME + flags data)
1962    //   All within a single 4 KB page.
1963
1964    fn build_extended_hive(flat_page: &mut Vec<u8>) {
1965        // First lay down the standard 5-level chain.
1966        build_deep_hive(flat_page);
1967
1968        // === Additional cell offsets ===
1969        let names_cell_off: u32 = 0x380;
1970        let names_list_off: u32 = 0x3C0;
1971        let uname_cell_off: u32 = 0x400;
1972        let uname_vlist_off: u32 = 0x440; // value list for username NK
1973        let uname_vk_off: u32 = 0x480; // VK for username NK (type = RID)
1974        let rid_vlist_off: u32 = 0x4C0; // value list for RID NK
1975        let f_vk_off: u32 = 0x500; // VK "F" cell
1976        let f_data_off: u32 = 0x540; // F value data cell
1977
1978        let addr = |off: u32| (off + 4) as usize;
1979
1980        // ── Users NK (at cell_off=0x280) needs Names subkey ──
1981        // Increase subkey_count from 1 (RID key) to 2 (RID key + Names)
1982        // and update the users lf list to hold both entries.
1983        // BUT: modifying users_list to add Names entry would require
1984        // rewriting the list with count=2 and two entries.
1985        // Simpler approach: use a *separate* hive where Users has 1 RID child
1986        // plus a Names subkey accessible at users_key + some other mechanism.
1987        // Actually, find_name_for_rid is called with names_key returned by
1988        // find_subkey_by_name(reader, flat_base, users_key, "Names").
1989        // So we need the users lf list to also point to a Names NK.
1990        //
1991        // Override the users lf list (at users_list_off=0x2E0, addr=0x2E4):
1992        //  - Change count from 1 to 2
1993        //  - Add Names cell entry at [1] = names_cell_off
1994        let ulo = addr(0x2E0); // users lf list addr offset
1995        flat_page[ulo + 2..ulo + 4].copy_from_slice(&2u16.to_le_bytes()); // count=2
1996                                                                          // entry[0] already = rid_cell_off (0x320)
1997                                                                          // entry[1] = names_cell_off (lf entry = 8 bytes)
1998        flat_page[ulo + 12..ulo + 16].copy_from_slice(&names_cell_off.to_le_bytes());
1999
2000        // Also bump Users NK subkey_count to 2.
2001        let uo = addr(0x280);
2002        flat_page[uo + 0x18..uo + 0x1C].copy_from_slice(&2u32.to_le_bytes());
2003
2004        // ── Names NK: name="Names", 1 subkey → uname_cell_off ──
2005        let no = addr(names_cell_off);
2006        flat_page[no + 0x18..no + 0x1C].copy_from_slice(&1u32.to_le_bytes());
2007        flat_page[no + 0x20..no + 0x24].copy_from_slice(&names_list_off.to_le_bytes());
2008        flat_page[no + 0x4A..no + 0x4C].copy_from_slice(&5u16.to_le_bytes()); // len("Names")=5
2009        flat_page[no + 0x4C..no + 0x4C + 5].copy_from_slice(b"Names");
2010
2011        // Names lf list: sig="lf", count=1, entry[0]=uname_cell_off
2012        let nlo = addr(names_list_off);
2013        flat_page[nlo] = b'l';
2014        flat_page[nlo + 1] = b'f';
2015        flat_page[nlo + 2..nlo + 4].copy_from_slice(&1u16.to_le_bytes());
2016        flat_page[nlo + 4..nlo + 8].copy_from_slice(&uname_cell_off.to_le_bytes());
2017
2018        // ── Username NK: name="Administrator" (matches RID 500 via VK type field) ──
2019        // val_count=1 at +0x28, val_list_off=uname_vlist_off at +0x2C
2020        let uno = addr(uname_cell_off);
2021        let uname = b"Administrator";
2022        flat_page[uno + 0x4A..uno + 0x4C].copy_from_slice(&(uname.len() as u16).to_le_bytes());
2023        flat_page[uno + 0x4C..uno + 0x4C + uname.len()].copy_from_slice(uname);
2024        flat_page[uno + 0x28..uno + 0x2C].copy_from_slice(&1u32.to_le_bytes()); // val_count=1
2025        flat_page[uno + 0x2C..uno + 0x30].copy_from_slice(&uname_vlist_off.to_le_bytes());
2026
2027        // Value list cell for username NK: one entry → uname_vk_off
2028        let uvlo = addr(uname_vlist_off);
2029        flat_page[uvlo..uvlo + 4].copy_from_slice(&uname_vk_off.to_le_bytes());
2030
2031        // VK for username NK: type=500 (=0x1F4, the RID), NameLength=0 (default value)
2032        // VK layout: sig[0x0]=0x76 'v', sig[0x1]=0x6B 'k',
2033        //   NameLength at 0x02 (u16), DataLength at 0x08 (u32), DataOffset at 0x0C (u32),
2034        //   Type at 0x10 (u32), Name at 0x18.
2035        let uvko = addr(uname_vk_off);
2036        flat_page[uvko] = b'v';
2037        flat_page[uvko + 1] = b'k';
2038        flat_page[uvko + 2..uvko + 4].copy_from_slice(&0u16.to_le_bytes()); // NameLength=0
2039        flat_page[uvko + 0x10..uvko + 0x14].copy_from_slice(&500u32.to_le_bytes()); // type=RID=500
2040
2041        // ── RID NK (at 0x320): add val_count=1 and val_list → F VK ──
2042        let rido = addr(0x320);
2043        flat_page[rido + 0x28..rido + 0x2C].copy_from_slice(&1u32.to_le_bytes()); // val_count=1
2044        flat_page[rido + 0x2C..rido + 0x30].copy_from_slice(&rid_vlist_off.to_le_bytes());
2045
2046        // Value list cell for RID NK: one entry → f_vk_off
2047        let rvlo = addr(rid_vlist_off);
2048        flat_page[rvlo..rvlo + 4].copy_from_slice(&f_vk_off.to_le_bytes());
2049
2050        // VK "F" cell: NameLength=1, Name[0]='F', DataLength=0x38 (>=0x38), DataOffset=f_data_off
2051        let fvko = addr(f_vk_off);
2052        flat_page[fvko] = b'v';
2053        flat_page[fvko + 1] = b'k';
2054        flat_page[fvko + 2..fvko + 4].copy_from_slice(&1u16.to_le_bytes()); // NameLength=1
2055        flat_page[fvko + 0x08..fvko + 0x0C].copy_from_slice(&0x38u32.to_le_bytes()); // DataLength=0x38
2056        flat_page[fvko + 0x0C..fvko + 0x10].copy_from_slice(&f_data_off.to_le_bytes()); // DataOffset
2057        flat_page[fvko + 0x10..fvko + 0x14].copy_from_slice(&0u32.to_le_bytes()); // type=0
2058        flat_page[fvko + 0x18] = b'F'; // Name[0]='F'
2059
2060        // F data cell: 0x38 bytes with timestamps and flags
2061        // Offsets within data cell (relative to data addr, not cell_off):
2062        //   +0x08: last_login   (FILETIME)
2063        //   +0x18: last_pw      (FILETIME)
2064        //   +0x20: created      (FILETIME)
2065        //   +0x30: flags        (u16)
2066        //   +0x38: login_count  (u16) — note: 0x38 bytes means this is at byte 56 from data start
2067        // But data_len = 0x38 means indices 0..0x38, so login_count at +0x38 is PAST the end.
2068        // read_f_value reads login_count at data_addr + 0x38 (the 57th byte).
2069        // Provide 0x40 bytes to be safe.
2070        let fdo = addr(f_data_off);
2071        let last_login: u64 = 132_000_000_000_000_000u64;
2072        let last_pw: u64 = 131_500_000_000_000_000u64;
2073        let created: u64 = 130_000_000_000_000_000u64;
2074        let flags: u16 = UAC_NORMAL_ACCOUNT as u16;
2075        let login_cnt: u16 = 7u16;
2076        flat_page[fdo + 0x08..fdo + 0x10].copy_from_slice(&last_login.to_le_bytes());
2077        flat_page[fdo + 0x18..fdo + 0x20].copy_from_slice(&last_pw.to_le_bytes());
2078        flat_page[fdo + 0x20..fdo + 0x28].copy_from_slice(&created.to_le_bytes());
2079        flat_page[fdo + 0x30..fdo + 0x32].copy_from_slice(&flags.to_le_bytes());
2080        flat_page[fdo + 0x38..fdo + 0x3A].copy_from_slice(&login_cnt.to_le_bytes());
2081    }

2345        let root_cell_off: u32 = 0x020;
2346
2347        let mut hive_page = vec![0u8; 0x1000];
2348        hive_page[0x10..0x18].copy_from_slice(&base_block.to_le_bytes());
2349        hive_page[0x30..0x38].copy_from_slice(&0u64.to_le_bytes());
2350
2351        let mut bb_page = vec![0u8; 0x1000];
2352        bb_page[0x24..0x28].copy_from_slice(&root_cell_off.to_le_bytes());
2353
2354        let mut flat_page = vec![0u8; 0x1000];
2355        build_extended_hive(&mut flat_page);
2356
2357        // Override F VK DataLength to 0x10 (< 0x38) → `if data_len < 0x38 { return default }`.
2358        let fvko = (0x500 + 4) as usize;
2359        flat_page[fvko + 0x08..fvko + 0x0C].copy_from_slice(&0x10u32.to_le_bytes());
2360
2361        let resolver = IsfResolver::from_value(&make_full_isf()).unwrap();
2362        let (cr3, mem) = PageTableBuilder::new()
2363            .map_4k(hive_vaddr, hive_paddr, flags::WRITABLE)
2364            .write_phys(hive_paddr, &hive_page)
2365            .map_4k(base_block, base_block_paddr, flags::WRITABLE)
2366            .write_phys(base_block_paddr, &bb_page)
2367            .map_4k(base_block + 0x1000, flat_base_paddr, flags::WRITABLE)
2368            .write_phys(flat_base_paddr, &flat_page)
2369            .build();
2370
2371        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
2372        let reader = ObjectReader::new(vas, Box::new(resolver));
2373
2374        let result = walk_sam_users(&reader, hive_vaddr).unwrap();
2375        // F data_len < 0x38 → return default → account_flags=0
2376        assert_eq!(result.len(), 1);
2377        assert_eq!(result[0].account_flags, 0, "short F data → default flags");
2378    }
