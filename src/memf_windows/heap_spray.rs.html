1//! Heap spray detection walker — MITRE ATT&CK T1203.
2//!
3//! Analyses process heaps for patterns consistent with heap spraying:
4//! large numbers of same-sized allocations, NOP sled patterns, and
5//! unusually high heap commit sizes.
6
7use memf_core::object_reader::ObjectReader;
8use memf_format::PhysicalMemoryProvider;
9
10use crate::{types::HeapSprayInfo, Result};
11
12/// Classify whether a heap's committed size exceeds the spray threshold.
13///
14/// Returns `true` if `heap_commit_size > threshold`, indicating that the heap
15/// has grown to a size consistent with a heap spray attack (large number of
16/// uniform allocations filling the heap).
17pub fn is_spray_pattern(heap_commit_size: u64, threshold: u64) -> bool {
18    heap_commit_size > threshold
19}
20
21/// Scan process heaps for heap spray indicators.
22///
23/// For each process, walks the `_PEB.ProcessHeaps` array of `_HEAP`
24/// structures. For each heap:
25/// - Counts allocations that match spray heuristics (uniform size,
26///   repeated byte pattern in the user data).
27/// - Scans for NOP sled sequences (`0x90` repeated for ≥16 bytes).
28/// - Records total committed bytes.
29///
30/// # MITRE ATT&CK
31/// T1203 — Exploitation for Client Execution
32pub fn scan_heap_spray<P: PhysicalMemoryProvider>(
33    reader: &ObjectReader<P>,
34) -> Result<Vec<HeapSprayInfo>> {
35    let _ = reader;
36    Ok(vec![])
37}

42    use memf_core::test_builders::PageTableBuilder;

44    use memf_symbols::isf::IsfResolver;

47    fn make_minimal_reader(
48    ) -> ObjectReader<memf_core::test_builders::SyntheticPhysMem> {
49        let isf = IsfBuilder::new().build_json();
50        let resolver = IsfResolver::from_value(&isf).unwrap();
51        let (cr3, mem) = PageTableBuilder::new().build();
52        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
53        ObjectReader::new(vas, Box::new(resolver))
54    }
55
56    #[test]
57    fn empty_memory_returns_ok_empty() {
58        let reader = make_minimal_reader();
59        let result = scan_heap_spray(&reader);
60        assert!(result.is_ok(), "should succeed with minimal reader");
61        assert!(result.unwrap().is_empty(), "empty memory → no heap spray hits");
62    }

64    #[test]
65    fn result_is_vec_of_heap_spray_info() {
66        let reader = make_minimal_reader();
67        let result: Result<Vec<HeapSprayInfo>> = scan_heap_spray(&reader);
68        assert!(result.is_ok());
69    }
