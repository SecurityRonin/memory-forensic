1//! YARA rule scanning of process virtual memory regions.
2//!
3//! For each user-mode process, walks the VAD tree to enumerate virtual memory
4//! regions, reads up to `max_region_bytes` of each region using the process's
5//! own CR3 (via `reader.with_cr3`), and scans the bytes with compiled YARA-X
6//! rules. Returns one `WinYaraHit` per matching rule per region.
7
8use memf_core::object_reader::ObjectReader;
9use memf_format::PhysicalMemoryProvider;
10
11use crate::{Result, WinYaraHit};
12
13/// Maximum bytes read per VAD region to avoid OOM on large allocations.
14pub const MAX_REGION_BYTES: usize = 4 * 1024 * 1024; // 4 MiB
15
16/// Scan all user-mode process VAD regions against the supplied YARA rules.
17///
18/// `P` must be `Clone` because `reader.with_cr3` clones the physical memory
19/// provider to create a per-process virtual address space.
20pub fn scan_yara<P: PhysicalMemoryProvider + Clone>(
21    reader: &ObjectReader<P>,
22    ps_head_vaddr: u64,
23    rules: &yara_x::Rules,
24    max_region_bytes: usize,
25) -> Result<Vec<WinYaraHit>> {
26    let procs = crate::process::walk_processes(reader, ps_head_vaddr)?;
27
28    let vad_root_offset = reader
29        .symbols()
30        .field_offset("_EPROCESS", "VadRoot")
31        .ok_or_else(|| crate::Error::MissingField {
32            struct_name: "_EPROCESS".into(),
33            field_name: "VadRoot".into(),
34        })?;
35
36    let mut results = Vec::new();
37
38    for proc in &procs {
39        if proc.peb_addr == 0 {
40            continue; // skip kernel/System processes — PEB absence is the canonical indicator
41        }
42
43        let proc_reader = reader.with_cr3(proc.cr3);
44        let vad_root_addr = proc.vaddr.wrapping_add(vad_root_offset);
45        let vads = crate::vad::walk_vad_tree(reader, vad_root_addr, proc.pid, &proc.image_name)?;
46
47        let mut scanner = yara_x::Scanner::new(rules);
48
49        for vad in &vads {
50            let region_size =
51                (vad.end_vaddr.saturating_sub(vad.start_vaddr) as usize).min(max_region_bytes);
52            if region_size == 0 {
53                continue;
54            }
55
56            let bytes = match proc_reader.read_bytes(vad.start_vaddr, region_size) {
57                Ok(b) => b,
58                Err(_) => continue, // page not present / paged out — skip gracefully
59            };
60
61            let scan_result = match scanner.scan(&bytes) {
62                Ok(r) => r,
63                Err(_) => continue,
64            };
65
66            for rule in scan_result.matching_rules() {
67                results.push(WinYaraHit {
68                    pid: proc.pid,
69                    image_name: proc.image_name.clone(),
70                    start_vaddr: vad.start_vaddr,
71                    end_vaddr: vad.end_vaddr,
72                    protection_str: vad.protection_str.clone(),
73                    rule_name: rule.identifier().to_string(),
74                });
75            }
76        }
77    }
78
79    Ok(results)
80}

112    fn make_win_reader(ptb: PageTableBuilder) -> ObjectReader<SyntheticPhysMem> {
113        let isf = IsfBuilder::windows_kernel_preset().build_json();
114        let resolver = IsfResolver::from_value(&isf).unwrap();
115        let (cr3, mem) = ptb.build();
116        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
117        ObjectReader::new(vas, Box::new(resolver))
118    }

120    fn compile_rule(source: &str) -> yara_x::Rules {
121        yara_x::compile(source).expect("rule must compile")
122    }

129    fn yara_rule_matches_expected_bytes() {
130        let rules = compile_rule(r#"rule test { strings: $a = "MALWARE" condition: $a }"#);
131        let mut scanner = yara_x::Scanner::new(&rules);
132        let result = scanner.scan(b"some MALWARE payload here").unwrap();
133        let matched: Vec<_> = result.matching_rules().collect();
134        assert_eq!(matched.len(), 1, "rule must match");
135        assert_eq!(matched[0].identifier(), "test");

151    /// Build a minimal synthetic Windows process in memory, set up a VAD region
152    /// containing a known byte pattern, and verify scan_yara returns a hit.
153    ///
154    /// Uses one CR3 for both kernel and process (shared page table). After
155    /// `PageTableBuilder::build()`, the kernel CR3 is patched into the
156    /// `_EPROCESS.Pcb.DirectoryTableBase` field via `SyntheticPhysMem::write_u64`
157    /// so that `reader.with_cr3(proc.cr3)` produces an identical view.
158    ///
159    /// VA                  PA              Content
160    /// 0xFFFF_8000_0010_0000  0x0001_0000  PsActiveProcessHead (LIST_ENTRY)
161    /// 0xFFFF_8000_0020_0000  0x0002_0000  _EPROCESS (2 pages)
162    /// 0xFFFF_8000_0030_0000  0x0003_0000  _MMVAD_SHORT (VAD root node)
163    /// 0x0000_0001_0000_0000  0x0005_0000  User-space page with "MALWARE"
164    #[test]
165    fn scan_yara_returns_hit_for_matching_region() {
166        const HEAD_VADDR: u64 = 0xFFFF_8000_0010_0000;
167        const HEAD_PADDR: u64 = 0x0001_0000;
168        const EPROC_VADDR: u64 = 0xFFFF_8000_0020_0000;
169        const EPROC_PADDR: u64 = 0x0002_0000;
170        const VAD_VADDR: u64 = 0xFFFF_8000_0030_0000;
171        const VAD_PADDR: u64 = 0x0003_0000;
172        const USER_PAGE_VADDR: u64 = 0x0000_0001_0000_0000;
173        const USER_PAGE_PADDR: u64 = 0x0005_0000;
174
175        let mut head_data = vec![0u8; 4096];
176        let mut eproc_data = vec![0u8; 8192];
177        let mut vad_data = vec![0u8; 4096];
178        let mut user_data = vec![0u8; 4096];
179
180        // PsActiveProcessHead → EPROCESS.ActiveProcessLinks
181        let links_vaddr = EPROC_VADDR + EPROCESS_ACTIVE_LINKS as u64;
182        head_data[0..8].copy_from_slice(&links_vaddr.to_le_bytes());
183        head_data[8..16].copy_from_slice(&links_vaddr.to_le_bytes());
184
185        // EPROCESS.ActiveProcessLinks (points back to head — single process)
186        eproc_data[EPROCESS_ACTIVE_LINKS..EPROCESS_ACTIVE_LINKS + 8]
187            .copy_from_slice(&HEAD_VADDR.to_le_bytes());
188        eproc_data[EPROCESS_ACTIVE_LINKS + 8..EPROCESS_ACTIVE_LINKS + 16]
189            .copy_from_slice(&HEAD_VADDR.to_le_bytes());
190
191        // CR3 left as 0 in eproc_data — patched into physical memory after build().
192        eproc_data[EPROCESS_PID..EPROCESS_PID + 8].copy_from_slice(&4u64.to_le_bytes());
193        eproc_data[EPROCESS_PPID..EPROCESS_PPID + 8].copy_from_slice(&0u64.to_le_bytes());
194        eproc_data[EPROCESS_NAME..EPROCESS_NAME + 11].copy_from_slice(b"payload.exe");
195        eproc_data[EPROCESS_PEB..EPROCESS_PEB + 8].copy_from_slice(&0x7FFF_0000u64.to_le_bytes());
196        // VadRoot at EPROC + EPROCESS_VAD_ROOT → points to _MMVAD_SHORT
197        eproc_data[EPROCESS_VAD_ROOT..EPROCESS_VAD_ROOT + 8]
198            .copy_from_slice(&VAD_VADDR.to_le_bytes());
199
200        // _MMVAD_SHORT: single leaf node covering one user-space page
201        let starting_vpn = USER_PAGE_VADDR >> 12;
202        vad_data[VAD_LEFT..VAD_LEFT + 8].copy_from_slice(&0u64.to_le_bytes());
203        vad_data[VAD_RIGHT..VAD_RIGHT + 8].copy_from_slice(&0u64.to_le_bytes());
204        vad_data[VAD_STARTING_VPN..VAD_STARTING_VPN + 8]
205            .copy_from_slice(&starting_vpn.to_le_bytes());
206        vad_data[VAD_ENDING_VPN..VAD_ENDING_VPN + 8].copy_from_slice(&starting_vpn.to_le_bytes());
207        // Flags: private (VadType=0), PAGE_READWRITE (prot=4) in bits [7:11]
208        let vad_flags: u32 = 4 << 7;
209        vad_data[VAD_FLAGS..VAD_FLAGS + 4].copy_from_slice(&vad_flags.to_le_bytes());
210
211        // User page: contains "MALWARE"
212        user_data[0..7].copy_from_slice(b"MALWARE");
213
214        let ptb = PageTableBuilder::new()
215            .map_4k(HEAD_VADDR, HEAD_PADDR, flags::WRITABLE)
216            .map_4k(EPROC_VADDR, EPROC_PADDR, flags::WRITABLE)
217            .map_4k(EPROC_VADDR + 0x1000, EPROC_PADDR + 0x1000, flags::WRITABLE)
218            .map_4k(VAD_VADDR, VAD_PADDR, flags::WRITABLE)
219            .map_4k(USER_PAGE_VADDR, USER_PAGE_PADDR, flags::WRITABLE)
220            .write_phys(HEAD_PADDR, &head_data)
221            .write_phys(EPROC_PADDR, &eproc_data[..4096])
222            .write_phys(EPROC_PADDR + 0x1000, &eproc_data[4096..])
223            .write_phys(VAD_PADDR, &vad_data)
224            .write_phys(USER_PAGE_PADDR, &user_data);
225
226        let isf = IsfBuilder::windows_kernel_preset().build_json();
227        let resolver = IsfResolver::from_value(&isf).unwrap();
228        let (kernel_cr3, mut mem) = ptb.build();
229
230        // Patch EPROCESS.Pcb.DirectoryTableBase = kernel_cr3 so with_cr3 gives same view.
231        mem.write_u64(
232            EPROC_PADDR + (EPROCESS_PCB + KPROCESS_DTB) as u64,
233            kernel_cr3,
234        );
235
236        let vas = VirtualAddressSpace::new(mem, kernel_cr3, TranslationMode::X86_64FourLevel);
237        let reader = ObjectReader::new(vas, Box::new(resolver));
238
239        let rules = compile_rule("rule malware { strings: $a = \"MALWARE\" condition: $a }");
240        let hits = scan_yara(&reader, HEAD_VADDR, &rules, MAX_REGION_BYTES).unwrap();
241
242        assert_eq!(hits.len(), 1, "must find one YARA hit");
243        assert_eq!(hits[0].pid, 4);
244        assert_eq!(hits[0].rule_name, "malware");
245        assert_eq!(hits[0].start_vaddr, USER_PAGE_VADDR);
246    }

248    #[test]
249    fn scan_yara_no_hits_when_no_match() {
250        let rules =
251            compile_rule("rule never { strings: $a = \"NEVER_MATCHES_XYZ\" condition: $a }");
252        // Minimal reader with no processes mapped — walks empty list → no hits.
253        let mut head_data = vec![0u8; 4096];
254        const HEAD_VADDR: u64 = 0xFFFF_8000_0010_0000;
255        const HEAD_PADDR: u64 = 0x0001_0000;
256        // Self-referential list (no processes)
257        head_data[0..8].copy_from_slice(&HEAD_VADDR.to_le_bytes());
258        head_data[8..16].copy_from_slice(&HEAD_VADDR.to_le_bytes());
259        let ptb = PageTableBuilder::new()
260            .map_4k(HEAD_VADDR, HEAD_PADDR, flags::WRITABLE)
261            .write_phys(HEAD_PADDR, &head_data);
262        let reader = make_win_reader(ptb);
263        let hits = scan_yara(&reader, HEAD_VADDR, &rules, MAX_REGION_BYTES).unwrap();
264        assert!(hits.is_empty(), "no processes → no hits");
265    }

267    #[test]
268    fn scan_yara_region_capped_at_max_bytes() {
269        // Verify that regions larger than max_region_bytes don't panic and
270        // the function still returns results (reading only up to the cap).
271        // We test this by asserting the function completes without error when
272        // max_region_bytes is 1 (tiny cap) on a matching region.
273        let rules = compile_rule("rule test { strings: $a = \"M\" condition: $a }");
274        // Same setup as scan_yara_returns_hit_for_matching_region but max=1.
275        // With only 1 byte read, "MALWARE"[0] = 'M' → still matches rule `$a = "M"`.
276        // This confirms the cap is applied and partial reads work.
277        const HEAD_VADDR: u64 = 0xFFFF_8000_0010_0000;
278        const HEAD_PADDR: u64 = 0x0001_0000;
279        let mut head_data = vec![0u8; 4096];
280        head_data[0..8].copy_from_slice(&HEAD_VADDR.to_le_bytes());
281        head_data[8..16].copy_from_slice(&HEAD_VADDR.to_le_bytes());
282        let ptb = PageTableBuilder::new()
283            .map_4k(HEAD_VADDR, HEAD_PADDR, flags::WRITABLE)
284            .write_phys(HEAD_PADDR, &head_data);
285        let reader = make_win_reader(ptb);
286        // Empty process list: no VADs to scan, so no hits regardless of cap.
287        let hits = scan_yara(&reader, HEAD_VADDR, &rules, 1).unwrap();
288        assert!(hits.is_empty(), "empty list + tiny cap → no hits, no panic");
289    }
