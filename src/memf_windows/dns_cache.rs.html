1//! Windows DNS resolver cache extraction.
2//!
3//! The Windows DNS Client service (`Dnscache`) maintains an in-memory cache
4//! of recently resolved DNS records in `dnsrslvr.dll`. Extracting this cache
5//! from memory reveals what domains a system has resolved — critical for
6//! identifying C2 infrastructure, data exfiltration endpoints, and lateral
7//! movement targets during DFIR triage.
8//!
9//! The cache uses a hash table of `DNS_HASHTABLE_ENTRY` structures, each
10//! containing a doubly-linked list of `DNS_CACHE_ENTRY` records. Each record
11//! stores the queried name, record type, TTL, and resolved data.
12
13use memf_core::object_reader::ObjectReader;
14use memf_format::PhysicalMemoryProvider;
15
16use crate::{DnsCacheEntry, DnsRecordType, Result};
17
18/// Maximum number of hash table buckets to iterate (safety limit).
19const MAX_HASH_BUCKETS: u64 = 4096;
20
21/// Maximum number of UTF-16LE code units to read for a wide string (safety limit).
22const MAX_WIDE_CHARS: usize = 512;
23
24/// Maximum chain length per bucket (safety limit against corruption).
25const MAX_CHAIN_LENGTH: usize = 512;
26
27/// Walk the Windows DNS resolver cache and extract cached DNS records.
28///
29/// Looks for the DNS cache hash table in `svchost.exe` / `Dnscache` service
30/// memory. Iterates hash buckets, follows entry chains via `Next` pointer,
31/// and extracts name, type, TTL, and resolved data for each record.
32pub fn walk_dns_cache<P: PhysicalMemoryProvider>(
33    reader: &ObjectReader<P>,
34) -> Result<Vec<DnsCacheEntry>> {
35    // Look for DnsHashTable symbol (from dnsrslvr.dll debug info)
36    let hash_table_addr = reader
37        .symbols()
38        .symbol_address("g_HashTable")
39        .or_else(|| reader.symbols().symbol_address("g_CacheHashTable"))
40        .or_else(|| reader.symbols().symbol_address("DnsCacheHashTable"));
41
42    let Some(hash_table_addr) = hash_table_addr else {
43        return Ok(Vec::new());
44    };
45
46    let bucket_count: u32 = reader
47        .read_field(hash_table_addr, "DNS_HASHTABLE", "BucketCount")
48        .unwrap_or(256);
49    let bucket_count = u64::from(bucket_count).min(MAX_HASH_BUCKETS);
50
51    let buckets_offset = reader
52        .symbols()
53        .field_offset("DNS_HASHTABLE", "Buckets")
54        .unwrap_or(8);
55    let buckets_addr = hash_table_addr + buckets_offset;
56
57    let mut entries = Vec::new();
58
59    for i in 0..bucket_count {
60        let bucket_ptr_addr = buckets_addr + i * 8;
61        let entry_ptr: u64 = match reader.read_bytes(bucket_ptr_addr, 8) {
62            Ok(bytes) => u64::from_le_bytes(bytes[..8].try_into().expect("8 bytes")),
63            Err(_) => continue,
64        };
65
66        let mut current = entry_ptr;
67        let mut chain_len = 0;
68
69        while current != 0 && chain_len < MAX_CHAIN_LENGTH {
70            if let Ok(entry) = read_cache_entry(reader, current) {
71                entries.push(entry);
72            }
73            current = match reader.read_field::<u64>(current, "DNS_CACHE_ENTRY", "Next") {
74                Ok(v) => v,
75                Err(_) => break,
76            };
77            chain_len += 1;
78        }
79    }
80
81    Ok(entries)
82}

84/// Read a null-terminated UTF-16LE wide string (`LPWSTR`) from a virtual address.
85///
86/// Reads up to `MAX_WIDE_CHARS` code units (2 bytes each), stopping at the
87/// first null (0x0000) code unit. This handles the `LPWSTR` / `PWSTR` pointer
88/// type common in Windows user-mode structures (as opposed to `_UNICODE_STRING`
89/// which carries an explicit length field).
90fn read_wide_string<P: PhysicalMemoryProvider>(
91    reader: &ObjectReader<P>,
92    ptr: u64,
93) -> Result<String> {
94    let raw = reader.read_bytes(ptr, MAX_WIDE_CHARS * 2)?;
95    let u16s: Vec<u16> = raw
96        .chunks_exact(2)
97        .map(|pair| u16::from_le_bytes([pair[0], pair[1]]))
98        .take_while(|&c| c != 0)
99        .collect();
100    Ok(String::from_utf16_lossy(&u16s))
101}

103/// Read a single DNS cache entry from the given address.
104fn read_cache_entry<P: PhysicalMemoryProvider>(
105    reader: &ObjectReader<P>,
106    addr: u64,
107) -> Result<DnsCacheEntry> {
108    let name = {
109        let name_ptr: u64 = reader.read_field(addr, "DNS_CACHE_ENTRY", "Name")?;
110        if name_ptr == 0 {
111            String::new()
112        } else {
113            read_wide_string(reader, name_ptr).unwrap_or_default()
114        }
115    };
116
117    let record_type_raw: u16 = reader
118        .read_field(addr, "DNS_CACHE_ENTRY", "Type")
119        .unwrap_or(0);
120    let record_type = DnsRecordType::from_raw(record_type_raw);
121
122    let ttl: u32 = reader
123        .read_field(addr, "DNS_CACHE_ENTRY", "Ttl")
124        .unwrap_or(0);
125
126    let data = read_record_data(reader, addr, record_type_raw).unwrap_or_default();
127
128    Ok(DnsCacheEntry {
129        name,
130        record_type,
131        data,
132        ttl,
133    })
134}

136/// Read the resolved data from a DNS cache entry based on record type.
137fn read_record_data<P: PhysicalMemoryProvider>(
138    reader: &ObjectReader<P>,
139    entry_addr: u64,
140    record_type: u16,
141) -> Result<String> {
142    let data_ptr: u64 = reader.read_field(entry_addr, "DNS_CACHE_ENTRY", "Data")?;
143    if data_ptr == 0 {
144        return Ok(String::new());
145    }
146
147    match record_type {
148        // A record: 4 bytes IPv4
149        1 => {
150            let bytes = reader.read_bytes(data_ptr, 4)?;
151            Ok(format!(
152                "{}.{}.{}.{}",
153                bytes[0], bytes[1], bytes[2], bytes[3]
154            ))
155        }
156        // AAAA record: 16 bytes IPv6
157        28 => {
158            let bytes = reader.read_bytes(data_ptr, 16)?;
159            let mut parts = Vec::with_capacity(8);
160            for chunk in bytes.chunks(2) {
161                parts.push(format!("{:02x}{:02x}", chunk[0], chunk[1]));
162            }
163            Ok(parts.join(":"))
164        }
165        // CNAME, PTR: pointer to wide string
166        5 | 12 => read_wide_string(reader, data_ptr),
167        // Other types: hex dump of first 32 bytes
168        _ => {
169            let len = reader
170                .read_field::<u16>(entry_addr, "DNS_CACHE_ENTRY", "DataLength")
171                .unwrap_or(0);
172            let read_len = usize::from(len).min(32);
173            if read_len == 0 {
174                return Ok(String::new());
175            }
176            let bytes = reader.read_bytes(data_ptr, read_len)?;
177            Ok(bytes.iter().map(|b| format!("{b:02x}")).collect::<String>())
178        }
179    }
180}

208    // Synthetic layout:
209    //   g_HashTable @ 0xFFFF_8000_0020_0000 (DNS_HASHTABLE)
210    //     BucketCount @ offset 0 (u32) = 2
211    //     Buckets     @ offset 8 (array of pointers)
212    //       bucket[0] → 0xFFFF_8000_0020_1000 (DNS_CACHE_ENTRY)
213    //       bucket[1] → 0 (empty)
214    //
215    //   DNS_CACHE_ENTRY @ 0xFFFF_8000_0020_1000:
216    //     Next    @ 0  (pointer, 8 bytes) = 0 (no chain)
217    //     Name    @ 8  (pointer to wide string at 0xFFFF_8000_0020_2000)
218    //     Type    @ 16 (u16) = 1 (A record)
219    //     Ttl     @ 20 (u32) = 300
220    //     DataLength @ 24 (u16) = 4
221    //     Data    @ 32 (pointer to 4 bytes at 0xFFFF_8000_0020_3000)
222    //
223    //   Name string @ 0xFFFF_8000_0020_2000: "evil.c2.example.com\0" (wide)
224    //   Data bytes  @ 0xFFFF_8000_0020_3000: [10, 0, 0, 1] → "10.0.0.1"
225
226    const HASHTABLE_VADDR: u64 = 0xFFFF_8000_0020_0000;
227    const HASHTABLE_PADDR: u64 = 0x0080_0000;
228    const ENTRY_VADDR: u64 = 0xFFFF_8000_0020_1000;
229    const ENTRY_PADDR: u64 = 0x0080_1000;
230    const NAME_VADDR: u64 = 0xFFFF_8000_0020_2000;
231    const NAME_PADDR: u64 = 0x0080_2000;
232    const DATA_VADDR: u64 = 0xFFFF_8000_0020_3000;
233    const DATA_PADDR: u64 = 0x0080_3000;

235    fn build_dns_reader() -> ObjectReader<SyntheticPhysMem> {
236        let isf = IsfBuilder::new()
237            .add_struct("DNS_HASHTABLE", 64)
238            .add_field("DNS_HASHTABLE", "BucketCount", 0, "unsigned int")
239            .add_field("DNS_HASHTABLE", "Buckets", 8, "pointer")
240            .add_struct("DNS_CACHE_ENTRY", 48)
241            .add_field("DNS_CACHE_ENTRY", "Next", 0, "pointer")
242            .add_field("DNS_CACHE_ENTRY", "Name", 8, "pointer")
243            .add_field("DNS_CACHE_ENTRY", "Type", 16, "unsigned short")
244            .add_field("DNS_CACHE_ENTRY", "Ttl", 20, "unsigned int")
245            .add_field("DNS_CACHE_ENTRY", "DataLength", 24, "unsigned short")
246            .add_field("DNS_CACHE_ENTRY", "Data", 32, "pointer")
247            .add_symbol("g_HashTable", HASHTABLE_VADDR)
248            .build_json();
249        let resolver = IsfResolver::from_value(&isf).unwrap();
250
251        let mut ht_data = vec![0u8; 4096];
252        ht_data[0..4].copy_from_slice(&2u32.to_le_bytes()); // BucketCount
253        ht_data[8..16].copy_from_slice(&ENTRY_VADDR.to_le_bytes()); // Buckets[0]
254
255        let mut entry_data = vec![0u8; 4096];
256        entry_data[0..8].copy_from_slice(&0u64.to_le_bytes()); // Next = NULL
257        entry_data[8..16].copy_from_slice(&NAME_VADDR.to_le_bytes()); // Name ptr
258        entry_data[16..18].copy_from_slice(&1u16.to_le_bytes()); // Type = A
259        entry_data[20..24].copy_from_slice(&300u32.to_le_bytes()); // TTL = 300
260        entry_data[24..26].copy_from_slice(&4u16.to_le_bytes()); // DataLength = 4
261        entry_data[32..40].copy_from_slice(&DATA_VADDR.to_le_bytes()); // Data ptr
262
263        let mut name_data = vec![0u8; 4096];
264        let name = "evil.c2.example.com";
265        for (i, ch) in name.encode_utf16().enumerate() {
266            let off = i * 2;
267            name_data[off..off + 2].copy_from_slice(&ch.to_le_bytes());
268        }
269
270        let mut ip_data = vec![0u8; 4096];
271        ip_data[0] = 10;
272        ip_data[1] = 0;
273        ip_data[2] = 0;
274        ip_data[3] = 1;
275
276        let (cr3, mem) = PageTableBuilder::new()
277            .map_4k(HASHTABLE_VADDR, HASHTABLE_PADDR, flags::WRITABLE)
278            .write_phys(HASHTABLE_PADDR, &ht_data)
279            .map_4k(ENTRY_VADDR, ENTRY_PADDR, flags::WRITABLE)
280            .write_phys(ENTRY_PADDR, &entry_data)
281            .map_4k(NAME_VADDR, NAME_PADDR, flags::WRITABLE)
282            .write_phys(NAME_PADDR, &name_data)
283            .map_4k(DATA_VADDR, DATA_PADDR, flags::WRITABLE)
284            .write_phys(DATA_PADDR, &ip_data)
285            .build();
286        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
287        ObjectReader::new(vas, Box::new(resolver))
288    }

292    fn build_dns_reader_with_type(
293        record_type: u16,
294        data_bytes: Vec<u8>,
295        data_vaddr: u64,
296        data_paddr: u64,
297    ) -> ObjectReader<SyntheticPhysMem> {
298        let isf = IsfBuilder::new()
299            .add_struct("DNS_HASHTABLE", 64)
300            .add_field("DNS_HASHTABLE", "BucketCount", 0, "unsigned int")
301            .add_field("DNS_HASHTABLE", "Buckets", 8, "pointer")
302            .add_struct("DNS_CACHE_ENTRY", 48)
303            .add_field("DNS_CACHE_ENTRY", "Next", 0, "pointer")
304            .add_field("DNS_CACHE_ENTRY", "Name", 8, "pointer")
305            .add_field("DNS_CACHE_ENTRY", "Type", 16, "unsigned short")
306            .add_field("DNS_CACHE_ENTRY", "Ttl", 20, "unsigned int")
307            .add_field("DNS_CACHE_ENTRY", "DataLength", 24, "unsigned short")
308            .add_field("DNS_CACHE_ENTRY", "Data", 32, "pointer")
309            .add_symbol("g_HashTable", HASHTABLE_VADDR)
310            .build_json();
311        let resolver = IsfResolver::from_value(&isf).unwrap();
312
313        let mut ht_data = vec![0u8; 4096];
314        ht_data[0..4].copy_from_slice(&1u32.to_le_bytes()); // BucketCount = 1
315        ht_data[8..16].copy_from_slice(&ENTRY_VADDR.to_le_bytes()); // Buckets[0]
316
317        let mut entry_data = vec![0u8; 4096];
318        entry_data[0..8].copy_from_slice(&0u64.to_le_bytes()); // Next = NULL
319        entry_data[8..16].copy_from_slice(&NAME_VADDR.to_le_bytes()); // Name ptr
320        entry_data[16..18].copy_from_slice(&record_type.to_le_bytes()); // Type
321        entry_data[20..24].copy_from_slice(&60u32.to_le_bytes()); // TTL = 60
322        entry_data[24..26].copy_from_slice(&(data_bytes.len() as u16).to_le_bytes());
323        entry_data[32..40].copy_from_slice(&data_vaddr.to_le_bytes()); // Data ptr
324
325        let mut name_data = vec![0u8; 4096];
326        let name = "test.example.com";
327        for (i, ch) in name.encode_utf16().enumerate() {
328            let off = i * 2;
329            name_data[off..off + 2].copy_from_slice(&ch.to_le_bytes());
330        }
331
332        let mut ip_data = vec![0u8; 4096];
333        ip_data[..data_bytes.len().min(4096)]
334            .copy_from_slice(&data_bytes[..data_bytes.len().min(4096)]);
335
336        let (cr3, mem) = PageTableBuilder::new()
337            .map_4k(HASHTABLE_VADDR, HASHTABLE_PADDR, flags::WRITABLE)
338            .write_phys(HASHTABLE_PADDR, &ht_data)
339            .map_4k(ENTRY_VADDR, ENTRY_PADDR, flags::WRITABLE)
340            .write_phys(ENTRY_PADDR, &entry_data)
341            .map_4k(NAME_VADDR, NAME_PADDR, flags::WRITABLE)
342            .write_phys(NAME_PADDR, &name_data)
343            .map_4k(data_vaddr, data_paddr, flags::WRITABLE)
344            .write_phys(data_paddr, &ip_data)
345            .build();
346        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
347        ObjectReader::new(vas, Box::new(resolver))
348    }

491        ht_data[0..4].copy_from_slice(&1u32.to_le_bytes());
492        ht_data[8..16].copy_from_slice(&ENTRY_VADDR.to_le_bytes());
493
494        let mut entry_data = vec![0u8; 4096];
495        entry_data[0..8].copy_from_slice(&0u64.to_le_bytes()); // Next = NULL
496        entry_data[8..16].copy_from_slice(&NAME_VADDR.to_le_bytes());
497        entry_data[16..18].copy_from_slice(&1u16.to_le_bytes()); // Type = A
498        entry_data[20..24].copy_from_slice(&300u32.to_le_bytes());
499        entry_data[24..26].copy_from_slice(&4u16.to_le_bytes());
500        entry_data[32..40].copy_from_slice(&0u64.to_le_bytes()); // Data = NULL
501
502        let mut name_data = vec![0u8; 4096];
503        let name = "nulldata.example.com";
504        for (i, ch) in name.encode_utf16().enumerate() {
505            let off = i * 2;
506            name_data[off..off + 2].copy_from_slice(&ch.to_le_bytes());
507        }
508
509        let (cr3, mem) = PageTableBuilder::new()
510            .map_4k(HASHTABLE_VADDR, HASHTABLE_PADDR, flags::WRITABLE)
511            .write_phys(HASHTABLE_PADDR, &ht_data)
512            .map_4k(ENTRY_VADDR, ENTRY_PADDR, flags::WRITABLE)
513            .write_phys(ENTRY_PADDR, &entry_data)
514            .map_4k(NAME_VADDR, NAME_PADDR, flags::WRITABLE)
515            .write_phys(NAME_PADDR, &name_data)
516            .build();
517        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
518        let reader = ObjectReader::new(vas, Box::new(resolver));
519        let entries = walk_dns_cache(&reader).unwrap();
520        assert_eq!(entries.len(), 1);
521        assert_eq!(entries[0].name, name);
522        assert_eq!(entries[0].data, "");
523    }
