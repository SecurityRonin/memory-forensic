1//! Windows hibernation file (`hiberfil.sys`) format provider.
2//!
3//! Parses hibernation files with `PO_MEMORY_IMAGE` header signatures:
4//! `hibr` (0x72626968), `wake` (0x656B6177), `RSTR` (0x52545352),
5//! `HORM` (0x4D524F48). Eagerly decompresses Xpress LZ77 blocks into a
6//! `HashMap<pfn, page>` for random-access reads. Extracts CR3 from the
7//! processor state page.
8
9use std::collections::HashMap;
10use std::path::Path;
11
12use crate::{DumpMetadata, Error, FormatPlugin, PhysicalMemoryProvider, PhysicalRange, Result};
13
14/// Magic values (little-endian u32).
15const HIBR_MAGIC: u32 = 0x7262_6968;
16const WAKE_MAGIC: u32 = 0x656B_6177;
17const RSTR_MAGIC: u32 = 0x5254_5352;
18const HORM_MAGIC: u32 = 0x4D52_4F48;
19
20/// Page size in bytes.
21const PAGE_SIZE: usize = 4096;
22
23/// Xpress block signature: `[0x81, 0x81, 'x', 'p', 'r', 'e', 's', 's']`.
24const XPRESS_SIG: [u8; 8] = [0x81, 0x81, b'x', b'p', b'r', b'e', b's', b's'];
25
26/// Block header size (padded to 0x20).
27const BLOCK_HEADER_SIZE: usize = 0x20;
28
29/// Offset of `LengthSelf` in the PO_MEMORY_IMAGE header.
30const OFF_LENGTH_SELF: usize = 0x0C;
31
32/// Offset of `FirstTablePage` in the PO_MEMORY_IMAGE header (u64).
33const OFF_FIRST_TABLE_PAGE: usize = 0x68;
34
35/// Offset of CR3 within the processor state page (page 1).
36const OFF_CR3_IN_PROC_STATE: usize = 0x28;
37
38/// Read a little-endian u32 from `data` at `offset`.
39fn read_u32(data: &[u8], offset: usize) -> u32 {
40    u32::from_le_bytes(data[offset..offset + 4].try_into().unwrap())
41}
42
43/// Read a little-endian u64 from `data` at `offset`.
44fn read_u64(data: &[u8], offset: usize) -> u64 {
45    u64::from_le_bytes(data[offset..offset + 8].try_into().unwrap())
46}

48/// Provider that exposes physical memory from a Windows hibernation file.
49///
50/// Stores decompressed pages in a `HashMap<pfn, Vec<u8>>` for O(1) lookup.
51pub struct HiberfilProvider {
52    pages: HashMap<u64, Vec<u8>>,
53    ranges: Vec<PhysicalRange>,
54    meta: DumpMetadata,
55}
56
57impl HiberfilProvider {
58    /// Parse a hibernation file from an in-memory byte slice.
59    pub fn from_bytes(bytes: &[u8]) -> Result<Self> {
60        // Validate minimum size: at least 3 header pages (0x3000 bytes).
61        if bytes.len() < 3 * PAGE_SIZE {
62            return Err(Error::Corrupt(
63                "hiberfil too short: need at least 3 header pages".into(),
64            ));
65        }
66
67        // Validate magic.
68        let magic = read_u32(bytes, 0);
69        if !is_hiberfil_magic(magic) {
70            return Err(Error::Corrupt(format!(
71                "invalid hiberfil magic: 0x{magic:08X}"
72            )));
73        }
74
75        // Check LengthSelf to confirm 64-bit format (value 256).
76        let _length_self = read_u32(bytes, OFF_LENGTH_SELF);
77
78        // Extract CR3 from processor state page (page 1, offset 0x28).
79        let cr3 = read_u64(bytes, PAGE_SIZE + OFF_CR3_IN_PROC_STATE);
80
81        // Read the page table from page indicated by FirstTablePage.
82        let first_table_page = read_u64(bytes, OFF_FIRST_TABLE_PAGE);
83        let table_offset = first_table_page as usize * PAGE_SIZE;
84
85        if table_offset + PAGE_SIZE > bytes.len() {
86            return Err(Error::Corrupt("first table page beyond file end".into()));
87        }
88
89        // Parse PFN entries from the page table until sentinel 0xFFFFFFFFFFFFFFFF.
90        let mut pfn_list = Vec::new();
91        let mut pos = table_offset;
92        while pos + 8 <= table_offset + PAGE_SIZE {
93            let pfn = read_u64(bytes, pos);
94            if pfn == u64::MAX {
95                break;
96            }
97            pfn_list.push(pfn);
98            pos += 8;
99        }
100
101        // Decompress Xpress blocks starting after header pages (3 * PAGE_SIZE).
102        let mut pages = HashMap::new();
103        let mut block_offset = 3 * PAGE_SIZE;
104        let mut pfn_idx = 0;
105
106        while block_offset + BLOCK_HEADER_SIZE <= bytes.len() && pfn_idx < pfn_list.len() {
107            // Check Xpress signature.
108            if bytes[block_offset..block_offset + 8] != XPRESS_SIG {
109                break;
110            }
111
112            // Parse block header.
113            let num_pages_minus_1 = bytes[block_offset + 8] as usize;
114            let num_pages = num_pages_minus_1 + 1;
115
116            // compressed_size_field: 3 bytes LE at offset 9.
117            let csf_b0 = u32::from(bytes[block_offset + 9]);
118            let csf_b1 = u32::from(bytes[block_offset + 10]);
119            let csf_b2 = u32::from(bytes[block_offset + 11]);
120            let compressed_size_field = csf_b0 | (csf_b1 << 8) | (csf_b2 << 16);
121
122            // Decode compressed size: (compressed_size_field + 1) / 4.
123            let compressed_len = ((compressed_size_field + 1) / 4) as usize;
124
125            let data_start = block_offset + BLOCK_HEADER_SIZE;
126            let data_end = data_start + compressed_len;
127
128            if data_end > bytes.len() {
129                return Err(Error::Corrupt(format!(
130                    "xpress block at 0x{block_offset:X} extends beyond file (need {compressed_len} bytes)"
131                )));
132            }
133
134            // Decompress the block.
135            let compressed_data = &bytes[data_start..data_end];
136            let decompressed = lzxpress::data::decompress(compressed_data).map_err(|e| {
137                Error::Decompression(format!("xpress decompress at 0x{block_offset:X}: {e:?}"))
138            })?;
139
140            // Split decompressed data into individual pages.
141            for i in 0..num_pages {
142                if pfn_idx >= pfn_list.len() {
143                    break;
144                }
145                let pfn = pfn_list[pfn_idx];
146                let page_start = i * PAGE_SIZE;
147                let page_end = page_start + PAGE_SIZE;
148
149                if page_end <= decompressed.len() {
150                    pages.insert(pfn, decompressed[page_start..page_end].to_vec());
151                }
152                pfn_idx += 1;
153            }
154
155            block_offset = data_end;
156        }
157
158        // Build sorted ranges from the page map.
159        let mut pfns: Vec<u64> = pages.keys().copied().collect();
160        pfns.sort_unstable();
161        let ranges = build_ranges(&pfns);
162
163        let meta = DumpMetadata {
164            cr3: Some(cr3),
165            dump_type: Some("Hibernation".into()),
166            ..DumpMetadata::default()
167        };
168
169        Ok(Self {
170            pages,
171            ranges,
172            meta,
173        })
174    }

176    /// Parse a hibernation file from a file path.
177    pub fn from_path(path: &Path) -> Result<Self> {
178        let data = std::fs::read(path)?;
179        Self::from_bytes(&data)
180    }
181}

183/// Build coalesced `PhysicalRange` entries from a sorted list of PFNs.
184fn build_ranges(sorted_pfns: &[u64]) -> Vec<PhysicalRange> {
185    let mut ranges = Vec::new();
186    let mut iter = sorted_pfns.iter().copied();
187
188    let Some(first) = iter.next() else {
189        return ranges;
190    };
191
192    let mut range_start = first * PAGE_SIZE as u64;
193    let mut range_end = range_start + PAGE_SIZE as u64;
194
195    for pfn in iter {
196        let addr = pfn * PAGE_SIZE as u64;
197        if addr == range_end {
198            // Contiguous — extend.
199            range_end = addr + PAGE_SIZE as u64;
200        } else {
201            // Gap — push current and start new.
202            ranges.push(PhysicalRange {
203                start: range_start,
204                end: range_end,
205            });
206            range_start = addr;
207            range_end = addr + PAGE_SIZE as u64;
208        }
209    }
210    ranges.push(PhysicalRange {
211        start: range_start,
212        end: range_end,
213    });
214
215    ranges
216}

218/// Check whether a u32 matches one of the known hiberfil magic values.
219fn is_hiberfil_magic(magic: u32) -> bool {
220    matches!(magic, HIBR_MAGIC | WAKE_MAGIC | RSTR_MAGIC | HORM_MAGIC)
221}

223impl PhysicalMemoryProvider for HiberfilProvider {
224    fn read_phys(&self, addr: u64, buf: &mut [u8]) -> Result<usize> {
225        if buf.is_empty() {
226            return Ok(0);
227        }
228
229        let pfn = addr / PAGE_SIZE as u64;
230        let offset = (addr % PAGE_SIZE as u64) as usize;
231
232        let Some(page) = self.pages.get(&pfn) else {
233            return Ok(0);
234        };
235
236        let available = page.len().saturating_sub(offset);
237        let to_read = buf.len().min(available);
238        buf[..to_read].copy_from_slice(&page[offset..offset + to_read]);
239        Ok(to_read)
240    }
241
242    fn ranges(&self) -> &[PhysicalRange] {
243        &self.ranges
244    }
245
246    fn format_name(&self) -> &str {
247        "Hiberfil.sys"
248    }
249
250    fn metadata(&self) -> Option<DumpMetadata> {
251        Some(self.meta.clone())
252    }
253}

255/// FormatPlugin implementation for Windows hibernation files.
256pub struct HiberfilPlugin;
257
258impl FormatPlugin for HiberfilPlugin {
259    fn name(&self) -> &str {
260        "Hiberfil.sys"
261    }
262
263    fn probe(&self, header: &[u8]) -> u8 {
264        if header.len() < 4 {
265            return 0;
266        }
267        let magic = read_u32(header, 0);
268        match magic {
269            HIBR_MAGIC | WAKE_MAGIC => 90,
270            RSTR_MAGIC | HORM_MAGIC => 85,
271            _ => 0,
272        }
273    }
274
275    fn open(&self, path: &Path) -> Result<Box<dyn PhysicalMemoryProvider>> {
276        Ok(Box::new(HiberfilProvider::from_path(path)?))
277    }
278}
