6/// Build a synthetic LiME dump byte-by-byte.
7///
8/// Each range is serialised as a 32-byte header followed by raw payload data.
9/// Header layout (all fields little-endian):
10/// - 0x00: magic  = 0x4C694D45 (4 bytes)
11/// - 0x04: version = 1          (4 bytes)
12/// - 0x08: s_addr               (8 bytes)
13/// - 0x10: e_addr (inclusive)   (8 bytes)
14/// - 0x18: reserved             (8 bytes zeros)
15#[derive(Default)]
16pub struct LimeBuilder {
17    ranges: Vec<(u64, Vec<u8>)>,
18}
19
20impl LimeBuilder {
21    /// Create an empty builder.
22    pub fn new() -> Self {
23        Self { ranges: Vec::new() }
24    }
25
26    /// Add a physical memory range starting at `start` with the given `data`.
27    pub fn add_range(mut self, start: u64, data: &[u8]) -> Self {
28        self.ranges.push((start, data.to_vec()));
29        self
30    }
31
32    /// Serialise all ranges into a complete LiME dump.
33    pub fn build(self) -> Vec<u8> {
34        const MAGIC: u32 = 0x4C694D45;
35        const VERSION: u32 = 1;
36
37        let mut out = Vec::new();
38        for (start, data) in &self.ranges {
39            let e_addr = start + data.len() as u64 - 1; // inclusive end
40            out.extend_from_slice(&MAGIC.to_le_bytes());
41            out.extend_from_slice(&VERSION.to_le_bytes());
42            out.extend_from_slice(&start.to_le_bytes());
43            out.extend_from_slice(&e_addr.to_le_bytes());
44            out.extend_from_slice(&0u64.to_le_bytes()); // reserved
45            out.extend_from_slice(data);
46        }
47        out
48    }
49}

51/// Build a synthetic AVML dump byte-by-byte.
52///
53/// Each range is serialised as a 32-byte header followed by a Snappy-compressed
54/// payload, then 8 trailing bytes encoding the uncompressed size as u64 LE.
55/// Header layout (all fields little-endian):
56/// - 0x00: magic   = 0x4C4D5641 (4 bytes)
57/// - 0x04: version = 2           (4 bytes)
58/// - 0x08: s_addr                (8 bytes)
59/// - 0x10: e_addr (exclusive)    (8 bytes)
60/// - 0x18: reserved              (8 bytes zeros)
61#[derive(Default)]
62pub struct AvmlBuilder {
63    ranges: Vec<(u64, Vec<u8>)>,
64}
65
66impl AvmlBuilder {
67    /// Create an empty builder.
68    pub fn new() -> Self {
69        Self { ranges: Vec::new() }
70    }
71
72    /// Add a physical memory range starting at `start` with the given `data`.
73    pub fn add_range(mut self, start: u64, data: &[u8]) -> Self {
74        self.ranges.push((start, data.to_vec()));
75        self
76    }
77
78    /// Serialise all ranges into a complete AVML dump.
79    pub fn build(self) -> Vec<u8> {
80        const MAGIC: u32 = 0x4C4D5641;
81        const VERSION: u32 = 2;
82
83        let mut out = Vec::new();
84        for (start, data) in &self.ranges {
85            let e_addr = start + data.len() as u64; // exclusive end
86            let uncompressed_size = data.len() as u64;
87
88            let mut encoder = snap::raw::Encoder::new();
89            let compressed = encoder.compress_vec(data).expect("snappy compress");
90
91            out.extend_from_slice(&MAGIC.to_le_bytes());
92            out.extend_from_slice(&VERSION.to_le_bytes());
93            out.extend_from_slice(&start.to_le_bytes());
94            out.extend_from_slice(&e_addr.to_le_bytes());
95            out.extend_from_slice(&0u64.to_le_bytes()); // reserved
96            out.extend_from_slice(&compressed);
97            out.extend_from_slice(&uncompressed_size.to_le_bytes()); // trailer
98        }
99        out
100    }
101}

103/// Build a synthetic Windows 64-bit crash dump (`_DUMP_HEADER64`).
104///
105/// Produces an 8192-byte header followed by physical memory page data.
106/// Supports both run-based (DumpType 0x01) and bitmap (DumpType 0x02/0x05) layouts.
107///
108/// Header layout (little-endian, key offsets):
109/// - 0x000: "PAGE" magic (u32 = 0x4547_4150)
110/// - 0x004: "DU64" signature (u32 = 0x3436_5544)
111/// - 0x010: DirectoryTableBase / CR3 (u64)
112/// - 0x020: PsLoadedModuleList (u64)
113/// - 0x028: PsActiveProcessHead (u64)
114/// - 0x030: MachineImageType (u32)
115/// - 0x034: NumberProcessors (u32)
116/// - 0x080: KdDebuggerDataBlock (u64)
117/// - 0x088: PhysicalMemoryBlockBuffer — NumberOfRuns(u32) + pad(u32) + NumberOfPages(u64) + Runs[]
118/// - 0xF98: DumpType (u32)
119/// - 0xFA8: SystemTime (u64)
120pub struct CrashDumpBuilder {
121    runs: Vec<(u64, Vec<u8>)>,
122    cr3: u64,
123    ps_active_process_head: u64,
124    ps_loaded_module_list: u64,
125    kd_debugger_data_block: u64,
126    machine_type: u32,
127    num_processors: u32,
128    dump_type: u32,
129    system_time: u64,
130}
131
132impl Default for CrashDumpBuilder {
133    fn default() -> Self {
134        Self {
135            runs: Vec::new(),
136            cr3: 0x0018_7000,
137            ps_active_process_head: 0xFFFFF802_1A2B3C40,
138            ps_loaded_module_list: 0xFFFFF802_1A2B3D60,
139            kd_debugger_data_block: 0xFFFFF802_1A000000,
140            machine_type: 0x8664, // AMD64
141            num_processors: 4,
142            dump_type: 0x01, // Full (run-based)
143            system_time: 0x01DA_5678_9ABC_DEF0,
144        }
145    }
146}

213    /// Build the complete crash dump as a byte vector.
214    pub fn build(self) -> Vec<u8> {
215        const PAGE_MAGIC: u32 = 0x4547_4150; // "PAGE"
216        const DU64_SIG: u32 = 0x3436_5544; // "DU64"
217        const HEADER_SIZE: usize = 0x2000; // 8192 bytes
218        const PAGE_SIZE: usize = 4096;
219
220        let mut header = vec![0u8; HEADER_SIZE];
221
222        // 0x000: PAGE magic
223        header[0x000..0x004].copy_from_slice(&PAGE_MAGIC.to_le_bytes());
224        // 0x004: DU64 signature
225        header[0x004..0x008].copy_from_slice(&DU64_SIG.to_le_bytes());
226        // 0x010: CR3 / DirectoryTableBase
227        header[0x010..0x018].copy_from_slice(&self.cr3.to_le_bytes());
228        // 0x020: PsLoadedModuleList
229        header[0x020..0x028].copy_from_slice(&self.ps_loaded_module_list.to_le_bytes());
230        // 0x028: PsActiveProcessHead
231        header[0x028..0x030].copy_from_slice(&self.ps_active_process_head.to_le_bytes());
232        // 0x030: MachineImageType
233        header[0x030..0x034].copy_from_slice(&self.machine_type.to_le_bytes());
234        // 0x034: NumberProcessors
235        header[0x034..0x038].copy_from_slice(&self.num_processors.to_le_bytes());
236        // 0x080: KdDebuggerDataBlock
237        header[0x080..0x088].copy_from_slice(&self.kd_debugger_data_block.to_le_bytes());
238
239        // 0x088: PhysicalMemoryBlockBuffer
240        let num_runs = self.runs.len() as u32;
241        let total_pages: u64 = self
242            .runs
243            .iter()
244            .map(|(_, d)| (d.len() / PAGE_SIZE) as u64)
245            .sum();
246        // NumberOfRuns (u32) at 0x088
247        header[0x088..0x08C].copy_from_slice(&num_runs.to_le_bytes());
248        // Padding (u32) at 0x08C
249        header[0x08C..0x090].copy_from_slice(&0u32.to_le_bytes());
250        // NumberOfPages (u64) at 0x090
251        header[0x090..0x098].copy_from_slice(&total_pages.to_le_bytes());
252        // Runs[] starting at 0x098, each run is 16 bytes: base_page(u64) + page_count(u64)
253        for (i, (base_page, data)) in self.runs.iter().enumerate() {
254            let page_count = (data.len() / PAGE_SIZE) as u64;
255            let off = 0x098 + i * 16;
256            header[off..off + 8].copy_from_slice(&base_page.to_le_bytes());
257            header[off + 8..off + 16].copy_from_slice(&page_count.to_le_bytes());
258        }
259
260        // 0xF98: DumpType
261        header[0xF98..0xF9C].copy_from_slice(&self.dump_type.to_le_bytes());
262        // 0xFA8: SystemTime
263        header[0xFA8..0xFB0].copy_from_slice(&self.system_time.to_le_bytes());
264
265        let is_bitmap = self.dump_type == 0x02 || self.dump_type == 0x05;
266
267        if is_bitmap {
268            self.build_bitmap(header)
269        } else {
270            self.build_run_based(header)
271        }
272    }
273
274    /// Build run-based layout: data pages follow header sequentially at offset 0x2000.
275    fn build_run_based(self, mut out: Vec<u8>) -> Vec<u8> {
276        for (_, data) in &self.runs {
277            out.extend_from_slice(data);
278        }
279        out
280    }
281
282    /// Build bitmap layout: summary header + bitmap + data pages at offset 0x2000.
283    fn build_bitmap(self, mut out: Vec<u8>) -> Vec<u8> {
284        const DUMP_VALID: u32 = 0x504D_5544; // "DUMP"
285        const PAGE_SIZE: usize = 4096;
286
287        // Find the highest PFN to determine bitmap size.
288        let max_pfn: u64 = self
289            .runs
290            .iter()
291            .map(|(base, data)| base + (data.len() / PAGE_SIZE) as u64)
292            .max()
293            .unwrap_or(0);
294
295        // Bitmap: one bit per page up to max_pfn, rounded up to 8 bytes.
296        let bitmap_bits = max_pfn as usize;
297        let bitmap_bytes = bitmap_bits.div_ceil(8);
298        // Align bitmap_bytes up to multiple of 8.
299        let bitmap_bytes_aligned = (bitmap_bytes + 7) & !7;
300
301        let mut bitmap = vec![0u8; bitmap_bytes_aligned];
302        for (base_page, data) in &self.runs {
303            let page_count = data.len() / PAGE_SIZE;
304            for p in 0..page_count {
305                let pfn = *base_page as usize + p;
306                bitmap[pfn / 8] |= 1 << (pfn % 8);
307            }
308        }
309
310        // Summary header at 0x2000:
311        // ValidDump (u32) = "DUMP"
312        // HeaderSize (u32) = offset from 0x2000 to start of page data
313        // BitmapSize (u32) = bitmap size in bytes
314        // Pages (u32) = total number of set bits
315        let total_set_pages: u32 = self
316            .runs
317            .iter()
318            .map(|(_, d)| (d.len() / PAGE_SIZE) as u32)
319            .sum();
320        let summary_header_size: u32 = 16; // 4 fields * 4 bytes
321        let data_offset = summary_header_size as usize + bitmap_bytes_aligned;
322
323        out.extend_from_slice(&DUMP_VALID.to_le_bytes());
324        out.extend_from_slice(&(data_offset as u32).to_le_bytes());
325        out.extend_from_slice(&(bitmap_bytes_aligned as u32).to_le_bytes());
326        out.extend_from_slice(&total_set_pages.to_le_bytes());
327        out.extend_from_slice(&bitmap);
328
329        // Write page data in PFN order.
330        // Build a map from PFN to page data for ordered output.
331        let mut pfn_data: Vec<(u64, &[u8])> = Vec::new();
332        for (base_page, data) in &self.runs {
333            let page_count = data.len() / PAGE_SIZE;
334            for p in 0..page_count {
335                let pfn = base_page + p as u64;
336                let start = p * PAGE_SIZE;
337                pfn_data.push((pfn, &data[start..start + PAGE_SIZE]));
338            }
339        }
340        pfn_data.sort_by_key(|(pfn, _)| *pfn);
341        for (_, page) in &pfn_data {
342            out.extend_from_slice(page);
343        }
344
345        out
346    }
347}

349/// Build a synthetic ELF core dump for testing.
350#[derive(Default)]
351pub struct ElfCoreBuilder {
352    segments: Vec<(u64, Vec<u8>)>,
353}
354
355impl ElfCoreBuilder {
356    /// Create a new empty builder.
357    pub fn new() -> Self {
358        Self {
359            segments: Vec::new(),
360        }
361    }
362
363    /// Add a PT_LOAD segment at the given physical address with the given data.
364    pub fn add_segment(mut self, paddr: u64, data: &[u8]) -> Self {
365        self.segments.push((paddr, data.to_vec()));
366        self
367    }
368
369    /// Build the ELF core dump as a byte vector.
370    pub fn build(self) -> Vec<u8> {
371        let ehdr_size: usize = 64;
372        let phdr_size: usize = 56;
373        let phdr_count = self.segments.len();
374        let phdr_total = phdr_count * phdr_size;
375        let data_start = (ehdr_size + phdr_total).div_ceil(0x1000) * 0x1000;
376        let mut out = vec![0u8; data_start];
377
378        // ELF header
379        out[0..4].copy_from_slice(&[0x7F, b'E', b'L', b'F']);
380        out[4] = 2; // ELFCLASS64
381        out[5] = 1; // ELFDATA2LSB
382        out[6] = 1; // EV_CURRENT
383        out[16..18].copy_from_slice(&4u16.to_le_bytes()); // ET_CORE
384        out[18..20].copy_from_slice(&62u16.to_le_bytes()); // EM_X86_64
385        out[20..24].copy_from_slice(&1u32.to_le_bytes()); // e_version
386        out[32..40].copy_from_slice(&(ehdr_size as u64).to_le_bytes()); // e_phoff
387        out[52..54].copy_from_slice(&(ehdr_size as u16).to_le_bytes()); // e_ehsize
388        out[54..56].copy_from_slice(&(phdr_size as u16).to_le_bytes()); // e_phentsize
389        out[56..58].copy_from_slice(&(phdr_count as u16).to_le_bytes()); // e_phnum
390
391        let mut current_offset = data_start;
392        for (i, (paddr, data)) in self.segments.iter().enumerate() {
393            let phdr_off = ehdr_size + i * phdr_size;
394            out[phdr_off..phdr_off + 4].copy_from_slice(&1u32.to_le_bytes()); // PT_LOAD
395            out[phdr_off + 4..phdr_off + 8].copy_from_slice(&6u32.to_le_bytes()); // PF_R|PF_W
396            out[phdr_off + 8..phdr_off + 16]
397                .copy_from_slice(&(current_offset as u64).to_le_bytes());
398            out[phdr_off + 24..phdr_off + 32].copy_from_slice(&paddr.to_le_bytes());
399            out[phdr_off + 32..phdr_off + 40].copy_from_slice(&(data.len() as u64).to_le_bytes());
400            out[phdr_off + 40..phdr_off + 48].copy_from_slice(&(data.len() as u64).to_le_bytes());
401            out[phdr_off + 48..phdr_off + 56].copy_from_slice(&0x1000u64.to_le_bytes());
402            out.resize(current_offset + data.len(), 0);
403            out[current_offset..current_offset + data.len()].copy_from_slice(data);
404            current_offset += data.len();
405        }
406        out
407    }
408}

410/// Build a synthetic VMware `.vmss`/`.vmsn` state file.
411///
412/// Produces the VMware group/tag binary format:
413/// - 12-byte file header: magic(u32=0xBED2BED0) + unknown(u32=0) + group_count(u32)
414/// - Group entries (80 bytes each): name(64 bytes null-terminated) + tags_offset(u64) + padding(8 bytes)
415/// - "memory" group with region tags containing paddr(u64) + data
416/// - Optional "cpu" group with CR3 tag
417///
418/// Tag format for memory regions:
419/// - flags: u8 (0x06 = large data with explicit size)
420/// - name_length: u8
421/// - name: bytes ("regionPPN", "regionBytes")
422/// - data_length: u32
423/// - payload: depends on tag name
424///
425/// Tag format for CPU CR3:
426/// - flags: u8 (0x46 = indexed + 8-byte data)
427/// - name_length: u8
428/// - name: "CR3"
429/// - index0: u8 (0 = CPU 0)
430/// - index1: u8 (3 = CR register 3)
431/// - value: u64
432///
433/// Tag terminator: flags byte = 0
434#[derive(Default)]
435pub struct VmwareStateBuilder {
436    memory_regions: Vec<(u64, Vec<u8>)>,
437    cr3: Option<u64>,
438}
439
440impl VmwareStateBuilder {
441    /// Create a new empty builder.
442    pub fn new() -> Self {
443        Self {
444            memory_regions: Vec::new(),
445            cr3: None,
446        }
447    }
448
449    /// Add a physical memory region at the given physical address.
450    pub fn add_region(mut self, paddr: u64, data: &[u8]) -> Self {
451        self.memory_regions.push((paddr, data.to_vec()));
452        self
453    }
454
455    /// Set the CR3 / DirectoryTableBase value (adds a "cpu" group).
456    pub fn cr3(mut self, cr3: u64) -> Self {
457        self.cr3 = Some(cr3);
458        self
459    }
460
461    /// Build the complete VMware state file as a byte vector.
462    pub fn build(self) -> Vec<u8> {
463        let group_count: u32 = if self.cr3.is_some() { 2 } else { 1 };
464
465        // Header: 12 bytes
466        let header_size = 12usize;
467        let group_entry_size = 80usize;
468        let groups_size = group_count as usize * group_entry_size;
469
470        let mut out = Vec::new();
471
472        // File header
473        out.extend_from_slice(&0xBED2BED0u32.to_le_bytes()); // magic
474        out.extend_from_slice(&0u32.to_le_bytes()); // unknown
475        out.extend_from_slice(&group_count.to_le_bytes()); // group_count
476
477        // Reserve space for group entries — we'll fill tags_offset later
478        let groups_start = out.len();
479        out.resize(header_size + groups_size, 0);
480
481        // --- "memory" group tags ---
482        let memory_tags_offset = out.len() as u64;
483
484        // Write region tags: for each region, emit regionPPN + regionBytes pair
485        for (paddr, data) in &self.memory_regions {
486            // regionPPN tag: flags=0x06, name="regionPPN", data=paddr as u64
487            let name = b"regionPPN";
488            out.push(0x06); // flags: large data with explicit size
489            out.push(name.len() as u8);
490            out.extend_from_slice(name);
491            out.extend_from_slice(&8u32.to_le_bytes()); // data_length = 8
492            out.extend_from_slice(&paddr.to_le_bytes());
493
494            // regionBytes tag: flags=0x06, name="regionBytes", data=raw bytes
495            let name = b"regionBytes";
496            out.push(0x06); // flags: large data with explicit size
497            out.push(name.len() as u8);
498            out.extend_from_slice(name);
499            let data_len = data.len() as u32;
500            out.extend_from_slice(&data_len.to_le_bytes()); // data_length
501            out.extend_from_slice(data);
502        }
503
504        // Tag terminator
505        out.push(0x00);
506
507        // Fill "memory" group entry
508        {
509            let entry_offset = groups_start;
510            let name = b"memory";
511            out[entry_offset..entry_offset + name.len()].copy_from_slice(name);
512            // name is null-terminated, rest of 64 bytes is already zero
513            let tags_off_pos = entry_offset + 64;
514            out[tags_off_pos..tags_off_pos + 8].copy_from_slice(&memory_tags_offset.to_le_bytes());
515            // padding 8 bytes already zero
516        }
517
518        // --- Optional "cpu" group ---
519        if let Some(cr3_val) = self.cr3 {
520            let cpu_tags_offset = out.len() as u64;
521
522            // CR3 tag: flags=0x46 (indexed + 8-byte data)
523            let name = b"CR3";
524            out.push(0x46); // flags
525            out.push(name.len() as u8);
526            out.extend_from_slice(name);
527            out.push(0x00); // index0 = CPU 0
528            out.push(0x03); // index1 = CR register 3
529            out.extend_from_slice(&cr3_val.to_le_bytes());
530
531            // Tag terminator
532            out.push(0x00);
533
534            // Fill "cpu" group entry (second entry)
535            let entry_offset = groups_start + group_entry_size;
536            let name = b"cpu";
537            out[entry_offset..entry_offset + name.len()].copy_from_slice(name);
538            let tags_off_pos = entry_offset + 64;
539            out[tags_off_pos..tags_off_pos + 8].copy_from_slice(&cpu_tags_offset.to_le_bytes());
540        }
541
542        out
543    }
544}

546/// Compress data using `lzxpress::data::compress`, falling back to a
547/// literal-only Xpress encoding if the library's round-trip is broken
548/// for the given input.
549///
550/// The literal-only format is: every 32 bytes of input get a 4-byte flags
551/// word (all zeros = all literals) followed by the 32 raw bytes.
552fn xpress_compress_safe(data: &[u8]) -> Vec<u8> {
553    // Try the library compressor first.
554    if let Ok(compressed) = lzxpress::data::compress(data) {
555        // Verify round-trip; the library has known issues with certain patterns.
556        if let Ok(decompressed) = lzxpress::data::decompress(&compressed) {
557            if decompressed == data {
558                return compressed;
559            }
560        }
561    }
562
563    // Fallback: produce literal-only Xpress output.
564    // Format: repeated blocks of [flags_u32_le=0x00000000][32 literal bytes].
565    // A flags word of 0 means all 32 bits are 0, so all 32 symbols are literals.
566    let mut out = Vec::with_capacity(data.len() + data.len() / 32 * 4 + 8);
567    let mut pos = 0;
568    while pos < data.len() {
569        let chunk_len = (data.len() - pos).min(32);
570        // flags = 0 means every bit is 0 = literal
571        out.extend_from_slice(&0u32.to_le_bytes());
572        out.extend_from_slice(&data[pos..pos + chunk_len]);
573        pos += chunk_len;
574    }
575    out
576}

578/// Build a synthetic Windows hibernation file (`hiberfil.sys`).
579///
580/// Produces a `PO_MEMORY_IMAGE` header with "hibr" magic, processor state
581/// page with CR3, a page table page with PFN entries, and Xpress LZ77
582/// compressed data blocks.
583///
584/// Layout:
585/// - Page 0 (offset 0x0000): PO\_MEMORY\_IMAGE header with "hibr" magic at 0x00,
586///   `LengthSelf` at 0x0C (256 = 64-bit), `FirstTablePage` at 0x68 (value 2).
587/// - Page 1 (offset 0x1000): Processor state page with CR3 at offset 0x28.
588/// - Page 2 (offset 0x2000): Page table -- array of PFN entries (u64),
589///   terminated by `0xFFFF_FFFF_FFFF_FFFF`.
590/// - After header pages: Xpress LZ77 compressed blocks.
591///   Block header: sig(8) + num\_pages\_minus\_1(1) + compressed\_size\_field(3 bytes)
592///   + padding to 0x20 + compressed data.
593pub struct HiberfilBuilder {
594    pages: Vec<(u64, [u8; 4096])>,
595    cr3: u64,
596}
597
598impl Default for HiberfilBuilder {
599    fn default() -> Self {
600        Self {
601            pages: Vec::new(),
602            cr3: 0x1ab000,
603        }
604    }
605}
606
607impl HiberfilBuilder {
608    /// Create a new builder with default CR3 (`0x1ab000`).
609    pub fn new() -> Self {
610        Self::default()
611    }
612
613    /// Set the CR3 / `DirectoryTableBase` value stored in the processor state page.
614    pub fn cr3(mut self, cr3: u64) -> Self {
615        self.cr3 = cr3;
616        self
617    }
618
619    /// Add a physical memory page at the given PFN.
620    pub fn add_page(mut self, pfn: u64, data: &[u8; 4096]) -> Self {
621        self.pages.push((pfn, *data));
622        self
623    }
624
625    /// Build the complete hibernation file as a byte vector.
626    pub fn build(self) -> Vec<u8> {
627        const PAGE_SIZE: usize = 4096;
628        const HIBR_MAGIC: u32 = 0x7262_6968; // "hibr" LE
629        const LENGTH_SELF_64: u32 = 256;
630        const XPRESS_SIG: [u8; 8] = [0x81, 0x81, b'x', b'p', b'r', b'e', b's', b's'];
631        const BLOCK_HEADER_SIZE: usize = 0x20;
632
633        let mut out = vec![0u8; 3 * PAGE_SIZE]; // pages 0, 1, 2
634
635        // --- Page 0: PO_MEMORY_IMAGE header ---
636        // Magic "hibr" at offset 0x00
637        out[0x00..0x04].copy_from_slice(&HIBR_MAGIC.to_le_bytes());
638        // LengthSelf at offset 0x0C: 256 indicates 64-bit
639        out[0x0C..0x10].copy_from_slice(&LENGTH_SELF_64.to_le_bytes());
640        // FirstTablePage at offset 0x68: page 2
641        out[0x68..0x70].copy_from_slice(&2u64.to_le_bytes());
642
643        // --- Page 1: Processor state ---
644        // CR3 at offset 0x28 within page 1
645        let cr3_offset = PAGE_SIZE + 0x28;
646        out[cr3_offset..cr3_offset + 8].copy_from_slice(&self.cr3.to_le_bytes());
647
648        // --- Page 2: Page table ---
649        // Array of PFN entries (u64), terminated by sentinel 0xFFFFFFFFFFFFFFFF
650        let table_base = 2 * PAGE_SIZE;
651        let mut table_offset = 0usize;
652        for (pfn, _) in &self.pages {
653            out[table_base + table_offset..table_base + table_offset + 8]
654                .copy_from_slice(&pfn.to_le_bytes());
655            table_offset += 8;
656        }
657        // Sentinel terminator
658        out[table_base + table_offset..table_base + table_offset + 8]
659            .copy_from_slice(&u64::MAX.to_le_bytes());
660
661        // --- Compressed data blocks ---
662        // Each page gets its own Xpress block.
663        for (_, page_data) in &self.pages {
664            let compressed = xpress_compress_safe(page_data);
665
666            // compressed_size_field = (compressed_len * 4) - 1, stored as 3 bytes LE
667            let compressed_size_field = (compressed.len() * 4 - 1) as u32;
668            let num_pages_minus_1: u8 = 0; // single page per block
669
670            let mut block = Vec::new();
671            // 8-byte signature
672            block.extend_from_slice(&XPRESS_SIG);
673            // 1 byte: num_pages_minus_1
674            block.push(num_pages_minus_1);
675            // 3 bytes: compressed_size_field (LE)
676            block.push(compressed_size_field as u8);
677            block.push((compressed_size_field >> 8) as u8);
678            block.push((compressed_size_field >> 16) as u8);
679            // Pad to BLOCK_HEADER_SIZE (0x20 = 32 bytes)
680            block.resize(BLOCK_HEADER_SIZE, 0);
681            // Compressed data
682            block.extend_from_slice(&compressed);
683
684            out.extend_from_slice(&block);
685        }
686
687        out
688    }
689}

826/// Build a synthetic kdump (makedumpfile) dump for testing.
827///
828/// Produces the `disk_dump_header` + `kdump_sub_header` + bitmaps + page
829/// descriptors + compressed page data layout used by makedumpfile and
830/// crash-utility.
831///
832/// File layout (block_size = 4096 by default):
833/// - Block 0: `disk_dump_header`
834/// - Block 1: `kdump_sub_header` (mostly zeros for test)
835/// - Blocks 2..2+N: 1st bitmap (valid PFNs)
836/// - Blocks 2+N..2+2N: 2nd bitmap (dumped PFNs)
837/// - After bitmaps: `page_desc[]` array (24 bytes each, block-aligned)
838/// - After descs: compressed page data
839///
840/// `disk_dump_header` (Block 0):
841/// - 0x00: signature "KDUMP   " (8 bytes, 3 trailing spaces)
842/// - 0x08: header_version = 6 (i32)
843/// - 0x0C: utsname (390 bytes = 6 fields * 65, zeros)
844/// - Aligned to 4 after utsname: block_size(i32) + sub_hdr_size(i32=1) +
845///   bitmap_blocks(u32) + max_mapnr(u32)
846///
847/// `page_desc` (24 bytes each):
848/// - offset: i64 (file offset of compressed data)
849/// - size: u32 (compressed size)
850/// - flags: u32 (compression method)
851/// - page_flags: u64 (kernel flags, 0 for test)
852pub struct KdumpBuilder {
853    pages: Vec<(u64, Vec<u8>)>,
854    compression: u32,
855    block_size: u32,
856}
857
858impl Default for KdumpBuilder {
859    fn default() -> Self {
860        Self {
861            pages: Vec::new(),
862            compression: 0x04, // snappy
863            block_size: 4096,
864        }
865    }
866}
867
868impl KdumpBuilder {
869    /// Create a new builder with defaults: block_size=4096, compression=snappy (0x04).
870    pub fn new() -> Self {
871        Self::default()
872    }
873
874    /// Set the block size (must be a power of 2, typically 4096).
875    pub fn block_size(mut self, bs: u32) -> Self {
876        self.block_size = bs;
877        self
878    }
879
880    /// Set the compression flags for page data.
881    /// - 0x00: uncompressed
882    /// - 0x01: zlib
883    /// - 0x04: snappy
884    /// - 0x20: zstd (stored as minimal zstd frame)
885    pub fn compression(mut self, flags: u32) -> Self {
886        self.compression = flags;
887        self
888    }
889
890    /// Add a physical page at the given PFN with `data` (must be exactly block_size bytes).
891    pub fn add_page(mut self, pfn: u64, data: &[u8]) -> Self {
892        self.pages.push((pfn, data.to_vec()));
893        self
894    }
895
896    /// Build the complete kdump file as a byte vector.
897    pub fn build(self) -> Vec<u8> {
898        let bs = self.block_size as usize;
899
900        // Determine max PFN for bitmap sizing.
901        let max_pfn = self
902            .pages
903            .iter()
904            .map(|(pfn, _)| *pfn)
905            .max()
906            .map_or(0, |p| p + 1);
907
908        // Bitmap: one bit per PFN up to max_pfn, padded to block_size.
909        let bitmap_bits = max_pfn as usize;
910        let bitmap_bytes_raw = bitmap_bits.div_ceil(8);
911        let bitmap_blocks = bitmap_bytes_raw.div_ceil(bs);
912        let bitmap_bytes = bitmap_blocks * bs;
913
914        // Build bitmaps: both bitmaps are identical (valid == dumped).
915        let mut bitmap = vec![0u8; bitmap_bytes];
916        for (pfn, _) in &self.pages {
917            let pfn = *pfn as usize;
918            if pfn / 8 < bitmap.len() {
919                bitmap[pfn / 8] |= 1 << (pfn % 8);
920            }
921        }
922
923        // Compress each page.
924        let mut compressed_pages: Vec<(u32, Vec<u8>)> = Vec::new(); // (flags, data)
925        for (_, page_data) in &self.pages {
926            let (flags, compressed) = self.compress_page(page_data);
927            compressed_pages.push((flags, compressed));
928        }
929
930        // Layout calculation:
931        // Block 0: disk_dump_header
932        // Block 1: kdump_sub_header
933        // Blocks 2..2+bitmap_blocks: 1st bitmap
934        // Blocks 2+bitmap_blocks..2+2*bitmap_blocks: 2nd bitmap
935        let desc_start_block = 2 + 2 * bitmap_blocks;
936        let desc_start = desc_start_block * bs;
937
938        // Sort pages by PFN for descriptor ordering.
939        let mut indexed_pages: Vec<(usize, u64)> = self
940            .pages
941            .iter()
942            .enumerate()
943            .map(|(i, (pfn, _))| (i, *pfn))
944            .collect();
945        indexed_pages.sort_by_key(|(_, pfn)| *pfn);
946
947        // page_desc array: 24 bytes each, block-aligned total.
948        let num_descs = indexed_pages.len();
949        let descs_raw_size = num_descs * 24;
950        let descs_padded = descs_raw_size.div_ceil(bs) * bs;
951        let data_start = desc_start + descs_padded;
952
953        // Compute file offsets for each page's compressed data.
954        let mut data_offsets: Vec<usize> = Vec::new();
955        let mut cur_offset = data_start;
956        for (orig_idx, _) in &indexed_pages {
957            data_offsets.push(cur_offset);
958            cur_offset += compressed_pages[*orig_idx].1.len();
959        }
960
961        // --- Assemble the file ---
962        let total_size = cur_offset;
963        let mut out = vec![0u8; total_size];
964
965        // Block 0: disk_dump_header
966        // Signature "KDUMP   " (8 bytes)
967        out[0x00..0x08].copy_from_slice(b"KDUMP   ");
968        // header_version = 6 (i32 LE)
969        out[0x08..0x0C].copy_from_slice(&6i32.to_le_bytes());
970        // utsname: 390 bytes of zeros (already zero)
971        // Aligned offset after utsname: (0x0C + 390 + 3) & !3 = 0x19C
972        let fields_off = (0x0C + 390 + 3) & !3; // 0x19C
973                                                // block_size (i32)
974        #[allow(clippy::cast_possible_wrap)]
975        let block_size_i32 = self.block_size as i32;
976        out[fields_off..fields_off + 4].copy_from_slice(&block_size_i32.to_le_bytes());
977        // sub_hdr_size (i32) = 1
978        out[fields_off + 4..fields_off + 8].copy_from_slice(&1i32.to_le_bytes());
979        // bitmap_blocks (u32)
980        out[fields_off + 8..fields_off + 12].copy_from_slice(&(bitmap_blocks as u32).to_le_bytes());
981        // max_mapnr (u32)
982        out[fields_off + 12..fields_off + 16].copy_from_slice(&(max_pfn as u32).to_le_bytes());
983
984        // Block 1: kdump_sub_header (all zeros, already done)
985
986        // Blocks 2..2+N: 1st bitmap
987        let bm1_start = 2 * bs;
988        out[bm1_start..bm1_start + bitmap.len()].copy_from_slice(&bitmap);
989
990        // Blocks 2+N..2+2N: 2nd bitmap (same as 1st)
991        let bm2_start = (2 + bitmap_blocks) * bs;
992        out[bm2_start..bm2_start + bitmap.len()].copy_from_slice(&bitmap);
993
994        // Page descriptors
995        for (desc_idx, (orig_idx, _)) in indexed_pages.iter().enumerate() {
996            let d_off = desc_start + desc_idx * 24;
997            let (flags, ref compressed) = compressed_pages[*orig_idx];
998            // offset: i64
999            out[d_off..d_off + 8].copy_from_slice(
1000                &{
1001                    #[allow(clippy::cast_possible_wrap)]
1002                    let offset_i64 = data_offsets[desc_idx] as i64;
1003                    offset_i64
1004                }
1005                .to_le_bytes(),
1006            );
1007            // size: u32
1008            out[d_off + 8..d_off + 12].copy_from_slice(&(compressed.len() as u32).to_le_bytes());
1009            // flags: u32
1010            out[d_off + 12..d_off + 16].copy_from_slice(&flags.to_le_bytes());
1011            // page_flags: u64 = 0 (already zero)
1012        }
1013
1014        // Compressed page data
1015        for (desc_idx, (orig_idx, _)) in indexed_pages.iter().enumerate() {
1016            let offset = data_offsets[desc_idx];
1017            let data = &compressed_pages[*orig_idx].1;
1018            out[offset..offset + data.len()].copy_from_slice(data);
1019        }
1020
1021        out
1022    }
1023
1024    /// Compress a single page based on the configured compression method.
1025    /// Returns (flags, compressed_data).
1026    fn compress_page(&self, data: &[u8]) -> (u32, Vec<u8>) {
1027        match self.compression {
1028            0x00 => {
1029                // Uncompressed: flags=0, data is raw
1030                (0x00, data.to_vec())
1031            }
1032            0x01 => {
1033                // Zlib
1034                use std::io::Write;
1035                let mut encoder =
1036                    flate2::write::ZlibEncoder::new(Vec::new(), flate2::Compression::default());
1037                encoder.write_all(data).expect("zlib compress write");
1038                let compressed = encoder.finish().expect("zlib compress finish");
1039                (0x01, compressed)
1040            }
1041            0x04 => {
1042                // Snappy
1043                let mut encoder = snap::raw::Encoder::new();
1044                let compressed = encoder.compress_vec(data).expect("snappy compress");
1045                (0x04, compressed)
1046            }
1047            0x20 => {
1048                // Zstd: produce a minimal valid zstd frame.
1049                let compressed = Self::zstd_compress_minimal(data);
1050                (0x20, compressed)
1051            }
1052            _ => (0x00, data.to_vec()),
1053        }
1054    }
1055
1056    /// Produce a minimal valid zstd frame that stores raw (uncompressed) data.
1057    ///
1058    /// Zstd frame format (minimal raw block):
1059    /// - 4-byte magic: 0xFD2FB528
1060    /// - Frame header descriptor byte
1061    /// - Window descriptor byte
1062    /// - Block header: 3 bytes (last_block=1, block_type=raw, block_size)
1063    /// - Raw data
1064    fn zstd_compress_minimal(data: &[u8]) -> Vec<u8> {
1065        let mut out = Vec::new();
1066        // Magic number
1067        out.extend_from_slice(&0xFD2FB528u32.to_le_bytes());
1068        // Frame header descriptor: 0x00
1069        // single_segment=0, so window descriptor is present
1070        // no checksum, no dict_id, fcs_field_size=0
1071        out.push(0x00);
1072        // Window descriptor: exponent=0, mantissa=0 => window_size = 1 << 10 = 1024
1073        out.push(0x00);
1074        // Block header: 3 bytes LE
1075        // bit 0: last_block = 1
1076        // bits 2-1: block_type = 0 (raw)
1077        // bits 23-3: block_size = data.len()
1078        let block_header: u32 = 1 | ((data.len() as u32) << 3);
1079        let bh_bytes = block_header.to_le_bytes();
1080        out.extend_from_slice(&bh_bytes[..3]);
1081        // Raw data
1082        out.extend_from_slice(data);
1083        out
1084    }
