6/// Builds a minimal ISF JSON document for testing.
7#[derive(Default)]
8pub struct IsfBuilder {
9    structs: HashMap<String, IsfStruct>,
10    symbols: HashMap<String, u64>,
11    base_types: HashMap<String, (u64, bool)>,
12}
13
14struct IsfStruct {
15    size: u64,
16    fields: Vec<(String, u64, String)>, // (name, offset, type_name)
17}
18
19impl IsfBuilder {
20    /// Create a new builder with common base types pre-registered.
21    pub fn new() -> Self {
22        let mut b = Self::default();
23        b.base_types.insert("int".into(), (4, true));
24        b.base_types.insert("unsigned int".into(), (4, false));
25        b.base_types.insert("long".into(), (8, true));
26        b.base_types.insert("unsigned long".into(), (8, false));
27        b.base_types.insert("short".into(), (2, true));
28        b.base_types.insert("unsigned short".into(), (2, false));
29        b.base_types.insert("char".into(), (1, true));
30        b.base_types.insert("unsigned char".into(), (1, false));
31        b.base_types.insert("pointer".into(), (8, false));
32        b
33    }

35    /// Add a struct type with its total size.
36    pub fn add_struct(mut self, name: &str, size: u64) -> Self {
37        self.structs.insert(
38            name.into(),
39            IsfStruct {
40                size,
41                fields: Vec::new(),
42            },
43        );
44        self
45    }
46
47    /// Add a field to a struct.
48    pub fn add_field(
49        mut self,
50        struct_name: &str,
51        field_name: &str,
52        offset: u64,
53        type_name: &str,
54    ) -> Self {
55        self.structs
56            .get_mut(struct_name)
57            .unwrap_or_else(|| panic!("struct {struct_name} not found"))
58            .fields
59            .push((field_name.into(), offset, type_name.into()));
60        self
61    }
62
63    /// Add a kernel symbol with its virtual address.
64    pub fn add_symbol(mut self, name: &str, address: u64) -> Self {
65        self.symbols.insert(name.into(), address);
66        self
67    }

69    /// Build the ISF JSON as a `serde_json::Value`.
70    pub fn build_json(&self) -> Value {
71        let mut base_types = serde_json::Map::new();
72        for (name, (size, signed)) in &self.base_types {
73            base_types.insert(
74                name.clone(),
75                json!({
76                    "size": size,
77                    "signed": signed,
78                    "kind": "int",
79                    "endian": "little"
80                }),
81            );
82        }
83
84        let mut user_types = serde_json::Map::new();
85        for (name, s) in &self.structs {
86            let mut fields = serde_json::Map::new();
87            for (fname, offset, tname) in &s.fields {
88                fields.insert(
89                    fname.clone(),
90                    json!({
91                        "offset": offset,
92                        "type": {
93                            "kind": "base",
94                            "name": tname
95                        }
96                    }),
97                );
98            }
99            user_types.insert(
100                name.clone(),
101                json!({
102                    "size": s.size,
103                    "fields": fields
104                }),
105            );
106        }
107
108        let mut symbols = serde_json::Map::new();
109        for (name, addr) in &self.symbols {
110            symbols.insert(name.clone(), json!({ "address": addr }));
111        }
112
113        json!({
114            "metadata": {
115                "format": "6.2.0",
116                "producer": {
117                    "name": "memf-test",
118                    "version": "0.1.0"
119                }
120            },
121            "base_types": base_types,
122            "user_types": user_types,
123            "enums": {},
124            "symbols": symbols
125        })
126    }
127
128    /// Build the ISF JSON as a byte vector (UTF-8 encoded).
129    pub fn build_bytes(&self) -> Vec<u8> {
130        serde_json::to_vec_pretty(&self.build_json()).expect("JSON serialization")
131    }
