5//! Provides the [`SymbolResolver`] trait and concrete backends:
6//! - ISF JSON (Volatility 3-compatible symbol tables)

12//! use memf_symbols::AutoProfile;
13//!
14//! # fn example() -> memf_symbols::Result<()> {
15//! // Scan a physical dump, download the matching PDB, return a resolver:
16//! // let dump = memf_format::open_dump("win10.dmp")?;
17//! // let resolver = AutoProfile::new()?.from_dump(dump.as_ref())?;

42/// Error type for memf-symbols operations.
43#[derive(Debug, thiserror::Error)]
44pub enum Error {
45    /// I/O error reading symbol files.
46    #[error("I/O error: {0}")]
47    Io(#[from] std::io::Error),
48
49    /// JSON deserialization error.
50    #[error("JSON error: {0}")]
51    Json(#[from] serde_json::Error),
52
53    /// Symbol file is malformed or unsupported.
54    #[error("malformed symbol data: {0}")]
55    Malformed(String),
56
57    /// Requested symbol or type not found.
58    #[error("symbol not found: {0}")]
59    NotFound(String),
60
61    /// PDB parsing error.
62    #[error("PDB error: {0}")]
63    Pdb(String),
64
65    /// Network/download error.
66    #[error("network error: {0}")]
67    Network(String),
68
69    /// Cache I/O error.
70    #[error("cache error: {0}")]
71    Cache(String),
72}
73
74impl From<pdb::Error> for Error {
75    fn from(e: pdb::Error) -> Self {
76        Self::Pdb(e.to_string())
77    }
78}

83/// Information about a struct field from the symbol table.
84#[derive(Debug, Clone)]
85pub struct FieldInfo {
86    /// Byte offset of this field within the struct.
87    pub offset: u64,
88    /// Name of the field's type (e.g., "unsigned int", "list_head").
89    pub type_name: String,
90}
91
92/// Information about a struct type from the symbol table.
93#[derive(Debug, Clone)]
94pub struct StructInfo {
95    /// Total size of the struct in bytes.
96    pub size: u64,
97    /// Fields keyed by field name.
98    pub fields: HashMap<String, FieldInfo>,
99}

101/// A provider of symbol/type information for kernel analysis.
102///
103/// Backends implement this trait to resolve struct layouts, field offsets,
104/// and symbol addresses from various sources (ISF JSON, BTF, DWARF, etc.).
105pub trait SymbolResolver: Send + Sync {
106    /// Return the byte offset of `field_name` within `struct_name`.
107    fn field_offset(&self, struct_name: &str, field_name: &str) -> Option<u64>;
108
109    /// Return the total size in bytes of `struct_name`.
110    fn struct_size(&self, struct_name: &str) -> Option<u64>;
111
112    /// Return the virtual address of a kernel symbol by name.
113    fn symbol_address(&self, symbol_name: &str) -> Option<u64>;
114
115    /// Return full struct information including all fields.
116    fn struct_info(&self, struct_name: &str) -> Option<StructInfo>;
117
118    /// Human-readable name for this backend (e.g., "ISF JSON", "BTF").
119    fn backend_name(&self) -> &str;
120
121    /// Clone this resolver into a new boxed trait object.
122    fn clone_boxed(&self) -> Box<dyn SymbolResolver>;
123}
124
125/// A [`SymbolResolver`] decorator that rebases global symbol addresses by a
126/// kernel image base virtual address (KVO). Windows ISF stores symbols as RVAs;
127/// wrapping the resolver here means **every** `symbol_address` caller receives a
128/// real virtual address by construction — there is no way to obtain an
129/// un-rebased RVA (secure-by-design). Struct/field offsets are image-relative
130/// and pass through unchanged.
131pub struct RebasedResolver {
132    inner: Box<dyn SymbolResolver>,
133    kernel_base: u64,
134}
135
136impl RebasedResolver {
137    /// Wrap `inner`, adding `kernel_base` to every resolved symbol address.
138    #[must_use]
139    pub fn new(inner: Box<dyn SymbolResolver>, kernel_base: u64) -> Self {
140        Self { inner, kernel_base }
141    }
142}
143
144impl SymbolResolver for RebasedResolver {
145    fn field_offset(&self, struct_name: &str, field_name: &str) -> Option<u64> {
146        self.inner.field_offset(struct_name, field_name)
147    }
148    fn struct_size(&self, struct_name: &str) -> Option<u64> {
149        self.inner.struct_size(struct_name)
150    }
151    fn symbol_address(&self, symbol_name: &str) -> Option<u64> {
152        self.inner
153            .symbol_address(symbol_name)
154            .map(|rva| self.kernel_base.wrapping_add(rva))
155    }
156    fn struct_info(&self, struct_name: &str) -> Option<StructInfo> {
157        self.inner.struct_info(struct_name)
158    }
159    fn backend_name(&self) -> &str {
160        self.inner.backend_name()
161    }
162    fn clone_boxed(&self) -> Box<dyn SymbolResolver> {
163        Box::new(RebasedResolver {
164            inner: self.inner.clone_boxed(),
165            kernel_base: self.kernel_base,
166        })
167    }
168}

187            (s == "_EPROCESS").then_some(0x500)

193            None

203    /// The decorator rebases `symbol_address` by the KVO but leaves field/struct
204    /// offsets (image-relative) untouched — so the EPROCESS walk dereferences a
205    /// real VA, not the bare `0x2b00a0` RVA.
206    #[test]
207    fn rebased_resolver_adds_base_to_symbols_only() {
208        let r = RebasedResolver::new(Box::new(MockResolver), 0xFFFF_F800_CBE0_0000);
209        assert_eq!(
210            r.symbol_address("PsActiveProcessHead"),
211            Some(0xFFFF_F800_CC0B_00A0)
212        );
213        assert_eq!(r.field_offset("_EPROCESS", "UniqueProcessId"), Some(0x2e0));
214        assert_eq!(r.struct_size("_EPROCESS"), Some(0x500));
215        assert_eq!(r.symbol_address("Missing"), None);
216    }

411        assert!(cache
412            .to_string_lossy()
