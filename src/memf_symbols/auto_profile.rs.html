1//! Automatic profile resolution for Windows kernel memory dumps.
2//!
3//! [`AutoProfile`] orchestrates PDB acquisition: it accepts a PDB identity,
4//! a raw PE image, or a physical memory dump, and returns a [`SymbolResolver`]
5//! backed by the matching PDB file.
6
7use std::path::PathBuf;
8
9use crate::pdb_resolver::PdbResolver;
10use crate::pe_debug::{extract_pdb_id, PdbId};
11use crate::symserver::{self, SymbolServerClient};
12use crate::SymbolResolver;
13
14/// Automatic symbol profile resolver for Windows kernels.
15///
16/// Locates and loads the correct PDB file given a kernel PDB identity,
17/// a PE binary, or a physical memory dump.
18pub struct AutoProfile {
19    /// Local directory used as the PDB cache.
20    pub(crate) cache_dir: PathBuf,
21    /// Whether a cache miss may fetch the PDB from the network symbol server.
22    /// Enabled by default; disable (`with_network(false)`) for air-gapped work.
23    network_enabled: bool,
24}
25
26impl AutoProfile {
27    /// Create an `AutoProfile` using the default cache directory (`~/.memf/symbols/`).
28    pub fn new() -> crate::Result<Self> {
29        let cache_dir = crate::symserver::default_cache_dir()
30            .ok_or_else(|| crate::Error::Cache("HOME not set".into()))?;
31        Ok(Self {
32            cache_dir,
33            network_enabled: true,
34        })
35    }
36
37    /// Create an `AutoProfile` with a custom cache directory.
38    pub fn with_cache_dir(dir: impl Into<PathBuf>) -> Self {
39        Self {
40            cache_dir: dir.into(),
41            network_enabled: true,
42        }
43    }
44
45    /// Enable or disable network symbol-server download (default: **enabled**).
46    ///
47    /// Pass `false` for air-gapped / offline forensic analysis: a cache miss
48    /// then returns [`crate::Error::NotFound`] instead of reaching the network.
49    /// This is a runtime setting, never a compile-time feature — the capability
50    /// is always present.
51    #[must_use]
52    pub fn with_network(mut self, enabled: bool) -> Self {
53        self.network_enabled = enabled;
54        self
55    }
56
57    /// Whether network symbol-server download is currently enabled.
58    #[must_use]
59    pub fn network_enabled(&self) -> bool {
60        self.network_enabled
61    }
62
63    /// Resolve a [`SymbolResolver`] for the given PDB identity.
64    ///
65    /// Checks the local cache first. On a miss, downloads the PDB from the
66    /// Microsoft symbol server **iff** network is enabled; otherwise returns
67    /// [`crate::Error::NotFound`] without touching the network.
68    pub fn from_pdb_id(&self, pdb_id: &PdbId) -> crate::Result<Box<dyn SymbolResolver>> {
69        let cached =
70            symserver::cache_path(&self.cache_dir, &pdb_id.pdb_name, &pdb_id.guid, pdb_id.age);
71
72        if !cached.exists() {
73            if !self.network_enabled {
74                return Err(crate::Error::NotFound(format!(
75                    "{} not cached and network symbol-server disabled (offline mode)",
76                    pdb_id.pdb_name
77                )));
78            }
79            let client = SymbolServerClient::new(symserver::default_server_url(), &self.cache_dir);
80            client.get_pdb(&pdb_id.pdb_name, &pdb_id.guid, pdb_id.age)?;
81        }
82
83        let resolver = PdbResolver::from_path(&cached)?;
84        Ok(Box::new(resolver))
85    }
86
87    /// Resolve a [`SymbolResolver`] by parsing the PDB identity from a PE binary.
88    ///
89    /// Extracts the PDB identity from `pe_bytes` then delegates to [`from_pdb_id`].
90    ///
91    /// [`from_pdb_id`]: AutoProfile::from_pdb_id
92    pub fn from_pe_bytes(&self, pe_bytes: &[u8]) -> crate::Result<Box<dyn SymbolResolver>> {
93        let pdb_id = extract_pdb_id(pe_bytes)?;
94        self.from_pdb_id(&pdb_id)
95    }
96
97    /// Resolve a [`SymbolResolver`] by scanning physical memory for the kernel.
98    ///
99    /// Scans `mem` for ntoskrnl.exe, extracts its PDB identity, then delegates
100    /// to [`from_pdb_id`].
101    ///
102    /// [`from_pdb_id`]: AutoProfile::from_pdb_id
103    pub fn from_dump<P: memf_format::PhysicalMemoryProvider>(
104        &self,
105        mem: &P,
106    ) -> crate::Result<Box<dyn SymbolResolver>> {
107        let pdb_id = crate::kernel_scanner::scan_for_kernel(mem)?;
108        self.from_pdb_id(&pdb_id)
109    }
110}

112#[cfg(test)]
113mod tests {
114    use super::*;
115
116    fn test_pdb_id() -> PdbId {
117        PdbId {
118            guid: "1B72224D-37B8-1792-2820-0ED8994498B2".into(),
119            age: 1,
120            pdb_name: "ntkrnlmp.pdb".into(),
121        }
122    }
123
124    /// AutoProfile::with_cache_dir stores the provided path.
125    #[test]
126    fn with_cache_dir_stores_path() {
127        let tmp = tempfile::tempdir().unwrap();
128        let profile = AutoProfile::with_cache_dir(tmp.path());
129        assert_eq!(profile.cache_dir, tmp.path());
130    }
131
132    /// Network symbol-server download is a RUNTIME setting, ENABLED by default
133    /// (forensic tool: full capability by default; examiners opt out at runtime
134    /// for air-gapped work — never a compile-time feature).
135    #[test]
136    fn network_enabled_by_default() {
137        let tmp = tempfile::tempdir().unwrap();
138        assert!(AutoProfile::with_cache_dir(tmp.path()).network_enabled());
139    }
140
141    /// In offline mode (`with_network(false)`) a cache miss returns NotFound
142    /// WITHOUT attempting the network — deterministic, no phone-home.
143    #[test]
144    fn offline_mode_cache_miss_returns_notfound_without_network() {
145        let tmp = tempfile::tempdir().unwrap();
146        let profile = AutoProfile::with_cache_dir(tmp.path()).with_network(false);
147        assert!(!profile.network_enabled());
148        let result = profile.from_pdb_id(&test_pdb_id());
149        assert!(
150            matches!(result, Err(crate::Error::NotFound(_))),
151            "offline cache-miss should be NotFound, got: {:?}",
152            result.err().map(|e| e.to_string())
153        );
154    }
155
156    /// Stub returns an error when cache is empty and no network is available.
157    #[test]
158    fn from_pdb_id_returns_error_when_cache_empty_and_no_network() {
159        let tmp = tempfile::tempdir().unwrap();
160        let profile = AutoProfile::with_cache_dir(tmp.path());
161        let result = profile.from_pdb_id(&test_pdb_id());
162        assert!(result.is_err(), "expected error from stub, got Ok");
163    }
164
165    /// Garbage PE bytes produce a Malformed error (from extract_pdb_id).
166    #[test]
167    fn from_pe_bytes_propagates_malformed_error() {
168        let tmp = tempfile::tempdir().unwrap();
169        let profile = AutoProfile::with_cache_dir(tmp.path());
170        let result = profile.from_pe_bytes(b"not a PE");
171        assert!(
172            matches!(result, Err(crate::Error::Malformed(_))),
173            "expected Malformed error, got: {}",
174            result.err().map(|e| e.to_string()).unwrap_or_default()
175        );
176    }
177
178    /// RED test: when a cached PDB file exists the real implementation should
179    /// attempt to open it (yielding Pdb/Malformed on a zero-byte file), NOT
180    /// NotFound.  The stub returns NotFound, so this test FAILS in RED.
181    #[test]
182    fn from_pdb_id_uses_cached_pdb_when_present() {
183        let tmp = tempfile::tempdir().unwrap();
184        let pdb_id = test_pdb_id();
185
186        // Pre-populate the cache with a zero-byte file at the expected path.
187        let cached =
188            crate::symserver::cache_path(tmp.path(), &pdb_id.pdb_name, &pdb_id.guid, pdb_id.age);
189        std::fs::create_dir_all(cached.parent().unwrap()).unwrap();
190        std::fs::write(&cached, b"").unwrap();
191
192        let profile = AutoProfile::with_cache_dir(tmp.path());
193        let result = profile.from_pdb_id(&pdb_id);
194
195        // With a real implementation the cache-hit path opens the file and
196        // gets Pdb or Malformed (zero-byte PDB).  The stub returns NotFound,
197        // which is neither — so this assertion fails in RED.
198        assert!(
199            matches!(
200                result,
201                Err(crate::Error::Pdb(_) | crate::Error::Malformed(_))
202            ),
203            "expected Pdb or Malformed (cache-hit path), got: {}",
204            result.err().map(|e| e.to_string()).unwrap_or_default()
205        );
206    }
