1//! PE debug info extraction.
2//!
3//! Parses CodeView RSDS records from PE debug directories to extract
4//! PDB identification (GUID, age, PDB filename) needed for symbol
5//! server downloads.
6
7/// PDB identification extracted from a PE debug directory.
8#[derive(Debug, Clone, PartialEq, Eq)]
9pub struct PdbId {
10    /// GUID as uppercase hex with dashes (e.g., "1B72224D-37B8-1792-2820-0ED8994498B2").
11    pub guid: String,
12    /// PDB age counter.
13    pub age: u32,
14    /// PDB filename (e.g., "ntkrnlmp.pdb").
15    pub pdb_name: String,
16}
17
18/// Format a 16-byte mixed-endian GUID as uppercase hex with dashes.
19///
20/// The GUID is stored in mixed-endian format:
21/// - Bytes 0-3: `Data1` (little-endian u32)
22/// - Bytes 4-5: `Data2` (little-endian u16)
23/// - Bytes 6-7: `Data3` (little-endian u16)
24/// - Bytes 8-15: `Data4` (big-endian, raw bytes)
25fn format_guid(bytes: &[u8; 16]) -> String {
26    let d1 = u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
27    let d2 = u16::from_le_bytes([bytes[4], bytes[5]]);
28    let d3 = u16::from_le_bytes([bytes[6], bytes[7]]);
29    format!(
30        "{:08X}-{:04X}-{:04X}-{:02X}{:02X}-{:02X}{:02X}{:02X}{:02X}{:02X}{:02X}",
31        d1,
32        d2,
33        d3,
34        bytes[8],
35        bytes[9],
36        bytes[10],
37        bytes[11],
38        bytes[12],
39        bytes[13],
40        bytes[14],
41        bytes[15],
42    )
43}

45/// Extract PDB identification from a PE binary.
46///
47/// Parses the PE debug directory to find a CodeView RSDS record containing
48/// the GUID, age, and PDB filename needed to download the matching PDB
49/// from a symbol server.
50pub fn extract_pdb_id(pe_bytes: &[u8]) -> crate::Result<PdbId> {
51    use goblin::pe::PE;
52
53    let pe = PE::parse(pe_bytes).map_err(|e| crate::Error::Malformed(format!("PE parse: {e}")))?;
54
55    let debug_data = pe
56        .debug_data
57        .ok_or_else(|| crate::Error::Malformed("no debug data".into()))?;
58
59    let cv = debug_data
60        .codeview_pdb70_debug_info
61        .ok_or_else(|| crate::Error::Malformed("no CodeView PDB70 debug info".into()))?;
62
63    let guid = format_guid(&cv.signature);
64
65    // Convert filename bytes to string, stripping null terminator.
66    let raw_name = std::str::from_utf8(cv.filename)
67        .map_err(|e| crate::Error::Malformed(format!("PDB filename not UTF-8: {e}")))?
68        .trim_end_matches('\0');
69
70    // Extract just the filename component (strip any directory path).
71    let pdb_name = std::path::Path::new(raw_name)
72        .file_name()
73        .and_then(|n| n.to_str())
74        .unwrap_or(raw_name)
75        .to_string();
76
77    Ok(PdbId {
78        guid,
79        age: cv.age,
80        pdb_name,
81    })
82}

88    /// Build a minimal PE (PE32+/AMD64) with an embedded CodeView RSDS debug record.
89    ///
90    /// The resulting buffer is a valid PE that goblin can parse, containing:
91    /// - DOS header with e_lfanew at 0x80
92    /// - PE signature + COFF header (AMD64, 1 section)
93    /// - PE32+ optional header with debug data directory
94    /// - One .rdata section mapping RVA 0x200 to file offset 0x200
95    /// - IMAGE_DEBUG_DIRECTORY pointing to CodeView data
96    /// - CodeView RSDS record with the given GUID, age, and PDB filename
97    fn build_pe_with_debug(guid_bytes: [u8; 16], age: u32, pdb_name: &str) -> Vec<u8> {
98        let mut buf = vec![0u8; 4096];
99
100        // DOS header
101        buf[0] = b'M';
102        buf[1] = b'Z';
103        let pe_offset: u32 = 0x80;
104        buf[0x3C..0x40].copy_from_slice(&pe_offset.to_le_bytes());
105
106        let mut pos = pe_offset as usize;
107
108        // PE signature
109        buf[pos..pos + 4].copy_from_slice(b"PE\0\0");
110        pos += 4;
111
112        // COFF header (20 bytes)
113        buf[pos..pos + 2].copy_from_slice(&0x8664u16.to_le_bytes()); // Machine: AMD64
114        buf[pos + 2..pos + 4].copy_from_slice(&1u16.to_le_bytes()); // NumberOfSections: 1
115        let optional_header_size: u16 = 240; // PE32+ optional header size (112 + 16*8)
116        buf[pos + 16..pos + 18].copy_from_slice(&optional_header_size.to_le_bytes());
117        buf[pos + 18..pos + 20].copy_from_slice(&0x0022u16.to_le_bytes()); // Characteristics
118        pos += 20;
119
120        // Optional header (PE32+)
121        let opt_start = pos;
122        buf[pos..pos + 2].copy_from_slice(&0x020Bu16.to_le_bytes()); // PE32+ magic
123
124        // SectionAlignment at offset 32 from opt_start
125        buf[opt_start + 32..opt_start + 36].copy_from_slice(&0x1000u32.to_le_bytes());
126        // FileAlignment at offset 36 from opt_start (must be power of 2)
127        buf[opt_start + 36..opt_start + 40].copy_from_slice(&0x200u32.to_le_bytes());
128        // SizeOfImage at offset 56 from opt_start
129        buf[opt_start + 56..opt_start + 60].copy_from_slice(&0x2000u32.to_le_bytes());
130        // SizeOfHeaders at offset 60 from opt_start
131        buf[opt_start + 60..opt_start + 64].copy_from_slice(&0x200u32.to_le_bytes());
132
133        // NumberOfRvaAndSizes at offset 108 from opt_start
134        buf[opt_start + 108..opt_start + 112].copy_from_slice(&16u32.to_le_bytes());
135
136        // Data directories start at offset 112 from opt_start.
137        // Debug directory is index 6 -> offset 112 + 6*8 = 160 from opt_start.
138        let debug_dir_rva: u32 = 0x200;
139        let debug_dir_size: u32 = 28; // One IMAGE_DEBUG_DIRECTORY entry
140        buf[opt_start + 160..opt_start + 164].copy_from_slice(&debug_dir_rva.to_le_bytes());
141        buf[opt_start + 164..opt_start + 168].copy_from_slice(&debug_dir_size.to_le_bytes());
142
143        pos = opt_start + optional_header_size as usize;
144
145        // Section header (40 bytes) — .rdata covering our debug data area
146        buf[pos..pos + 8].copy_from_slice(b".rdata\0\0");
147        buf[pos + 8..pos + 12].copy_from_slice(&0x1000u32.to_le_bytes()); // VirtualSize
148        buf[pos + 12..pos + 16].copy_from_slice(&0x200u32.to_le_bytes()); // VirtualAddress
149        buf[pos + 16..pos + 20].copy_from_slice(&0x200u32.to_le_bytes()); // SizeOfRawData
150        buf[pos + 20..pos + 24].copy_from_slice(&0x200u32.to_le_bytes()); // PointerToRawData
151
152        // IMAGE_DEBUG_DIRECTORY at file offset 0x200
153        let debug_dir_offset = 0x200usize;
154        let cv_rva: u32 = 0x220;
155        let pdb_name_bytes = pdb_name.as_bytes();
156        let cv_size: u32 = (24 + pdb_name_bytes.len() + 1) as u32;
157
158        // Type field is at offset +12 in IMAGE_DEBUG_DIRECTORY
159        buf[debug_dir_offset + 12..debug_dir_offset + 16].copy_from_slice(&2u32.to_le_bytes()); // IMAGE_DEBUG_TYPE_CODEVIEW
160        buf[debug_dir_offset + 16..debug_dir_offset + 20].copy_from_slice(&cv_size.to_le_bytes()); // SizeOfData
161        buf[debug_dir_offset + 20..debug_dir_offset + 24].copy_from_slice(&cv_rva.to_le_bytes()); // AddressOfRawData
162        buf[debug_dir_offset + 24..debug_dir_offset + 28].copy_from_slice(&cv_rva.to_le_bytes()); // PointerToRawData (= RVA, 1:1 mapping)
163
164        // CodeView RSDS record at file offset 0x220
165        let cv_offset = 0x220usize;
166        buf[cv_offset..cv_offset + 4].copy_from_slice(b"RSDS");
167        buf[cv_offset + 4..cv_offset + 20].copy_from_slice(&guid_bytes);
168        buf[cv_offset + 20..cv_offset + 24].copy_from_slice(&age.to_le_bytes());
169        let name_start = cv_offset + 24;
170        buf[name_start..name_start + pdb_name_bytes.len()].copy_from_slice(pdb_name_bytes);
171        buf[name_start + pdb_name_bytes.len()] = 0; // null terminator
172
173        buf
174    }

227    // Known GUID bytes for testing:
228    // Stored as mixed-endian: Data1(LE) Data2(LE) Data3(LE) Data4(BE)
229    // Target: "1B72224D-37B8-1792-2820-0ED8994498B2"
230    //   Data1 = 0x1B72224D -> LE bytes: [0x4D, 0x22, 0x72, 0x1B]
231    //   Data2 = 0x37B8     -> LE bytes: [0xB8, 0x37]
232    //   Data3 = 0x1792     -> LE bytes: [0x92, 0x17]
233    //   Data4 = [0x28, 0x20, 0x0E, 0xD8, 0x99, 0x44, 0x98, 0xB2]
234    const TEST_GUID_BYTES: [u8; 16] = [
235        0x4D, 0x22, 0x72, 0x1B, // Data1 LE
236        0xB8, 0x37, // Data2 LE
237        0x92, 0x17, // Data3 LE
238        0x28, 0x20, 0x0E, 0xD8, 0x99, 0x44, 0x98, 0xB2, // Data4 BE
239    ];
240    const TEST_GUID_STR: &str = "1B72224D-37B8-1792-2820-0ED8994498B2";

251    #[test]
252    fn extract_pdb_id_guid_format() {
253        // Verify GUID has correct dash positions and uppercase hex.
254        let pe = build_pe_with_debug(TEST_GUID_BYTES, 1, "test.pdb");
255        let id = extract_pdb_id(&pe).expect("should parse");
256
257        let parts: Vec<&str> = id.guid.split('-').collect();
258        assert_eq!(parts.len(), 5, "GUID should have 5 dash-separated parts");
259        assert_eq!(parts[0].len(), 8, "Data1 should be 8 hex chars");
260        assert_eq!(parts[1].len(), 4, "Data2 should be 4 hex chars");
261        assert_eq!(parts[2].len(), 4, "Data3 should be 4 hex chars");
262        assert_eq!(parts[3].len(), 4, "Data4a should be 4 hex chars");
263        assert_eq!(parts[4].len(), 12, "Data4b should be 12 hex chars");
264
265        // Verify all uppercase hex
266        assert_eq!(id.guid, id.guid.to_uppercase());
267    }
