1//! PE debug info extraction.
2//!
3//! Parses CodeView RSDS records from PE debug directories to extract
4//! PDB identification (GUID, age, PDB filename) needed for symbol
5//! server downloads.
6
7/// PDB identification extracted from a PE debug directory.
8#[derive(Debug, Clone, PartialEq, Eq)]
9pub struct PdbId {
10    /// GUID as uppercase hex with dashes (e.g., "1B72224D-37B8-1792-2820-0ED8994498B2").
11    pub guid: String,
12    /// PDB age counter.
13    pub age: u32,
14    /// PDB filename (e.g., "ntkrnlmp.pdb").
15    pub pdb_name: String,
16}
17
18/// Format a 16-byte mixed-endian GUID as uppercase hex with dashes.
19///
20/// The GUID is stored in mixed-endian format:
21/// - Bytes 0-3: `Data1` (little-endian u32)
22/// - Bytes 4-5: `Data2` (little-endian u16)
23/// - Bytes 6-7: `Data3` (little-endian u16)
24/// - Bytes 8-15: `Data4` (big-endian, raw bytes)
25fn format_guid(bytes: &[u8; 16]) -> String {
26    let d1 = u32::from_le_bytes([bytes[0], bytes[1], bytes[2], bytes[3]]);
27    let d2 = u16::from_le_bytes([bytes[4], bytes[5]]);
28    let d3 = u16::from_le_bytes([bytes[6], bytes[7]]);
29    format!(
30        "{:08X}-{:04X}-{:04X}-{:02X}{:02X}-{:02X}{:02X}{:02X}{:02X}{:02X}{:02X}",
31        d1,
32        d2,
33        d3,
34        bytes[8],
35        bytes[9],
36        bytes[10],
37        bytes[11],
38        bytes[12],
39        bytes[13],
40        bytes[14],
41        bytes[15],
42    )
43}

45/// Extract PDB identification from a PE binary.
46///
47/// Parses the PE debug directory to find a CodeView RSDS record containing
48/// the GUID, age, and PDB filename needed to download the matching PDB
49/// from a symbol server.
50pub fn extract_pdb_id(pe_bytes: &[u8]) -> crate::Result<PdbId> {
51    use goblin::pe::PE;
52
53    let pe = PE::parse(pe_bytes).map_err(|e| crate::Error::Malformed(format!("PE parse: {e}")))?;
54
55    let debug_data = pe
56        .debug_data
57        .ok_or_else(|| crate::Error::Malformed("no debug data".into()))?;
58
59    let cv = debug_data
60        .codeview_pdb70_debug_info
61        .ok_or_else(|| crate::Error::Malformed("no CodeView PDB70 debug info".into()))?;
62
63    let guid = format_guid(&cv.signature);
64
65    // Convert filename bytes to string, stripping null terminator.
66    let raw_name = std::str::from_utf8(cv.filename)
67        .map_err(|e| crate::Error::Malformed(format!("PDB filename not UTF-8: {e}")))?
68        .trim_end_matches('\0');
69
70    // Extract just the filename component (strip any directory path).
71    let pdb_name = std::path::Path::new(raw_name)
72        .file_name()
73        .and_then(|n| n.to_str())
74        .unwrap_or(raw_name)
75        .to_string();
76
77    Ok(PdbId {
78        guid,
79        age: cv.age,
80        pdb_name,
81    })
82}

84/// CodeView RSDS record signature.
85const RSDS_MAGIC: &[u8; 4] = b"RSDS";
86/// Minimum bytes in an RSDS record: 4 (magic) + 16 (GUID) + 4 (age) + 1 (name nul).
87const RSDS_MIN_LEN: usize = 4 + 16 + 4 + 1;
88/// Cap on the PDB-name length scanned after the age field (defensive bound).
89const MAX_PDB_NAME_LEN: usize = 256;
90
91/// Extract PDB identification by scanning raw bytes for a CodeView RSDS record.
92///
93/// Tolerant fallback for [`extract_pdb_id`]: it does **not** parse the PE
94/// structure (section table, data directories, attribute-certificate table),
95/// so it works on an *in-memory* kernel image whose on-disk file offsets are no
96/// longer valid (goblin rejects such images with errors like "End of attribute
97/// certificates table is after the end of the PE binary").
98///
99/// Scans `bytes` for the ASCII signature `RSDS`, then reads the 16-byte
100/// mixed-endian GUID, the 4-byte age, and the NUL-terminated PDB filename that
101/// follow. The first record with a non-empty, valid-UTF-8 filename wins.
102///
103/// All reads are bounds-checked; malformed or truncated input yields
104/// [`Error::NotFound`] rather than a panic (Paranoid Gatekeeper).
105pub fn extract_pdb_id_tolerant(bytes: &[u8]) -> crate::Result<PdbId> {
106    extract_pdb_id_tolerant_where(bytes, |_| true)
107}
108
109/// Like [`extract_pdb_id_tolerant`], but returns the first RSDS record whose
110/// recovered PDB name satisfies `accept`. Used to pick a *kernel* RSDS out of a
111/// physical-memory window that may also contain driver RSDS records — when
112/// ntoskrnl's PE header is paged out, its RSDS in `.rdata` is still the anchor.
113pub fn extract_pdb_id_tolerant_where(
114    bytes: &[u8],
115    accept: impl Fn(&str) -> bool,
116) -> crate::Result<PdbId> {
117    let mut search = 0usize;
118    while let Some(rel) = find_subslice(&bytes[search..], RSDS_MAGIC) {
119        let pos = search + rel;
120        // Advance past this candidate so the next iteration cannot re-match it.
121        search = pos + 1;
122
123        let Some(record) = bytes.get(pos..) else {
124            continue; // cov:unreachable: pos < bytes.len() because find_subslice matched here
125        };
126        if record.len() < RSDS_MIN_LEN {
127            continue;
128        }
129
130        let Some(guid_src) = record.get(4..20) else {
131            continue; // cov:unreachable: record.len() >= RSDS_MIN_LEN guarantees 20 bytes
132        };
133        let mut guid = [0u8; 16];
134        guid.copy_from_slice(guid_src);
135
136        let Some(age_src) = record.get(20..24) else {
137            continue; // cov:unreachable: record.len() >= RSDS_MIN_LEN guarantees 24 bytes
138        };
139        let age = u32::from_le_bytes([age_src[0], age_src[1], age_src[2], age_src[3]]);
140
141        // Name is NUL-terminated, bounded to MAX_PDB_NAME_LEN.
142        let Some(name_region) = record.get(24..) else {
143            continue; // cov:unreachable: record.len() >= RSDS_MIN_LEN guarantees byte 24 exists
144        };
145        let name_region = &name_region[..name_region.len().min(MAX_PDB_NAME_LEN)];
146        let name_len = name_region
147            .iter()
148            .position(|&b| b == 0)
149            .unwrap_or(name_region.len());
150        let Ok(raw_name) = std::str::from_utf8(&name_region[..name_len]) else {
151            continue; // not a UTF-8 name — keep scanning
152        };
153        if raw_name.is_empty() {
154            continue;
155        }
156
157        // PDB names are Windows paths; split on both separators (std::path::Path
158        // does not treat '\\' as a separator on Unix, so do it explicitly).
159        let pdb_name = raw_name
160            .rsplit(['/', '\\'])
161            .next()
162            .unwrap_or(raw_name)
163            .to_string();
164
165        if !accept(&pdb_name) {
166            continue; // an RSDS, but not the one the caller wants — keep scanning
167        }
168
169        return Ok(PdbId {
170            guid: format_guid(&guid),
171            age,
172            pdb_name,
173        });
174    }
175    Err(crate::Error::NotFound(
176        "no CodeView RSDS record found in image".into(),
177    ))
178}

180/// Find the first occurrence of `needle` in `haystack`, returning its offset.
181fn find_subslice(haystack: &[u8], needle: &[u8]) -> Option<usize> {
182    if needle.is_empty() || haystack.len() < needle.len() {
183        return None;
184    }
185    haystack
186        .windows(needle.len())
187        .position(|window| window == needle)
188}

194    /// Build a minimal PE (PE32+/AMD64) with an embedded CodeView RSDS debug record.
195    ///
196    /// The resulting buffer is a valid PE that goblin can parse, containing:
197    /// - DOS header with e_lfanew at 0x80
198    /// - PE signature + COFF header (AMD64, 1 section)
199    /// - PE32+ optional header with debug data directory
200    /// - One .rdata section mapping RVA 0x200 to file offset 0x200
201    /// - IMAGE_DEBUG_DIRECTORY pointing to CodeView data
202    /// - CodeView RSDS record with the given GUID, age, and PDB filename
203    fn build_pe_with_debug(guid_bytes: [u8; 16], age: u32, pdb_name: &str) -> Vec<u8> {
204        let mut buf = vec![0u8; 4096];
205
206        // DOS header
207        buf[0] = b'M';
208        buf[1] = b'Z';
209        let pe_offset: u32 = 0x80;
210        buf[0x3C..0x40].copy_from_slice(&pe_offset.to_le_bytes());
211
212        let mut pos = pe_offset as usize;
213
214        // PE signature
215        buf[pos..pos + 4].copy_from_slice(b"PE\0\0");
216        pos += 4;
217
218        // COFF header (20 bytes)
219        buf[pos..pos + 2].copy_from_slice(&0x8664u16.to_le_bytes()); // Machine: AMD64
220        buf[pos + 2..pos + 4].copy_from_slice(&1u16.to_le_bytes()); // NumberOfSections: 1
221        let optional_header_size: u16 = 240; // PE32+ optional header size (112 + 16*8)
222        buf[pos + 16..pos + 18].copy_from_slice(&optional_header_size.to_le_bytes());
223        buf[pos + 18..pos + 20].copy_from_slice(&0x0022u16.to_le_bytes()); // Characteristics
224        pos += 20;
225
226        // Optional header (PE32+)
227        let opt_start = pos;
228        buf[pos..pos + 2].copy_from_slice(&0x020Bu16.to_le_bytes()); // PE32+ magic
229
230        // SectionAlignment at offset 32 from opt_start
231        buf[opt_start + 32..opt_start + 36].copy_from_slice(&0x1000u32.to_le_bytes());
232        // FileAlignment at offset 36 from opt_start (must be power of 2)
233        buf[opt_start + 36..opt_start + 40].copy_from_slice(&0x200u32.to_le_bytes());
234        // SizeOfImage at offset 56 from opt_start
235        buf[opt_start + 56..opt_start + 60].copy_from_slice(&0x2000u32.to_le_bytes());
236        // SizeOfHeaders at offset 60 from opt_start
237        buf[opt_start + 60..opt_start + 64].copy_from_slice(&0x200u32.to_le_bytes());
238
239        // NumberOfRvaAndSizes at offset 108 from opt_start
240        buf[opt_start + 108..opt_start + 112].copy_from_slice(&16u32.to_le_bytes());
241
242        // Data directories start at offset 112 from opt_start.
243        // Debug directory is index 6 -> offset 112 + 6*8 = 160 from opt_start.
244        let debug_dir_rva: u32 = 0x200;
245        let debug_dir_size: u32 = 28; // One IMAGE_DEBUG_DIRECTORY entry
246        buf[opt_start + 160..opt_start + 164].copy_from_slice(&debug_dir_rva.to_le_bytes());
247        buf[opt_start + 164..opt_start + 168].copy_from_slice(&debug_dir_size.to_le_bytes());
248
249        pos = opt_start + optional_header_size as usize;
250
251        // Section header (40 bytes) — .rdata covering our debug data area
252        buf[pos..pos + 8].copy_from_slice(b".rdata\0\0");
253        buf[pos + 8..pos + 12].copy_from_slice(&0x1000u32.to_le_bytes()); // VirtualSize
254        buf[pos + 12..pos + 16].copy_from_slice(&0x200u32.to_le_bytes()); // VirtualAddress
255        buf[pos + 16..pos + 20].copy_from_slice(&0x200u32.to_le_bytes()); // SizeOfRawData
256        buf[pos + 20..pos + 24].copy_from_slice(&0x200u32.to_le_bytes()); // PointerToRawData
257
258        // IMAGE_DEBUG_DIRECTORY at file offset 0x200
259        let debug_dir_offset = 0x200usize;
260        let cv_rva: u32 = 0x220;
261        let pdb_name_bytes = pdb_name.as_bytes();
262        let cv_size: u32 = (24 + pdb_name_bytes.len() + 1) as u32;
263
264        // Type field is at offset +12 in IMAGE_DEBUG_DIRECTORY
265        buf[debug_dir_offset + 12..debug_dir_offset + 16].copy_from_slice(&2u32.to_le_bytes()); // IMAGE_DEBUG_TYPE_CODEVIEW
266        buf[debug_dir_offset + 16..debug_dir_offset + 20].copy_from_slice(&cv_size.to_le_bytes()); // SizeOfData
267        buf[debug_dir_offset + 20..debug_dir_offset + 24].copy_from_slice(&cv_rva.to_le_bytes()); // AddressOfRawData
268        buf[debug_dir_offset + 24..debug_dir_offset + 28].copy_from_slice(&cv_rva.to_le_bytes()); // PointerToRawData (= RVA, 1:1 mapping)
269
270        // CodeView RSDS record at file offset 0x220
271        let cv_offset = 0x220usize;
272        buf[cv_offset..cv_offset + 4].copy_from_slice(b"RSDS");
273        buf[cv_offset + 4..cv_offset + 20].copy_from_slice(&guid_bytes);
274        buf[cv_offset + 20..cv_offset + 24].copy_from_slice(&age.to_le_bytes());
275        let name_start = cv_offset + 24;
276        buf[name_start..name_start + pdb_name_bytes.len()].copy_from_slice(pdb_name_bytes);
277        buf[name_start + pdb_name_bytes.len()] = 0; // null terminator
278
279        buf
280    }

437    /// Build a bare CodeView RSDS record: "RSDS" + 16-byte GUID + 4-byte age +
438    /// NUL-terminated name. No surrounding PE — exactly the bytes the tolerant
439    /// scanner must locate inside an in-memory image.
440    fn rsds_record(guid: [u8; 16], age: u32, name: &str) -> Vec<u8> {
441        let mut v = Vec::new();
442        v.extend_from_slice(b"RSDS");
443        v.extend_from_slice(&guid);
444        v.extend_from_slice(&age.to_le_bytes());
445        v.extend_from_slice(name.as_bytes());
446        v.push(0);
447        v
448    }
