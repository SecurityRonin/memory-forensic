1//! Linux ARP cache extraction from the kernel neighbour table.
2//!
3//! Walks the `arp_tbl` (neigh_table) hash buckets to enumerate all
4//! ARP cache entries. Each `neighbour` struct holds the IP address,
5//! MAC address, NUD state, and associated network device.
6
7use memf_core::object_reader::ObjectReader;
8use memf_format::PhysicalMemoryProvider;
9
10use crate::{ArpEntryInfo, Error, NeighState, Result};
11
12/// Walk the kernel ARP neighbour table and extract all entries.
13///
14/// Reads the `arp_tbl` symbol (type `neigh_table`), dereferences
15/// the `nht` pointer to get the `neigh_hash_table`, then iterates
16/// hash buckets reading `neighbour` structs linked via `next`.
17pub fn walk_arp_cache<P: PhysicalMemoryProvider>(
18    reader: &ObjectReader<P>,
19) -> Result<Vec<ArpEntryInfo>> {
20    let arp_tbl_addr = reader
21        .symbols()
22        .symbol_address("arp_tbl")
23        .ok_or_else(|| Error::Walker("symbol 'arp_tbl' not found".into()))?;
24
25    // neigh_table.nht → pointer to neigh_hash_table
26    let nht_ptr: u64 = reader.read_field(arp_tbl_addr, "neigh_table", "nht")?;
27    if nht_ptr == 0 {
28        return Ok(Vec::new());
29    }
30
31    // neigh_hash_table.hash_buckets → pointer to array of neighbour*
32    let buckets_ptr: u64 = reader.read_field(nht_ptr, "neigh_hash_table", "hash_buckets")?;
33    // neigh_hash_table.hash_shift → log2(bucket_count)
34    let hash_shift: u32 = reader.read_field(nht_ptr, "neigh_hash_table", "hash_shift")?;
35    let bucket_count: u64 = 1u64 << hash_shift;
36
37    if buckets_ptr == 0 {
38        return Ok(Vec::new());
39    }
40
41    let mut entries = Vec::new();
42
43    for i in 0..bucket_count {
44        // Each bucket is a pointer (8 bytes) to the first neighbour
45        let bucket_addr = buckets_ptr + i * 8;
46        let neigh_ptr: u64 = match reader.read_bytes(bucket_addr, 8) {
47            Ok(bytes) => u64::from_le_bytes(bytes[..8].try_into().expect("8 bytes")),
48            Err(_) => continue,
49        };
50
51        let mut current = neigh_ptr;
52        let mut chain_len = 0;
53        while current != 0 && chain_len < 1000 {
54            if let Ok(entry) = read_neighbour(reader, current) {
55                entries.push(entry);
56            }
57
58            // Follow neighbour.next pointer
59            current = match reader.read_field::<u64>(current, "neighbour", "next") {
60                Ok(v) => v,
61                Err(_) => break,
62            };
63            chain_len += 1;
64        }
65    }
66
67    Ok(entries)
68}

70fn read_neighbour<P: PhysicalMemoryProvider>(
71    reader: &ObjectReader<P>,
72    neigh_addr: u64,
73) -> Result<ArpEntryInfo> {
74    // Read the 4-byte IPv4 address from primary_key
75    let ip_raw: u32 = reader.read_field(neigh_addr, "neighbour", "primary_key")?;
76    let ip_bytes = ip_raw.to_le_bytes();
77    let ip_addr = format!(
78        "{}.{}.{}.{}",
79        ip_bytes[0], ip_bytes[1], ip_bytes[2], ip_bytes[3]
80    );
81
82    // Read the 6-byte MAC address from ha field
83    let ha_offset = reader
84        .symbols()
85        .field_offset("neighbour", "ha")
86        .ok_or_else(|| Error::Walker("neighbour.ha field not found".into()))?;
87    let mac_bytes = reader.read_bytes(neigh_addr + ha_offset, 6)?;
88    let mac_addr = format!(
89        "{:02x}:{:02x}:{:02x}:{:02x}:{:02x}:{:02x}",
90        mac_bytes[0], mac_bytes[1], mac_bytes[2], mac_bytes[3], mac_bytes[4], mac_bytes[5]
91    );
92
93    // Read NUD state
94    let nud_state: u8 = reader.read_field(neigh_addr, "neighbour", "nud_state")?;
95
96    // Read device name via dev pointer → net_device.name
97    let dev_ptr: u64 = reader.read_field(neigh_addr, "neighbour", "dev")?;
98    let dev_name = if dev_ptr != 0 {
99        reader.read_field_string(dev_ptr, "net_device", "name", 16)?
100    } else {
101        String::from("?")
102    };
103
104    Ok(ArpEntryInfo {
105        ip_addr,
106        mac_addr,
107        dev_name,
108        state: NeighState::from_raw(nud_state),
109    })
110}

122    // Synthetic layout:
123    //   arp_tbl (neigh_table):
124    //     nht @ 0 (pointer to neigh_hash_table)
125    //
126    //   neigh_hash_table:
127    //     hash_buckets @ 0 (pointer to array of neighbour*)
128    //     hash_shift   @ 8 (u32) — log2(bucket_count)
129    //
130    //   neighbour:
131    //     next          @ 0  (pointer — next in hash chain)
132    //     primary_key   @ 8  (4 bytes — IPv4 address)
133    //     ha            @ 12 (6 bytes — MAC address)
134    //     nud_state     @ 18 (u8)
135    //     dev           @ 24 (pointer to net_device)
136    //     total: 64 bytes
137    //
138    //   net_device:
139    //     name @ 0 (char[16])
140
141    const NHT_PTR_OFF: usize = 0;
142    // neigh_hash_table offsets
143    const HASH_BUCKETS_OFF: usize = 0;
144    const HASH_SHIFT_OFF: usize = 8;
145    // neighbour offsets
146    const NEIGH_NEXT_OFF: usize = 0;
147    const NEIGH_KEY_OFF: usize = 8;
148    const NEIGH_HA_OFF: usize = 12;
149    const NEIGH_NUD_OFF: usize = 18;
150    const NEIGH_DEV_OFF: usize = 24;
151
152    fn build_arp_isf() -> serde_json::Value {
153        IsfBuilder::new()
154            .add_struct("neigh_table", 64)
155            .add_field("neigh_table", "nht", 0, "pointer")
156            .add_struct("neigh_hash_table", 16)
157            .add_field("neigh_hash_table", "hash_buckets", 0, "pointer")
158            .add_field("neigh_hash_table", "hash_shift", 8, "unsigned int")
159            .add_struct("neighbour", 64)
160            .add_field("neighbour", "next", 0, "pointer")
161            .add_field("neighbour", "primary_key", 8, "unsigned int")
162            .add_field("neighbour", "ha", 12, "char")
163            .add_field("neighbour", "nud_state", 18, "unsigned char")
164            .add_field("neighbour", "dev", 24, "pointer")
165            .add_struct("net_device", 256)
166            .add_field("net_device", "name", 0, "char")
167            .add_symbol("arp_tbl", 0xFFFF_8000_0010_0000)
168            .build_json()
169    }
170
171    fn make_reader(pages: &[(u64, u64, &[u8])]) -> ObjectReader<SyntheticPhysMem> {
172        let isf = build_arp_isf();
173        let resolver = IsfResolver::from_value(&isf).unwrap();
174
175        let mut builder = PageTableBuilder::new();
176        for &(vaddr, paddr, data) in pages {
177            builder = builder
178                .map_4k(vaddr, paddr, flags::WRITABLE)
179                .write_phys(paddr, data);
180        }
181        let (cr3, mem) = builder.build();
182        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
183        ObjectReader::new(vas, Box::new(resolver))
184    }

186    /// Single ARP entry: 192.168.1.1 -> aa:bb:cc:dd:ee:ff on eth0
187    #[test]
188    fn walk_single_arp_entry() {
189        let arp_tbl_vaddr: u64 = 0xFFFF_8000_0010_0000;
190        let arp_tbl_paddr: u64 = 0x0080_0000;
191
192        let nht_vaddr: u64 = 0xFFFF_8000_0020_0000;
193        let nht_paddr: u64 = 0x0090_0000;
194
195        let neigh_vaddr: u64 = 0xFFFF_8000_0030_0000;
196        let neigh_paddr: u64 = 0x00A0_0000;
197
198        let dev_vaddr: u64 = 0xFFFF_8000_0040_0000;
199        let dev_paddr: u64 = 0x00B0_0000;
200
201        // bucket array lives at nht_vaddr + 0x100
202        let bucket_array_vaddr: u64 = nht_vaddr + 0x100;
203
204        // -- arp_tbl page: nht pointer
205        let mut arp_data = vec![0u8; 4096];
206        arp_data[NHT_PTR_OFF..NHT_PTR_OFF + 8].copy_from_slice(&nht_vaddr.to_le_bytes());
207
208        // -- nht page: bucket array pointer + hash_shift
209        let mut nht_data = vec![0u8; 4096];
210        nht_data[HASH_BUCKETS_OFF..HASH_BUCKETS_OFF + 8]
211            .copy_from_slice(&bucket_array_vaddr.to_le_bytes());
212        // hash_shift = 0 means 1 bucket (2^0 = 1)
213        nht_data[HASH_SHIFT_OFF..HASH_SHIFT_OFF + 4].copy_from_slice(&0u32.to_le_bytes());
214        // bucket[0] = pointer to neighbour
215        nht_data[0x100..0x108].copy_from_slice(&neigh_vaddr.to_le_bytes());
216
217        // -- neighbour page
218        let mut neigh_data = vec![0u8; 4096];
219        neigh_data[NEIGH_NEXT_OFF..NEIGH_NEXT_OFF + 8].copy_from_slice(&0u64.to_le_bytes()); // null = end of chain
220        let ip: u32 = u32::from_le_bytes([192, 168, 1, 1]);
221        neigh_data[NEIGH_KEY_OFF..NEIGH_KEY_OFF + 4].copy_from_slice(&ip.to_le_bytes());
222        neigh_data[NEIGH_HA_OFF..NEIGH_HA_OFF + 6]
223            .copy_from_slice(&[0xaa, 0xbb, 0xcc, 0xdd, 0xee, 0xff]);
224        neigh_data[NEIGH_NUD_OFF] = 0x02; // REACHABLE
225        neigh_data[NEIGH_DEV_OFF..NEIGH_DEV_OFF + 8].copy_from_slice(&dev_vaddr.to_le_bytes());
226
227        // -- net_device page
228        let mut dev_data = vec![0u8; 4096];
229        dev_data[..4].copy_from_slice(b"eth0");
230
231        let reader = make_reader(&[
232            (arp_tbl_vaddr, arp_tbl_paddr, &arp_data),
233            (nht_vaddr, nht_paddr, &nht_data),
234            (neigh_vaddr, neigh_paddr, &neigh_data),
235            (dev_vaddr, dev_paddr, &dev_data),
236        ]);
237
238        let entries = walk_arp_cache(&reader).unwrap();
239        assert_eq!(entries.len(), 1);
240        assert_eq!(entries[0].ip_addr, "192.168.1.1");
241        assert_eq!(entries[0].mac_addr, "aa:bb:cc:dd:ee:ff");
242        assert_eq!(entries[0].dev_name, "eth0");
243        assert_eq!(entries[0].state, NeighState::Reachable);
244    }

246    /// Empty ARP table (hash_shift=0, bucket[0] is null).
247    #[test]
248    fn walk_empty_arp_table() {
249        let arp_tbl_vaddr: u64 = 0xFFFF_8000_0010_0000;
250        let arp_tbl_paddr: u64 = 0x0080_0000;
251        let nht_vaddr: u64 = 0xFFFF_8000_0020_0000;
252        let nht_paddr: u64 = 0x0090_0000;
253        let bucket_array_vaddr: u64 = nht_vaddr + 0x100;
254
255        let mut arp_data = vec![0u8; 4096];
256        arp_data[NHT_PTR_OFF..NHT_PTR_OFF + 8].copy_from_slice(&nht_vaddr.to_le_bytes());
257
258        let mut nht_data = vec![0u8; 4096];
259        nht_data[HASH_BUCKETS_OFF..HASH_BUCKETS_OFF + 8]
260            .copy_from_slice(&bucket_array_vaddr.to_le_bytes());
261        nht_data[HASH_SHIFT_OFF..HASH_SHIFT_OFF + 4].copy_from_slice(&0u32.to_le_bytes());
262        // bucket[0] = 0 (null)
263        nht_data[0x100..0x108].copy_from_slice(&0u64.to_le_bytes());
264
265        let reader = make_reader(&[
266            (arp_tbl_vaddr, arp_tbl_paddr, &arp_data),
267            (nht_vaddr, nht_paddr, &nht_data),
268        ]);
269
270        let entries = walk_arp_cache(&reader).unwrap();
271        assert!(entries.is_empty());
272    }

274    /// Two ARP entries chained in same bucket.
275    #[test]
276    fn walk_two_entries_in_chain() {
277        let arp_tbl_vaddr: u64 = 0xFFFF_8000_0010_0000;
278        let arp_tbl_paddr: u64 = 0x0080_0000;
279        let nht_vaddr: u64 = 0xFFFF_8000_0020_0000;
280        let nht_paddr: u64 = 0x0090_0000;
281        let neigh1_vaddr: u64 = 0xFFFF_8000_0030_0000;
282        let neigh1_paddr: u64 = 0x00A0_0000;
283        let neigh2_vaddr: u64 = 0xFFFF_8000_0050_0000;
284        let neigh2_paddr: u64 = 0x00C0_0000;
285        let dev_vaddr: u64 = 0xFFFF_8000_0040_0000;
286        let dev_paddr: u64 = 0x00B0_0000;
287        let bucket_array_vaddr: u64 = nht_vaddr + 0x100;
288
289        let mut arp_data = vec![0u8; 4096];
290        arp_data[NHT_PTR_OFF..NHT_PTR_OFF + 8].copy_from_slice(&nht_vaddr.to_le_bytes());
291
292        let mut nht_data = vec![0u8; 4096];
293        nht_data[HASH_BUCKETS_OFF..HASH_BUCKETS_OFF + 8]
294            .copy_from_slice(&bucket_array_vaddr.to_le_bytes());
295        nht_data[HASH_SHIFT_OFF..HASH_SHIFT_OFF + 4].copy_from_slice(&0u32.to_le_bytes());
296        nht_data[0x100..0x108].copy_from_slice(&neigh1_vaddr.to_le_bytes());
297
298        // neigh1 -> neigh2
299        let mut neigh1_data = vec![0u8; 4096];
300        neigh1_data[NEIGH_NEXT_OFF..NEIGH_NEXT_OFF + 8]
301            .copy_from_slice(&neigh2_vaddr.to_le_bytes());
302        let ip1: u32 = u32::from_le_bytes([10, 0, 0, 1]);
303        neigh1_data[NEIGH_KEY_OFF..NEIGH_KEY_OFF + 4].copy_from_slice(&ip1.to_le_bytes());
304        neigh1_data[NEIGH_HA_OFF..NEIGH_HA_OFF + 6]
305            .copy_from_slice(&[0x11, 0x22, 0x33, 0x44, 0x55, 0x66]);
306        neigh1_data[NEIGH_NUD_OFF] = 0x04; // STALE
307        neigh1_data[NEIGH_DEV_OFF..NEIGH_DEV_OFF + 8].copy_from_slice(&dev_vaddr.to_le_bytes());
308
309        // neigh2 -> null
310        let mut neigh2_data = vec![0u8; 4096];
311        neigh2_data[NEIGH_NEXT_OFF..NEIGH_NEXT_OFF + 8].copy_from_slice(&0u64.to_le_bytes());
312        let ip2: u32 = u32::from_le_bytes([10, 0, 0, 2]);
313        neigh2_data[NEIGH_KEY_OFF..NEIGH_KEY_OFF + 4].copy_from_slice(&ip2.to_le_bytes());
314        neigh2_data[NEIGH_HA_OFF..NEIGH_HA_OFF + 6]
315            .copy_from_slice(&[0xAA, 0xBB, 0xCC, 0xDD, 0xEE, 0x00]);
316        neigh2_data[NEIGH_NUD_OFF] = 0x80; // PERMANENT
317        neigh2_data[NEIGH_DEV_OFF..NEIGH_DEV_OFF + 8].copy_from_slice(&dev_vaddr.to_le_bytes());
318
319        let mut dev_data = vec![0u8; 4096];
320        dev_data[..5].copy_from_slice(b"ens33");
321
322        let reader = make_reader(&[
323            (arp_tbl_vaddr, arp_tbl_paddr, &arp_data),
324            (nht_vaddr, nht_paddr, &nht_data),
325            (neigh1_vaddr, neigh1_paddr, &neigh1_data),
326            (neigh2_vaddr, neigh2_paddr, &neigh2_data),
327            (dev_vaddr, dev_paddr, &dev_data),
328        ]);
329
330        let entries = walk_arp_cache(&reader).unwrap();
331        assert_eq!(entries.len(), 2);
332        assert_eq!(entries[0].ip_addr, "10.0.0.1");
333        assert_eq!(entries[0].mac_addr, "11:22:33:44:55:66");
334        assert_eq!(entries[0].state, NeighState::Stale);
335        assert_eq!(entries[1].ip_addr, "10.0.0.2");
336        assert_eq!(entries[1].mac_addr, "aa:bb:cc:dd:ee:00");
337        assert_eq!(entries[1].state, NeighState::Permanent);
338        assert!(entries.iter().all(|e| e.dev_name == "ens33"));
339    }
