1//! Linux kernel inline hook detector.
2//!
3//! Checks the first bytes of key kernel functions for JMP/CALL
4//! trampolines that indicate inline hooking. Reads the function
5//! prologue and checks for x86_64 patterns like:
6//!   - `0xE9` (relative JMP)
7//!   - `0xFF 0x25` (absolute indirect JMP)
8//!   - `0x48 0xB8 ... 0xFF 0xE0` (MOV RAX, imm64; JMP RAX)
9
10use memf_core::object_reader::ObjectReader;
11use memf_format::PhysicalMemoryProvider;
12
13use crate::{Error, KernelHookInfo, Result};
14
15/// Number of prologue bytes to read from each function.
16const PROLOGUE_SIZE: usize = 16;
17
18/// Well-known kernel functions to check for inline hooks.
19const FUNCTIONS_TO_CHECK: &[&str] = &[
20    "sys_read",
21    "sys_write",
22    "sys_open",
23    "sys_close",
24    "vfs_read",
25    "vfs_write",
26    "tcp4_seq_show",
27    "filldir",
28    "filldir64",
29];
30
31/// Check key kernel functions for inline hooks.
32///
33/// Reads the first [`PROLOGUE_SIZE`] bytes of each function in
34/// [`FUNCTIONS_TO_CHECK`] and looks for JMP/CALL trampoline patterns.
35pub fn check_inline_hooks<P: PhysicalMemoryProvider>(
36    reader: &ObjectReader<P>,
37) -> Result<Vec<KernelHookInfo>> {
38    let stext = reader
39        .symbols()
40        .symbol_address("_stext")
41        .ok_or_else(|| Error::Walker("symbol '_stext' not found".into()))?;
42    let etext = reader
43        .symbols()
44        .symbol_address("_etext")
45        .ok_or_else(|| Error::Walker("symbol '_etext' not found".into()))?;
46
47    let mut results = Vec::new();
48
49    for &func_name in FUNCTIONS_TO_CHECK {
50        let Some(func_addr) = reader.symbols().symbol_address(func_name) else {
51            continue; // Symbol not present, skip
52        };
53
54        let Ok(prologue) = reader.read_bytes(func_addr, PROLOGUE_SIZE) else {
55            continue;
56        };
57
58        let (hook_type, target) = analyze_prologue(&prologue, func_addr);
59        // Suspicious only when a hook IS present AND the target is outside kernel text.
60        // A jmp into a legitimate kernel function is not suspicious.
61        let suspicious = hook_type != "none" && target.map_or(true, |t| t < stext || t > etext);
62
63        results.push(KernelHookInfo {
64            symbol: func_name.to_string(),
65            address: func_addr,
66            hook_type,
67            target,
68            suspicious,
69        });
70    }
71
72    Ok(results)
73}

75/// Analyze a function prologue for hook patterns.
76///
77/// Returns `(hook_type, target)` if a hook is detected, or `("none", None)`.
78#[allow(clippy::cast_possible_wrap, clippy::cast_sign_loss)]
79fn analyze_prologue(bytes: &[u8], func_addr: u64) -> (String, Option<u64>) {
80    if bytes.len() < PROLOGUE_SIZE {
81        return ("none".to_string(), None);
82    }
83
84    // Pattern 1: E9 xx xx xx xx — relative JMP (5 bytes)
85    if bytes[0] == 0xE9 {
86        let offset = i32::from_le_bytes(bytes[1..5].try_into().unwrap());
87        let target = (func_addr as i64 + 5 + i64::from(offset)) as u64;
88        return ("jmp_rel32".to_string(), Some(target));
89    }
90
91    // Pattern 2: FF 25 xx xx xx xx — absolute indirect JMP [rip+disp32]
92    if bytes[0] == 0xFF && bytes[1] == 0x25 {
93        let offset = i32::from_le_bytes(bytes[2..6].try_into().unwrap());
94        let target = (func_addr as i64 + 6 + i64::from(offset)) as u64;
95        return ("jmp_indirect".to_string(), Some(target));
96    }
97
98    // Pattern 3: 48 B8 <imm64> FF E0 — MOV RAX, imm64; JMP RAX (12 bytes)
99    if bytes.len() >= 12
100        && bytes[0] == 0x48
101        && bytes[1] == 0xB8
102        && bytes[10] == 0xFF
103        && bytes[11] == 0xE0
104    {
105        let target = u64::from_le_bytes(bytes[2..10].try_into().unwrap());
106        return ("mov_rax_jmp".to_string(), Some(target));
107    }
108
109    ("none".to_string(), None)
110}

120    fn make_test_reader(
121        data: &[u8],
122        func_vaddr: u64,
123        func_paddr: u64,
124        stext: u64,
125        etext: u64,
126        func_symbols: &[(&str, u64)],
127    ) -> ObjectReader<SyntheticPhysMem> {
128        let mut builder = IsfBuilder::new()
129            .add_struct("task_struct", 64)
130            .add_field("task_struct", "pid", 0, "int")
131            .add_symbol("_stext", stext)
132            .add_symbol("_etext", etext);
133
134        for &(name, addr) in func_symbols {
135            builder = builder.add_symbol(name, addr);
136        }
137
138        let isf = builder.build_json();
139        let resolver = IsfResolver::from_value(&isf).unwrap();
140        let (cr3, mem) = PageTableBuilder::new()
141            .map_4k(func_vaddr, func_paddr, ptflags::WRITABLE)
142            .write_phys(func_paddr, data)
143            .build();
144        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
145        ObjectReader::new(vas, Box::new(resolver))
146    }

148    #[test]
149    fn clean_function_no_hook() {
150        // Normal function prologue: push rbp; mov rbp, rsp; sub rsp, 0x20
151        let mut prologue = vec![0u8; 4096];
152        prologue[0] = 0x55; // push rbp
153        prologue[1] = 0x48; // REX.W
154        prologue[2] = 0x89; // mov
155        prologue[3] = 0xE5; // rbp, rsp
156
157        let func_vaddr: u64 = 0xFFFF_8000_0001_0000;
158        let func_paddr: u64 = 0x0080_0000;
159        let stext: u64 = 0xFFFF_8000_0000_0000;
160        let etext: u64 = 0xFFFF_8000_00FF_FFFF;
161
162        let reader = make_test_reader(
163            &prologue,
164            func_vaddr,
165            func_paddr,
166            stext,
167            etext,
168            &[("sys_read", func_vaddr)],
169        );
170        let results = check_inline_hooks(&reader).unwrap();
171
172        assert_eq!(results.len(), 1);
173        assert!(!results[0].suspicious);
174        assert_eq!(results[0].symbol, "sys_read");
175        assert_eq!(results[0].hook_type, "none");
176    }

178    #[test]
179    fn detects_relative_jmp_hook() {
180        // Hooked: E9 xx xx xx xx (relative JMP)
181        // Target lands inside kernel text → hook detected but NOT suspicious
182        // (jmp to a legitimate kernel function should not be flagged).
183        let mut prologue = vec![0u8; 4096];
184        prologue[0] = 0xE9; // JMP rel32
185        prologue[1..5].copy_from_slice(&0x1000i32.to_le_bytes());
186
187        let func_vaddr: u64 = 0xFFFF_8000_0001_0000;
188        let func_paddr: u64 = 0x0080_0000;
189        let stext: u64 = 0xFFFF_8000_0000_0000;
190        let etext: u64 = 0xFFFF_8000_00FF_FFFF;
191
192        let reader = make_test_reader(
193            &prologue,
194            func_vaddr,
195            func_paddr,
196            stext,
197            etext,
198            &[("sys_read", func_vaddr)],
199        );
200        let results = check_inline_hooks(&reader).unwrap();
201
202        // target = func_vaddr + 5 + 0x1000 = 0xFFFF_8000_0002_1005, inside [stext, etext)
203        assert_eq!(results.len(), 1);
204        assert_eq!(results[0].hook_type, "jmp_rel32");
205        assert!(results[0].target.is_some());
206        // Hook detected but target inside kernel text → not suspicious
207        assert!(
208            !results[0].suspicious,
209            "jmp into kernel text should not be suspicious"
210        );
211    }
