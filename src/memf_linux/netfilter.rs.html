1//! Linux netfilter (iptables) rule extraction from kernel memory.
2//!
3//! Reads the kernel's iptables rule structures from the `xt_table` chain.
4//! The kernel organizes rules into tables (filter, nat, mangle) and chains
5//! (INPUT, OUTPUT, FORWARD, PREROUTING, POSTROUTING).
6
7use memf_core::object_reader::ObjectReader;
8use memf_format::PhysicalMemoryProvider;
9
10use crate::{Error, NetfilterRuleInfo, Result};
11
12/// Known iptables table names and their kernel symbols.
13const TABLE_NAMES: &[(&str, &str)] = &[
14    ("filter", "iptable_filter_net_ops"),
15    ("nat", "iptable_nat_net_ops"),
16    ("mangle", "iptable_mangle_net_ops"),
17];
18
19/// Walk kernel iptables tables and extract rules.
20///
21/// Attempts to find the `init_net` namespace, then reads each registered
22/// iptables table (filter, nat, mangle) and parses rule entries.
23pub fn walk_netfilter_rules<P: PhysicalMemoryProvider>(
24    reader: &ObjectReader<P>,
25) -> Result<Vec<NetfilterRuleInfo>> {
26    // Look for init_net symbol
27    let init_net_addr = reader
28        .symbols()
29        .symbol_address("init_net")
30        .ok_or_else(|| Error::Walker("symbol 'init_net' not found".into()))?;
31
32    let mut rules = Vec::new();
33
34    // Try to read xt_table for each known table
35    for &(table_name, _symbol) in TABLE_NAMES {
36        // Find the table's xt_table by walking net->xt.tables[AF_INET]
37        // AF_INET = 2, so offset into xt.tables array
38        if let Ok(table_rules) = read_xt_table(reader, init_net_addr, table_name) {
39            rules.extend(table_rules);
40        }
41    }
42
43    Ok(rules)
44}
45
46fn read_xt_table<P: PhysicalMemoryProvider>(
47    reader: &ObjectReader<P>,
48    init_net_addr: u64,
49    table_name: &str,
50) -> Result<Vec<NetfilterRuleInfo>> {
51    // Read net.xt offset
52    let xt_offset = reader
53        .symbols()
54        .field_offset("net", "xt")
55        .ok_or_else(|| Error::Walker("net.xt field not found".into()))?;
56    let xt_addr = init_net_addr + xt_offset;
57
58    // xt contains tables_lock and tables array.
59    // netns_xt.tables is an array of list_head, indexed by protocol family.
60    // AF_INET = 2
61    let tables_offset = reader
62        .symbols()
63        .field_offset("netns_xt", "tables")
64        .ok_or_else(|| Error::Walker("netns_xt.tables not found".into()))?;
65
66    let list_head_size = reader.symbols().struct_size("list_head").unwrap_or(16);
67    let af_inet_list = xt_addr + tables_offset + 2 * list_head_size; // AF_INET = 2
68
69    // Walk the list to find xt_table with matching name
70    let table_addrs = reader.walk_list(af_inet_list, "xt_table", "list")?;
71
72    for &table_addr in &table_addrs {
73        let name = reader.read_field_string(table_addr, "xt_table", "name", 32)?;
74        if name == table_name {
75            return parse_table_rules(reader, table_addr, table_name);
76        }
77    }
78
79    Ok(Vec::new())
80}

82fn parse_table_rules<P: PhysicalMemoryProvider>(
83    reader: &ObjectReader<P>,
84    table_addr: u64,
85    table_name: &str,
86) -> Result<Vec<NetfilterRuleInfo>> {
87    // Read xt_table.private → pointer to xt_table_info.
88    let private_ptr = reader.read_pointer(table_addr, "xt_table", "private")?;
89    if private_ptr == 0 {
90        return Ok(Vec::new());
91    }
92
93    // Read xt_table_info.entries → virtual address of the flat ipt_entry region.
94    let entries_vaddr = reader.read_pointer(private_ptr, "xt_table_info", "entries")?;
95    // Read xt_table_info.size → byte length of the region.
96    let size: u64 = reader.read_field::<u64>(private_ptr, "xt_table_info", "size")?;
97
98    parse_ipt_entries(reader, entries_vaddr, size, table_name)
99}

101/// Parse a flat region of `ipt_entry` structures from raw memory.
102///
103/// `data_vaddr` is the virtual address of the first entry; `data_len` is the
104/// byte length of the region.  Entries are walked via `next_offset` until it
105/// is 0 or the end of the region is reached.
106///
107/// `ipt_entry` field offsets (kernel ABI, x86-64):
108///   0x00: src_ip (u32)
109///   0x04: dst_ip (u32)
110///   0x10: protocol (u16)
111///   0x58: target_offset (u16) — offset within entry to `ipt_entry_target`
112///   0x5A: next_offset (u16) — stride to next entry; 0 = end of table
113///
114/// `ipt_entry_target` at `entry_base + target_offset`:
115///   +0: name (29 bytes, null-terminated ASCII)
116pub fn parse_ipt_entries<P: PhysicalMemoryProvider>(
117    reader: &ObjectReader<P>,
118    data_vaddr: u64,
119    data_len: u64,
120    table_name: &str,
121) -> Result<Vec<NetfilterRuleInfo>> {
122    const MAX_RULES: usize = 10_000;
123    let data_len = data_len as usize;
124    // Read the entire data region at once.
125    let data = reader.read_bytes(data_vaddr, data_len).unwrap_or_default();
126    if data.is_empty() {
127        return Ok(Vec::new());
128    }
129
130    let mut rules = Vec::new();
131    let mut offset = 0usize;
132
133    for _ in 0..MAX_RULES {
134        // Need at least 0x5C bytes to read target_offset + next_offset.
135        if offset + 0x5C > data.len() {
136            break;
137        }
138
139        let src_ip = u32::from_le_bytes(data[offset..offset + 4].try_into().unwrap_or([0; 4]));
140        let dst_ip = u32::from_le_bytes(data[offset + 4..offset + 8].try_into().unwrap_or([0; 4]));
141        let proto = u16::from_le_bytes(
142            data[offset + 0x10..offset + 0x12]
143                .try_into()
144                .unwrap_or([0; 2]),
145        );
146        let target_off = u16::from_le_bytes(
147            data[offset + 0x58..offset + 0x5A]
148                .try_into()
149                .unwrap_or([0; 2]),
150        ) as usize;
151        let next_off = u16::from_le_bytes(
152            data[offset + 0x5A..offset + 0x5C]
153                .try_into()
154                .unwrap_or([0; 2]),
155        ) as usize;
156
157        // Parse ipt_entry_target.name (29 bytes at entry_base + target_offset).
158        let target_name = if target_off > 0 && offset + target_off + 29 <= data.len() {
159            let name_bytes = &data[offset + target_off..offset + target_off + 29];
160            let end = name_bytes.iter().position(|&b| b == 0).unwrap_or(29);
161            String::from_utf8_lossy(&name_bytes[..end]).into_owned()
162        } else {
163            String::new()
164        };
165
166        let source = if src_ip != 0 {
167            Some(format_ipv4(src_ip))
168        } else {
169            None
170        };
171        let destination = if dst_ip != 0 {
172            Some(format_ipv4(dst_ip))
173        } else {
174            None
175        };
176
177        rules.push(NetfilterRuleInfo {
178            table: table_name.to_string(),
179            chain: String::new(), // chain resolution requires hook_entry offsets
180            target: target_name,
181            protocol: protocol_name(proto),
182            source,
183            destination,
184        });
185
186        if next_off == 0 {
187            break;
188        }
189        offset += next_off;
190        if offset >= data.len() {
191            break;
192        }
193    }
194
195    Ok(rules)
196}

198/// Format a raw u32 IPv4 address (little-endian stored) as a dotted string.
199fn format_ipv4(ip: u32) -> String {
200    let b = ip.to_le_bytes();
201    format!("{}.{}.{}.{}", b[0], b[1], b[2], b[3])
202}

204/// Parse a protocol number to name.
205pub fn protocol_name(proto: u16) -> String {
206    match proto {
207        0 => "all".to_string(),
208        6 => "tcp".to_string(),
209        17 => "udp".to_string(),
210        1 => "icmp".to_string(),
211        _ => format!("proto:{proto}"),
212    }
213}

241    /// Build a minimal reader that maps a fake table data region and exposes
242    /// `parse_ipt_entries` directly.
243    fn make_ipt_entry_data(src_ip: u32, dst_ip: u32, proto: u16, target_name: &str) -> Vec<u8> {
244        // ipt_entry layout (offsets per kernel ABI):
245        //   0x00: src_ip (u32)
246        //   0x04: dst_ip (u32)
247        //   0x10: protocol (u16)
248        //   0x58: target_offset (u16) — relative offset to ipt_entry_target within entry
249        //   0x5A: next_offset (u16) — stride to next entry (0 = end)
250        //
251        // ipt_entry_target (at base + target_offset):
252        //   +0: name (29 bytes, null-terminated)
253        //
254        // We place one entry at the start.  target_offset = 0x60 (96 bytes into entry).
255        // next_offset = 0 means no more entries.
256        let mut data = vec![0u8; 256];
257
258        // src_ip at 0x00
259        data[0x00..0x04].copy_from_slice(&src_ip.to_le_bytes());
260        // dst_ip at 0x04
261        data[0x04..0x08].copy_from_slice(&dst_ip.to_le_bytes());
262        // protocol at 0x10
263        data[0x10..0x12].copy_from_slice(&proto.to_le_bytes());
264        // target_offset at 0x58: 0x60 (96 bytes in)
265        let target_off: u16 = 0x60;
266        data[0x58..0x5A].copy_from_slice(&target_off.to_le_bytes());
267        // next_offset at 0x5A: 0 = end
268        data[0x5A..0x5C].copy_from_slice(&0u16.to_le_bytes());
269        // target name at base + target_offset
270        let name_bytes = target_name.as_bytes();
271        let len = name_bytes.len().min(28);
272        data[0x60..0x60 + len].copy_from_slice(&name_bytes[..len]);
273
274        data
275    }

277    fn make_ipt_reader(
278        entry_data: &[u8],
279        entry_vaddr: u64,
280        entry_paddr: u64,
281    ) -> ObjectReader<SyntheticPhysMem> {
282        let isf = IsfBuilder::new().build_json();
283        let resolver = IsfResolver::from_value(&isf).unwrap();
284        let (cr3, mut mem) = PageTableBuilder::new()
285            .map_4k(entry_vaddr, entry_paddr, flags::PRESENT | flags::WRITABLE)
286            .build();
287        mem.write_bytes(entry_paddr, entry_data);
288        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
289        ObjectReader::new(vas, Box::new(resolver))
290    }

530    fn parse_ipt_entries_two_chained_entries() {
531        // Two ipt_entry records back to back. First has next_off = 128 (size of one record).
532        // Second has next_off = 0 (terminates).
533        let entry_size = 128usize;
534        let mut data = vec![0u8; entry_size * 2];
535
536        // Entry 1: src=1.2.3.4 (stored as bytes [1,2,3,4] → u32), proto=6, target="ACCEPT", next=128
537        let src1 = u32::from_le_bytes([1, 2, 3, 4]);
538        data[0x00..0x04].copy_from_slice(&src1.to_le_bytes());
539        data[0x10..0x12].copy_from_slice(&6u16.to_le_bytes()); // tcp
540                                                               // target_offset at 0x58: 0x60 → but 0x60 > entry_size(128=0x80), still within 2*128=256
541        let target_off1: u16 = 0x60;
542        data[0x58..0x5A].copy_from_slice(&target_off1.to_le_bytes());
543        // next_offset at 0x5A = 128
544        data[0x5A..0x5C].copy_from_slice(&(entry_size as u16).to_le_bytes());
545        // target name at 0x60: "ACCEPT"
546        data[0x60..0x66].copy_from_slice(b"ACCEPT");
547
548        // Entry 2 at offset 128: dst=5.6.7.8, proto=17, target="DROP", next=0
549        let dst2 = u32::from_le_bytes([5, 6, 7, 8]);
550        data[entry_size + 0x04..entry_size + 0x08].copy_from_slice(&dst2.to_le_bytes());
551        data[entry_size + 0x10..entry_size + 0x12].copy_from_slice(&17u16.to_le_bytes()); // udp
552                                                                                          // target_offset for entry 2: since entry 2 starts at 128, target at 128+0x60=0xE0
553                                                                                          // but we need it relative to entry 2's base: 0x60 places it at offset 96 within entry
554                                                                                          // data[entry_size+0x60..] is within our 256-byte buffer
555        let target_off2: u16 = 0x60;
556        data[entry_size + 0x58..entry_size + 0x5A].copy_from_slice(&target_off2.to_le_bytes());
557        data[entry_size + 0x5A..entry_size + 0x5C].copy_from_slice(&0u16.to_le_bytes()); // next=0
558        data[entry_size + 0x60..entry_size + 0x64].copy_from_slice(b"DROP");
559
560        let entry_vaddr: u64 = 0xFFFF_8000_0080_0000;
561        let entry_paddr: u64 = 0x00F0_0000;
562        let reader = make_ipt_reader(&data, entry_vaddr, entry_paddr);
563
564        let rules = parse_ipt_entries(&reader, entry_vaddr, data.len() as u64, "filter").unwrap();
565        assert_eq!(
566            rules.len(),
567            2,
568            "two chained entries should produce two rules"
569        );
570        assert_eq!(rules[0].target, "ACCEPT");
571        assert_eq!(rules[0].protocol, "tcp");
572        assert!(rules[0].source.is_some(), "entry 1 has src_ip");
573        assert_eq!(rules[1].target, "DROP");
574        assert_eq!(rules[1].protocol, "udp");
575        assert!(rules[1].destination.is_some(), "entry 2 has dst_ip");
576    }

603        //   init_net symbol, net.xt at offset 0, netns_xt.tables at offset 0,
604        //   list_head for AF_INET (index 2 × 16 = offset 32) that points to an
605        //   xt_table entry whose "name" field contains "filter".
606        //
607        // init_net_vaddr layout:
608        //   [0..16]  = net.xt (embedded netns_xt, tables at offset 0 within it)
609        //   [32..40] = AF_INET list_head.next → xt_table_vaddr (pointing at the table)
610        //   [40..48] = AF_INET list_head.prev → af_inet_list_vaddr (self)
611        //
612        // xt_table_vaddr layout (same page, offset 0x100):
613        //   list.next @ offset 0 → af_inet_list_vaddr  (back to head → list terminates after one entry)
614        //   list.prev @ offset 8 → af_inet_list_vaddr
615        //   name      @ offset 0x10 → "filter\0"
616        //   (no private/entries fields needed — parse_table_rules stub returns empty)
617
618        let init_net_vaddr: u64 = 0xFFFF_8800_00A1_0000;
619        let init_net_paddr: u64 = 0x00A1_0000;
620
621        let af_inet_offset: u64 = 32; // 2 * list_head_size(16)
622        let af_inet_list_vaddr = init_net_vaddr + af_inet_offset;
623        let xt_table_off: u64 = 0x100;
624        let xt_table_vaddr = init_net_vaddr + xt_table_off;
625
626        let mut page = [0u8; 4096];
627
628        // AF_INET list_head at [32..48]: next=xt_table_vaddr, prev=af_inet_list_vaddr
629        page[32..40].copy_from_slice(&xt_table_vaddr.to_le_bytes()); // list_head.next → xt_table
630        page[40..48].copy_from_slice(&af_inet_list_vaddr.to_le_bytes()); // list_head.prev
631
632        // xt_table at [0x100..]:
633        //   list.next @ 0x100 = af_inet_list_vaddr (back to head, terminates walk_list)
634        //   list.prev @ 0x108 = af_inet_list_vaddr
635        //   name @ 0x110 = "filter\0"
636        page[0x100..0x108].copy_from_slice(&af_inet_list_vaddr.to_le_bytes()); // list.next
637        page[0x108..0x110].copy_from_slice(&af_inet_list_vaddr.to_le_bytes()); // list.prev
638        let name_bytes = b"filter\0";
639        page[0x110..0x110 + name_bytes.len()].copy_from_slice(name_bytes);
640
641        let isf = IsfBuilder::new()
642            .add_symbol("init_net", init_net_vaddr)
643            .add_struct("net", 256)
644            .add_field("net", "xt", 0x00u64, "netns_xt")
645            .add_struct("netns_xt", 256)
646            .add_field("netns_xt", "tables", 0x00u64, "pointer")
647            .add_struct("list_head", 16)
648            .add_field("list_head", "next", 0x00u64, "pointer")
649            .add_field("list_head", "prev", 0x08u64, "pointer")
650            .add_struct("xt_table", 128)
651            .add_field("xt_table", "list", 0x00u64, "list_head")
652            .add_field("xt_table", "name", 0x10u64, "char")
653            .build_json();
654        let resolver = IsfResolver::from_value(&isf).unwrap();
655
656        let (cr3, mem) = PageTableBuilder::new()
657            .map_4k(init_net_vaddr, init_net_paddr, flags::WRITABLE)
658            .write_phys(init_net_paddr, &page)
659            .build();
660        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
661        let reader = ObjectReader::new(vas, Box::new(resolver));
662
663        // parse_table_rules is a stub that returns Ok([]) → total result is empty

675    /// Build a reader wired up with `xt_table` and `xt_table_info` ISF types plus
676    /// the real entry data so that `parse_table_rules` can follow the full chain:
677    ///   xt_table.private → xt_table_info → (entries vaddr, size) → ipt_entry region
678    fn make_parse_table_rules_reader(
679        private_ptr: u64, // value stored in xt_table.private (0 = null test)
680        table_vaddr: u64,
681        table_paddr: u64,
682        table_info_vaddr: u64,
683        table_info_paddr: u64,
684        entries_vaddr: u64,
685        entries_paddr: u64,
686        entry_data: &[u8],
687    ) -> ObjectReader<SyntheticPhysMem> {
688        // ISF: add xt_table.private (pointer), xt_table_info.entries (pointer),
689        //      xt_table_info.size (unsigned long)
690        let isf = IsfBuilder::new()
691            // xt_table: only need the fields parse_table_rules reads
692            .add_struct("xt_table", 256)
693            .add_field("xt_table", "private", 0x00u64, "pointer")
694            // xt_table_info: entries at offset 0, size at offset 8
695            .add_struct("xt_table_info", 256)
696            .add_field("xt_table_info", "entries", 0x00u64, "pointer")
697            .add_field("xt_table_info", "size", 0x08u64, "unsigned long")
698            .build_json();
699        let resolver = IsfResolver::from_value(&isf).unwrap();
700
701        let entry_size = entry_data.len() as u64;
702
703        // xt_table page: private pointer at offset 0
704        let mut table_page = [0u8; 4096];
705        table_page[0x00..0x08].copy_from_slice(&private_ptr.to_le_bytes());
706
707        // xt_table_info page: entries pointer at offset 0, size at offset 8
708        let mut info_page = [0u8; 4096];
709        info_page[0x00..0x08].copy_from_slice(&entries_vaddr.to_le_bytes());
710        info_page[0x08..0x10].copy_from_slice(&entry_size.to_le_bytes());
711
712        let mut builder = PageTableBuilder::new()
713            .map_4k(table_vaddr, table_paddr, flags::PRESENT | flags::WRITABLE)
714            .write_phys(table_paddr, &table_page);
715
716        if private_ptr != 0 {
717            builder = builder
718                .map_4k(
719                    table_info_vaddr,
720                    table_info_paddr,
721                    flags::PRESENT | flags::WRITABLE,
722                )
723                .write_phys(table_info_paddr, &info_page)
724                .map_4k(
725                    entries_vaddr,
726                    entries_paddr,
727                    flags::PRESENT | flags::WRITABLE,
728                )
729                .write_phys(entries_paddr, entry_data);
730        }
731
732        let (cr3, mem) = builder.build();
733        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
734        ObjectReader::new(vas, Box::new(resolver))
735    }
