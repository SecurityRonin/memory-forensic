1//! Linux network protocol handler (`seq_afinfo`) hook detector.
2//!
3//! Linux rootkits commonly replace the `seq_show` function pointer in
4//! `tcp_seq_afinfo`, `udp_seq_afinfo`, and similar protocol handler
5//! structures to hide network connections from `/proc/net/tcp` and
6//! `/proc/net/udp`. This module reads those structs from memory and
7//! compares each `seq_ops` function pointer against the kernel text
8//! range (`_stext`..`_etext`) to detect hooks.
9
10use memf_core::object_reader::ObjectReader;
11use memf_format::PhysicalMemoryProvider;
12
13use crate::Result;
14
15/// Protocol handler symbols to check, paired with human-readable names.
16const AFINFO_SYMBOLS: &[(&str, &str)] = &[
17    ("tcp_seq_afinfo", "tcp"),
18    ("udp_seq_afinfo", "udp"),
19    ("tcp6_seq_afinfo", "tcp6"),
20    ("udp6_seq_afinfo", "udp6"),
21    ("raw_seq_afinfo", "raw"),
22];
23
24/// Function pointer field names within the `seq_operations` struct.
25const SEQ_OPS_FIELDS: &[&str] = &["show", "start", "next", "stop"];
26
27/// Information about a network protocol handler with potential hooks.
28#[derive(Debug, Clone, serde::Serialize)]
29pub struct AfInfoHookInfo {
30    /// Protocol name, e.g. "tcp", "udp", "tcp6", "udp6", "raw".
31    pub protocol: String,
32    /// Kernel symbol name, e.g. "tcp_seq_afinfo".
33    pub struct_name: String,
34    /// Field path that was checked, e.g. "seq_ops.show".
35    pub field: String,
36    /// Virtual address the function pointer targets.
37    pub hook_address: u64,
38    /// Expected module (should be kernel text).
39    pub expected_module: String,
40    /// Where the hook actually points.
41    pub actual_module: String,
42    /// Whether this function pointer is considered hooked.
43    pub is_hooked: bool,
44}
45
46/// Classify whether a function pointer in a `seq_afinfo` struct is hooked.
47///
48/// - Address of `0` is not considered hooked (null/unset pointer).
49/// - Address within `[kernel_start, kernel_end]` is benign (kernel text).
50/// - Address outside that range is suspicious (hooked).
51pub fn classify_afinfo_hook(hook_addr: u64, kernel_start: u64, kernel_end: u64) -> bool {
52    if hook_addr == 0 {
53        return false;
54    }
55    !(kernel_start <= hook_addr && hook_addr <= kernel_end)
56}
57
58/// Walk network protocol handler structs and check for hooks.
59///
60/// Looks up `tcp_seq_afinfo`, `udp_seq_afinfo`, `tcp6_seq_afinfo`,
61/// `udp6_seq_afinfo`, and `raw_seq_afinfo` symbols. For each, reads
62/// the `seq_ops` function pointers (`show`, `start`, `next`, `stop`)
63/// and compares against the kernel text range.
64///
65/// Returns `Ok(Vec::new())` if no afinfo symbols are found (graceful
66/// degradation).
67pub fn walk_check_afinfo<P: PhysicalMemoryProvider>(
68    reader: &ObjectReader<P>,
69) -> Result<Vec<AfInfoHookInfo>> {
70    let symbols = reader.symbols();
71
72    // Resolve kernel text boundaries; if missing, we cannot classify anything.
73    let Some(kernel_start) = symbols.symbol_address("_stext") else {
74        return Ok(Vec::new());
75    };
76    let Some(kernel_end) = symbols.symbol_address("_etext") else {
77        return Ok(Vec::new());
78    };
79
80    let mut results = Vec::new();
81
82    for &(sym_name, protocol) in AFINFO_SYMBOLS {
83        // Graceful degradation: skip symbols that aren't in this profile.
84        let Some(afinfo_addr) = symbols.symbol_address(sym_name) else {
85            continue;
86        };
87
88        // Read the seq_ops pointer from the seq_afinfo struct.
89        let seq_ops_addr: u64 = match reader.read_pointer(afinfo_addr, "seq_afinfo", "seq_ops") {
90            Ok(addr) => addr,
91            Err(_) => continue, // struct layout unavailable, skip
92        };
93
94        if seq_ops_addr == 0 {
95            continue; // No seq_ops set for this protocol
96        }
97
98        // Read each function pointer from the seq_operations struct.
99        for &field_name in SEQ_OPS_FIELDS {
100            let ptr: u64 = match reader.read_pointer(seq_ops_addr, "seq_operations", field_name) {
101                Ok(p) => p,
102                Err(_) => continue,
103            };
104
105            let is_hooked = classify_afinfo_hook(ptr, kernel_start, kernel_end);
106
107            let actual_module = if ptr == 0 {
108                "null".to_string()
109            } else if is_hooked {
110                format!("unknown (0x{ptr:016x})")
111            } else {
112                "kernel".to_string()
113            };
114
115            results.push(AfInfoHookInfo {
116                protocol: protocol.to_string(),
117                struct_name: sym_name.to_string(),
118                field: format!("seq_ops.{field_name}"),
119                hook_address: ptr,
120                expected_module: "kernel".to_string(),
121                actual_module,
122                is_hooked,
123            });
124        }
125    }
126
127    Ok(results)
128}

142    #[test]
143    fn hook_outside_kernel_suspicious() {
144        let kernel_start = 0xFFFF_8000_0000_0000u64;
145        let kernel_end = 0xFFFF_8000_00FF_FFFFu64;
146
147        // Address in module space, well outside kernel text
148        assert!(classify_afinfo_hook(
149            0xFFFF_C900_DEAD_BEEF,
150            kernel_start,
151            kernel_end
152        ));
153        // Address just below kernel start
154        assert!(classify_afinfo_hook(
155            kernel_start - 1,
156            kernel_start,
157            kernel_end
158        ));
159        // Address just above kernel end
160        assert!(classify_afinfo_hook(
161            kernel_end + 1,
162            kernel_start,
163            kernel_end
164        ));
165    }

167    #[test]
168    fn hook_inside_kernel_benign() {
169        let kernel_start = 0xFFFF_8000_0000_0000u64;
170        let kernel_end = 0xFFFF_8000_00FF_FFFFu64;
171
172        // Exactly at start
173        assert!(!classify_afinfo_hook(
174            kernel_start,
175            kernel_start,
176            kernel_end
177        ));
178        // In the middle
179        assert!(!classify_afinfo_hook(
180            kernel_start + 0x1000,
181            kernel_start,
182            kernel_end
183        ));
184        // Exactly at end
185        assert!(!classify_afinfo_hook(kernel_end, kernel_start, kernel_end));
186    }

198    fn classify_multiple_protocols() {
199        let kernel_start = 0xFFFF_8000_0000_0000u64;
200        let kernel_end = 0xFFFF_8000_00FF_FFFFu64;
201        let kernel_func = kernel_start + 0x5000;
202        let module_func = 0xFFFF_C900_1234_0000u64;
203
204        // Simulate checking several protocol handler pointers:
205        // tcp show → kernel (benign)
206        assert!(!classify_afinfo_hook(kernel_func, kernel_start, kernel_end));
207        // udp show → module space (hooked)
208        assert!(classify_afinfo_hook(module_func, kernel_start, kernel_end));
209        // tcp6 show → null (benign)
210        assert!(!classify_afinfo_hook(0, kernel_start, kernel_end));
211        // raw show → just past kernel end (hooked)
212        assert!(classify_afinfo_hook(
213            kernel_end + 0x100,
214            kernel_start,
215            kernel_end
216        ));
217    }

268    #[test]
269    fn walk_check_afinfo_detects_hooked_seq_ops() {
270        let kernel_start: u64 = 0xFFFF_8000_0000_0000;
271        let kernel_end: u64 = 0xFFFF_8000_00FF_FFFF;
272        let afinfo_vaddr: u64 = 0xFFFF_8000_0010_0000;
273        let afinfo_paddr: u64 = 0x0080_0000;
274        let kernel_func: u64 = kernel_start + 0x1000;
275        let hooked_func: u64 = 0xFFFF_C900_DEAD_0000; // Outside kernel text
276
277        // Build seq_afinfo struct: contains a pointer to seq_operations.
278        // seq_operations contains 4 function pointers (show, start, next, stop).
279        // Layout: afinfo has seq_ops at offset 0 (pointer to seq_operations struct).
280        // seq_operations: show=0, start=8, next=16, stop=24.
281        let seq_ops_vaddr: u64 = 0xFFFF_8000_0020_0000;
282        let seq_ops_paddr: u64 = 0x0090_0000;
283
284        // seq_operations data: show is hooked, rest are kernel
285        let mut seq_ops_data = vec![0u8; 4096];
286        seq_ops_data[0..8].copy_from_slice(&hooked_func.to_le_bytes()); // show → hooked
287        seq_ops_data[8..16].copy_from_slice(&kernel_func.to_le_bytes()); // start → kernel
288        seq_ops_data[16..24].copy_from_slice(&kernel_func.to_le_bytes()); // next → kernel
289        seq_ops_data[24..32].copy_from_slice(&kernel_func.to_le_bytes()); // stop → kernel
290
291        // afinfo data: seq_ops pointer at offset 0
292        let mut afinfo_data = vec![0u8; 4096];
293        afinfo_data[0..8].copy_from_slice(&seq_ops_vaddr.to_le_bytes());
294
295        let isf = IsfBuilder::new()
296            .add_struct("seq_afinfo", 64)
297            .add_field("seq_afinfo", "seq_ops", 0, "pointer")
298            .add_struct("seq_operations", 32)
299            .add_field("seq_operations", "show", 0, "pointer")
300            .add_field("seq_operations", "start", 8, "pointer")
301            .add_field("seq_operations", "next", 16, "pointer")
302            .add_field("seq_operations", "stop", 24, "pointer")
303            .add_symbol("_stext", kernel_start)
304            .add_symbol("_etext", kernel_end)
305            .add_symbol("tcp_seq_afinfo", afinfo_vaddr)
306            .build_json();
307
308        let resolver = IsfResolver::from_value(&isf).unwrap();
309        let (cr3, mem) = PageTableBuilder::new()
310            .map_4k(afinfo_vaddr, afinfo_paddr, ptflags::WRITABLE)
311            .write_phys(afinfo_paddr, &afinfo_data)
312            .map_4k(seq_ops_vaddr, seq_ops_paddr, ptflags::WRITABLE)
313            .write_phys(seq_ops_paddr, &seq_ops_data)
314            .build();
315        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
316        let reader = ObjectReader::new(vas, Box::new(resolver));
317
318        let results = walk_check_afinfo(&reader).unwrap();
319
320        // Should find results for tcp protocol
321        assert!(!results.is_empty(), "expected non-empty results");
322
323        // Find the hooked entry (show)
324        let hooked: Vec<_> = results.iter().filter(|r| r.is_hooked).collect();
325        assert_eq!(hooked.len(), 1, "expected exactly one hooked entry");
326        assert_eq!(hooked[0].protocol, "tcp");
327        assert_eq!(hooked[0].field, "seq_ops.show");
328        assert_eq!(hooked[0].hook_address, hooked_func);
329
330        // The other 3 (start, next, stop) should not be hooked
331        let benign: Vec<_> = results.iter().filter(|r| !r.is_hooked).collect();
332        assert_eq!(benign.len(), 3, "expected 3 benign entries");
333    }
