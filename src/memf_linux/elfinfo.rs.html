1//! Linux ELF header extraction from process memory.
2//!
3//! Walks process VMAs and checks for the ELF magic (`\x7fELF`) at the
4//! start of file-backed regions. Extracts ELF header fields to identify
5//! loaded binaries and shared libraries.
6
7use memf_core::object_reader::ObjectReader;
8use memf_format::PhysicalMemoryProvider;
9
10use crate::{ElfInfo, ElfType, Error, Result};
11
12/// ELF magic bytes.
13const ELF_MAGIC: [u8; 4] = [0x7f, b'E', b'L', b'F'];
14
15/// Minimum ELF header size (64-bit).
16const ELF64_HEADER_SIZE: usize = 64;
17
18/// Walk all process VMAs and extract ELF headers.
19///
20/// For each process, walks the VMA list and reads the first
21/// [`ELF64_HEADER_SIZE`] bytes from each region. Regions starting
22/// with the ELF magic are parsed and returned.
23pub fn walk_elfinfo<P: PhysicalMemoryProvider>(reader: &ObjectReader<P>) -> Result<Vec<ElfInfo>> {
24    let init_task_addr = reader
25        .symbols()
26        .symbol_address("init_task")
27        .ok_or_else(|| Error::Walker("symbol 'init_task' not found".into()))?;
28
29    let tasks_offset = reader
30        .symbols()
31        .field_offset("task_struct", "tasks")
32        .ok_or_else(|| Error::Walker("task_struct.tasks field not found".into()))?;
33
34    let head_vaddr = init_task_addr + tasks_offset;
35    let task_addrs = reader.walk_list(head_vaddr, "task_struct", "tasks")?;
36
37    let mut results = Vec::new();
38
39    scan_process_elfs(reader, init_task_addr, &mut results);
40
41    for &task_addr in &task_addrs {
42        scan_process_elfs(reader, task_addr, &mut results);
43    }
44
45    Ok(results)
46}
47
48/// Scan a single process's VMAs for ELF headers.
49fn scan_process_elfs<P: PhysicalMemoryProvider>(
50    reader: &ObjectReader<P>,
51    task_addr: u64,
52    out: &mut Vec<ElfInfo>,
53) {
54    let mm_ptr: u64 = match reader.read_field(task_addr, "task_struct", "mm") {
55        Ok(v) => v,
56        Err(_) => return,
57    };
58    if mm_ptr == 0 {
59        return;
60    }
61
62    let pid: u32 = match reader.read_field(task_addr, "task_struct", "pid") {
63        Ok(v) => v,
64        Err(_) => return,
65    };
66    let comm = reader
67        .read_field_string(task_addr, "task_struct", "comm", 16)
68        .unwrap_or_default();
69
70    let mmap_ptr: u64 = match reader.read_field(mm_ptr, "mm_struct", "mmap") {
71        Ok(v) => v,
72        Err(_) => return,
73    };
74
75    let mut vma_addr = mmap_ptr;
76    while vma_addr != 0 {
77        let vm_start: u64 = match reader.read_field(vma_addr, "vm_area_struct", "vm_start") {
78            Ok(v) => v,
79            Err(_) => break,
80        };
81
82        // Read the first 64 bytes and check for ELF magic
83        if let Ok(header_bytes) = reader.read_bytes(vm_start, ELF64_HEADER_SIZE) {
84            if let Some(info) = parse_elf64_header(&header_bytes, u64::from(pid), &comm, vm_start) {
85                out.push(info);
86            }
87        }
88
89        vma_addr = reader
90            .read_field(vma_addr, "vm_area_struct", "vm_next")
91            .unwrap_or(0);
92    }
93}

95/// Parse a 64-bit ELF header from raw bytes.
96///
97/// Returns `None` if the magic doesn't match or the header is too short.
98fn parse_elf64_header(bytes: &[u8], pid: u64, comm: &str, vma_start: u64) -> Option<ElfInfo> {
99    if bytes.len() < ELF64_HEADER_SIZE {
100        return None;
101    }
102    if bytes[0..4] != ELF_MAGIC {
103        return None;
104    }
105    // Verify ELFCLASS64 (e_ident[4] == 2)
106    if bytes[4] != 2 {
107        return None;
108    }
109
110    let e_type = u16::from_le_bytes(bytes[16..18].try_into().unwrap());
111    let e_machine = u16::from_le_bytes(bytes[18..20].try_into().unwrap());
112    let e_entry = u64::from_le_bytes(bytes[24..32].try_into().unwrap());
113
114    Some(ElfInfo {
115        pid,
116        comm: comm.to_string(),
117        vma_start,
118        elf_type: ElfType::from_raw(e_type),
119        machine: e_machine,
120        entry_point: e_entry,
121    })
122}

132    fn make_test_reader(
133        data: &[u8],
134        vaddr: u64,
135        paddr: u64,
136        extra_mappings: &[(u64, u64, &[u8])],
137    ) -> ObjectReader<SyntheticPhysMem> {
138        let isf = IsfBuilder::new()
139            .add_struct("task_struct", 128)
140            .add_field("task_struct", "pid", 0, "int")
141            .add_field("task_struct", "state", 4, "long")
142            .add_field("task_struct", "tasks", 16, "list_head")
143            .add_field("task_struct", "comm", 32, "char")
144            .add_field("task_struct", "mm", 48, "pointer")
145            .add_struct("list_head", 16)
146            .add_field("list_head", "next", 0, "pointer")
147            .add_field("list_head", "prev", 8, "pointer")
148            .add_struct("mm_struct", 128)
149            .add_field("mm_struct", "pgd", 0, "pointer")
150            .add_field("mm_struct", "mmap", 8, "pointer")
151            .add_struct("vm_area_struct", 64)
152            .add_field("vm_area_struct", "vm_start", 0, "unsigned long")
153            .add_field("vm_area_struct", "vm_end", 8, "unsigned long")
154            .add_field("vm_area_struct", "vm_next", 16, "pointer")
155            .add_field("vm_area_struct", "vm_flags", 24, "unsigned long")
156            .add_field("vm_area_struct", "vm_pgoff", 32, "unsigned long")
157            .add_field("vm_area_struct", "vm_file", 40, "pointer")
158            .add_symbol("init_task", vaddr)
159            .build_json();
160
161        let resolver = IsfResolver::from_value(&isf).unwrap();
162        let mut builder = PageTableBuilder::new()
163            .map_4k(vaddr, paddr, ptflags::WRITABLE)
164            .write_phys(paddr, data);
165
166        for &(ev, ep, edata) in extra_mappings {
167            builder = builder
168                .map_4k(ev, ep, ptflags::WRITABLE)
169                .write_phys(ep, edata);
170        }
171
172        let (cr3, mem) = builder.build();
173        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
174        ObjectReader::new(vas, Box::new(resolver))
175    }

177    /// Build a minimal ELF64 header for testing.
178    fn build_elf64_header(elf_type: u16, machine: u16, entry: u64) -> Vec<u8> {
179        let mut hdr = vec![0u8; 4096];
180        // e_ident
181        hdr[0..4].copy_from_slice(&ELF_MAGIC);
182        hdr[4] = 2; // ELFCLASS64
183        hdr[5] = 1; // ELFDATA2LSB
184        hdr[6] = 1; // EV_CURRENT
185                    // e_type (offset 16)
186        hdr[16..18].copy_from_slice(&elf_type.to_le_bytes());
187        // e_machine (offset 18)
188        hdr[18..20].copy_from_slice(&machine.to_le_bytes());
189        // e_version (offset 20)
190        hdr[20..24].copy_from_slice(&1u32.to_le_bytes());
191        // e_entry (offset 24)
192        hdr[24..32].copy_from_slice(&entry.to_le_bytes());
193        hdr
194    }

196    #[test]
197    fn detects_elf_in_process_vma() {
198        let vaddr: u64 = 0xFFFF_8000_0010_0000;
199        let paddr: u64 = 0x0080_0000;
200        let mut data = vec![0u8; 4096];
201
202        // init_task (PID 1, "cat")
203        data[0..4].copy_from_slice(&1u32.to_le_bytes());
204        let tasks_addr = vaddr + 16;
205        data[16..24].copy_from_slice(&tasks_addr.to_le_bytes());
206        data[24..32].copy_from_slice(&tasks_addr.to_le_bytes());
207        data[32..35].copy_from_slice(b"cat");
208        let mm_addr = vaddr + 0x200;
209        data[48..56].copy_from_slice(&mm_addr.to_le_bytes());
210
211        // mm_struct
212        data[0x200..0x208].copy_from_slice(&0x1000u64.to_le_bytes()); // pgd
213        let vma_addr = vaddr + 0x300;
214        data[0x208..0x210].copy_from_slice(&vma_addr.to_le_bytes()); // mmap
215
216        // VMA: code segment (r-x, file-backed)
217        let code_vaddr: u64 = 0x0000_5555_0000_0000;
218        let code_paddr: u64 = 0x0090_0000;
219        data[0x300..0x308].copy_from_slice(&code_vaddr.to_le_bytes());
220        data[0x308..0x310].copy_from_slice(&(code_vaddr + 0x1000).to_le_bytes());
221        data[0x310..0x318].copy_from_slice(&0u64.to_le_bytes()); // vm_next = NULL
222        data[0x318..0x320].copy_from_slice(&0x5u64.to_le_bytes()); // r-x
223        data[0x328..0x330].copy_from_slice(&0xABCDu64.to_le_bytes()); // vm_file non-null
224
225        let elf = build_elf64_header(
226            3,  // ET_DYN (PIE executable)
227            62, // EM_X86_64
228            0x0000_5555_0000_1000,
229        );
230
231        let reader = make_test_reader(&data, vaddr, paddr, &[(code_vaddr, code_paddr, &elf)]);
232        let results = walk_elfinfo(&reader).unwrap();
233
234        assert_eq!(results.len(), 1);
235        assert_eq!(results[0].pid, 1);
236        assert_eq!(results[0].comm, "cat");
237        assert_eq!(results[0].elf_type, ElfType::SharedObject);
238        assert_eq!(results[0].machine, 62);
239        assert_eq!(results[0].entry_point, 0x0000_5555_0000_1000);
240    }
