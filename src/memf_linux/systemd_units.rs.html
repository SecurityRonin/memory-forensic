1//! In-memory systemd unit analysis.
2//!
3//! Scans the `systemd` (PID 1) process VMAs for unit file content patterns
4//! (`.service`, `.timer` strings and associated `ExecStart=` commands) to
5//! detect malicious persistence (MITRE ATT&CK T1543.002).
6
7use memf_core::object_reader::ObjectReader;
8use memf_format::PhysicalMemoryProvider;
9
10use crate::Result;
11
12/// Page-sized chunk for VMA scanning.
13const SCAN_CHUNK: usize = 4096;
14
15/// How many bytes to search forward/backward for ExecStart.
16const EXEC_SEARCH_WINDOW: usize = 512;
17
18/// Information about a systemd unit found in memory.
19#[derive(Debug, Clone)]
20pub struct SystemdUnitInfo {
21    /// Unit name, e.g. "evil.service".
22    pub unit_name: String,
23    /// ExecStart command found nearby in memory.
24    pub exec_start: String,
25    /// Virtual address of the VMA where the unit name was found.
26    pub vma_start: u64,
27    /// Unit type: "service", "timer", "socket", "path", "mount".
28    pub unit_type: String,
29    /// True if the unit is considered suspicious.
30    pub is_suspicious: bool,
31}
32
33/// Suspicious ExecStart patterns.
34const SUSPICIOUS_EXEC_PATTERNS: &[&str] = &[
35    "/tmp/",
36    "/dev/shm/",
37    "/var/tmp/",
38    "curl",
39    "wget",
40    "bash -c",
41    "sh -c",
42    "python",
43    "perl",
44    "ruby",
45    "nc ",
46    "ncat",
47    "base64",
48];
49
50/// ExecStart prefixes considered safe.
51const SAFE_EXEC_PREFIXES: &[&str] = &["/usr/", "/bin/", "/sbin/", "/lib/"];
52
53/// Known safe unit name prefixes.
54const KNOWN_SAFE_UNITS: &[&str] = &["systemd-", "NetworkManager", "dbus", "cron", "ssh"];
55
56/// Unit file extensions we look for.
57const UNIT_EXTENSIONS: &[&str] = &[".service", ".timer", ".socket", ".path", ".mount"];
58
59/// Classify whether a systemd unit is suspicious.
60///
61/// Suspicious if:
62/// - `exec_start` contains a suspicious pattern, OR
63/// - `unit_name` looks like a randomized hex name (8+ lowercase hex chars + extension), OR
64/// - `exec_start` contains base64 indicators.
65///
66/// Not suspicious if exec_start starts with a safe prefix or the unit name
67/// is from a known system service.
68pub fn classify_systemd_unit(unit_name: &str, exec_start: &str) -> bool {
69    // Known safe units are never suspicious.
70    if KNOWN_SAFE_UNITS
71        .iter()
72        .any(|prefix| unit_name.starts_with(prefix))
73    {
74        return false;
75    }
76
77    // Safe ExecStart prefix — not suspicious.
78    if SAFE_EXEC_PREFIXES
79        .iter()
80        .any(|prefix| exec_start.starts_with(prefix))
81    {
82        return false;
83    }
84
85    // Suspicious ExecStart patterns.
86    if SUSPICIOUS_EXEC_PATTERNS
87        .iter()
88        .any(|pat| exec_start.contains(pat))
89    {
90        return true;
91    }
92
93    // Randomized name: strip extension, check if remainder is 8+ lowercase hex chars.
94    let stem = UNIT_EXTENSIONS
95        .iter()
96        .find_map(|ext| unit_name.strip_suffix(ext))
97        .unwrap_or(unit_name);
98    if stem.len() >= 8
99        && stem
100            .chars()
101            .all(|c| c.is_ascii_hexdigit() && !c.is_ascii_uppercase())
102    {
103        return true;
104    }
105
106    false
107}

109/// Walk the systemd process VMAs and extract unit information from memory strings.
110///
111/// Returns `Ok(vec![])` if `init_task` symbol is missing.
112pub fn walk_systemd_units<P: PhysicalMemoryProvider>(
113    reader: &ObjectReader<P>,
114) -> Result<Vec<SystemdUnitInfo>> {
115    let init_task_addr = match reader.symbols().symbol_address("init_task") {
116        Some(a) => a,
117        None => return Ok(vec![]),
118    };
119
120    let tasks_offset = match reader.symbols().field_offset("task_struct", "tasks") {
121        Some(o) => o,
122        None => return Ok(vec![]),
123    };
124
125    let head_vaddr = init_task_addr + tasks_offset;
126    let task_addrs = reader.walk_list(head_vaddr, "task_struct", "tasks")?;
127
128    // Include init_task itself (PID 1 = systemd on modern Linux).
129    let all_tasks = std::iter::once(init_task_addr).chain(task_addrs);
130
131    for task_addr in all_tasks {
132        let pid: u32 = match reader.read_field(task_addr, "task_struct", "pid") {
133            Ok(v) => v,
134            Err(_) => continue,
135        };
136        let comm = reader
137            .read_field_string(task_addr, "task_struct", "comm", 16)
138            .unwrap_or_default();
139
140        // Find systemd: comm == "systemd" and pid == 1.
141        if pid == 1 && comm == "systemd" {
142            return Ok(scan_systemd_vmas(reader, task_addr));
143        }
144    }
145
146    Ok(vec![])
147}

149/// Scan the systemd process's VMAs for unit content strings.
150fn scan_systemd_vmas<P: PhysicalMemoryProvider>(
151    reader: &ObjectReader<P>,
152    task_addr: u64,
153) -> Vec<SystemdUnitInfo> {
154    let mm_ptr: u64 = match reader.read_field(task_addr, "task_struct", "mm") {
155        Ok(v) => v,
156        Err(_) => return vec![],
157    };
158    if mm_ptr == 0 {
159        return vec![];
160    }
161
162    let mmap_ptr: u64 = match reader.read_field(mm_ptr, "mm_struct", "mmap") {
163        Ok(v) => v,
164        Err(_) => return vec![],
165    };
166
167    let mut findings = Vec::new();
168    let mut vma_addr = mmap_ptr;
169
170    while vma_addr != 0 {
171        let vm_start: u64 = reader
172            .read_field(vma_addr, "vm_area_struct", "vm_start")
173            .unwrap_or(0);
174        let vm_end: u64 = reader
175            .read_field(vma_addr, "vm_area_struct", "vm_end")
176            .unwrap_or(0);
177        let vm_flags: u64 = reader
178            .read_field(vma_addr, "vm_area_struct", "vm_flags")
179            .unwrap_or(0);
180
181        // Only scan readable, non-execute VMAs (data/heap, not code).
182        let readable = (vm_flags & 0x1) != 0;
183        let executable = (vm_flags & 0x4) != 0;
184        if readable && !executable && vm_start < vm_end {
185            scan_vma_for_units(reader, vm_start, vm_end, &mut findings);
186        }
187
188        vma_addr = reader
189            .read_field(vma_addr, "vm_area_struct", "vm_next")
190            .unwrap_or(0);
191    }
192
193    findings
194}

196/// Scan a VMA's address range in chunks for unit name strings.
197fn scan_vma_for_units<P: PhysicalMemoryProvider>(
198    reader: &ObjectReader<P>,
199    vm_start: u64,
200    vm_end: u64,
201    out: &mut Vec<SystemdUnitInfo>,
202) {
203    let mut offset: u64 = 0;
204    let total = vm_end - vm_start;
205
206    while offset < total {
207        let chunk_size = SCAN_CHUNK.min((total - offset) as usize);
208        let chunk_addr = vm_start + offset;
209        let bytes = if let Ok(b) = reader.read_bytes(chunk_addr, chunk_size) { b } else {
210            offset += chunk_size as u64;
211            continue;
212        };
213
214        // Scan chunk for unit name markers.
215        for ext in UNIT_EXTENSIONS {
216            let ext_bytes = ext.as_bytes();
217            let mut search_start = 0usize;
218            while let Some(pos) = find_subsequence(&bytes[search_start..], ext_bytes) {
219                let abs_pos = search_start + pos;
220                // Walk backwards from abs_pos to find the start of the unit name.
221                let name_start = find_name_start(&bytes, abs_pos);
222                let name_end = abs_pos + ext_bytes.len();
223                if let Ok(unit_name) = std::str::from_utf8(&bytes[name_start..name_end]) {
224                    let unit_name = unit_name.to_string();
225                    let unit_type = ext.trim_start_matches('.').to_string();
226
227                    // Search forward/backward in the chunk for ExecStart=.
228                    let exec_start = find_exec_start(&bytes, abs_pos);
229
230                    let is_suspicious = classify_systemd_unit(&unit_name, &exec_start);
231                    out.push(SystemdUnitInfo {
232                        unit_name,
233                        exec_start,
234                        vma_start: vm_start,
235                        unit_type,
236                        is_suspicious,
237                    });
238                }
239                search_start = abs_pos + 1;
240            }
241        }
242
243        offset += chunk_size as u64;
244    }
245}

247/// Find the first occurrence of `needle` in `haystack`.
248fn find_subsequence(haystack: &[u8], needle: &[u8]) -> Option<usize> {
249    if needle.is_empty() || needle.len() > haystack.len() {
250        return None;
251    }
252    haystack.windows(needle.len()).position(|w| w == needle)
253}
254
255/// Walk backwards from `pos` to find the start of a unit file name (stops at
256/// whitespace, NUL, `=`, or `\n`).
257fn find_name_start(bytes: &[u8], pos: usize) -> usize {
258    let mut i = pos;
259    while i > 0 {
260        let c = bytes[i - 1];
261        if c == 0 || c == b'\n' || c == b'\r' || c == b' ' || c == b'\t' || c == b'=' {
262            break;
263        }
264        i -= 1;
265    }
266    i
267}
268
269/// Search `±EXEC_SEARCH_WINDOW` bytes around `pos` in `bytes` for an
270/// `ExecStart=` marker and extract the command value.
271fn find_exec_start(bytes: &[u8], pos: usize) -> String {
272    let search_start = pos.saturating_sub(EXEC_SEARCH_WINDOW);
273    let search_end = (pos + EXEC_SEARCH_WINDOW).min(bytes.len());
274    let window = &bytes[search_start..search_end];
275
276    let marker = b"ExecStart=";
277    if let Some(idx) = find_subsequence(window, marker) {
278        let value_start = idx + marker.len();
279        let value_bytes = &window[value_start..];
280        let end = value_bytes
281            .iter()
282            .position(|&b| b == 0 || b == b'\n' || b == b'\r')
283            .unwrap_or(value_bytes.len());
284        return String::from_utf8_lossy(&value_bytes[..end]).into_owned();
285    }
286
287    String::new()
288}

392            .add_field("task_struct", "comm", 32, "char")
393            .add_field("task_struct", "mm", 48, "pointer")
394            .add_struct("list_head", 16)
395            .add_field("list_head", "next", 0, "pointer")
396            .add_field("list_head", "prev", 8, "pointer")
397            .add_struct("mm_struct", 64)
398            .add_field("mm_struct", "mmap", 8, "pointer")

400            .add_field("vm_area_struct", "vm_start", 0, "unsigned long")

716    // ---------------------------------------------------------------------------
717    // walk_systemd_units: full path — systemd found, mm non-null, VMA with
718    // readable+non-exec flags, VMA data contains a unit extension string.
719    // ---------------------------------------------------------------------------
720
721    #[test]
722    fn walk_systemd_units_scans_readable_vma_for_units() {
723        // Build a synthetic memory where:
724        //   init_task (pid=1, comm="systemd") → mm → VMA (readable, non-exec)
725        //   VMA data contains "evil.service\0"
726        let task_vaddr: u64 = 0xFFFF_8800_0100_0000;
727        let task_paddr: u64 = 0x00F0_0000;
728        let mm_vaddr: u64 = 0xFFFF_8800_0101_0000;
729        let mm_paddr: u64 = 0x00F1_0000;
730        let vma_vaddr: u64 = 0xFFFF_8800_0102_0000;
731        let vma_paddr: u64 = 0x00F2_0000;
732        // The actual data page that the VMA points at
733        let data_vaddr: u64 = 0xFFFF_8800_0103_0000;
734        let data_paddr: u64 = 0x00F3_0000;
735
736        let tasks_offset: u64 = 16;
737
738        // task_struct page
739        let mut task_page = [0u8; 4096];
740        // pid = 1
741        task_page[0..4].copy_from_slice(&1u32.to_le_bytes());
742        // tasks: self-pointing (only init_task in list)
743        let list_self = task_vaddr + tasks_offset;
744        task_page[tasks_offset as usize..tasks_offset as usize + 8]
745            .copy_from_slice(&list_self.to_le_bytes());
746        task_page[tasks_offset as usize + 8..tasks_offset as usize + 16]
747            .copy_from_slice(&list_self.to_le_bytes());
748        // comm = "systemd"
749        task_page[32..39].copy_from_slice(b"systemd");
750        // mm pointer at offset 48
751        task_page[48..56].copy_from_slice(&mm_vaddr.to_le_bytes());
752
753        // mm_struct page: mmap at offset 8
754        let mut mm_page = [0u8; 4096];
755        mm_page[8..16].copy_from_slice(&vma_vaddr.to_le_bytes());
756
757        // vm_area_struct page
758        let mut vma_page = [0u8; 4096];
759        vma_page[0..8].copy_from_slice(&data_vaddr.to_le_bytes()); // vm_start
760        let data_end = data_vaddr + 4096u64;
761        vma_page[8..16].copy_from_slice(&data_end.to_le_bytes()); // vm_end
762        vma_page[16..24].copy_from_slice(&0u64.to_le_bytes()); // vm_next = 0
763                                                               // vm_flags: readable (bit 0) = 1, not executable (bit 2) = 0 → 0x1
764        vma_page[24..32].copy_from_slice(&0x1u64.to_le_bytes());
765
766        // Data page: put "evil.service\0" near the start
767        let mut data_page = [0u8; 4096];
768        let unit = b"evil.service\0";
769        data_page[100..100 + unit.len()].copy_from_slice(unit);
770
771        let isf = IsfBuilder::new()
772            .add_struct("task_struct", 128)
773            .add_field("task_struct", "pid", 0x00u64, "unsigned int")
774            .add_field("task_struct", "tasks", 16u64, "list_head")
775            .add_field("task_struct", "comm", 32u64, "char")
776            .add_field("task_struct", "mm", 48u64, "pointer")
777            .add_struct("list_head", 16)
778            .add_field("list_head", "next", 0x00u64, "pointer")
779            .add_field("list_head", "prev", 0x08u64, "pointer")
780            .add_struct("mm_struct", 64)
781            .add_field("mm_struct", "mmap", 8u64, "pointer")
782            .add_struct("vm_area_struct", 64)
783            .add_field("vm_area_struct", "vm_start", 0x00u64, "unsigned long")
784            .add_field("vm_area_struct", "vm_end", 0x08u64, "unsigned long")
785            .add_field("vm_area_struct", "vm_next", 0x10u64, "pointer")
786            .add_field("vm_area_struct", "vm_flags", 0x18u64, "unsigned long")
787            .add_symbol("init_task", task_vaddr)
788            .build_json();
789        let resolver = IsfResolver::from_value(&isf).unwrap();
790
791        let (cr3, mem) = PageTableBuilder::new()
792            .map_4k(task_vaddr, task_paddr, ptflags::WRITABLE)
793            .write_phys(task_paddr, &task_page)
794            .map_4k(mm_vaddr, mm_paddr, ptflags::WRITABLE)
795            .write_phys(mm_paddr, &mm_page)
796            .map_4k(vma_vaddr, vma_paddr, ptflags::WRITABLE)
797            .write_phys(vma_paddr, &vma_page)
798            .map_4k(data_vaddr, data_paddr, ptflags::WRITABLE)
799            .write_phys(data_paddr, &data_page)
800            .build();
801        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
802        let reader = ObjectReader::new(vas, Box::new(resolver));
803
804        let result = walk_systemd_units(&reader).unwrap_or_default();
805        // We should find "evil.service" in the VMA
806        assert!(
807            result.iter().any(|u| u.unit_name.contains(".service")),
808            "should detect .service extension in VMA data; got: {:?}",
809            result.iter().map(|u| &u.unit_name).collect::<Vec<_>>()
810        );
811    }

817    #[test]
818    fn walk_systemd_units_exec_vma_skipped() {
819        // VMA has readable+executable flags → should not be scanned → no unit found
820        let task_vaddr: u64 = 0xFFFF_8800_0200_0000;
821        let task_paddr: u64 = 0x00F4_0000;
822        let mm_vaddr: u64 = 0xFFFF_8800_0201_0000;
823        let mm_paddr: u64 = 0x00F5_0000;
824        let vma_vaddr: u64 = 0xFFFF_8800_0202_0000;
825        let vma_paddr: u64 = 0x00F6_0000;
826        let data_vaddr: u64 = 0xFFFF_8800_0203_0000;
827        let data_paddr: u64 = 0x00F7_0000;
828
829        let tasks_offset: u64 = 16;
830
831        let mut task_page = [0u8; 4096];
832        task_page[0..4].copy_from_slice(&1u32.to_le_bytes());
833        let list_self = task_vaddr + tasks_offset;
834        task_page[tasks_offset as usize..tasks_offset as usize + 8]
835            .copy_from_slice(&list_self.to_le_bytes());
836        task_page[tasks_offset as usize + 8..tasks_offset as usize + 16]
837            .copy_from_slice(&list_self.to_le_bytes());
838        task_page[32..39].copy_from_slice(b"systemd");
839        task_page[48..56].copy_from_slice(&mm_vaddr.to_le_bytes());
840
841        let mut mm_page = [0u8; 4096];
842        mm_page[8..16].copy_from_slice(&vma_vaddr.to_le_bytes());
843
844        let mut vma_page = [0u8; 4096];
845        vma_page[0..8].copy_from_slice(&data_vaddr.to_le_bytes());
846        vma_page[8..16].copy_from_slice(&(data_vaddr + 4096).to_le_bytes());
847        vma_page[16..24].copy_from_slice(&0u64.to_le_bytes());
848        // vm_flags: readable (bit 0) + executable (bit 2) = 0x5
849        vma_page[24..32].copy_from_slice(&0x5u64.to_le_bytes());
850
851        let mut data_page = [0u8; 4096];
852        data_page[100..113].copy_from_slice(b"evil.service\0");
853
854        let isf = IsfBuilder::new()
855            .add_struct("task_struct", 128)
856            .add_field("task_struct", "pid", 0x00u64, "unsigned int")
857            .add_field("task_struct", "tasks", 16u64, "list_head")
858            .add_field("task_struct", "comm", 32u64, "char")
859            .add_field("task_struct", "mm", 48u64, "pointer")
860            .add_struct("list_head", 16)
861            .add_field("list_head", "next", 0x00u64, "pointer")
862            .add_field("list_head", "prev", 0x08u64, "pointer")
863            .add_struct("mm_struct", 64)
864            .add_field("mm_struct", "mmap", 8u64, "pointer")
865            .add_struct("vm_area_struct", 64)
866            .add_field("vm_area_struct", "vm_start", 0x00u64, "unsigned long")
867            .add_field("vm_area_struct", "vm_end", 0x08u64, "unsigned long")
868            .add_field("vm_area_struct", "vm_next", 0x10u64, "pointer")
869            .add_field("vm_area_struct", "vm_flags", 0x18u64, "unsigned long")
870            .add_symbol("init_task", task_vaddr)
871            .build_json();
872        let resolver = IsfResolver::from_value(&isf).unwrap();
873
874        let (cr3, mem) = PageTableBuilder::new()
875            .map_4k(task_vaddr, task_paddr, ptflags::WRITABLE)
876            .write_phys(task_paddr, &task_page)
877            .map_4k(mm_vaddr, mm_paddr, ptflags::WRITABLE)
878            .write_phys(mm_paddr, &mm_page)
879            .map_4k(vma_vaddr, vma_paddr, ptflags::WRITABLE)
880            .write_phys(vma_paddr, &vma_page)
881            .map_4k(data_vaddr, data_paddr, ptflags::WRITABLE)
882            .write_phys(data_paddr, &data_page)
883            .build();
884        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
885        let reader = ObjectReader::new(vas, Box::new(resolver));
886
887        let result = walk_systemd_units(&reader).unwrap_or_default();
888        assert!(
889            result.is_empty(),
890            "executable VMA must not be scanned; found: {:?}",
891            result.iter().map(|u| &u.unit_name).collect::<Vec<_>>()
892        );
893    }
