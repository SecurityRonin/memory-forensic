1//! Linux process memory map (VMA) walker.
2//!
3//! Enumerates virtual memory areas by walking the `vm_area_struct` singly-linked
4//! list from `mm_struct.mmap` for each process in the task list.
5
6use memf_core::object_reader::ObjectReader;
7use memf_format::PhysicalMemoryProvider;
8
9use crate::{Error, Result, VmaFlags, VmaInfo};
10
11/// Walk all process VMAs from the task list.
12///
13/// For each process, follows `task_struct.mm → mm_struct.mmap` to the head
14/// of the `vm_area_struct` chain, then traverses via `vm_next` pointers.
15pub fn walk_maps<P: PhysicalMemoryProvider>(reader: &ObjectReader<P>) -> Result<Vec<VmaInfo>> {
16    let init_task_addr = reader
17        .symbols()
18        .symbol_address("init_task")
19        .ok_or_else(|| Error::Walker("symbol 'init_task' not found".into()))?;
20
21    let tasks_offset = reader
22        .symbols()
23        .field_offset("task_struct", "tasks")
24        .ok_or_else(|| Error::Walker("task_struct.tasks field not found".into()))?;
25
26    let head_vaddr = init_task_addr + tasks_offset;
27    let task_addrs = reader.walk_list(head_vaddr, "task_struct", "tasks")?;
28
29    let mut all_vmas = Vec::new();
30
31    // Include init_task itself
32    collect_process_vmas(reader, init_task_addr, &mut all_vmas);
33
34    for &task_addr in &task_addrs {
35        collect_process_vmas(reader, task_addr, &mut all_vmas);
36    }
37
38    Ok(all_vmas)
39}

41/// Collect VMAs for a single process, silently skipping kernel threads (mm == NULL).
42fn collect_process_vmas<P: PhysicalMemoryProvider>(
43    reader: &ObjectReader<P>,
44    task_addr: u64,
45    out: &mut Vec<VmaInfo>,
46) {
47    let mm_ptr: u64 = match reader.read_field(task_addr, "task_struct", "mm") {
48        Ok(v) => v,
49        Err(_) => return,
50    };
51    if mm_ptr == 0 {
52        return; // kernel thread — no user VMAs
53    }
54
55    if let Ok(vmas) = walk_process_maps(reader, task_addr) {
56        out.extend(vmas);
57    }
58}

60/// Walk VMAs for a single process given its `task_struct` address.
61pub fn walk_process_maps<P: PhysicalMemoryProvider>(
62    reader: &ObjectReader<P>,
63    task_addr: u64,
64) -> Result<Vec<VmaInfo>> {
65    let pid: u32 = reader.read_field(task_addr, "task_struct", "pid")?;
66    let comm = reader.read_field_string(task_addr, "task_struct", "comm", 16)?;
67    let mm_ptr: u64 = reader.read_field(task_addr, "task_struct", "mm")?;
68
69    if mm_ptr == 0 {
70        return Err(Error::Walker(format!(
71            "task {comm} (PID {pid}) has NULL mm (kernel thread)"
72        )));
73    }
74
75    let mmap_ptr: u64 = reader.read_field(mm_ptr, "mm_struct", "mmap")?;
76
77    let mut vmas = Vec::new();
78    let mut vma_addr = mmap_ptr;
79
80    // Walk the singly-linked vm_next chain (NULL-terminated)
81    while vma_addr != 0 {
82        let vm_start: u64 = reader.read_field(vma_addr, "vm_area_struct", "vm_start")?;
83        let vm_end: u64 = reader.read_field(vma_addr, "vm_area_struct", "vm_end")?;
84        let vm_flags: u64 = reader.read_field(vma_addr, "vm_area_struct", "vm_flags")?;
85        let vm_pgoff: u64 = reader.read_field(vma_addr, "vm_area_struct", "vm_pgoff")?;
86        let vm_file: u64 = reader.read_field(vma_addr, "vm_area_struct", "vm_file")?;
87
88        vmas.push(VmaInfo {
89            pid: u64::from(pid),
90            comm: comm.clone(),
91            start: vm_start,
92            end: vm_end,
93            flags: VmaFlags::from_raw(vm_flags),
94            pgoff: vm_pgoff,
95            file_backed: vm_file != 0,
96        });
97
98        vma_addr = reader.read_field(vma_addr, "vm_area_struct", "vm_next")?;
99    }
100
101    Ok(vmas)
102}

113        let isf = IsfBuilder::new()
114            // task_struct
115            .add_struct("task_struct", 128)
116            .add_field("task_struct", "pid", 0, "int")
117            .add_field("task_struct", "state", 4, "long")
118            .add_field("task_struct", "tasks", 16, "list_head")
119            .add_field("task_struct", "comm", 32, "char")
120            .add_field("task_struct", "mm", 48, "pointer")
121            .add_field("task_struct", "real_parent", 56, "pointer")
122            // list_head
123            .add_struct("list_head", 16)
124            .add_field("list_head", "next", 0, "pointer")
125            .add_field("list_head", "prev", 8, "pointer")
126            // mm_struct
127            .add_struct("mm_struct", 128)
128            .add_field("mm_struct", "pgd", 0, "pointer")
129            .add_field("mm_struct", "mmap", 8, "pointer")
130            // vm_area_struct
131            .add_struct("vm_area_struct", 64)
132            .add_field("vm_area_struct", "vm_start", 0, "unsigned long")
133            .add_field("vm_area_struct", "vm_end", 8, "unsigned long")
134            .add_field("vm_area_struct", "vm_next", 16, "pointer")
135            .add_field("vm_area_struct", "vm_flags", 24, "unsigned long")
136            .add_field("vm_area_struct", "vm_pgoff", 32, "unsigned long")
137            .add_field("vm_area_struct", "vm_file", 40, "pointer")
138            // symbol
139            .add_symbol("init_task", vaddr)
140            .build_json();
141
142        let resolver = IsfResolver::from_value(&isf).unwrap();
143        let (cr3, mem) = PageTableBuilder::new()
144            .map_4k(vaddr, paddr, flags::WRITABLE)
145            .write_phys(paddr, data)
146            .build();
147        let vas = VirtualAddressSpace::new(mem, cr3, TranslationMode::X86_64FourLevel);
148        ObjectReader::new(vas, Box::new(resolver))
149    }

151    #[test]
152    fn walk_single_process_two_vmas() {
153        let vaddr: u64 = 0xFFFF_8000_0010_0000;
154        let paddr: u64 = 0x0080_0000;
155        let mut data = vec![0u8; 4096];
156
157        // init_task (PID 1, "systemd")
158        data[0..4].copy_from_slice(&1u32.to_le_bytes()); // pid
159        data[4..12].copy_from_slice(&0i64.to_le_bytes()); // state
160        let tasks_addr = vaddr + 16;
161        data[16..24].copy_from_slice(&tasks_addr.to_le_bytes()); // tasks.next → self
162        data[24..32].copy_from_slice(&tasks_addr.to_le_bytes()); // tasks.prev → self
163        data[32..39].copy_from_slice(b"systemd"); // comm
164        let mm_addr = vaddr + 0x200;
165        data[48..56].copy_from_slice(&mm_addr.to_le_bytes()); // mm → mm_struct
166
167        // mm_struct at +0x200
168        data[0x200..0x208].copy_from_slice(&0x1000u64.to_le_bytes()); // pgd (unused here)
169        let vma1_addr = vaddr + 0x300;
170        data[0x208..0x210].copy_from_slice(&vma1_addr.to_le_bytes()); // mmap → first VMA
171
172        // VMA #1 at +0x300: code segment (r-x)
173        data[0x300..0x308].copy_from_slice(&0x0040_0000u64.to_le_bytes()); // vm_start
174        data[0x308..0x310].copy_from_slice(&0x0040_1000u64.to_le_bytes()); // vm_end
175        let vma2_addr = vaddr + 0x400;
176        data[0x310..0x318].copy_from_slice(&vma2_addr.to_le_bytes()); // vm_next → VMA #2
177        data[0x318..0x320].copy_from_slice(&0x5u64.to_le_bytes()); // vm_flags: r-x
178        data[0x320..0x328].copy_from_slice(&0u64.to_le_bytes()); // vm_pgoff
179        data[0x328..0x330].copy_from_slice(&0x9999u64.to_le_bytes()); // vm_file (non-null)
180
181        // VMA #2 at +0x400: heap (rw-)
182        data[0x400..0x408].copy_from_slice(&0x7FFF_0000u64.to_le_bytes()); // vm_start
183        data[0x408..0x410].copy_from_slice(&0x7FFF_2000u64.to_le_bytes()); // vm_end
184        data[0x410..0x418].copy_from_slice(&0u64.to_le_bytes()); // vm_next: NULL (end)
185        data[0x418..0x420].copy_from_slice(&0x3u64.to_le_bytes()); // vm_flags: rw-
186        data[0x420..0x428].copy_from_slice(&0u64.to_le_bytes()); // vm_pgoff
187        data[0x428..0x430].copy_from_slice(&0u64.to_le_bytes()); // vm_file: NULL (anon)
188
189        let reader = make_test_reader(&data, vaddr, paddr);
190        let vmas = walk_maps(&reader).unwrap();
191
192        assert_eq!(vmas.len(), 2);
193
194        assert_eq!(vmas[0].pid, 1);
195        assert_eq!(vmas[0].comm, "systemd");
196        assert_eq!(vmas[0].start, 0x0040_0000);
197        assert_eq!(vmas[0].end, 0x0040_1000);
198        assert!(vmas[0].flags.read);
199        assert!(!vmas[0].flags.write);
200        assert!(vmas[0].flags.exec);
201        assert!(vmas[0].file_backed);
202
203        assert_eq!(vmas[1].start, 0x7FFF_0000);
204        assert_eq!(vmas[1].end, 0x7FFF_2000);
205        assert!(vmas[1].flags.read);
206        assert!(vmas[1].flags.write);
207        assert!(!vmas[1].flags.exec);
208        assert!(!vmas[1].file_backed);
209    }

212    fn walk_maps_skips_kernel_threads() {
213        // Kernel threads have mm == NULL — should produce no VMAs
214        let vaddr: u64 = 0xFFFF_8000_0010_0000;
215        let paddr: u64 = 0x0080_0000;
216        let mut data = vec![0u8; 4096];
217

221        data[16..24].copy_from_slice(&tasks_addr.to_le_bytes());
222        data[24..32].copy_from_slice(&tasks_addr.to_le_bytes());
223        data[32..41].copy_from_slice(b"swapper/0");
224        data[48..56].copy_from_slice(&0u64.to_le_bytes()); // mm = NULL
225
226        let reader = make_test_reader(&data, vaddr, paddr);
227        let vmas = walk_maps(&reader).unwrap();
228
229        assert!(vmas.is_empty());
230    }
