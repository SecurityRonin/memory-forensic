152/// Parse a `--pid N` flag out of plugin-specific trailing args.
153pub fn parse_pid(plugin_args: &[String]) -> Option<u64> {
154    let mut iter = plugin_args.iter();
155    while let Some(arg) = iter.next() {
156        if arg == "--pid" {
157            return iter.next().and_then(|v| v.parse().ok());
158        }
159        if let Some(val) = arg.strip_prefix("--pid=") {
160            return val.parse().ok();
161        }
162    }
163    None
164}

166/// Exec the real `vol` binary with the same arguments.
167/// Returns an error if `vol` is not found in PATH.
168pub fn proxy_to_vol(
169    dump: &Path,
170    renderer: crate::VolRenderer,
171    output_dir: Option<&Path>,
172    quiet: bool,
173    symbol_dirs: &[PathBuf],
174    plugin_dirs: &[PathBuf],
175    plugin_args: &[String],
176) -> anyhow::Result<()> {
177    let plugin_name = plugin_args.first().map_or("(unknown)", std::string::String::as_str);
178
179    let mut cmd = std::process::Command::new("vol");
180    cmd.arg("-f").arg(dump);
181
182    // Forward renderer if non-default
183    match renderer {
184        crate::VolRenderer::Json  => { cmd.args(["-r", "json"]); }
185        crate::VolRenderer::Csv   => { cmd.args(["-r", "csv"]); }
186        crate::VolRenderer::Pretty => { cmd.args(["-r", "pretty"]); }
187        crate::VolRenderer::Html  => { cmd.args(["-r", "html"]); }
188        crate::VolRenderer::Xlsx  => { cmd.args(["-r", "xlsx"]); }
189        crate::VolRenderer::Dot   => { cmd.args(["-r", "dote"]); }
190        crate::VolRenderer::Text  => {}  // default, omit
191    }
192    if let Some(od) = output_dir {
193        cmd.arg("-o").arg(od);
194    }
195    if quiet {
196        cmd.arg("-q");
197    }
198    for sd in symbol_dirs {
199        cmd.arg("-s").arg(sd);
200    }
201    for pd in plugin_dirs {
202        cmd.arg("-p").arg(pd);
203    }
204    cmd.args(plugin_args);
205
206    let result = cmd.status();
207    match result {
208        Ok(status) => {
209            if !status.success() {
210                std::process::exit(status.code().unwrap_or(1));
211            }
212            Ok(())
213        }
214        Err(e) if e.kind() == std::io::ErrorKind::NotFound => {
215            anyhow::bail!(
216                "Plugin '{plugin_name}' is not natively implemented in memf \
217                 and `vol` was not found in PATH.\n\
218                 Install volatility3 (`pip install volatility3`) to enable proxy support,\n\
219                 or wait for native implementation in a future memf release."
220            )
221        }
222        Err(e) => anyhow::bail!("failed to exec vol: {e}"),
223    }
224}

230// NOTE: format_vol3_filetime returns "N/A" for zero, while the native
231// format_filetime in main.rs returns "-". Both are intentional — vol3 uses
232// "N/A" in its own text output; the native formatter uses "-" for compactness.
233
234/// Format a Windows FILETIME as `"YYYY-MM-DD HH:MM:SS.ffffff"` (vol3 style,
235/// microsecond precision). Returns `"N/A"` for zero (unset field).
236// h/m/s are standard date-time component abbreviations in this formatter.
237#[allow(clippy::many_single_char_names)]
238fn format_vol3_filetime(ft: u64) -> String {
239    if ft == 0 {
240        return "N/A".to_string();
241    }
242    if ft < forensicnomicon::temporal::FILETIME_EPOCH_OFFSET {
243        return format!("pre-1970 ({ft:#x})");
244    }
245    let total_us = (ft - forensicnomicon::temporal::FILETIME_EPOCH_OFFSET) / 10; // 100-ns → microseconds
246    let unix_secs = total_us / 1_000_000;
247    let us = total_us % 1_000_000;
248
249    let days = unix_secs / 86400;
250    let rem  = unix_secs % 86400;
251    let h = rem / 3600;
252    let m = (rem % 3600) / 60;
253    let s = rem % 60;
254    let (y, mo, d) = days_to_ymd(days);
255    format!("{y:04}-{mo:02}-{d:02} {h:02}:{m:02}:{s:02}.{us:06}")
256}

258/// Convert days since Unix epoch (1970-01-01) to (year, month, day).
259/// Algorithm: civil_from_days (Howard Hinnant).
260fn days_to_ymd(days: u64) -> (u32, u32, u32) {
261    let z = days as i64 + 719468;
262    let era = z.div_euclid(146097);
263    let doe = z.rem_euclid(146097) as u64;
264    let yoe = (doe - doe / 1460 + doe / 36524 - doe / 146096) / 365;
265    let y = yoe as i64 + era * 400;
266    let doy = doe - (365 * yoe + yoe / 4 - yoe / 100);
267    let mp = (5 * doy + 2) / 153;
268    let d = doy - (153 * mp + 2) / 5 + 1;
269    let m = if mp < 10 { mp + 3 } else { mp - 9 };
270    let y = if m <= 2 { y + 1 } else { y };
271    (y as u32, m as u32, d as u32)
272}

274/// Helper to encode a FILETIME for JSON: `null` when zero, quoted string otherwise.
275fn ft_json(ft: u64) -> serde_json::Value {
276    if ft == 0 {
277        serde_json::Value::Null
278    } else {
279        serde_json::Value::String(format_vol3_filetime(ft))
280    }
281}

283/// Format `&[WinProcessInfo]` as a vol3-exact JSON array string.
284/// Uses serde_json to guarantee correct escaping of all string fields.
285fn vol3_processes_json(procs: &[memf_windows::WinProcessInfo]) -> String {
286    let arr: Vec<serde_json::Value> = procs.iter().map(|p| {
287        serde_json::json!({
288            "PID":           p.pid,
289            "PPID":          p.ppid,
290            "ImageFileName": p.image_name,
291            "Offset(V)":     format!("{:#x}", p.vaddr),
292            "Threads":       p.thread_count,
293            "Handles":       p.handle_count,
294            "SessionId":     p.session_id,
295            "Wow64":         p.is_wow64,
296            "CreateTime":    ft_json(p.create_time),
297            "ExitTime":      ft_json(p.exit_time),
298            "__children":    serde_json::Value::Array(vec![]),
299        })
300    }).collect();
301    serde_json::to_string_pretty(&arr).unwrap_or_default()
302}

304/// Format `&[WinProcessInfo]` as a vol3-style TSV table string.
305/// String fields are sanitized — tabs and newlines are replaced with spaces.
306fn vol3_processes_text(procs: &[memf_windows::WinProcessInfo]) -> String {
307    let mut out = "PID\tPPID\tImageFileName\tOffset(V)\tThreads\tHandles\tSessionId\tWow64\tCreateTime\tExitTime\n".to_string();
308    for p in procs {
309        let create = format_vol3_filetime(p.create_time);
310        let exit   = ft_json(p.exit_time).as_str().map_or_else(|| "N/A".to_string(), str::to_string);
311        let _ = writeln!(
312            out,
313            "{}\t{}\t{}\t{:#x}\t{}\t{}\t{}\t{}\t{}\t{}",
314            p.pid, p.ppid, tsv_safe(&p.image_name), p.vaddr,
315            p.thread_count, p.handle_count, p.session_id,
316            p.is_wow64, create, exit,
317        );
318    }
319    out
320}

342/// Format `&[WinConnectionInfo]` as vol3-style TSV.
343/// String fields are sanitized — tabs and newlines replaced with spaces.
344fn vol3_connections_text(conns: &[memf_windows::WinConnectionInfo]) -> String {

354            c.local_port,
355            tsv_safe(&c.remote_addr),
356            c.remote_port,
357            c.state,
358            c.pid,
359            tsv_safe(&c.process_name),
360            created,
361        );
362    }
363    out

366/// Format `&[WinPsTreeEntry]` as a vol3-style TSV tree (child rows prefixed with `*`).
367fn vol3_pstree_text(entries: &[memf_windows::WinPsTreeEntry]) -> String {
368    let mut out = "PID\tPPID\tImageFileName\tOffset(V)\tThreads\tHandles\tSessionId\tWow64\tCreateTime\tExitTime\n".to_string();
369    for e in entries {
370        let prefix = "*".repeat(e.depth as usize);
371        let p = &e.process;
372        let create = format_vol3_filetime(p.create_time);
373        let exit   = ft_json(p.exit_time).as_str().map_or_else(|| "N/A".to_string(), str::to_string);
374        let _ = writeln!(
375            out,
376            "{prefix}{pid}\t{ppid}\t{name}\t{offset:#x}\t{threads}\t{handles}\t{session}\t{wow64}\t{create}\t{exit}",
377            pid     = p.pid,
378            ppid    = p.ppid,
379            name    = tsv_safe(&p.image_name),
380            offset  = p.vaddr,
381            threads = p.thread_count,
382            handles = p.handle_count,
383            session = p.session_id,
384            wow64   = p.is_wow64,
385        );
386    }
387    out
388}
