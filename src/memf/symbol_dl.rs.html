1//! ISF symbol auto-download from the community server.
2//!
3//! Cache directory: `~/.cache/memf/symbols/`
4//! Server: see [`forensicnomicon::toolchain::VOLATILITY3_VOLATILITY3_ISF_SERVER`]
5//! URL pattern: `<server>/windows/<pdb_name>/<GUID><AGE>.json.xz`

11/// Returns the default ISF cache directory (`~/.cache/memf/symbols/`).
12pub fn default_cache_dir() -> PathBuf {
13    dirs_next_cache().join("memf/symbols")
14}
15
16/// Returns a cache directory rooted at `base` (for testing without touching `~/.cache`).
17#[cfg(test)]
18pub fn cache_dir_for_testing(base: &Path) -> PathBuf {
19    base.join("memf/symbols")
20}

22/// Create the cache directory (and all parents) if it does not exist.
23pub fn ensure_cache_dir(dir: &Path) -> anyhow::Result<()> {
24    std::fs::create_dir_all(dir)?;
25    Ok(())
26}

28/// Build the full ISF download URL for a given PDB name + GUID + age.
29///
30/// `guid` must be an uppercase hex string without hyphens (32 chars).
31/// `age` is the PDB age (decimal).
32pub fn build_isf_url(pdb_name: &str, guid: &str, age: u32) -> String {
33    format!("{VOLATILITY3_ISF_SERVER}/windows/{pdb_name}/{guid}{age}.json.xz")
34}
35
36/// Returns the filename used to cache an ISF for a given pdb+guid+age.
37pub fn cache_filename(pdb_name: &str, guid: &str, age: u32) -> String {
38    format!("{pdb_name}-{guid}{age}.json")
39}
40
41/// Check if an ISF is already cached; returns the path if so.
42pub fn find_cached(cache_dir: &Path, pdb_name: &str, guid: &str, age: u32) -> Option<PathBuf> {
43    let path = cache_dir.join(cache_filename(pdb_name, guid, age));
44    if path.is_file() { Some(path) } else { None }
45}

47/// Download an ISF from the community server, decompress it, and store it in `cache_dir`.
48/// Returns the path to the cached `.json` file on success.
49pub fn download_isf(
50    cache_dir: &Path,
51    pdb_name: &str,
52    guid: &str,
53    age: u32,
54) -> anyhow::Result<PathBuf> {
55    let url = build_isf_url(pdb_name, guid, age);
56    let resp = ureq::get(&url)
57        .call()
58        .map_err(|e| anyhow::anyhow!("ISF download failed for {url}: {e}"))?;
59
60    // Check HTTP status before reading the body — a 404 body is HTML, not XZ,
61    // and would produce a confusing "XZ decompression failed" error otherwise.
62    let status = resp.status();
63    if status != 200 {
64        anyhow::bail!("ISF server returned HTTP {status} for {url}\nHint: check the PDB GUID/age match the dump's kernel version.");
65    }
66
67    let compressed = resp
68        .into_body()
69        .read_to_vec()
70        .map_err(|e| anyhow::anyhow!("failed to read ISF response body from {url}: {e}"))?;
71
72    // Decompress .xz stream → raw JSON bytes
73    let mut json_bytes = Vec::new();
74    lzma_rs::xz_decompress(&mut compressed.as_slice(), &mut json_bytes)
75        .map_err(|e| anyhow::anyhow!("XZ decompression failed for {url}: {e}"))?;
76
77    ensure_cache_dir(cache_dir)?;
78    let dest = cache_dir.join(cache_filename(pdb_name, guid, age));
79    // Write atomically: write to a temp file then rename, so a partial download
80    // never leaves a corrupt cache entry that would be served on the next run.
81    let tmp = dest.with_extension("json.tmp");
82    std::fs::write(&tmp, &json_bytes)
83        .map_err(|e| anyhow::anyhow!("failed to write temp ISF to {}: {e}", tmp.display()))?;
84    std::fs::rename(&tmp, &dest)
85        .map_err(|e| anyhow::anyhow!("failed to install ISF to {}: {e}", dest.display()))?;
86    Ok(dest)
87}

89/// Discover a cached ISF or auto-download one, searching `cache_dir` first.
90///
91/// `allow_network` gates the phone-home: when `false` (offline mode) a cache
92/// miss returns an error instead of reaching the ISF community server. A cache
93/// hit always resolves regardless of `allow_network`.
94pub fn resolve_isf(
95    cache_dir: &Path,
96    pdb_name: &str,
97    guid: &str,
98    age: u32,
99    allow_network: bool,
100) -> anyhow::Result<PathBuf> {
101    if let Some(cached) = find_cached(cache_dir, pdb_name, guid, age) {
102        return Ok(cached);
103    }
104    if !allow_network {
105        anyhow::bail!(
106            "{pdb_name} not in symbol cache and network download disabled (offline mode)\n\
107             Hint: run `memf symserver <dump>` while online to populate the cache, then retry."
108        );
109    }
110    download_isf(cache_dir, pdb_name, guid, age)
111}

113/// Resolve a symbol file for the kernel, trying both sources in order.
114///
115/// 1. **ISF first** — a pre-built Volatility-3 ISF from the community server
116///    (cached or downloaded). Fast, no conversion.
117/// 2. **PDB fallback** — when no ISF is available, a PDB straight from the
118///    Microsoft symbol server (`msdl.microsoft.com`), which covers *every*
119///    Windows build the community ISF set may lack. The returned `.pdb` path is
120///    consumed by the same loader (it dispatches `.pdb` → `PdbResolver`).
121///
122/// `allow_network` gates both downloads; offline resolution is cache-only.
123/// Returns the path to whichever symbol file was found (`.json` ISF or `.pdb`).
124pub fn resolve_symbols(
125    cache_dir: &Path,
126    pdb_name: &str,
127    guid: &str,
128    age: u32,
129    allow_network: bool,
130) -> anyhow::Result<PathBuf> {
131    // 1. Prefer a pre-built ISF.
132    if let Ok(isf) = resolve_isf(cache_dir, pdb_name, guid, age, allow_network) {
133        return Ok(isf);
134    }
135    // 2. Fall back to a PDB from the Microsoft symbol server.
136    let pdb_cached = memf_symbols::symserver::cache_path(cache_dir, pdb_name, guid, age);
137    if pdb_cached.exists() {
138        return Ok(pdb_cached);
139    }
140    if !allow_network {
141        anyhow::bail!(
142            "no symbols for {pdb_name}: neither a community ISF nor a Microsoft PDB is cached, \
143             and network download is disabled (offline mode)"
144        );
145    }
146    let client =
147        memf_symbols::symserver::SymbolServerClient::new(memf_symbols::symserver::default_server_url(), cache_dir);
148    client
149        .get_pdb(pdb_name, guid, age)
150        .map_err(|e| anyhow::anyhow!("ISF unavailable and PDB download from the symbol server failed: {e}"))
151}

157fn dirs_next_cache() -> PathBuf {
158    // Replicate XDG_CACHE_HOME / macOS ~/Library/Caches without pulling in `dirs`.
159    #[cfg(target_os = "macos")]
160    {
161        if let Some(home) = std::env::var_os("HOME") {
162            return PathBuf::from(home).join("Library/Caches");
163        }
164    }
165    if let Some(xdg) = std::env::var_os("XDG_CACHE_HOME") {
166        return PathBuf::from(xdg);
167    }
168    if let Some(home) = std::env::var_os("HOME") {
169        return PathBuf::from(home).join(".cache");
170    }
171    PathBuf::from(".cache")
172}
