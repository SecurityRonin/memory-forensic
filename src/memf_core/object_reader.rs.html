10/// Maximum number of iterations when walking a linked list (cycle protection).
11const MAX_LIST_ITERATIONS: usize = 100_000;
12
13/// Reads kernel objects from a physical memory dump using symbol information.
14///
15/// Combines a [`VirtualAddressSpace`] with a [`SymbolResolver`] to provide
16/// high-level access to kernel data structures like task_struct, modules, etc.
17pub struct ObjectReader<P: PhysicalMemoryProvider> {
18    vas: VirtualAddressSpace<P>,
19    symbols: Box<dyn SymbolResolver>,
20    /// Kernel image base virtual address (KVO). Windows ISF symbols are RVAs
21    /// relative to this; it is added in [`Self::required_symbol`] so no caller
22    /// can obtain an un-rebased address. Zero for resolvers whose symbols are
23    /// already absolute VAs (e.g. Linux kallsyms), leaving them unchanged.
24    kernel_base: u64,
25}
26
27impl<P: PhysicalMemoryProvider> ObjectReader<P> {
28    /// Create a new object reader (kernel base unset → symbols used verbatim).
29    pub fn new(vas: VirtualAddressSpace<P>, symbols: Box<dyn SymbolResolver>) -> Self {
30        Self {
31            vas,
32            symbols,
33            kernel_base: 0,
34        }
35    }
36
37    /// Set the kernel image base VA (KVO) so RVA-based symbols are rebased to
38    /// real virtual addresses. Builder form; the default (0) is a no-op.
39    #[must_use]
40    pub fn with_kernel_base(mut self, kernel_base: u64) -> Self {
41        self.kernel_base = kernel_base;
42        self
43    }
44
45    /// Access the underlying symbol resolver.
46    pub fn symbols(&self) -> &dyn SymbolResolver {
47        self.symbols.as_ref()
48    }
49
50    /// Access the underlying virtual address space.
51    pub fn vas(&self) -> &VirtualAddressSpace<P> {
52        &self.vas
53    }
54
55    /// Create a new reader sharing the same physical memory and symbols but
56    /// using a different page table root (CR3). Useful for switching to a
57    /// process's user-mode address space.
58    pub fn with_cr3(&self, cr3: u64) -> Self
59    where
60        P: Clone,
61    {
62        let vas = VirtualAddressSpace::new(self.vas.physical().clone(), cr3, self.vas.mode());
63        Self {
64            vas,
65            symbols: self.symbols.clone_boxed(),
66            kernel_base: self.kernel_base,
67        }
68    }

70    /// Read a field from a struct at `base_vaddr` and interpret it as type `T`.
71    ///
72    /// Looks up the field offset from the symbol resolver, reads `size_of::<T>()`
73    /// bytes from virtual memory, and casts via `bytemuck::from_bytes`.
74    pub fn read_field<T: Pod + Default>(
75        &self,
76        base_vaddr: u64,
77        struct_name: &str,
78        field_name: &str,
79    ) -> Result<T> {
80        let offset = self
81            .symbols
82            .field_offset(struct_name, field_name)
83            .ok_or_else(|| Error::MissingSymbol(format!("{struct_name}.{field_name}")))?;
84
85        let size = std::mem::size_of::<T>();
86        let mut buf = vec![0u8; size];
87        self.vas
88            .read_virt(base_vaddr.wrapping_add(offset), &mut buf)?;
89
90        if buf.len() != size {
91            return Err(Error::SizeMismatch {
92                expected: size,
93                got: buf.len(),
94            });
95        }
96
97        Ok(*bytemuck::from_bytes::<T>(&buf))
98    }
99
100    /// Read a pointer (u64) from a struct field.
101    pub fn read_pointer(
102        &self,
103        base_vaddr: u64,
104        struct_name: &str,
105        field_name: &str,
106    ) -> Result<u64> {
107        self.read_field::<u64>(base_vaddr, struct_name, field_name)
108    }

110    /// Read a null-terminated string from virtual memory, up to `max_len` bytes.
111    pub fn read_string(&self, vaddr: u64, max_len: usize) -> Result<String> {
112        let mut buf = vec![0u8; max_len];
113        self.vas.read_virt(vaddr, &mut buf)?;
114
115        let end = buf.iter().position(|&b| b == 0).unwrap_or(buf.len());
116        Ok(String::from_utf8_lossy(&buf[..end]).into_owned())
117    }

119    /// Read a string from a struct field (the field contains inline char data, not a pointer).
120    pub fn read_field_string(
121        &self,
122        base_vaddr: u64,
123        struct_name: &str,
124        field_name: &str,
125        max_len: usize,
126    ) -> Result<String> {
127        let offset = self
128            .symbols
129            .field_offset(struct_name, field_name)
130            .ok_or_else(|| Error::MissingSymbol(format!("{struct_name}.{field_name}")))?;
131
132        self.read_string(base_vaddr.wrapping_add(offset), max_len)
133    }

135    /// Walk a Linux `list_head` doubly-linked list.
136    ///
137    /// Starting from `head_vaddr` (the address of the list_head embedded in the
138    /// head/sentinel node), follows `next` pointers and returns the virtual address
139    /// of each containing struct (using container_of logic with `list_field` offset).
140    ///
141    /// Stops when the walk loops back to `head_vaddr` or hits `MAX_LIST_ITERATIONS`.
142    pub fn walk_list(
143        &self,
144        head_vaddr: u64,
145        struct_name: &str,
146        list_field: &str,
147    ) -> Result<Vec<u64>> {
148        self.walk_list_with(head_vaddr, "list_head", "next", struct_name, list_field)
149    }
150
151    /// Walk a doubly-linked list with configurable list struct and field names.
152    ///
153    /// This is a generalized version of [`walk_list`](Self::walk_list) that works
154    /// with any linked-list structure, not just Linux `list_head`.
155    ///
156    /// For example, Windows uses `_LIST_ENTRY` with `Flink`/`Blink` fields
157    /// instead of `list_head` with `next`/`prev`.
158    ///
159    /// # Arguments
160    /// * `head_vaddr` — virtual address of the list head (sentinel node)
161    /// * `list_struct` — name of the list-link struct (e.g., `"list_head"`, `"_LIST_ENTRY"`)
162    /// * `next_field` — name of the forward pointer field (e.g., `"next"`, `"Flink"`)
163    /// * `container_struct` — name of the containing struct (e.g., `"_EPROCESS"`)
164    /// * `list_field` — name of the list-link field in the container struct (e.g., `"ActiveProcessLinks"`)
165    pub fn walk_list_with(
166        &self,
167        head_vaddr: u64,
168        list_struct: &str,
169        next_field: &str,
170        container_struct: &str,
171        list_field: &str,
172    ) -> Result<Vec<u64>> {
173        let list_offset = self
174            .symbols
175            .field_offset(container_struct, list_field)
176            .ok_or_else(|| Error::MissingSymbol(format!("{container_struct}.{list_field}")))?;
177
178        let next_offset = self
179            .symbols
180            .field_offset(list_struct, next_field)
181            .ok_or_else(|| Error::MissingSymbol(format!("{list_struct}.{next_field}")))?;
182
183        // Read the first forward pointer from head
184        let mut current = self.read_u64_at(head_vaddr.wrapping_add(next_offset))?;
185
186        let mut result = Vec::new();
187
188        for _ in 0..MAX_LIST_ITERATIONS {
189            // If we've looped back to head, the walk is complete
190            if current == head_vaddr {
191                return Ok(result);
192            }
193
194            // Smear tolerance: a live-acquired dump can contain a torn-down node
195            // whose link reads 0 (a null terminus). Stop rather than dereference
196            // null or fabricate a container from `0 - offset`.
197            if current == 0 {
198                return Ok(result);
199            }
200
201            // Peek-before-record: read this node's forward pointer FIRST. If its
202            // LIST_ENTRY page is not mapped, `current` is not a real node — a
203            // torn-down node's link can hold garbage (e.g. the user-half value
204            // 0x5a289000 seen on DESKTOP-SDN1RPT.mem, which is canonical but
205            // unmapped). Terminate without fabricating a container that a later
206            // field read would fault on. This works for BOTH kernel object lists
207            // and user-space lists (PEB/LDR modules), so it must NOT assume a
208            // kernel-half address.
209            let Ok(next) = self.read_u64_at(current.wrapping_add(next_offset)) else {
210                return Ok(result);
211            };
212
213            // container_of: subtract list_offset to get the containing struct base
214            result.push(current.wrapping_sub(list_offset));
215            current = next;
216        }
217
218        Err(Error::ListCycle(MAX_LIST_ITERATIONS))
219    }
220
221    /// Walk a doubly-linked list in BOTH directions and return the union of
222    /// containers, deduplicated (forward order first, then backward-only nodes).
223    ///
224    /// On a live-acquired dump a single torn-down node can break the forward
225    /// (`next_field`/Flink) chain, orphaning every node beyond it from a
226    /// forward-only walk — yet those nodes remain reachable from the head via the
227    /// backward (`prev_field`/Blink) chain. Walking both directions recovers them
228    /// without resorting to pool-tag scanning. (A node unlinked from *both*
229    /// directions — full DKOM hiding — still requires a pool scan.)
230    pub fn walk_list_bidirectional(
231        &self,
232        head_vaddr: u64,
233        list_struct: &str,
234        next_field: &str,
235        prev_field: &str,
236        container_struct: &str,
237        list_field: &str,
238    ) -> Result<Vec<u64>> {
239        let mut forward =
240            self.walk_list_with(head_vaddr, list_struct, next_field, container_struct, list_field)?;
241        let backward =
242            self.walk_list_with(head_vaddr, list_struct, prev_field, container_struct, list_field)?;
243        let mut seen: std::collections::HashSet<u64> = forward.iter().copied().collect();
244        for container in backward {
245            if seen.insert(container) {
246                forward.push(container);
247            }
248        }
249        Ok(forward)
250    }

252    /// Read `len` raw bytes from virtual memory at `vaddr`.
253    pub fn read_bytes(&self, vaddr: u64, len: usize) -> Result<Vec<u8>> {
254        let mut buf = vec![0u8; len];
255        self.vas.read_virt(vaddr, &mut buf)?;
256        Ok(buf)

259    /// Resolve a global kernel symbol to a real virtual address, returning an
260    /// error if absent. The resolver yields an RVA (Windows ISF); the kernel
261    /// base (KVO) is added here so callers cannot obtain an un-rebased address.
262    pub fn required_symbol(&self, name: &str) -> Result<u64> {
263        self.symbols()
264            .symbol_address(name)
265            .map(|rva| self.kernel_base.wrapping_add(rva))
266            .ok_or_else(|| Error::MissingSymbol(name.to_owned()))
267    }
268
269    /// Resolve a struct field offset, returning an error if absent.
270    pub fn required_field_offset(&self, struct_name: &str, field_name: &str) -> Result<usize> {
271        self.symbols()
272            .field_offset(struct_name, field_name)
273            .map(|v| v as usize)
274            .ok_or_else(|| Error::MissingSymbol(format!("{struct_name}.{field_name}")))
275    }

277    fn read_u64_at(&self, vaddr: u64) -> Result<u64> {
278        let mut buf = [0u8; 8];
279        self.vas.read_virt(vaddr, &mut buf)?;
280        Ok(u64::from_le_bytes(buf))
281    }

283    /// Returns a lazy iterator over a kernel linked list (Linux `list_head`).
284    ///
285    /// Yields the virtual address of each containing struct (container_of adjusted),
286    /// same as [`walk_list`](Self::walk_list). Unlike `walk_list`, this does not
287    /// allocate a `Vec` — entries are yielded one at a time. Use `.take(n)` for
288    /// early termination or filter with `.filter_map`.
289    ///
290    /// # Errors
291    ///
292    /// Each yielded item is `Result<u64>`. The iterator stops (returning `None`) on
293    /// cycle or when the list loops back to `head_vaddr`. If a pointer read fails,
294    /// the failing `Err` is yielded as the last item.
295    pub fn iter_list<'a>(
296        &'a self,
297        head_vaddr: u64,
298        container_struct: &'a str,
299        list_field: &'a str,
300    ) -> ListIter<'a, P> {
301        let list_offset = self
302            .symbols
303            .field_offset(container_struct, list_field)
304            .unwrap_or(0);
305        let next_offset = self
306            .symbols
307            .field_offset("list_head", "next")
308            .unwrap_or(0);
309
310        let current = match self.read_u64_at(head_vaddr.wrapping_add(next_offset)) {
311            Ok(v) => v,
312            Err(_) => head_vaddr, // will immediately return None (current == head)
313        };
314
315        ListIter {
316            reader: self,
317            head_vaddr,
318            current,
319            list_offset,
320            next_offset,
321            seen: std::collections::HashSet::new(),
322            done: false,
323        }
324    }
325}

327/// Streaming iterator over a kernel doubly-linked list.
328///
329/// Returned by [`ObjectReader::iter_list`]. Yields the virtual address of each
330/// container struct (using container_of logic, same as [`ObjectReader::walk_list`]).
331pub struct ListIter<'a, P: PhysicalMemoryProvider> {
332    reader: &'a ObjectReader<P>,
333    head_vaddr: u64,
334    current: u64,
335    list_offset: u64,
336    next_offset: u64,
337    seen: std::collections::HashSet<u64>,
338    done: bool,
339}
340
341impl<P: PhysicalMemoryProvider> Iterator for ListIter<'_, P> {
342    type Item = crate::Result<u64>;
343
344    fn next(&mut self) -> Option<Self::Item> {
345        if self.done {
346            return None;
347        }
348        // Termination: looped back to head
349        if self.current == self.head_vaddr {
350            return None;
351        }
352        // Cycle detection
353        if !self.seen.insert(self.current) {
354            self.done = true;
355            return None; // silently stop on detected cycle without valid entry
356        }
357        if self.seen.len() > MAX_LIST_ITERATIONS {
358            self.done = true;
359            return Some(Err(crate::Error::ListCycle(MAX_LIST_ITERATIONS)));
360        }
361
362        // container_of: subtract list_offset to get containing struct base
363        let container = self.current.wrapping_sub(self.list_offset);
364
365        // Advance: follow next pointer
366        match self
367            .reader
368            .read_u64_at(self.current.wrapping_add(self.next_offset))
369        {
370            Ok(next) => self.current = next,
371            Err(e) => {
372                self.done = true;
373                return Some(Err(e));
374            }
375        }
376
377        Some(Ok(container))
378    }
379}

505    #[test]
506    fn walk_list_simple() {
507        // Create a simplified task_struct layout:
508        //   offset 0: pid (u32)
509        //   offset 8: tasks.next (u64)  -- list_head embedded at offset 8
510        //   offset 16: comm (16 bytes)
511        //   struct size: 128
512        let isf = IsfBuilder::new()
513            .add_struct("task_struct", 128)
514            .add_field("task_struct", "pid", 0, "int")
515            .add_field("task_struct", "tasks", 8, "list_head")
516            .add_field("task_struct", "comm", 16, "char")
517            .add_struct("list_head", 16)
518            .add_field("list_head", "next", 0, "pointer")
519            .add_field("list_head", "prev", 8, "pointer");
520
521        // Physical layout:
522        //   paddr 0x0080_0000: head task_struct (init_task)
523        //   paddr 0x0080_1000: task A
524        //   paddr 0x0080_2000: task B
525        //
526        // Circular list:
527        //   head.tasks.next -> A.tasks -> B.tasks -> head.tasks
528        let head_paddr: u64 = 0x0080_0000;
529        let a_paddr: u64 = 0x0080_1000;
530        let b_paddr: u64 = 0x0080_2000;
531
532        let head_vaddr: u64 = 0xFFFF_8000_0010_0000;
533        let a_vaddr: u64 = 0xFFFF_8000_0010_1000;
534        let b_vaddr: u64 = 0xFFFF_8000_0010_2000;
535
536        let list_offset: u64 = 8; // tasks field offset
537
538        // head.tasks.next = &A.tasks
539        // A.tasks.next = &B.tasks
540        // B.tasks.next = &head.tasks
541        let ptb = PageTableBuilder::new()
542            .map_4k(head_vaddr, head_paddr, flags::WRITABLE)
543            .map_4k(a_vaddr, a_paddr, flags::WRITABLE)
544            .map_4k(b_vaddr, b_paddr, flags::WRITABLE)
545            // head: pid=0, tasks.next = a_vaddr + list_offset
546            .write_phys_u64(head_paddr, 0) // pid
547            .write_phys_u64(head_paddr + list_offset, a_vaddr + list_offset) // tasks.next
548            // A: pid=100, tasks.next = b_vaddr + list_offset
549            .write_phys_u64(a_paddr, 100) // pid
550            .write_phys_u64(a_paddr + list_offset, b_vaddr + list_offset) // tasks.next
551            // B: pid=200, tasks.next = head_vaddr + list_offset (loops back)
552            .write_phys_u64(b_paddr, 200) // pid
553            .write_phys_u64(b_paddr + list_offset, head_vaddr + list_offset); // tasks.next
554
555        let reader = make_reader(&isf, ptb);
556
557        let containers = reader
558            .walk_list(head_vaddr + list_offset, "task_struct", "tasks")
559            .unwrap();
560        assert_eq!(containers.len(), 2);
561        assert_eq!(containers[0], a_vaddr);
562        assert_eq!(containers[1], b_vaddr);
563    }
